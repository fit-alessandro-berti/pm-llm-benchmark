import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Define all individual activity Transitions
SR = Transition(label="SR")
CS = Transition(label="CS")
DA = Transition(label="DA")
ACI = Transition(label="ACI")
BNC = Transition(label="BNC")
# Use a variable name compatible with Python syntax for "D&R"
D_R = Transition(label="D&R") 
CD = Transition(label="CD")
MD = Transition(label="MD")
QA = Transition(label="QA")
SA = Transition(label="SA")
GLA = Transition(label="GLA")

# 2. Define the composite parts of the process

# Initial Part: SR followed by CS
initial_part = StrictPartialOrder(nodes=[SR, CS])
initial_part.order.add_edge(SR, CS)

# Parallel Part: DA and ACI in parallel
# This StrictPartialOrder node represents the concurrent execution block.
parallel_part = StrictPartialOrder(nodes=[DA, ACI])
# No edges needed between DA and ACI as they are concurrent.

# Loop Part: BNC followed by optional D&R, repeating BNC
# OperatorPOWL represents the loop structure.
loop_part = OperatorPOWL(operator=Operator.LOOP, children=[BNC, D_R])

# Deployment Choice Part: XOR between CD and MD
# OperatorPOWL represents the exclusive choice.
deployment_part = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# QA/SA Part: QA and SA in parallel
# Another StrictPartialOrder for concurrency.
qa_sa_part = StrictPartialOrder(nodes=[QA, SA])
# No edges needed between QA and SA as they are concurrent.

# Final node GLA is already defined as a Transition.

# 3. Define the root StrictPartialOrder connecting the main parts

# The root node contains all the major process steps/phases as its children.
# These children can be simple Transitions or complex POWL models themselves.
root = StrictPartialOrder(nodes=[
    initial_part,
    parallel_part,
    loop_part,
    deployment_part,
    qa_sa_part,
    GLA  # GLA is a direct node in the root sequence
])

# 4. Add the ordering constraints between the main parts in the root model
# This defines the high-level sequence of the process phases.
root.order.add_edge(initial_part, parallel_part) # CS must complete before DA/ACI start
root.order.add_edge(parallel_part, loop_part)   # DA and ACI must complete before loop starts
root.order.add_edge(loop_part, deployment_part) # Loop must complete before deployment choice
root.order.add_edge(deployment_part, qa_sa_part) # Deployment must complete before QA/SA start
root.order.add_edge(qa_sa_part, GLA)            # QA and SA must complete before GLA starts

# The 'root' variable now holds the complete POWL model for the described process.
# You can print it or use it with other pm4py functions if needed.
