import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Define individual activities (Transitions) based on the scenario description
sr = Transition("SR")  # Service Request Verification
cs = Transition("CS")  # Contract Setup
da = Transition("DA")  # Datacenter Allocation
aci = Transition("ACI") # Access Control Initialization
bnc = Transition("BNC") # Base Network Configuration
d_r = Transition("D&R") # Debug & Reconfigure
cd = Transition("CD")  # Containerized Deployment
md = Transition("MD")  # Monolithic Deployment
qa = Transition("QA")  # Quality Assurance
sa = Transition("SA")  # Security Audit
gla = Transition("GLA") # Go-Live Approval

# Define composite structures using Operators

# Loop structure: Execute BNC, then either exit or execute D&R and loop back to BNC.
# This matches the *(A, B) definition where A=BNC, B=D&R.
loop_config = OperatorPOWL(operator=Operator.LOOP, children=[bnc, d_r])

# Exclusive choice structure: Either CD or MD is chosen.
xor_deploy = OperatorPOWL(operator=Operator.XOR, children=[cd, md])

# Define the root structure for the entire process using StrictPartialOrder.
# All main activities and composite structures are nodes in this partial order.
root_nodes = [
    sr, cs, da, aci, loop_config, xor_deploy, qa, sa, gla
]
root = StrictPartialOrder(nodes=root_nodes)

# Add the ordering constraints (edges) between the nodes based on the process description:

# 1. SR must be completed before CS can start.
root.order.add_edge(sr, cs)

# 2. DA and ACI can start after CS is completed. They can run concurrently.
root.order.add_edge(cs, da)
root.order.add_edge(cs, aci)
# No edge between da and aci implies they are concurrent relative to each other.

# 3. The loop_config phase can only start after BOTH DA and ACI are completed.
# The partial order semantics ensure this: loop_config waits for all its predecessors.
root.order.add_edge(da, loop_config)
root.order.add_edge(aci, loop_config)

# 4. The deployment choice (xor_deploy) starts after the loop_config is completed (i.e., loop exited).
root.order.add_edge(loop_config, xor_deploy)

# 5. QA and SA can start after the deployment choice (xor_deploy) is completed.
# They can run concurrently.
root.order.add_edge(xor_deploy, qa)
root.order.add_edge(xor_deploy, sa)
# No edge between qa and sa implies they are concurrent relative to each other.

# 6. GLA can only start after BOTH QA and SA are completed.
# The partial order semantics ensure this: gla waits for all its predecessors.
root.order.add_edge(qa, gla)
root.order.add_edge(sa, gla)
