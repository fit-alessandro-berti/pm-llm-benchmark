```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- Define Activities (Transitions) ---
# These are based on the process description
parse_resume = Transition(label="ParseResume")
data_check = Transition(label="DataCheck") # Part of the loop condition
request_more_info = Transition(label="RequestMoreInfo") # Part of the loop body
skill_assessment = Transition(label="SkillAssessment")
cultural_fit_check = Transition(label="CulturalFitCheck") # Standard check
community_affiliation_check = Transition(label="CommunityAffiliationCheck") # Biased check
managerial_review = Transition(label="ManagerialReview")
final_decision = Transition(label="FinalDecision")

# --- Model 1: POWL Model with Potential Unfairness ---

# 1. Define the Loop for Data Completeness
# The loop structure is * ( A, B ), where A is executed, then a choice to exit or execute B and then A again.
# Here, A = DataCheck, B = RequestMoreInfo.
# So, DataCheck runs. If incomplete, RequestMoreInfo runs, then DataCheck runs again.
loop_data_completeness = OperatorPOWL(operator=Operator.LOOP, children=[data_check, request_more_info])

# 2. Define the XOR for Cultural Fit / Affiliation Check
# This XOR represents the choice between the standard path and the potentially biased path.
xor_cultural_step = OperatorPOWL(operator=Operator.XOR, children=[cultural_fit_check, community_affiliation_check])

# 3. Define the overall process structure as a sequence (StrictPartialOrder)
# The nodes are the main steps/operators in the process flow.
unfair_process_nodes = [
    parse_resume,
    loop_data_completeness,
    skill_assessment,
    xor_cultural_step,
    managerial_review,
    final_decision
]
unfair_powl_model = StrictPartialOrder(nodes=unfair_process_nodes)

# 4. Add edges to enforce the sequential order
unfair_powl_model.order.add_edge(parse_resume, loop_data_completeness)
unfair_powl_model.order.add_edge(loop_data_completeness, skill_assessment)
unfair_powl_model.order.add_edge(skill_assessment, xor_cultural_step)
unfair_powl_model.order.add_edge(xor_cultural_step, managerial_review)
unfair_powl_model.order.add_edge(managerial_review, final_decision)

# --- Model 2: POWL Model Without Unfairness ---

# 1. Reuse the Loop definition (same loop mechanism needed)
# loop_data_completeness is the same as in Model 1.

# 2. Define the overall process structure as a sequence (StrictPartialOrder)
# The nodes are similar, but the XOR is replaced by the standard CulturalFitCheck activity.
fair_process_nodes = [
    parse_resume,
    loop_data_completeness,
    skill_assessment,
    cultural_fit_check, # Replaces the XOR node
    managerial_review,
    final_decision
]
fair_powl_model = StrictPartialOrder(nodes=fair_process_nodes)

# 3. Add edges to enforce the sequential order
fair_powl_model.order.add_edge(parse_resume, loop_data_completeness)
fair_powl_model.order.add_edge(loop_data_completeness, skill_assessment)
fair_powl_model.order.add_edge(skill_assessment, cultural_fit_check) # Edge points to the single activity now
fair_powl_model.order.add_edge(cultural_fit_check, managerial_review)
fair_powl_model.order.add_edge(managerial_review, final_decision)
```
