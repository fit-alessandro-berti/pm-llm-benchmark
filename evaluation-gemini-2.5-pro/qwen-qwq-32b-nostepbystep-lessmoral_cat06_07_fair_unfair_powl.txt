import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition

from pm4py.objects.process_tree.obj import Operator

# Define the loop operator for data completeness check
# Loop structure: Execute data_check_1. If incomplete, execute request_info_1, then loop back to data_check_1.
data_check_loop_1 = OperatorPOWL(operator=Operator.LOOP, children=[data_check_1, request_info_1])

# Define the XOR operator for the biased cultural fit evaluation
# XOR choice: Either standard CulturalFitCheck or biased CommunityAffiliationCheck
cultural_xor_1 = OperatorPOWL(operator=Operator.XOR, children=[cultural_fit_check_1, community_affiliation_check_1])

# Define the nodes for the main partial order
nodes_unfair = [
    receive_app_1,
    data_check_loop_1,
    skill_assessment_1,
    cultural_xor_1,
    managerial_review_1,
    final_decision_1
]

# Create the StrictPartialOrder for the unfair process
unfair_process = StrictPartialOrder(nodes=nodes_unfair)

# Add order edges to define the sequence
unfair_process.order.add_edge(receive_app_1, data_check_loop_1)
unfair_process.order.add_edge(data_check_loop_1, skill_assessment_1)
unfair_process.order.add_edge(skill_assessment_1, cultural_xor_1)
unfair_process.order.add_edge(cultural_xor_1, managerial_review_1)
unfair_process.order.add_edge(managerial_review_1, final_decision_1)

print("--- POWL Model 1 (With Potential Unfairness) ---")
