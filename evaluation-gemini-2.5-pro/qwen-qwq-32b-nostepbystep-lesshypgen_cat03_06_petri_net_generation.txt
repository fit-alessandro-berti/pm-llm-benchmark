from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# 1. Initialize the Petri Net
net = PetriNet("InsuranceClaimManagement")

# 2. Create Places (representing states)
p_start = PetriNet.Place("p_start")                     # State: Claim Filed, Ready for Registration
p_after_CR = PetriNet.Place("p_after_CR")               # State: Registered, Ready for DV
p_after_DV = PetriNet.Place("p_after_DV")               # State: DV Done, Ready for FC or RDS
# p_after_RDS implicitly loops back to p_after_CR state via transition t_RDS
p_after_FC = PetriNet.Place("p_after_FC")               # State: FC Done, Ready for CA or II
# p_after_II implicitly loops back to p_after_DV state via transition t_II
p_after_CA = PetriNet.Place("p_after_CA")               # State: Assessed, Ready for AP or P
p_after_AP = PetriNet.Place("p_after_AP")               # State: Approved, Ready for P
p_after_P = PetriNet.Place("p_after_P")                 # State: Paid, Ready for N
p_after_N = PetriNet.Place("p_after_N")                 # State: Notified, Ready for CL
p_end = PetriNet.Place("p_end")                         # State: Closed

# Add places to the net
places = [p_start, p_after_CR, p_after_DV, p_after_FC, p_after_CA, p_after_AP, p_after_P, p_after_N, p_end]
for p in places:
    net.places.add(p)

# 3. Create Transitions (representing activities)
t_CR = PetriNet.Transition(name="CR", label="Claim Registration")
t_DV = PetriNet.Transition(name="DV", label="Document Verification")
t_RDS = PetriNet.Transition(name="RDS", label="Re-Document Submission")
t_FC = PetriNet.Transition(name="FC", label="Fraud Check")
t_II = PetriNet.Transition(name="II", label="In-depth Investigation")
t_CA = PetriNet.Transition(name="CA", label="Claim Assessment")
t_AP = PetriNet.Transition(name="AP", label="Approval")
t_P = PetriNet.Transition(name="P", label="Payment")
t_N = PetriNet.Transition(name="N", label="Notification")
t_CL = PetriNet.Transition(name="CL", label="Closure")

# Add transitions to the net
transitions = [t_CR, t_DV, t_RDS, t_FC, t_II, t_CA, t_AP, t_P, t_N, t_CL]
for t in transitions:
    net.transitions.add(t)

# 4. Add Arcs (defining the flow)
# Standard flow
petri_utils.add_arc_from_to(p_start, t_CR, net)
petri_utils.add_arc_from_to(t_CR, p_after_CR, net)
petri_utils.add_arc_from_to(p_after_CR, t_DV, net)
petri_utils.add_arc_from_to(t_DV, p_after_DV, net)

# DV Loop (RDS)
# Choice after DV: Either proceed to FC or loop back via RDS
petri_utils.add_arc_from_to(p_after_DV, t_RDS, net)      # If incomplete, start RDS
petri_utils.add_arc_from_to(t_RDS, p_after_CR, net)      # After RDS, go back before DV

petri_utils.add_arc_from_to(p_after_DV, t_FC, net)

# FC Loop (II)
# Choice after FC: Either proceed to CA or loop back via II
petri_utils.add_arc_from_to(p_after_FC, t_II, net)       # If doubts, start II
petri_utils.add_arc_from_to(t_II, p_after_DV, net)       # After II, go back before FC (needs FC check again)

petri_utils.add_arc_from_to(t_CA, p_after_CA, net)

petri_utils.add_arc_from_to(p_after_P, t_N, net)
petri_utils.add_arc_from_to(t_N, p_after_N, net)
petri_utils.add_arc_from_to(p_after_N, t_CL, net)
