```sql
WITH case_event_counts AS (
    SELECT case_id, COUNT(*) AS event_count
    FROM event_log
    GROUP BY case_id
    HAVING COUNT(*) <= 6
)
SELECT el.*
FROM event_log el
INNER JOIN case_event_counts cec
    ON el.case_id = cec.case_id;
```
