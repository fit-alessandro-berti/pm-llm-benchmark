```sql
WITH case_event_counts AS (
  SELECT case_id, COUNT(*) AS event_count
  FROM event_log
  GROUP BY case_id
)
SELECT e.*
FROM event_log e
JOIN case_event_counts c ON e.case_id = c.case_id
WHERE c.event_count <= 6
ORDER BY e.case_id, e.timestamp;
```
