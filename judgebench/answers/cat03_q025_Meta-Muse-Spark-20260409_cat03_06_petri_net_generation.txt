Here’s an accepting Petri net for the insurance claim process, with loops for re-submitting documents and repeated in-depth investigations. 

The net models the main sequence C → CR → DV → FC → CA → AP → P → N → CL, plus two optional loops:
1. **DV ↔ RDS**: if documents are incomplete, loop back until verification passes
2. **FC ↔ II**: if fraud check raises doubts, do in-depth investigation, possibly multiple times

### Key structure

**Places** represent states between activities. **Transitions** are your labeled activities. The initial marking has 1 token in `start`, final marking has 1 token in `closed`.

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

def build_claim_accepting_petri_net():
    net = PetriNet(name="insurance_claim_process")
    
    # --- Places: states between activities ---
    p_start = PetriNet.Place("p_start")           # new claim arrived
    p_after_c = PetriNet.Place("p_after_c")       # after Claim filed
    p_after_cr = PetriNet.Place("p_after_cr")     # after Claim Registration
    p_dv_ready = PetriNet.Place("p_dv_ready")     # ready for Document Verification
    p_dv_done = PetriNet.Place("p_dv_done")       # DV completed, decision point
    p_fc_ready = PetriNet.Place("p_fc_ready")     # ready for Fraud Check
    p_fc_done = PetriNet.Place("p_fc_done")       # FC completed, decision point
    p_after_ca = PetriNet.Place("p_after_ca")     # after Claim Assessment
    p_after_ap = PetriNet.Place("p_after_ap")     # after Approval
    p_after_p = PetriNet.Place("p_after_p")       # after Payment
    p_after_n = PetriNet.Place("p_after_n")       # after Notification
    p_closed = PetriNet.Place("p_closed")         # claim closed and archived
    
    net.places.update([
        p_start, p_after_c, p_after_cr, p_dv_ready, p_dv_done,
        p_fc_ready, p_fc_done, p_after_ca, p_after_ap, p_after_p, 
        p_after_n, p_closed
    ])
    
    # --- Transitions: activities ---
    t_c = PetriNet.Transition("C", label="Insurance Claim")
    t_cr = PetriNet.Transition("CR", label="Claim Registration")
    t_dv = PetriNet.Transition("DV", label="Document Verification")
    t_rds = PetriNet.Transition("RDS", label="Re-Document Submission")
    t_dv_ok = PetriNet.Transition("dv_ok", label="Documents OK")      # silent: DV passed
    t_fc = PetriNet.Transition("FC", label="Fraud Check")
    t_ii = PetriNet.Transition("II", label="In-depth Investigation")
    t_fc_ok = PetriNet.Transition("fc_ok", label="Fraud Check OK")    # silent: FC passed
    t_ca = PetriNet.Transition("CA", label="Claim Assessment")
    t_ap = PetriNet.Transition("AP", label="Approval")
    t_p = PetriNet.Transition("P", label="Payment")
    t_n = PetriNet.Transition("N", label="Notification")
    t_cl = PetriNet.Transition("CL", label="Closure")
    
    net.transitions.update([
        t_c, t_cr, t_dv, t_rds, t_dv_ok, t_fc, t_ii, t_fc_ok,
        t_ca, t_ap, t_p, t_n, t_cl
    ])
    
    # --- Arcs: main sequence ---
    petri_utils.add_arc_from_to(p_start, t_c, net)
    petri_utils.add_arc_from_to(t_c, p_after_c, net)
    
    petri_utils.add_arc_from_to(p_after_c, t_cr, net)
    petri_utils.add_arc_from_to(t_cr, p_after_cr, net)
    
    petri_utils.add_arc_from_to(p_after_cr, t_dv, net)
    petri_utils.add_arc_from_to(t_dv, p_dv_done, net)
    
    # Document Verification loop: if incomplete, do RDS and retry DV
    petri_utils.add_arc_from_to(p_dv_done, t_rds, net)
    petri_utils.add_arc_from_to(t_rds, p_dv_ready, net)
    petri_utils.add_arc_from_to(p_dv_ready, t_dv, net)  # loop back to DV
    
    # If documents OK, proceed to Fraud Check
    petri_utils.add_arc_from_to(p_dv_done, t_dv_ok, net)
    petri_utils.add_arc_from_to(t_dv_ok, p_fc_ready, net)
    
    petri_utils.add_arc_from_to(p_fc_ready, t_fc, net)
    petri_utils.add_arc_from_to(t_fc, p_fc_done, net)
    
    # Fraud Check loop: if doubts, do II and retry FC
    petri_utils.add_arc_from_to(p_fc_done, t_ii, net)
    petri_utils.add_arc_from_to(t_ii, p_fc_ready, net)  # loop back to FC
    
    # If fraud check OK, proceed to Claim Assessment
    petri_utils.add_arc_from_to(p_fc_done, t_fc_ok, net)
    petri_utils.add_arc_from_to(t_fc_ok, p_after_ca, net)
    
    petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
    petri_utils.add_arc_from_to(t_ap, p_after_ap, net)
    
    petri_utils.add_arc_from_to(p_after_ap, t_p, net)
    petri_utils.add_arc_from_to(t_p, p_after_p, net)
    
    petri_utils.add_arc_from_to(p_after_p, t_n, net)
    petri_utils.add_arc_from_to(t_n, p_after_n, net)
    
    petri_utils.add_arc_from_to(p_after_n, t_cl, net)
    petri_utils.add_arc_from_to(t_cl, p_closed, net)
    
    # --- Initial and Final Markings ---
    im = Marking()
    im[p_start] = 1   # process starts with a new claim
    
    fm = Marking()
    fm[p_closed] = 1  # process ends with claim closed
    
    return net, im, fm

# Usage
net, im, fm = build_claim_accepting_petri_net()
```

### How the loops work

- **Document re-submission loop**: After `DV`, you’re in `p_dv_done`. Either fire `dv_ok` to continue to `FC`, or fire `RDS` which puts a token in `p_dv_ready` to retry `DV`. This can repeat.
- **In-depth investigation loop**: After `FC`, you’re in `p_fc_done`. Either fire `fc_ok` to continue to `CA`, or fire `II` which puts a token back in `p_fc_ready` to retry `FC`. This can also repeat multiple times.

**Note**: `dv_ok` and `fc_ok` are silent transitions that model the decision “passed” without a business activity. If you prefer to keep only business activities, you can model the choice directly on `DV` and `FC` with two output places each, but the silent-transition version is cleaner and common in pm4py.
