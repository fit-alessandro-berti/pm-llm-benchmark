    ```sql
    SELECT DISTINCT ce.claim_id
    FROM claim_events ce
    WHERE ce.activity = 'C'
    AND NOT EXISTS (SELECT 1 FROM claim_events ce2 WHERE ce2.claim_id = ce.claim_id AND ce2.activity IN ('E', 'P'));
    ```

    ```sql
    SELECT claim_id, COUNT(*) as approval_count
    FROM claim_events
    WHERE activity = 'P'
    ```
