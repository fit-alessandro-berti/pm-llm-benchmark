from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# 1. Create the PetriNet object
net = PetriNet(name="Insurance Claim Management Process")

# 2. Define Places (States)
# Initial State
P_NEW_CLAIM = PetriNet.Place(name="New Claim Arrived")

# States after activities
P_CLAIM_REGISTERED = PetriNet.Place(name="Claim Registered")
P_DOCUMENTS_VERIFIED = PetriNet.Place(name="Documents Verified (OK)")
P_DOCUMENTS_INCOMPLETE = PetriNet.Place(name="Documents Incomplete")
P_DOCUMENTS_SUBMITTED = PetriNet.Place(name="Documents Re-Submitted")
P_FRAUD_CHECK_PASSED = PetriNet.Place(name="Fraud Check Passed")
P_FRAUD_DOUBTS = PetriNet.Place(name="Fraud Doubts Raised")
P_INVESTIGATION_COMPLETE = PetriNet.Place(name="Investigation Complete")
P_CLAIM_ASSESSED = PetriNet.Place(name="Claim Assessed")
P_CLAIM_APPROVED = PetriNet.Place(name="Claim Approved")
P_CLAIM_PAID = PetriNet.Place(name="Claim Paid")
P_CLAIM_NOTIFIED = PetriNet.Place(name="Claim Notified")
P_CLAIM_CLOSED = PetriNet.Place(name="Claim Closed") # Final State

# Add all places to the PetriNet
places = [
    P_NEW_CLAIM,
    P_CLAIM_REGISTERED,
    P_DOCUMENTS_VERIFIED,
    P_DOCUMENTS_INCOMPLETE,
    P_DOCUMENTS_SUBMITTED,
    P_FRAUD_CHECK_PASSED,
    P_FRAUD_DOUBTS,
    P_INVESTIGATION_COMPLETE,
    P_CLAIM_ASSESSED,
    P_CLAIM_APPROVED,
    P_CLAIM_PAID,
    P_CLAIM_NOTIFIED,
    P_CLAIM_CLOSED,
]
for place in places:
    net.places.add(place)

# 3. Define Transitions (Activities)
T_CR = PetriNet.Transition(name="Claim Registration", label="CR")
T_DV = PetriNet.Transition(name="Document Verification", label="DV")
T_RDS = PetriNet.Transition(name="Re-Document Submission", label="RDS")
T_FC = PetriNet.Transition(name="Fraud Check", label="FC")
T_II = PetriNet.Transition(name="In-depth Investigation", label="II")
T_CA = PetriNet.Transition(name="Claim Assessment", label="CA")
T_AP = PetriNet.Transition(name="Approval", label="AP")
T_P = PetriNet.Transition(name="Payment", label="P")
T_N = PetriNet.Transition(name="Notification", label="N")
T_CL = PetriNet.Transition(name="Closure", label="CL")

# Add all transitions to the PetriNet
transitions = [
    T_CR, T_DV, T_RDS, T_FC, T_II, T_CA, T_AP, T_P, T_N, T_CL
]
for transition in transitions:
    net.transitions.add(transition)

# 4. Define Arcs (Connections between places and transitions)

# Initial Claim -> Claim Registration
petri_utils.add_arc_from_to(P_NEW_CLAIM, T_CR, net)
petri_utils.add_arc_from_to(T_CR, P_CLAIM_REGISTERED, net)

# Claim Registration -> Document Verification
petri_utils.add_arc_from_to(P_CLAIM_REGISTERED, T_DV, net)

# Document Verification can lead to two paths:
# Path 1: Documents Verified (OK) -> Fraud Check
petri_utils.add_arc_from_to(T_DV, P_DOCUMENTS_VERIFIED, net)
petri_utils.add_arc_from_to(P_DOCUMENTS_VERIFIED, T_FC, net)

# Path 2: Documents Incomplete -> Re-Document Submission
petri_utils.add_arc_from_to(T_DV, P_DOCUMENTS_INCOMPLETE, net)
petri_utils.add_arc_from_to(P_DOCUMENTS_INCOMPLETE, T_RDS, net)
# Re-Submitted Documents -> Document Verification (Loop)
petri_utils.add_arc_from_to(T_RDS, P_DOCUMENTS_SUBMITTED, net)
petri_utils.add_arc_from_to(P_DOCUMENTS_SUBMITTED, T_DV, net)

# Fraud Check can lead to two paths:
# Path 1: Fraud Check Passed -> Claim Assessment
petri_utils.add_arc_from_to(T_FC, P_FRAUD_CHECK_PASSED, net)
petri_utils.add_arc_from_to(P_FRAUD_CHECK_PASSED, T_CA, net)

# Path 2: Fraud Doubts Raised -> In-depth Investigation
petri_utils.add_arc_from_to(T_FC, P_FRAUD_DOUBTS, net)
petri_utils.add_arc_from_to(P_FRAUD_DOUBTS, T_II, net)
# In-depth Investigation Complete -> Fraud Check (Loop)
petri_utils.add_arc_from_to(T_II, P_INVESTIGATION_COMPLETE, net)
petri_utils.add_arc_from_to(P_INVESTIGATION_COMPLETE, T_FC, net)

# Claim Assessment -> Approval
petri_utils.add_arc_from_to(P_CLAIM_ASSESSED, T_AP, net)
petri_utils.add_arc_from_to(T_CA, P_CLAIM_ASSESSED, net) # CA output

# Approval -> Payment
petri_utils.add_arc_from_to(T_AP, P_CLAIM_APPROVED, net)
petri_utils.add_arc_from_to(P_CLAIM_APPROVED, T_P, net) # AP output

# Payment -> Notification
petri_utils.add_arc_from_to(T_P, P_CLAIM_PAID, net)
petri_utils.add_arc_from_to(P_CLAIM_PAID, T_N, net)

# Notification -> Closure
petri_utils.add_arc_from_to(T_N, P_CLAIM_NOTIFIED, net)
petri_utils.add_arc_from_to(P_CLAIM_NOTIFIED, T_CL, net)

# Closure leads to the final state
petri_utils.add_arc_from_to(T_CL, P_CLAIM_CLOSED, net)


# 5. Define Initial Marking (im)
# The process starts with one token in the "New Claim Arrived" place.
im = Marking()
im[P_NEW_CLAIM] = 1

# 6. Define Final Marking (fm)
# The process successfully ends when there is one token in the "Claim Closed" place.
fm = Marking()
fm[P_CLAIM_CLOSED] = 1

# You can optionally print the Petri Net structure to verify

print(f"PetriNet: {net.name}")
print(f"Number of Places: {len(net.places)}")
print(f"Number of Transitions: {len(net.transitions)}")
print(f"Initial Marking: {im}")
print(f"Final Marking: {fm}")

**Explanation:**

1.  **PetriNet Initialization:** A `PetriNet` object is created to hold all components.
2.  **Places Definition:** Each significant state or condition in the process is defined as a `PetriNet.Place`.
    *   `P_NEW_CLAIM`: Represents a new claim arriving, serving as the starting point.
    *   `P_CLAIM_REGISTERED`: The state after the claim has been logged.
    *   `P_DOCUMENTS_VERIFIED`: Represents successful document verification.
    *   `P_DOCUMENTS_INCOMPLETE`: Represents the case where documents are not satisfactory.
    *   `P_DOCUMENTS_SUBMITTED`: Represents the state after documents have been re-submitted, ready to be verified again.
    *   `P_FRAUD_CHECK_PASSED`: State where the initial fraud check is clear.
    *   `P_FRAUD_DOUBTS`: State where further investigation is warranted.
    *   `P_INVESTIGATION_COMPLETE`: State after an in-depth investigation, ready for another fraud check.
    *   `P_CLAIM_ASSESSED`: State after the claim's financial assessment.
    *   `P_CLAIM_APPROVED`: State after the claim has been approved (either by manager or automatically).
    *   `P_CLAIM_PAID`: State after the payment has been processed.
    *   `P_CLAIM_NOTIFIED`: State after the customer has been notified.
    *   `P_CLAIM_CLOSED`: The final state of the process.
3.  **Transitions Definition:** Each activity in the process is defined as a `PetriNet.Transition`. The `label` attribute is used to match the activity names mentioned in the scenario.
    *   `T_CR` (Claim Registration), `T_DV` (Document Verification), `T_RDS` (Re-Document Submission), `T_FC` (Fraud Check), `T_II` (In-depth Investigation), `T_CA` (Claim Assessment), `T_AP` (Approval), `T_P` (Payment), `T_N` (Notification), `T_CL` (Closure).
4.  **Arcs Definition:** `petri_utils.add_arc_from_to()` is used to connect places to transitions (input arcs) and transitions to places (output arcs).
    *   The main sequence is established: `New Claim` -> `CR` -> `Claim Registered` -> `DV` -> ... -> `CL` -> `Claim Closed`.
    *   **Document Verification Loop:** `T_DV` can lead to `P_DOCUMENTS_INCOMPLETE`. From `P_DOCUMENTS_INCOMPLETE`, `T_RDS` is triggered, leading to `P_DOCUMENTS_SUBMITTED`. A token in `P_DOCUMENTS_SUBMITTED` then loops back to `T_DV`, allowing re-verification.
    *   **Fraud Check Loop:** Similarly, `T_FC` can lead to `P_FRAUD_DOUBTS`. From `P_FRAUD_DOUBTS`, `T_II` is triggered, leading to `P_INVESTIGATION_COMPLETE`. A token in `P_INVESTIGATION_COMPLETE` loops back to `T_FC`.
    *   **Approval Logic:** The `T_AP` transition represents the approval step. While the prompt mentions a threshold, this model abstracts the decision logic within the `T_AP` transition itself. A token arriving at `P_CLAIM_ASSESSED` enables `T_AP`, and upon completion, it results in `P_CLAIM_APPROVED`.
5.  **Initial Marking (`im`):** A `Marking` object is created. A token (represented by the integer `1`) is placed in `P_NEW_CLAIM` to signify that the process begins with a new claim.
6.  **Final Marking (`fm`):** Another `Marking` object is created. A token is placed in `P_CLAIM_CLOSED`, indicating that the successful completion of the process results in a closed claim.

This code defines a complete accepting Petri net model for the described insurance claim management process.
