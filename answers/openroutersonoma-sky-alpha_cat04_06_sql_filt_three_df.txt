```sql
WITH with_sequence AS (
  SELECT 
    case_id,
    LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp ASC) AS prev_activity,
    LEAD(activity) OVER (PARTITION BY case_id ORDER BY timestamp ASC) AS next_activity,
    LEAD(timestamp) OVER (PARTITION BY case_id ORDER BY timestamp ASC) AS next_timestamp,
    timestamp AS create_ts
  FROM event_log
  WHERE activity = 'Create Order'
),
bad_cases AS (
  SELECT DISTINCT case_id
  FROM with_sequence
  WHERE prev_activity = 'Approve Purchase Requisition'
    AND next_activity = 'Confirm Order'
    AND (next_timestamp - create_ts) > INTERVAL '5' DAY
)
SELECT *
FROM event_log
WHERE case_id NOT IN (SELECT case_id FROM bad_cases)
ORDER BY case_id, timestamp;
```
