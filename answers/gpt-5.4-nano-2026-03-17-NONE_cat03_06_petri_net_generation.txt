```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# -----------------------------
# Build accepting Petri net
# -----------------------------
net = PetriNet(name="insurance_claim_management")

# Places (states)
p_start = PetriNet.Place("start_new_claim")

p_after_cr = PetriNet.Place("after_CR")
p_after_dv = PetriNet.Place("after_DV")
p_after_fc = PetriNet.Place("after_FC")
p_after_ca = PetriNet.Place("after_CA")
p_after_ap = PetriNet.Place("after_AP")
p_after_p = PetriNet.Place("after_P")
p_after_n = PetriNet.Place("after_N")

p_closed = PetriNet.Place("closed")

# Loop-related places (to allow repeated attempts)
p_dv_loop = PetriNet.Place("DV_loop_ready")  # token can re-enter DV
p_fc_loop = PetriNet.Place("FC_loop_ready")  # token can re-enter FC

# Optional improvement: start FC loop from after DV, and return after II
# (keeps places explicit and makes looping clear)

# Add places to net
for p in [
    p_start, p_after_cr, p_after_dv, p_after_fc, p_after_ca, p_after_ap,
    p_after_p, p_after_n, p_closed, p_dv_loop, p_fc_loop
]:
    net.places.add(p)

# Transitions (activities)
t_cr = PetriNet.Transition("t_CR", "Claim Registration")
t_dv = PetriNet.Transition("t_DV", "Document Verification")
t_rds = PetriNet.Transition("t_RDS", "Re-Document Submission")
t_fc = PetriNet.Transition("t_FC", "Fraud Check")
t_ii = PetriNet.Transition("t_II", "In-depth Investigation")
t_ca = PetriNet.Transition("t_CA", "Claim Assessment")
t_ap = PetriNet.Transition("t_AP", "Approval")
t_p = PetriNet.Transition("t_P", "Payment")
t_n = PetriNet.Transition("t_N", "Notification")
t_cl = PetriNet.Transition("t_CL", "Closure")

# Add transitions to net
for t in [t_cr, t_dv, t_rds, t_fc, t_ii, t_ca, t_ap, t_p, t_n, t_cl]:
    net.transitions.add(t)

# -----------------------------
# Linear flow (with entry/exit places)
# -----------------------------
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_after_cr, net)

petri_utils.add_arc_from_to(p_after_cr, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

petri_utils.add_arc_from_to(p_after_dv, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

petri_utils.add_arc_from_to(p_after_fc, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_after_ap, net)

petri_utils.add_arc_from_to(p_after_ap, t_p, net)
petri_utils.add_arc_from_to(t_p, p_after_p, net)

petri_utils.add_arc_from_to(p_after_p, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_n, net)

petri_utils.add_arc_from_to(p_after_n, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# -----------------------------
# Loops
# -----------------------------
# If DV finds documents incomplete -> re-document submission -> return to DV
petri_utils.add_arc_from_to(p_after_dv, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_dv_loop, net)
petri_utils.add_arc_from_to(p_dv_loop, t_dv, net)  # loop back into DV

# If FC raises doubts -> in-depth investigation (can happen multiple times)
# after II, return back to FC to re-check
petri_utils.add_arc_from_to(p_after_fc, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_fc_loop, net)
petri_utils.add_arc_from_to(p_fc_loop, t_fc, net)  # loop back into FC

# (Optional) allow FC to start after DV in normal flow is already handled above.
# The loop places ensure multiple iterations are possible.

# -----------------------------
# Initial and final markings
# -----------------------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1

# The Petri net (net) is now an accepting Petri net with (im, fm).
# You can return or use these variables:
# net, im, fm
```
