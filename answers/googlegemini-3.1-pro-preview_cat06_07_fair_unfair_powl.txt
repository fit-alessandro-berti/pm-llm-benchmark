```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# ==========================================================
# MODEL 1: POWL Model with Potential Unfairness
# ==========================================================

# 1. Define Activities
receive_app_1 = Transition(label="ReceiveApplication")
data_check_1 = Transition(label="DataCompletenessCheck")
req_info_1 = Transition(label="RequestMoreInfo")
skill_test_1 = Transition(label="SkillAssessment")
cultural_fit_1 = Transition(label="CulturalFitCheck")
community_aff_1 = Transition(label="CommunityAffiliationCheck")
managerial_review_1 = Transition(label="ManagerialReview")
final_decision_1 = Transition(label="FinalDecision")

# 2. Define Control-Flow Operators
# Loop: Keep asking for additional info until data is complete
loop_completeness_1 = OperatorPOWL(operator=Operator.LOOP, children=[data_check_1, req_info_1])

# XOR (Bias Point): Diverge between Standard Cultural Fit Check or Biased Community Affiliation Check
xor_bias_1 = OperatorPOWL(operator=Operator.XOR, children=[cultural_fit_1, community_aff_1])

# 3. Create Partial Order / Set Execution Order
model_with_unfairness = StrictPartialOrder(nodes=[
    receive_app_1, 
    loop_completeness_1, 
    skill_test_1, 
    xor_bias_1, 
    managerial_review_1, 
    final_decision_1
])

model_with_unfairness.order.add_edge(receive_app_1, loop_completeness_1)
model_with_unfairness.order.add_edge(loop_completeness_1, skill_test_1)
model_with_unfairness.order.add_edge(skill_test_1, xor_bias_1)
model_with_unfairness.order.add_edge(xor_bias_1, managerial_review_1)
model_with_unfairness.order.add_edge(managerial_review_1, final_decision_1)


# ==========================================================

    final_decision_2

# Here, directly routing every applicant into matching standard cultural fit check
```
