import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Create all the transition nodes
SR = Transition(label="SR")       # Service Request verification
CS = Transition(label="CS")       # Contract Setup
DA = Transition(label="DA")       # Datacenter Allocation
ACI = Transition(label="ACI")     # Access Control Initialization
BNC = Transition(label="BNC")     # Base Network Configuration
DR = Transition(label="D&R")      # Debug & Reconfigure
CD = Transition(label="CD")       # Containerized Deployment
MD = Transition(label="MD")       # Monolithic Deployment
QA = Transition(label="QA")       # Quality Assurance
SA = Transition(label="SA")       # Security Audit
GLA = Transition(label="GLA")     # Go-Live Approval

# Step 3: Create the loop for configuration phase (BNC followed by optional D&R)
# Loop structure: execute BNC, then optionally D&R, then repeat BNC until exit
config_loop = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DR])

# Step 4: Create the exclusive choice between deployment methods
deployment_choice = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# Step 5: Create partial order for QA and SA (can be executed in any order or concurrently)
final_parallel = StrictPartialOrder(nodes=[QA, SA])
# No ordering constraints between QA and SA - they are parallel

# Step 6: Build the overall workflow structure

# First level: SR and CS are sequential
# Combine them in a partial order: SR must come before CS
initial_sequence = StrictPartialOrder(nodes=[SR, CS])
initial_sequence.order.add_edge(SR, CS)

# Second level: After CS, we have DA and ACI in parallel
# These are part of a larger parallel structure
parallel_setup = StrictPartialOrder(nodes=[DA, ACI])
# DA and ACI have no ordering between them - they're concurrent

# Third level: After DA and ACI, we have the configuration loop
# All of: CS -> DA/ACI -> config_loop need to be connected

# Step 4: After config_loop, we have deployment_choice
# Then final_parallel (QA and SA), then GLA

# Build the complete root structure
# We need to carefully order:
# 1. SR -> CS (sequential)
# 2. CS enables both DA and ACI (which run in parallel)
# 3. Both DA and ACI must complete before config_loop
# 4. config_loop enables deployment_choice
# 5. deployment_choice enables the QA/SA parallel block
# 6. Both QA and SA must complete before GLA

root = StrictPartialOrder(nodes=[SR, CS, DA, ACI, config_loop, deployment_choice, final_parallel, GLA])

# Add ordering constraints:
# SR must come before CS
root.order.add_edge(SR, CS)

# CS must come before both DA and ACI
root.order.add_edge(CS, DA)
root.order.add_edge(CS, ACI)

# Both DA and ACI must complete before config_loop starts
root.order.add_edge(DA, config_loop)
root.order.add_edge(ACI, config_loop)

# config_loop must complete before deployment_choice
root.order.add_edge(config_loop, deployment_choice)

# deployment_choice must complete before the final parallel tasks (QA and SA)
root.order.add_edge(deployment_choice, final_parallel)

# Both QA and SA (which are part of final_parallel) must complete before GLA
# We need to add edges from QA and SA nodes to GLA
# Since final_parallel contains QA and SA, we reference those nodes directly
root.order.add_edge(QA, GLA)
root.order.add_edge(SA, GLA)
