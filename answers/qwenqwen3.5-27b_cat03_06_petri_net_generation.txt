    # 6. Payment
    petri_utils.add_arc_from_to(p_ready_for_P, t_P, net)
    petri_utils.add_arc_from_to(t_P, p_after_P, net)

    # 7. Notification
    petri_utils.add_arc_from_to(p_after_P, p_ready_for_N, net)
    petri_utils.add_arc_from_to(p_ready_for_N, t_N, net)
    petri_utils.add_arc_from_to(t_N, p_after_N, net)

    # 8. Closure
    petri_utils.add_arc_from_to(p_after_N, p_ready_for_CL, net)
    petri_utils.add_arc_from_to(p_ready_for_CL, t_CL, net)
    petri_utils.add_arc_from_to(t_CL, end_place, net)

    # --- Define Markings ---
    # Initial Marking: Token in 'start' place (New claim arrives)
    im = Marking()
    im[start_place] = 1

    # Final Marking: Token in 'end' place (Claim closed)
    fm = Marking()
    fm[end_place] = 1

    return net, im, fm
