from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

def build_insurance_claim_petri_net():
    """
    Constructs an accepting Petri net for the Insurance Claim Management process.
    
    Returns:
        tuple: (net, initial_marking, final_marking)
    """
    # 1. Initialize Petri Net
    net = PetriNet(name="InsuranceClaimManagement")

    # 2. Define Places (States)
    # Start and End
    p_start = PetriNet.Place(name="start")
    p_end = PetriNet.Place(name="closed")

    # Places between main activities
    p_after_cr = PetriNet.Place(name="after_Claim_Registration")
    p_after_dv = PetriNet.Place(name="after_Document_Verification")
    p_after_rds = PetriNet.Place(name="after_Re_Document_Submission")
    p_after_fc = PetriNet.Place(name="after_Fraud_Check")
    p_after_ii = PetriNet.Place(name="after_In_depth_Investigation")
    p_after_ca = PetriNet.Place(name="after_Claim_Assessment")
    p_after_ap = PetriNet.Place(name="after_Approval")
    p_after_p = PetriNet.Place(name="after_Payment")
    p_after_n = PetriNet.Place(name="after_Notification")

    # Add all places to the net
    places = [
        p_start, p_after_cr, p_after_dv, p_after_rds, 
        p_after_fc, p_after_ii, p_after_ca, p_after_ap, 
        p_after_p, p_after_n, p_end
    ]
    for p in places:
        net.places.add(p)

    # 3. Define Transitions (Activities)
    t_cr = PetriNet.Transition(name="CR", label="Claim Registration")
    t_dv = PetriNet.Transition(name="DV", label="Document Verification")
    t_rds = PetriNet.Transition(name="RDS", label="Re-Document Submission")
    t_fc = PetriNet.Transition(name="FC", label="Fraud Check")
    t_ii = PetriNet.Transition(name="II", label="In-depth Investigation")
    t_ca = PetriNet.Transition(name="CA", label="Claim Assessment")
    t_ap = PetriNet.Transition(name="AP", label="Approval")
    t_p = PetriNet.Transition(name="P", label="Payment")
    t_n = PetriNet.Transition(name="N", label="Notification")
    t_cl = PetriNet.Transition(name="CL", label="Closure")

    # Add all transitions to the net
    transitions = [t_cr, t_dv, t_rds, t_fc, t_ii, t_ca, t_ap, t_p, t_n, t_cl]
    for t in transitions:
        net.transitions.add(t)

    # 4. Define Arcs (Flow Relations) using petri_utils
    
    # Main Sequence: Start -> CR -> DV
    petri_utils.add_arc_from_to(p_start, t_cr, net)
    petri_utils.add_arc_from_to(t_cr, p_after_cr, net)
    petri_utils.add_arc_from_to(p_after_cr, t_dv, net)
    petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

    # Loop: Document Verification (DV) <-> Re-Document Submission (RDS)
    # If docs incomplete (p_after_dv) -> RDS -> back to DV
    petri_utils.add_arc_from_to(p_after_dv, t_rds, net)      # Branch to RDS
    petri_utils.add_arc_from_to(t_rds, p_after_rds, net)
    petri_utils.add_arc_from_to(p_after_rds, t_dv, net)      # Loop back to DV
    
    # Main Sequence: DV -> Fraud Check (FC)
    petri_utils.add_arc_from_to(p_after_dv, t_fc, net)       # Branch to FC (if docs ok)
    petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

    # Loop: Fraud Check (FC) <-> In-depth Investigation (II)
    # If doubts (p_after_fc) -> II -> back to FC
    petri_utils.add_arc_from_to(p_after_fc, t_ii, net)       # Branch to II
    petri_utils.add_arc_from_to(t_ii, p_after_ii, net)
    petri_utils.add_arc_from_to(p_after_ii, t_fc, net)       # Loop back to FC

    # Main Sequence: FC -> Claim Assessment (CA) -> Approval (AP) -> Payment (P) -> Notification (N) -> Closure (CL) -> End
    petri_utils.add_arc_from_to(p_after_fc, t_ca, net)       # Branch to CA (if FC clear)
    petri_utils.add_arc_from_to(t_ca, p_after_ca, net)
    petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
    petri_utils.add_arc_from_to(t_ap, p_after_ap, net)
    petri_utils.add_arc_from_to(p_after_ap, t_p, net)
    petri_utils.add_arc_from_to(t_p, p_after_p, net)
    petri_utils.add_arc_from_to(p_after_p, t_n, net)
    petri_utils.add_arc_from_to(t_n, p_after_n, net)
    petri_utils.add_arc_from_to(p_after_n, t_cl, net)
    petri_utils.add_arc_from_to(t_cl, p_end, net)

    # 5. Define Initial and Final Markings
    # Initial Marking: 1 token in 'start' (New claim arrives)
    im = Marking()
    im[p_start] = 1

    # Final Marking: 1 token in 'closed' (Claim archived)
    fm = Marking()
    fm[p_end] = 1

    return net, im, fm

# --- Execution / Verification ---
if __name__ == "__main__":
    net, im, fm = build_insurance_claim_petri_net()
    
    print(f"Petri Net Name: {net.name}")
    print(f"Places ({len(net.places)}): {[p.name for p in net.places]}")
    print(f"Transitions ({len(net.transitions)}): {[t.name for t in net.transitions]}")
