```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# -------------------------------------------------------------------
# Activities / transitions
# -------------------------------------------------------------------

# 1. Initial business setup
SR = Transition(label="SR")      # Service Request verification
CS = Transition(label="CS")      # Contract Setup

# 2. Parallel environment preparation
DA = Transition(label="DA")      # Datacenter Allocation
ACI = Transition(label="ACI")    # Access Control Initialization

# 3. Looped configuration phase
BNC = Transition(label="BNC")    # Base Network Configuration
DR = Transition(label="D&R")     # Debug & Reconfigure

# 4. Exclusive deployment choice
CD = Transition(label="CD")      # Containerized Deployment
MD = Transition(label="MD")      # Monolithic Deployment

# 5. Final validation tasks
QA = Transition(label="QA")      # Quality Assurance
SA = Transition(label="SA")      # Security Audit

# 6. Final approval
GLA = Transition(label="GLA")    # Go-Live Approval


# -------------------------------------------------------------------
# Loop: perform BNC, then either exit, or perform D&R and repeat BNC
#
# Semantics of LOOP(A, B):
#   execute A,
#   then either exit,
#   or execute B and then A again,
#   repeating until exit.
# -------------------------------------------------------------------
configuration_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[BNC, DR]
)


# -------------------------------------------------------------------
# Exclusive choice: either CD or MD
# -------------------------------------------------------------------
deployment_choice = OperatorPOWL(
    operator=Operator.XOR,
    children=[CD, MD]
)


# -------------------------------------------------------------------
# Root partial order for the complete process
# -------------------------------------------------------------------
root = StrictPartialOrder(
    nodes=[
        SR,
        CS,
        DA,
        ACI,
        configuration_loop,
        deployment_choice,
        QA,
        SA,
        GLA
    ]
)

# -------------------------------------------------------------------
# Ordering constraints
# -------------------------------------------------------------------

# Service Request verification must happen before Contract Setup
root.order.add_edge(SR, CS)

# Contract Setup is required before technical preparation starts
root.order.add_edge(CS, DA)
root.order.add_edge(CS, ACI)

# DA and ACI are concurrent with each other, so no edge between them.
# Both must complete before the looped configuration phase starts.
root.order.add_edge(DA, configuration_loop)
root.order.add_edge(ACI, configuration_loop)

# Deployment choice happens after successful configuration
root.order.add_edge(configuration_loop, deployment_choice)

# QA and SA both happen after deployment choice is completed.
# They can be performed in any order or concurrently, so no edge between QA and SA.
root.order.add_edge(deployment_choice, QA)
root.order.add_edge(deployment_choice, SA)

# Both QA and SA must be completed before Go-Live Approval
root.order.add_edge(QA, GLA)
root.order.add_edge(SA, GLA)
```
