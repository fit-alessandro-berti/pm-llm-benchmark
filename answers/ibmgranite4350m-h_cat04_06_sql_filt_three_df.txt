```sql
SELECT *
FROM event_log
WHERE (CASE 
       WHEN activity = 'Approve Purchase Requisition' AND case_id = 1 THEN true 
       ELSE false 
     END) OR (CASE 
         WHEN activity = 'Create Order' AND case_id = 2 THEN true 
         ELSE false 
       END)
AND (CASE 
       WHEN timestamp >= DATE_SUB(NOW(), INTERVAL 5 DAY) 
         THEN true 
       ELSE false 
     END) AS meets_condition;
```
