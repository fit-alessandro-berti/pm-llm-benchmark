### 1) POWL Model with Potential Unfairness (bias via community-affiliation XOR branch)

```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- Core activities ---
ReceiveApplication = Transition(label="ReceiveApplication")

ResumeParsingAndInitialDataCheck = Transition(
    label="ResumeParsing & InitialDataCheck"
)
RequestMoreInfo = Transition(label="RequestMoreInfo")

PreliminarySkillAssessment = Transition(label="PreliminarySkillAssessment")

StandardCulturalFitCheck = Transition(label="CulturalFitCheck (Standard)")
CommunityAffiliationCheck = Transition(
    label="CommunityAffiliationCheck (ImplicitUpliftPossible)"
)  # <-- potential unfairness source

ManagerialReview = Transition(label="ManagerialReview")
FinalDecision_Hire = Transition(label="FinalDecision:ApproveHire")
FinalDecision_Reject = Transition(label="FinalDecision:Reject")

# --- Loop for data completeness ---
# parse -> (request more info -> parse again)* until loop exits
DataCompletenessLoop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[ResumeParsingAndInitialDataCheck, RequestMoreInfo]
)

# --- XOR after skill check: either standard cultural fit OR community-affiliation-based uplift ---
CulturalChoiceWithBias = OperatorPOWL(
    operator=Operator.XOR,
    children=[StandardCulturalFitCheck, CommunityAffiliationCheck]
)

# --- Proceed path (skill threshold met) ---
ProceedPath = StrictPartialOrder(
    nodes=[CulturalChoiceWithBias, ManagerialReview, FinalDecision_Hire]
)
ProceedPath.order.add_edge(CulturalChoiceWithBias, ManagerialReview)
ProceedPath.order.add_edge(ManagerialReview, FinalDecision_Hire)

# --- Skill-threshold outcome (skill threshold not met => reject directly) ---
SkillOutcome = OperatorPOWL(
    operator=Operator.XOR,
    children=[FinalDecision_Reject, ProceedPath]
)

# --- Root process ordering ---
root_unfair = StrictPartialOrder(
    nodes=[ReceiveApplication, DataCompletenessLoop, PreliminarySkillAssessment, SkillOutcome]
)
root_unfair.order.add_edge(ReceiveApplication, DataCompletenessLoop)
root_unfair.order.add_edge(DataCompletenessLoop, PreliminarySkillAssessment)
root_unfair.order.add_edge(PreliminarySkillAssessment, SkillOutcome)
```

### 2) POWL Model Without Unfairness (no community-affiliation XOR branch; everyone uses same cultural fit check)

```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- Core activities ---
ReceiveApplication = Transition(label="ReceiveApplication")

ResumeParsingAndInitialDataCheck = Transition(
    label="ResumeParsing & InitialDataCheck"
)
RequestMoreInfo = Transition(label="RequestMoreInfo")

PreliminarySkillAssessment = Transition(label="PreliminarySkillAssessment")

CulturalFitCheck = Transition(label="CulturalFitCheck (SameForAll)")  # <-- no bias branch

ManagerialReview = Transition(label="ManagerialReview")
FinalDecision_Hire = Transition(label="FinalDecision:ApproveHire")
FinalDecision_Reject = Transition(label="FinalDecision:Reject")

# --- Loop for data completeness ---
DataCompletenessLoop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[ResumeParsingAndInitialDataCheck, RequestMoreInfo]
)

# --- Proceed path (skill threshold met): always the same cultural fit check ---
ProceedPath_noBias = StrictPartialOrder(
    nodes=[CulturalFitCheck, ManagerialReview, FinalDecision_Hire]
)
ProceedPath_noBias.order.add_edge(CulturalFitCheck, ManagerialReview)
ProceedPath_noBias.order.add_edge(ManagerialReview, FinalDecision_Hire)

# --- Skill-threshold outcome (same as in model 1) ---
SkillOutcome_noBias = OperatorPOWL(
    operator=Operator.XOR,
    children=[FinalDecision_Reject, ProceedPath_noBias]
)

# --- Root process ordering ---
root_fair = StrictPartialOrder(
    nodes=[ReceiveApplication, DataCompletenessLoop, PreliminarySkillAssessment, SkillOutcome_noBias]
)
root_fair.order.add_edge(ReceiveApplication, DataCompletenessLoop)
root_fair.order.add_edge(DataCompletenessLoop, PreliminarySkillAssessment)
root_fair.order.add_edge(PreliminarySkillAssessment, SkillOutcome_noBias)
```
