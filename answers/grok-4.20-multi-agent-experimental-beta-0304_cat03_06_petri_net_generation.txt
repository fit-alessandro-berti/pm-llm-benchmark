```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

net = PetriNet("insurance_claim_process")

# Places
start = PetriNet.Place("start")
p_after_c = PetriNet.Place("p_after_c")
p_dv_in = PetriNet.Place("p_dv_in")
p_dv_out = PetriNet.Place("p_dv_out")
p_fc_in = PetriNet.Place("p_fc_in")
p_fc_out = PetriNet.Place("p_fc_out")
p_ca_in = PetriNet.Place("p_ca_in")
p_ca_out = PetriNet.Place("p_ca_out")
p_p_in = PetriNet.Place("p_p_in")
p_n_in = PetriNet.Place("p_n_in")
p_cl_in = PetriNet.Place("p_cl_in")
end = PetriNet.Place("end")

for place in [start, p_after_c, p_dv_in, p_dv_out, p_fc_in, p_fc_out, p_ca_in, p_ca_out, p_p_in, p_n_in, p_cl_in, end]:
    net.places.add(place)

# Transitions
t_c = PetriNet.Transition("t_c", "C")
t_cr = PetriNet.Transition("t_cr", "CR")
t_dv = PetriNet.Transition("t_dv", "DV")
t_rds = PetriNet.Transition("t_rds", "RDS")
t_ok_dv = PetriNet.Transition("t_ok_dv", None)
t_fc = PetriNet.Transition("t_fc", "FC")
t_ii = PetriNet.Transition("t_ii", "II")
t_ok_fc = PetriNet.Transition("t_ok_fc", None)
t_ca = PetriNet.Transition("t_ca", "CA")
t_ap = PetriNet.Transition("t_ap", "AP")
t_skip_ap = PetriNet.Transition("t_skip_ap", None)
t_p = PetriNet.Transition("t_p", "P")
t_n = PetriNet.Transition("t_n", "N")
t_cl = PetriNet.Transition("t_cl", "CL")

for trans in [t_c, t_cr, t_dv, t_rds, t_ok_dv, t_fc, t_ii, t_ok_fc, t_ca, t_ap, t_skip_ap, t_p, t_n, t_cl]:
    net.transitions.add(trans)

# Arcs
petri_utils.add_arc_from_to(start, t_c, net)
petri_utils.add_arc_from_to(t_c, p_after_c, net)
petri_utils.add_arc_from_to(p_after_c, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_dv_in, net)
petri_utils.add_arc_from_to(p_dv_in, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_dv_out, net)
petri_utils.add_arc_from_to(p_dv_out, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_dv_in, net)
petri_utils.add_arc_from_to(p_dv_out, t_ok_dv, net)
petri_utils.add_arc_from_to(t_ok_dv, p_fc_in, net)
petri_utils.add_arc_from_to(p_fc_in, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_fc_out, net)
petri_utils.add_arc_from_to(p_fc_out, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_fc_in, net)
petri_utils.add_arc_from_to(p_fc_out, t_ok_fc, net)
petri_utils.add_arc_from_to(t_ok_fc, p_ca_in, net)
petri_utils.add_arc_from_to(p_ca_in, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_ca_out, net)
petri_utils.add_arc_from_to(p_ca_out, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_p_in, net)
petri_utils.add_arc_from_to(p_ca_out, t_skip_ap, net)
petri_utils.add_arc_from_to(t_skip_ap, p_p_in, net)
petri_utils.add_arc_from_to(p_p_in, t_p, net)
petri_utils.add_arc_from_to(t_p, p_n_in, net)
petri_utils.add_arc_from_to(p_n_in, t_n, net)
petri_utils.add_arc_from_to(t_n, p_cl_in, net)
petri_utils.add_arc_from_to(p_cl_in, t_cl, net)
petri_utils.add_arc_from_to(t_cl, end, net)

# Markings
im = Marking()
im[start] = 1
fm = Marking()
fm[end] = 1
```

This Petri net models the full insurance claim process with:
- Sequential flow through all core activities.
- Loop for RDS (re-document submission after DV, returning to re-verify).
- Loop for II (in-depth investigation after FC, returning to re-check).
- Optional approval (AP) via a silent transition for cases below threshold.
- Initial token in the start place and final token in the end place after Closure (CL).
