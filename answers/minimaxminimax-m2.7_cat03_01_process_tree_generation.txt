init_test = ProcessTree(operator=Operator.XOR, parent=root)
leaf('CE', init_test)
leaf('FBP', init_test)

loop = ProcessTree(operator=Operator.LOOP, parent=root)

#   4a. Loop body = SC  (TM X SPT)  (FCE X FBA)
loop_body = ProcessTree(operator=Operator.SEQUENCE, parent=loop)
