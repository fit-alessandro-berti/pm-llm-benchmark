```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Basic activities
SR  = Transition(label="SR")   # Service Request verification
CS  = Transition(label="CS")   # Contract Setup

DA  = Transition(label="DA")   # Datacenter Allocation
ACI = Transition(label="ACI")  # Access Control Initialization

BNC = Transition(label="BNC")  # Base Network Configuration
DR  = Transition(label="D&R")  # Debug & Reconfigure

CD  = Transition(label="CD")   # Containerized Deployment
MD  = Transition(label="MD")   # Monolithic Deployment

QA  = Transition(label="QA")   # Quality Assurance
SA  = Transition(label="SA")   # Security Audit

GLA = Transition(label="GLA")  # Go-Live Approval

# 2. Loop for configuration: *( BNC, D&R )
loop_config = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DR])

# 3. Exclusive choice for deployment: X( CD, MD )
xor_deploy = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# 4. Partial order over the main process elements
#    Elements: SR -> CS -> (DA, ACI in parallel) -> loop_config
#              -> xor_deploy -> (QA, SA in parallel) -> GLA
root = StrictPartialOrder(
    nodes=[
        SR,
        CS,
        DA,
        ACI,
        loop_config,
        xor_deploy,
        QA,
        SA,
        GLA
    ]
)

# 5. Add ordering constraints

# Request verification before contract setup
root.order.add_edge(SR, CS)

# Contract setup before environment preparation
root.order.add_edge(CS, DA)
root.order.add_edge(CS, ACI)

# Both DA and ACI must complete before looped configuration
root.order.add_edge(DA, loop_config)
root.order.add_edge(ACI, loop_config)

# Looped configuration must finish before deployment choice
root.order.add_edge(loop_config, xor_deploy)

# Deployment (CD or MD) must complete before QA and SA
root.order.add_edge(xor_deploy, QA)
root.order.add_edge(xor_deploy, SA)

# Both QA and SA must complete before Go-Live Approval
root.order.add_edge(QA, GLA)
root.order.add_edge(SA, GLA)
```
