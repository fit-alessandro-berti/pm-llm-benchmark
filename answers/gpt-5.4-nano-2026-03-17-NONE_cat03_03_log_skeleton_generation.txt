    # (e.g., every received order is eventually dispatched exactly once per occurrence pairing).
    'equivalence': {('RO', 'D')},

    'always_before': {('CA', 'RO')},

    'always_after': {('QC', 'D')},

    'never_together': {('RI', 'PI')},

        'D':  {1},          # exactly once
        'IC': {1},          # exactly once
        'RP': {0, 1}        # may occur or not depending on payment timing
    },

    # Directly-follows constraints:
    # RO -> CA, CA -> RI, QC -> PO, PO -> GSL, GSL -> D, D -> IC, IC -> RP
    'directly_follows': {
