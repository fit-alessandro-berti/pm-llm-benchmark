```sql
WITH case_variants AS (
  SELECT
    case_id,
    list(activity ORDER BY timestamp) AS variant
  FROM event_log
  GROUP BY case_id
),
variant_freq AS (
  SELECT
    variant,
    COUNT(*) AS freq,
    ROW_NUMBER() OVER (ORDER BY COUNT(*) DESC) AS rnk
  FROM case_variants
  GROUP BY variant
),
top_k_variants AS (
  SELECT variant
  FROM variant_freq
  WHERE rnk <= $1  -- $1 or ? is the parameter for K
),
filtered_cases AS (
  SELECT cv.case_id
  FROM case_variants cv
  JOIN top_k_variants tv ON cv.variant = tv.variant
)
SELECT 
  el.*
FROM event_log el
JOIN filtered_cases fc ON el.case_id = fc.case_id
ORDER BY el.case_id, el.timestamp;
```

This query uses LIST (with ORDER BY) to capture ordered activity sequences per case, groups to count variant frequencies, applies ROW_NUMBER for top-K selection (parameterized via $1), and finally returns original events only for cases matching the top K variants. DuckDB natively supports array/list comparison and aggregation as used here.
