Here are several targeted queries to investigate these anomalies further:

### **1. Find all cases with activity sequence violations**
```sql
WITH ordered_activities AS (
    SELECT 
        case_id,
        activity,
        timestamp,
        LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_activity,
        LAG(timestamp) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_timestamp
    FROM order_event_log
)
SELECT 
    case_id,
    activity,
    prev_activity,
    timestamp as current_time,
    prev_timestamp,
    EXTRACT(EPOCH FROM (timestamp - prev_timestamp))/3600 as hours_since_prev
FROM ordered_activities
WHERE 
    -- Ship before credit check
    (activity = 'Ship Goods' AND prev_activity != 'Validate Stock')
    -- Confirm shipment after shipping
    OR (activity = 'Confirm Shipment' AND 
        EXISTS (SELECT 1 FROM order_event_log e2 
                WHERE e2.case_id = ordered_activities.case_id 
                AND e2.activity = 'Ship Goods' 
                AND e2.timestamp < ordered_activities.timestamp))
    -- Payment before invoice
    OR (activity = 'Receive Payment' AND 
        NOT EXISTS (SELECT 1 FROM order_event_log e2 
                    WHERE e2.case_id = ordered_activities.case_id 
                    AND e2.activity = 'Issue Invoice' 
                    AND e2.timestamp < ordered_activities.timestamp))
ORDER BY case_id, timestamp;
```

### **2. Identify cases missing critical process steps**
```sql
WITH required_activities AS (
    SELECT unnest(ARRAY['Register Order', 'Perform Credit Check', 
                        'Validate Stock', 'Confirm Shipment', 
                        'Ship Goods', 'Issue Invoice', 'Receive Payment']) as activity
),
case_activities AS (
    SELECT DISTINCT case_id, activity 
    FROM order_event_log
)
SELECT 
    r.case_id,
    STRING_AGG(CASE WHEN ca.activity IS NULL THEN r.activity END, ', ') as missing_activities
FROM (SELECT DISTINCT case_id FROM order_event_log) r
CROSS JOIN required_activities ca
LEFT JOIN case_activities ca2 ON r.case_id = ca2.case_id AND ca.activity = ca2.activity
WHERE ca2.activity IS NULL
GROUP BY r.case_id
HAVING COUNT(CASE WHEN ca2.activity IS NULL THEN 1 END) > 0;
```

### **3. Analyze timing patterns for anomalies**
```sql
SELECT 
    case_id,
    -- Time from registration to credit check (if exists)
    EXTRACT(EPOCH FROM (
        (SELECT MIN(timestamp) FROM order_event_log e2 
         WHERE e2.case_id = e.case_id AND e2.activity = 'Perform Credit Check') -
        (SELECT MIN(timestamp) FROM order_event_log e3 
         WHERE e3.case_id = e.case_id AND e3.activity = 'Register Order')
    ))/3600 as hours_to_credit_check,
    -- Time from registration to first shipment
    EXTRACT(EPOCH FROM (
        (SELECT MIN(timestamp) FROM order_event_log e2 
         WHERE e2.case_id = e.case_id AND e2.activity = 'Ship Goods') -
        (SELECT MIN(timestamp) FROM order_event_log e3 
         WHERE e3.case_id = e.case_id AND e3.activity = 'Register Order')
    ))/3600 as hours_to_shipment,
    -- Check if payment received before invoice
    CASE WHEN 
        (SELECT MIN(timestamp) FROM order_event_log e2 
         WHERE e2.case_id = e.case_id AND e2.activity = 'Receive Payment') <
        (SELECT MIN(timestamp) FROM order_event_log e2 
         WHERE e2.case_id = e.case_id AND e2.activity = 'Issue Invoice')
        THEN 'YES' ELSE 'NO' END as payment_before_invoice,
    -- Check for missing stock validation
    CASE WHEN 
        NOT EXISTS (SELECT 1 FROM order_event_log e2 
                    WHERE e2.case_id = e.case_id AND e2.activity = 'Validate Stock')
        THEN 'YES' ELSE 'NO' END as missing_stock_validation
```

### **4. Investigate resource patterns in anomalies**
```sql
SELECT 
    r.resource_id,
    r.role,
    r.department,
    COUNT(DISTINCT el.case_id) as cases_involved,
    COUNT(*) as total_activities,
    -- Count specific anomalous activities
    SUM(CASE WHEN el.activity = 'Ship Goods' AND 
                  NOT EXISTS (SELECT 1 FROM order_event_log e2 
                              WHERE e2.case_id = el.case_id 
                              AND e2.activity = 'Validate Stock'
                              AND e2.timestamp < el.timestamp) 
             THEN 1 ELSE 0 END) as ships_without_stock_check,
    SUM(CASE WHEN el.activity = 'Confirm Shipment' AND
                  EXISTS (SELECT 1 FROM order_event_log e2 
                          WHERE e2.case_id = el.case_id 
                          AND e2.activity = 'Ship Goods'
                          AND e2.timestamp < el.timestamp)
             THEN 1 ELSE 0 END) as late_confirmations
FROM order_event_log el
JOIN resources r ON el.resource = r.resource_id
GROUP BY r.resource_id, r.role, r.department
ORDER BY cases_involved DESC;
```
