```sql
-- Top-K process variants (by number of cases) and return all events
-- from cases belonging only to those variants.

WITH case_variants AS (
    -- 1) Build ordered activity sequence per case
    SELECT
        case_id,
        string_agg(activity, ' -> ' ORDER BY "timestamp", activity) AS variant_seq
    FROM event_log
    GROUP BY case_id
),
variant_counts AS (
    -- 2) Count how many cases belong to each complete variant sequence
    SELECT
        variant_seq,
        COUNT(*) AS case_cnt
    FROM case_variants
    GROUP BY variant_seq
),
top_variants AS (
    -- 3) Select the top K variants by frequency
    SELECT
        variant_seq
    FROM variant_counts
    ORDER BY case_cnt DESC
    LIMIT :K
),
top_cases AS (
    -- Identify the case_ids that belong to the top K variants
    SELECT cv.case_id
    FROM case_variants cv
    JOIN top_variants tv
      ON cv.variant_seq = tv.variant_seq
)

-- 4) Return all events from only those top-K cases
SELECT el.*
FROM event_log el
JOIN top_cases tc
  ON el.case_id = tc.case_id
ORDER BY el.case_id, el."timestamp";
```

*Replace `:K` with your desired integer (e.g., `:K = 10`).*
