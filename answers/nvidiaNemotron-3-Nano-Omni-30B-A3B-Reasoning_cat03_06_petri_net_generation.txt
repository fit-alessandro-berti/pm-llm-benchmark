```python
# --------------------------------------------------------------
# Accepting Petri net for the insuranceclaim management process
# --------------------------------------------------------------
from pm4py.objects.petri_net.obj import PetriNet, Place, Transition, Marking
from pm4py.objects.petri_net.utils import petri_utils

# ------------------------------------------------------------------
# 1.  Build the net (places + transitions)
# ------------------------------------------------------------------
net = PetriNet(name="InsuranceClaimProcess")

# ---------- Places (states between activities) ----------
p_start   = Place(name="Claim_New")                # claim just arrived
p_cr      = Place(name="Registration_Completed")   # after CR
p_dv      = Place(name="Documents_Verified")       # after DV
p_rds     = Place(name="ReDocument_Submission")    # resubmission needed
p_fc      = Place(name="Fraud_Check_Completed")    # after FC
p_ii      = Place(name="InDepth_Investigation")    # indepth investigation
p_ca      = Place(name="Claim_Assessment_Completed") # after CA
p_ap      = Place(name="Approval_Completed")       # after AP
p_p       = Place(name="Payment_Completed")        # after P
p_n       = Place(name="Customer_Notified")        # after N
p_cl      = Place(name="Claim_Closed")             # final state

# add places to the net
for pl in [p_start, p_cr, p_dv, p_rds, p_fc, p_ii, p_ca, p_ap, p_p, p_n, p_cl]:
    net.places.add(pl)

# ---------- Transitions (activities) ----------
t_cr   = Transition(name="Claim_Registration")
t_dv   = Transition(name="Document_Verification")
t_rds  = Transition(name="ReDocument_Submission")
t_fc   = Transition(name="Fraud_Check")
t_ii   = Transition(name="InDepth_Investigation")
t_ca   = Transition(name="Claim_Assessment")
t_ap   = Transition(name="Approval")
t_p    = Transition(name="Payment")
t_n    = Transition(name="Notification")
t_cl   = Transition(name="Closure")

# add transitions to the net
for tr in [t_cr, t_dv, t_rds, t_fc, t_ii, t_ca, t_ap, t_p, t_n, t_cl]:
    net.transitions.add(tr)

# ------------------------------------------------------------------
# 2.  Connect the places and transitions with arcs
# ------------------------------------------------------------------
# CR
petri_utils.add_arc_from_to(p_start, t_cr, net)

# DV
petri_utils.add_arc_from_to(p_cr, t_dv, net)

# Redocument submission (if documents are incomplete)
petri_utils.add_arc_from_to(p_dv, t_rds, net)   # DV  RDS
petri_utils.add_arc_from_to(p_rds, t_dv, net)   # RDS  DV (after resubmission)

# Fraud check
petri_utils.add_arc_from_to(p_dv, t_fc, net)    # DV  FC (documents OK)

# Indepth investigation (if fraud doubts)
petri_utils.add_arc_from_to(p_fc, t_ii, net)   # FC  II
petri_utils.add_arc_from_to(p_ii, t_fc, net)   # II  FC (investigation finished)

# Claim assessment (can be reached from FC or from II)
petri_utils.add_arc_from_to(p_fc, t_ca, net)   # FC  CA (no further investigation)
petri_utils.add_arc_from_to(p_ii, t_ca, net)   # II  CA (after investigation)

# Approval
petri_utils.add_arc_from_to(p_ca, t_ap, net)   # CA  AP

# Payment
petri_utils.add_arc_from_to(p_ap, t_p, net)    # AP  P

# Notification
petri_utils.add_arc_from_to(p_p, t_n, net)     # P  N

# Closure
petri_utils.add_arc_from_to(p_n, t_cl, net)    # N  CL

# ------------------------------------------------------------------
# 3.  Define the initial and final markings
# ------------------------------------------------------------------
# Initial marking: a token in the "Claim_New" place (a new claim has just arrived)
im = Marking()
im[p_start] = 1

# Final marking: a token in the "Claim_Closed" place (the process has finished)
fm = Marking()
fm[p_cl] = 1
```
