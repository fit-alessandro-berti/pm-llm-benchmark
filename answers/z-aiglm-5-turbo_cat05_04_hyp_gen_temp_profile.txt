**1. Identifying Premature Closures (A to C Anomaly)**
This query identifies claims closed within 2 hours of assignment and checks whether the mandatory Evaluation (`E`) and Approval (`P`) steps are missing.
```sql
WITH event_times AS (
    SELECT claim_id,
           MAX(CASE WHEN activity = 'A' THEN timestamp END) AS assign_ts,
           MAX(CASE WHEN activity = 'C' THEN timestamp END) AS close_ts,
           MAX(CASE WHEN activity = 'E' THEN timestamp END) AS eval_ts,
           MAX(CASE WHEN activity = 'P' THEN timestamp END) AS approve_ts
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    et.claim_id, 
    c.claim_type, 
    c.claim_amount,
    EXTRACT(EPOCH FROM (et.close_ts - et.assign_ts)) / 3600 AS hours_to_close,
    CASE WHEN et.eval_ts IS NULL THEN 'Missing' ELSE 'Present' END AS evaluation_status,
    CASE WHEN et.approve_ts IS NULL THEN 'Missing' ELSE 'Present' END AS approval_status
FROM event_times et
JOIN claims c ON et.claim_id = c.claim_id
WHERE et.assign_ts IS NOT NULL 
  AND et.close_ts IS NOT NULL
  AND EXTRACT(EPOCH FROM (et.close_ts - et.assign_ts)) <= 7200;
```

**2. Detecting Skipped Approvals (E to N Anomaly)**
This query isolates claims that transition directly from Evaluation to Notify, entirely skipping the Approval step, to see if this correlates with specific claim types or amounts.

    FROM claim_events
    GROUP BY claim_id
    HAVING COUNT(CASE WHEN activity = 'P' THEN 1 END) = 0

SELECT
