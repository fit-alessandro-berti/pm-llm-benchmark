```python
# -*- coding: utf-8 -*-
"""
POWL model for the “telecom corporate network provisioning” scenario
"""

# ----------------------------------------------------------------------
# Imports from pm4py
# ----------------------------------------------------------------------
from pm4py.objects.powl.obj import (
    StrictPartialOrder,
    OperatorPOWL,
    Transition,
    SilentTransition,
)
from pm4py.objects.process_tree.obj import Operator

# ----------------------------------------------------------------------
# 1.  Atomic activities (transitions)
# ----------------------------------------------------------------------
# Request / contract
sr = Transition(label="Service Request (SR)")
cs = Transition(label="Contract Setup (CS)")

# Parallel preparation
da = Transition(label="Datacenter Allocation (DA)")
aci = Transition(label="Access Control Init (ACI)")

# Loop body
bnc = Transition(label="Base Network Config (BNC)")
dr = Transition(label="Debug & Reconfigure (D&R)")

# Exclusive deployment choice
cd = Transition(label="Containerized Deployment (CD)")
md = Transition(label="Monolithic Deployment (MD)")

# Final partialorder tasks
qa = Transition(label="Quality Assurance (QA)")
sa = Transition(label="Security Audit (SA)")
gla = Transition(label="GoLive Approval (GLA)")

# ----------------------------------------------------------------------
# 2.  Loop node (BNC followed by optional D&R, repeated until stable)
# ----------------------------------------------------------------------
# OperatorPOWL nodes are immutable – children are fixed at construction time.
loop_cfg = OperatorPOWL(operator=Operator.LOOP, children=[bnc, dr])

# ----------------------------------------------------------------------
# 3.  Exclusive choice node (deployment method)
# ----------------------------------------------------------------------
xor_deploy = OperatorPOWL(operator=Operator.XOR, children=[cd, md])

# ----------------------------------------------------------------------
# 4.  Assemble the whole process as a StrictPartialOrder
# ----------------------------------------------------------------------
# All toplevel nodes that have ordering constraints are listed here.
root = StrictPartialOrder(
    nodes=[
        sr,          # 1. request verification
        cs,          # 2. contract setup
        da,          # 3a. datacenter allocation (parallel)
        aci,         # 3b. accesscontrol init (parallel)
        loop_cfg,    # 4. configuration loop
        xor_deploy,  # 5. deployment choice
        qa,          # 6a. quality assurance (parallel)
        sa,          # 6b. security audit (parallel)
        gla,        # 7. golive approval
    ]
)

# ----------------------------------------------------------------------
# 5.  Define the partialorder edges (execution constraints)
# ----------------------------------------------------------------------
# 1  2  (request must be verified before contract)
root.order.add_edge(sr, cs)

# 2  3a & 2  3b  (contract must finish before any preparation)
root.order.add_edge(cs, da)
root.order.add_edge(cs, aci)

# 3a & 3b  4  (both preparation steps must finish before the loop)
root.order.add_edge(da, loop_cfg)
root.order.add_edge(aci, loop_cfg)

# 4  5  (configuration loop must finish before deployment choice)
root.order.add_edge(loop_cfg, xor_deploy)

# 5  6a & 5  6b  (deployment must finish before QA and SA)
root.order.add_edge(xor_deploy, qa)
root.order.add_edge(xor_deploy, sa)

# 6a & 6b  7  (both QA and SA must finish before golive approval)
root.order.add_edge(qa, gla)
root.order.add_edge(sa, gla)

# ----------------------------------------------------------------------
# 6.  (Optional) Silent transition example – not required for this scenario
# ----------------------------------------------------------------------
# If you ever need a silent step, you can create it as follows:
# silent_step = SilentTransition()
# and add it to the order just like any other node.

# ----------------------------------------------------------------------
# The variable `root` now holds the complete POWL model for the
# described telecom provisioning process.
# ----------------------------------------------------------------------
```
