Below is one possible accepting Petri net for the described insuranceclaim process.  
It models the DVRDS loop and the FCII loop, with a single token in a start place and a final token in the closedclaim place.

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# create net
net = PetriNet("insurance_claim_process")

# ----- Places (states) -----
p_start   = PetriNet.Place("start")          # new claim arrives
p_after_C = PetriNet.Place("after_C")
p_after_CR = PetriNet.Place("after_CR")      # registration done
p_after_DV = PetriNet.Place("after_DV")
p_after_FC = PetriNet.Place("after_FC")
p_after_II = PetriNet.Place("after_II")
p_after_CA = PetriNet.Place("after_CA")
p_after_AP = PetriNet.Place("after_AP")
p_after_P  = PetriNet.Place("after_P")
p_after_N  = PetriNet.Place("after_N")
p_end      = PetriNet.Place("end")           # claim closed

net.places.update({
    p_start, p_after_C, p_after_CR, p_after_DV, p_after_FC,
    p_after_II, p_after_CA, p_after_AP, p_after_P, p_after_N, p_end
})

# ----- Transitions (activities) -----
t_C   = PetriNet.Transition("t_C",   "Insurance Claim")
t_CR  = PetriNet.Transition("t_CR",  "Claim Registration")
t_DV  = PetriNet.Transition("t_DV",  "Document Verification")
t_RDS = PetriNet.Transition("t_RDS", "Re-Document Submission")
t_FC1 = PetriNet.Transition("t_FC1", "Fraud Check")            # FC after DV
t_II  = PetriNet.Transition("t_II",  "In-depth Investigation")
t_FC2 = PetriNet.Transition("t_FC2", "Fraud Check")            # FC after II
t_CA  = PetriNet.Transition("t_CA",  "Claim Assessment")
t_AP  = PetriNet.Transition("t_AP",  "Approval")
t_P   = PetriNet.Transition("t_P",   "Payment")
t_N   = PetriNet.Transition("t_N",   "Notification")
t_CL  = PetriNet.Transition("t_CL",  "Closure")

net.transitions.update({
    t_C, t_CR, t_DV, t_RDS, t_FC1, t_II, t_FC2,
    t_CA, t_AP, t_P, t_N, t_CL
})

# ----- Arcs -----
petri_utils.add_arc_from_to(p_start, t_C, net)
petri_utils.add_arc_from_to(t_C, p_after_C, net)

petri_utils.add_arc_from_to(p_after_C, t_CR, net)
petri_utils.add_arc_from_to(t_CR, p_after_CR, net)

petri_utils.add_arc_from_to(p_after_CR, t_DV, net)
petri_utils.add_arc_from_to(t_DV, p_after_DV, net)

# DV  RDS loop
petri_utils.add_arc_from_to(p_after_DV, t_RDS, net)
petri_utils.add_arc_from_to(t_RDS, p_after_CR, net)

# FC after DV
petri_utils.add_arc_from_to(p_after_DV, t_FC1, net)
petri_utils.add_arc_from_to(t_FC1, p_after_FC, net)

# FC  II loop
petri_utils.add_arc_from_to(p_after_FC, t_II, net)
petri_utils.add_arc_from_to(t_II, p_after_II, net)
petri_utils.add_arc_from_to(p_after_II, t_FC2, net)
petri_utils.add_arc_from_to(t_FC2, p_after_FC, net)

# Continue after fraud check
petri_utils.add_arc_from_to(p_after_FC, t_CA, net)
petri_utils.add_arc_from_to(t_CA, p_after_CA, net)

petri_utils.add_arc_from_to(p_after_CA, t_AP, net)
petri_utils.add_arc_from_to(t_AP, p_after_AP, net)

petri_utils.add_arc_from_to(p_after_AP, t_P, net)
petri_utils.add_arc_from_to(t_P, p_after_P, net)

petri_utils.add_arc_from_to(p_after_P, t_N, net)
petri_utils.add_arc_from_to(t_N, p_after_N, net)

petri_utils.add_arc_from_to(p_after_N, t_CL, net)
petri_utils.add_arc_from_to(t_CL, p_end, net)

# ----- Markings -----
im = Marking({p_start: 1})  # initial marking
fm = Marking({p_end: 1})    # final marking
```

This Petri net is an **accepting Petri net** with a single token at the start and a final token in the closed-claim place. The DVRDS and FCII loops allow multiple repetitions as required.
