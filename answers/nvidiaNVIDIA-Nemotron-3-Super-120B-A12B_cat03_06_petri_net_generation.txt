```python
# Accepting Petri net for the insurance claim management process
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="Insurance Claim Management")

# ----- Places (states) -----
p_start   = PetriNet.Place(name="Start")
p_CR      = PetriNet.Place(name="Claim Registration")
p_DV      = PetriNet.Place(name="Document Verification")
p_FC      = PetriNet.Place(name="Fraud Check")
p_CA      = PetriNet.Place(name="Claim Assessment")
p_AP      = PetriNet.Place(name="Approval")
p_P       = PetriNet.Place(name="Payment")
p_N       = PetriNet.Place(name="Notification")
p_CL      = PetriNet.Place(name="Closure")
p_RDS     = PetriNet.Place(name="Re-Document Submission")
p_II      = PetriNet.Place(name="In-depth Investigation")
p_end     = PetriNet.Place(name="End")   # final place (closed claim)

# Add all places to the net
for place in (p_start, p_CR, p_DV, p_FC, p_CA, p_AP, p_P, p_N, p_CL, p_RDS, p_II, p_end):
    net.places.add(place)

# ----- Transitions (arcs) -----
# Main flow
petri_utils.add_arc_from_to(p_start, p_CR, net, "Claim Registration")
petri_utils.add_arc_from_to(p_CR, p_DV, net, "Document Verification")
petri_utils.add_arc_from_to(p_DV, p_FC, net, "Fraud Check")
petri_utils.add_arc_from_to(p_FC, p_CA, net, "Claim Assessment")
petri_utils.add_arc_from_to(p_CA, p_AP, net, "Approval")
petri_utils.add_arc_from_to(p_AP, p_P, net, "Payment")
petri_utils.add_arc_from_to(p_P, p_N, net, "Notification")
petri_utils.add_arc_from_to(p_N, p_CL, net, "Closure")

# Loop: incomplete documents  resubmit and retry verification
petri_utils.add_arc_from_to(p_DV, p_RDS, net, "Re-Document Submission")
petri_utils.add_arc_from_to(p_RDS, p_DV, net, "Document Verification (retry)")

# Loop: fraud doubts  indepth investigation and retry fraud check
petri_utils.add_arc_from_to(p_FC, p_II, net, "In-depth Investigation")
petri_utils.add_arc_from_to(p_II, p_FC, net, "Fraud Check (retry)")

# Final step to closed claim
petri_utils.add_arc_from_to(p_CL, p_end, net, "Closure Completed")

# ----- Markings (tokens) -----
# Initial marking: one token in the start place (new claim arrives)
im = Marking()
im[p_start] = 1

# Final marking: one token in the end place (claim closed)
fm = Marking()
fm[p_end] = 1

# (Optional) remove the sink place and its token to mimic the example in the prompt
petri_utils.remove_place(net, p_end)
del fm[p_end]

# The Petri net `net` now represents an accepting Petri net for the scenario,
# with places for each activity, transitions labeled by the activity names,
# possible loops (DV  RDS and FC  II), and initial/final markings as required.
```
