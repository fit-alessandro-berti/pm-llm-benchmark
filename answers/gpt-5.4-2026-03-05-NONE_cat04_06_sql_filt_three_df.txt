```sql
WITH ordered_events AS (
    SELECT
        case_id,
        activity,
        "timestamp",
        LEAD(activity, 1) OVER (
            PARTITION BY case_id
            ORDER BY "timestamp"
        ) AS next_activity_1,
        LEAD(activity, 2) OVER (
            PARTITION BY case_id
            ORDER BY "timestamp"
        ) AS next_activity_2,
        LEAD("timestamp", 1) OVER (
            PARTITION BY case_id
            ORDER BY "timestamp"
        ) AS next_timestamp_1,
        LEAD("timestamp", 2) OVER (
            PARTITION BY case_id
            ORDER BY "timestamp"
        ) AS next_timestamp_2
    FROM event_log
),
cases_to_exclude AS (
    SELECT DISTINCT case_id
    FROM ordered_events
    WHERE activity = 'Approve Purchase Requisition'
      AND next_activity_1 = 'Create Order'
      AND next_activity_2 = 'Confirm Order'
      AND next_timestamp_2 - next_timestamp_1 > INTERVAL 5 DAY
)
SELECT e.*
FROM event_log e
WHERE e.case_id NOT IN (
    SELECT case_id
    FROM cases_to_exclude
)
ORDER BY e.case_id, e."timestamp";
```

If you want to make the ordering deterministic when multiple events in the same case share the exact same timestamp, add a tie-breaker column in the `ORDER BY` inside the window functions. For example, if you have an event id:

```sql
ORDER BY "timestamp", event_id
```
