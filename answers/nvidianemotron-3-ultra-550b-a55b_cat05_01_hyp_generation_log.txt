### Proposed SQL Investigation Queries

The following queries are designed to quantify these anomalies across the full dataset, test the hypotheses (Priority vs. Standard, Missing Events, Resource Behavior), and identify systemic issues.

#### 1. Detect Sequence Violations (Conformance Checking)
Identifies all cases where the actual activity order deviates from the defined "Happy Path" sequence (1-7). Uses `LAG` to compare current activity rank vs previous activity rank.

```sql
WITH activity_rank AS (
    -- Define the expected sequence order
    SELECT 
        oel.*,
        CASE activity
            WHEN 'Register Order'       THEN 1
            WHEN 'Perform Credit Check' THEN 2
            WHEN 'Validate Stock'       THEN 3
            WHEN 'Confirm Shipment'     THEN 4
            WHEN 'Ship Goods'           THEN 5
            WHEN 'Issue Invoice'        THEN 6
            WHEN 'Receive Payment'      THEN 7
            ELSE 99
        END AS expected_seq
    FROM order_event_log oel
),
sequence_violations AS (
    SELECT
        case_id,
        event_id,
        activity,
        timestamp,
        resource,
        expected_seq,
        LAG(expected_seq) OVER (PARTITION BY case_id ORDER BY timestamp) AS prev_seq,
        LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp) AS prev_activity
    FROM activity_rank
)
SELECT 
    case_id,
    event_id,
    prev_activity,
    activity,
    timestamp,
    resource,
    CASE 
        WHEN expected_seq < prev_seq THEN 'BACKWARD_JUMP (Skipped Step?)'
        WHEN expected_seq = prev_seq THEN 'DUPLICATE_STEP'
        WHEN expected_seq > prev_seq + 1 THEN 'FORWARD_JUMP (Skipped Mandatory Steps)'
        ELSE 'OK'
    END AS violation_type
FROM sequence_violations
WHERE prev_seq IS NOT NULL 
  AND (expected_seq < prev_seq OR expected_seq > prev_seq + 1)
ORDER BY case_id, timestamp;
```

#### 2. Identify Missing Mandatory Activities per Case
Checks for the absence of critical control steps (`Perform Credit Check`, `Validate Stock`) grouped by Order Type to test the "Priority Order Bypass" hypothesis.

```sql
SELECT 
    o.order_type,
    oel.case_id,
    -- Flags for mandatory steps
    MAX(CASE WHEN oel.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) AS has_credit_check,
    MAX(CASE WHEN oel.activity = 'Validate Stock' THEN 1 ELSE 0 END) AS has_stock_validation,
    MAX(CASE WHEN oel.activity = 'Confirm Shipment' THEN 1 ELSE 0 END) AS has_shipment_confirm,
    COUNT(*) AS total_events
FROM orders o
JOIN order_event_log oel ON o.case_id = oel.case_id
GROUP BY o.order_type, oel.case_id
HAVING 
    MAX(CASE WHEN oel.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) = 0
    OR MAX(CASE WHEN oel.activity = 'Validate Stock' THEN 1 ELSE 0 END) = 0
ORDER BY o.order_type, oel.case_id;
```

#### 3. Analyze Payment vs. Invoice Sequence & Logistics Contradictions
Specifically targets Case 1004 patterns: Payment before Invoice, and `Ship Goods` after `Confirm Shipment = 'N'`.

```sql
WITH case_event_flags AS (
    SELECT
        case_id,
        -- Payment/Invoice Timing
        MIN(CASE WHEN activity = 'Receive Payment' THEN timestamp END) AS payment_ts,
        MIN(CASE WHEN activity = 'Issue Invoice' THEN timestamp END) AS invoice_ts,
        -- Shipment Logic
        MAX(CASE WHEN activity = 'Confirm Shipment' AND additional_info ILIKE '%shipment_scheduled=N%' THEN 1 ELSE 0 END) AS confirm_negative,
        MAX(CASE WHEN activity = 'Ship Goods' THEN 1 ELSE 0 END) AS has_ship_goods,
        MAX(CASE WHEN activity = 'Confirm Shipment' THEN timestamp END) AS confirm_ts,
        MIN(CASE WHEN activity = 'Ship Goods' THEN timestamp END) AS ship_ts
    FROM order_event_log
    GROUP BY case_id
)
SELECT
    case_id,
    payment_ts,
    invoice_ts,
    confirm_ts,
    ship_ts,
    CASE 
        WHEN payment_ts < invoice_ts THEN 'PAYMENT_BEFORE_INVOICE' 
        ELSE 'OK' 
    END AS payment_anomaly,
    CASE 
        WHEN confirm_negative = 1 AND has_ship_goods = 1 THEN 'SHIPPED_DESPITE_NEGATIVE_CONFIRM'
        WHEN confirm_ts > ship_ts THEN 'CONFIRM_AFTER_SHIP' -- Case 1003 pattern
        ELSE 'OK'
    END AS logistics_anomaly
FROM case_event_flags
WHERE 
    payment_ts < invoice_ts
    OR (confirm_negative = 1 AND has_ship_goods = 1)
    OR confirm_ts > ship_ts;
```

#### 4. Resource/Role Compliance Analysis (Segregation of Duties)
Verifies if activities are performed by the correct Department/Role (e.g., Finance doing Credit Checks, Logistics doing Shipping). Flags Case 1002 where Logistics acted before Finance.

```sql
SELECT 
    oel.case_id,
    oel.activity,
    oel.resource,
    r.role,
    r.department,
    oel.timestamp,
    -- Expected Department per Activity
    CASE oel.activity
        WHEN 'Register Order'       THEN 'Sales'
        WHEN 'Perform Credit Check' THEN 'Finance'
        WHEN 'Validate Stock'       THEN 'Logistics'
        WHEN 'Confirm Shipment'     THEN 'Logistics'
        WHEN 'Ship Goods'           THEN 'Logistics'
        WHEN 'Issue Invoice'        THEN 'Finance'
        WHEN 'Receive Payment'      THEN 'Finance'
    END AS expected_department,
    CASE 
        WHEN r.department <> 
            CASE oel.activity
                WHEN 'Register Order'       THEN 'Sales'
                WHEN 'Perform Credit Check' THEN 'Finance'
                WHEN 'Validate Stock'       THEN 'Logistics'
                WHEN 'Confirm Shipment'     THEN 'Logistics'
                WHEN 'Ship Goods'           THEN 'Logistics'
                WHEN 'Issue Invoice'        THEN 'Finance'
                WHEN 'Receive Payment'      THEN 'Finance'
            END THEN 'ROLE_VIOLATION'
        ELSE 'COMPLIANT'
    END AS duty_segregation_status
FROM order_event_log oel
JOIN resources r ON oel.resource = r.resource_id
ORDER BY oel.case_id, oel.timestamp;
```

#### 5. "Fast Track" Hypothesis Test: Priority Orders Throughput vs. Compliance
Compares Cycle Time (Register to Ship) and Compliance Rate between Standard and Priority orders.

```sql
WITH case_milestones AS (
    SELECT
        o.case_id,
        o.order_type,
        o.order_value,
        MIN(CASE WHEN oel.activity = 'Register Order' THEN oel.timestamp END) AS reg_ts,
        MIN(CASE WHEN oel.activity = 'Ship Goods' THEN oel.timestamp END) AS ship_ts,
        MAX(CASE WHEN oel.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) AS did_credit,
        MAX(CASE WHEN oel.activity = 'Validate Stock' THEN 1 ELSE 0 END) AS did_stock
    FROM orders o
    JOIN order_event_log oel ON o.case_id = oel.case_id
    GROUP BY o.case_id, o.order_type, o.order_value
)
SELECT
    order_type,
    COUNT(*) AS total_orders,
    AVG(EXTRACT(EPOCH FROM (ship_ts - reg_ts))/3600) AS avg_hours_to_ship,
    SUM(did_credit)::float / COUNT(*) AS credit_check_compliance_rate,
    SUM(did_stock)::float / COUNT(*) AS stock_check_compliance_rate
FROM case_milestones
WHERE reg_ts IS NOT NULL AND ship_ts IS NOT NULL
GROUP BY order_type;
```

#### 6. Forensic Audit on `additional_info` Flags
Parses the unstructured `additional_info` column to quantify specific operational flags mentioned in the logs (`attempted_early`, `late_confirmation`, `shipment_scheduled=N`).

```sql
SELECT
    activity,
    additional_info,
    CASE
        WHEN additional_info ILIKE '%attempted_early=Y%' THEN 'EARLY_SHIP_ATTEMPT'
        WHEN additional_info ILIKE '%late_confirmation=Y%' THEN 'LATE_CONFIRMATION'
        WHEN additional_info ILIKE '%shipment_scheduled=N%' THEN 'NEGATIVE_SCHEDULE_FLAG'
        WHEN additional_info ILIKE '%credit_score=%' THEN 'CREDIT_SCORE_LOGGED'
        ELSE 'OTHER'
    END AS parsed_flag,
    COUNT(*) AS frequency,
    COUNT(DISTINCT case_id) AS affected_cases
FROM order_event_log
WHERE additional_info IS NOT NULL AND additional_info <> ''
GROUP BY activity, additional_info, parsed_flag
ORDER BY activity, frequency DESC;
```
