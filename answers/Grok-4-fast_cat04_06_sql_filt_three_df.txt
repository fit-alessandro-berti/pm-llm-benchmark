```sql
WITH ordered_events AS (
  SELECT 
    case_id, 
    activity, 
    timestamp, 
    ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS rn
  FROM event_log
),
bad_cases AS (
  SELECT DISTINCT oe.case_id
  FROM ordered_events oe
  WHERE oe.activity = 'Approve Purchase Requisition'
    AND LEAD(oe.activity, 1) OVER (PARTITION BY oe.case_id ORDER BY oe.rn) = 'Create Order'
    AND LEAD(oe.activity, 2) OVER (PARTITION BY oe.case_id ORDER BY oe.rn) = 'Confirm Order'
    AND (LEAD(oe.timestamp, 2) OVER (PARTITION BY oe.case_id ORDER BY oe.rn) - 
         LEAD(oe.timestamp, 1) OVER (PARTITION BY oe.case_id ORDER BY oe.rn)) > INTERVAL '5' DAY
)
SELECT el.*
FROM event_log el
LEFT JOIN bad_cases bc ON el.case_id = bc.case_id
WHERE bc.case_id IS NULL
ORDER BY el.case_id, el.timestamp;
```
