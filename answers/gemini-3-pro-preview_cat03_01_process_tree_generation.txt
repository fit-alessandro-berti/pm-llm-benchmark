```python
from pm4py.objects.process_tree.obj import ProcessTree, Operator

def create_hospital_process_tree():
    """
    Constructs the Process Tree for the complex hospital patient management scenario.
    """

    # --- ROOT NODE ---
    # The overall process is a sequence of stages (1 through 7).
    root = ProcessTree(operator=Operator.SEQUENCE)

    # ---------------------------------------------------------
    # STAGE 1 & 2: Arrival, Registration, Intake, Triage
    # Logic: Sequence A -> RG -> IA -> TI -> TC
    # We attach these directly to the root sequence.
    # ---------------------------------------------------------
    
    # A: Arrival
    node_a = ProcessTree(label="A", parent=root)
    root.children.append(node_a)

    # RG: Registration
    node_rg = ProcessTree(label="RG", parent=root)
    root.children.append(node_rg)

    # IA: Intake Assessment
    node_ia = ProcessTree(label="IA", parent=root)
    root.children.append(node_ia)

    # TI: Triage Interview
    node_ti = ProcessTree(label="TI", parent=root)
    root.children.append(node_ti)

    # TC: Triage Classification
    node_tc = ProcessTree(label="TC", parent=root)
    root.children.append(node_tc)

    # ---------------------------------------------------------
    # STAGE 3: Initial Diagnostic Test
    # Logic: Exclusive Choice (XOR) between CE and FBP
    # ---------------------------------------------------------
    
    step3_xor = ProcessTree(operator=Operator.XOR, parent=root)
    root.children.append(step3_xor)

    # CE: Comprehensive ECG
    node_ce = ProcessTree(label="CE", parent=step3_xor)
    step3_xor.children.append(node_ce)

    # FBP: Full Blood Panel
    node_fbp = ProcessTree(label="FBP", parent=step3_xor)
    step3_xor.children.append(node_fbp)

    # ---------------------------------------------------------
    # STAGE 4: Treatment and Re-Diagnosis Loop
    # Logic: Loop * ( A, B )
    # The loop commences if results are abnormal. 
    # A (Body) = Silent Step (Tau). This allows the loop to be skipped (exit immediately) 
    # if the initial test results were normal.
    # B (Redo) = The Sequence of Specialist -> Treatment -> Re-test.
    # ---------------------------------------------------------

    step4_loop = ProcessTree(operator=Operator.LOOP, parent=root)
    root.children.append(step4_loop)

    # Loop Body (A): Silent Step (Tau)
    # Represented by label=None
    loop_body_tau = ProcessTree(label=None, parent=step4_loop)
    step4_loop.children.append(loop_body_tau)

    # Loop Redo (B): Sequence of SC -> Choice(TM, SPT) -> Choice(FCE, FBA)
    loop_redo_seq = ProcessTree(operator=Operator.SEQUENCE, parent=step4_loop)
    step4_loop.children.append(loop_redo_seq)

    # Inside Redo: Specialist Consultation (SC)
    node_sc = ProcessTree(label="SC", parent=loop_redo_seq)
    loop_redo_seq.children.append(node_sc)

    # Inside Redo: Treatment Choice (XOR: TM or SPT)
    treat_xor = ProcessTree(operator=Operator.XOR, parent=loop_redo_seq)
    loop_redo_seq.children.append(treat_xor)

    node_tm = ProcessTree(label="TM", parent=treat_xor)
    treat_xor.children.append(node_tm)

    node_spt = ProcessTree(label="SPT", parent=treat_xor)
    treat_xor.children.append(node_spt)

    # Inside Redo: Re-Diagnostic Test Choice (XOR: FCE or FBA)
    rdt_xor = ProcessTree(operator=Operator.XOR, parent=loop_redo_seq)
    loop_redo_seq.children.append(rdt_xor)

    node_fce = ProcessTree(label="FCE", parent=rdt_xor)
    rdt_xor.children.append(node_fce)

    node_fba = ProcessTree(label="FBA", parent=rdt_xor)
    rdt_xor.children.append(node_fba)

    # ---------------------------------------------------------
    # STAGE 5: Advanced Imaging & Lifestyle Support
    # Logic: True Concurrency (Parallel) between AI and NC
    # ---------------------------------------------------------

    step5_par = ProcessTree(operator=Operator.PARALLEL, parent=root)
    root.children.append(step5_par)

    # AI: Advanced Imaging
    node_ai = ProcessTree(label="AI", parent=step5_par)
    step5_par.children.append(node_ai)

    # NC: Nutritional Counseling
    node_nc = ProcessTree(label="NC", parent=step5_par)
    step5_par.children.append(node_nc)

    # ---------------------------------------------------------
    # STAGE 6: Discharge Instructions
    # Logic: Sequence FAS -> DDI
    # We attach these directly to the root sequence (continuing the main flow).
    # ---------------------------------------------------------

    # FAS: Follow-up Appointment Scheduling
    node_fas = ProcessTree(label="FAS", parent=root)
    root.children.append(node_fas)

    # DDI: Detailed Discharge Instructions
    node_ddi = ProcessTree(label="DDI", parent=root)
    root.children.append(node_ddi)

    # ---------------------------------------------------------
    # STAGE 7: Final Admin & Financials
    # Logic: True Concurrency (Parallel) between BL, ICP, MRF
    # ---------------------------------------------------------

    step7_par = ProcessTree(operator=Operator.PARALLEL, parent=root)
    root.children.append(step7_par)

    # BL: Billing
    node_bl = ProcessTree(label="BL", parent=step7_par)
    step7_par.children.append(node_bl)

    # ICP: Insurance Claims Processing
    node_icp = ProcessTree(label="ICP", parent=step7_par)
    step7_par.children.append(node_icp)

    # MRF: Medical Records Finalization
    node_mrf = ProcessTree(label="MRF", parent=step7_par)
    step7_par.children.append(node_mrf)

    return root

# Construct the tree
process_tree = create_hospital_process_tree()
```
