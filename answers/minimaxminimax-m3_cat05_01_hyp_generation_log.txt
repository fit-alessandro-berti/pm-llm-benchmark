### 3.1 High-Level: Build the Activity Sequence per Case

Reconstruct the actual sequence each case followed so that any deviation from the assumed process flow becomes immediately visible.

```sql
WITH case_sequence AS (
  SELECT
    case_id,
    array_agg(activity ORDER BY timestamp) AS activity_sequence,
    array_agg(timestamp ORDER BY timestamp) AS timestamps,
    COUNT(*) AS num_events
  FROM order_event_log
  GROUP BY case_id
)
SELECT
  c.case_id,
  o.order_type,
  o.order_value,
  c.activity_sequence,
  c.num_events
FROM case_sequence c
JOIN orders o USING (case_id)
ORDER BY c.case_id;
```

### 3.5 Find Cases with Missing Activities (Hypothesis H7)

```sql
WITH expected(activity) AS (
  VALUES
    ('Register Order'),
    ('Perform Credit Check'),
    ('Validate Stock'),
    ('Confirm Shipment'),
    ('Ship Goods'),
    ('Issue Invoice'),
    ('Receive Payment')
),
actual AS (
  SELECT DISTINCT case_id, activity
  FROM order_event_log
)
SELECT
  a.case_id,
  e.activity AS missing_activity
FROM actual a
CROSS JOIN expected e
WHERE NOT EXISTS (
  SELECT 1 FROM actual a2
  WHERE a2.case_id = a.case_id AND a2.activity = e.activity
)
ORDER BY a.case_id, e.activity;
```

### 3.6 Time-Gap Analysis: Identify Unusually Short / Long Gaps (Hypothesis H6)

```sql
WITH timeline AS (
  SELECT
    case_id,
    activity,
    timestamp,
    LAG(timestamp) OVER (PARTITION BY case_id ORDER BY timestamp) AS prev_timestamp,
    LAG(activity)  OVER (PARTITION BY case_id ORDER BY timestamp) AS prev_activity
  FROM order_event_log
)
SELECT
  case_id,
  prev_activity,
  activity,
  prev_timestamp,
  timestamp,
  EXTRACT(EPOCH FROM (timestamp - prev_timestamp)) AS gap_seconds
FROM timeline
WHERE prev_timestamp IS NOT NULL
ORDER BY gap_seconds ASC
LIMIT 20;
```

### 3.7 Resource Pattern: Who Performs Out-of-Order Activities? (Hypothesis H4, H8)

```sql
WITH ranked AS (
  SELECT
    case_id,
    activity,
    timestamp,
    resource,
    ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS step_order
  FROM order_event_log
),
violators AS (
  SELECT r.*
  FROM ranked r
  WHERE (r.activity = 'Ship Goods'
         AND EXISTS (SELECT 1 FROM ranked r2
                     WHERE r2.case_id = r.case_id
                       AND r2.activity = 'Perform Credit Check'
                       AND r2.timestamp > r.timestamp))
     OR (r.activity = 'Receive Payment'
         AND EXISTS (SELECT 1 FROM ranked r2
                     WHERE r2.case_id = r.case_id
                       AND r2.activity = 'Ship Goods'
                       AND r2.timestamp > r.timestamp))
)
SELECT
  v.resource,
  res.role,
  res.department,
  COUNT(*) AS anomaly_count
FROM violators v
JOIN resources res ON res.resource_id = v.resource
GROUP BY v.resource, res.role, res.department
ORDER BY anomaly_count DESC;
```

### 3.8 Cross-Reference Anomalies with Order Type (Hypothesis H2)

```sql
WITH anomaly_cases AS (
  SELECT DISTINCT case_id
  FROM order_event_log
  WHERE additional_info IN ('attempted_early=Y', 'late_confirmation=Y', 'shipment_scheduled=N')
     OR activity = 'Receive Payment'
)
SELECT
  o.order_type,
  COUNT(DISTINCT a.case_id) AS anomalous_cases,
  COUNT(DISTINCT o.case_id) AS total_cases,
  ROUND(100.0 * COUNT(DISTINCT a.case_id) / COUNT(DISTINCT o.case_id), 2) AS anomaly_rate_pct
FROM orders o
LEFT JOIN anomaly_cases a ON a.case_id = o.case_id
GROUP BY o.order_type
ORDER BY anomaly_rate_pct DESC;
```

### 3.12 Cycle-Time vs. Anomaly Indicator

```sql
WITH case_durations AS (
  SELECT
    case_id,
    MIN(timestamp) AS start_time,
    MAX(timestamp) AS end_time,
    MAX(timestamp) - MIN(timestamp) AS total_duration
  FROM order_event_log
  GROUP BY case_id
)
SELECT
  cd.case_id,
  o.order_type,
  cd.total_duration,
  EXISTS (
    SELECT 1 FROM order_event_log e
```
