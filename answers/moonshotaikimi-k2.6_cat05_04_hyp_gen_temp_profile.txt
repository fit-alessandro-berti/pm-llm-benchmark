### 1. Detect R  P Claims with Suspiciously Rigid Timing
Identifies approvals that deviate from the 25-hour mean by more than 3 hours, which should be rare if the process were normally distributed.

```sql

  SELECT 
    ce.claim_id,
    cl.claim_type,
    MIN(ce.timestamp) FILTER (WHERE ce.activity = 'R') AS t_r,
    MIN(ce.timestamp) FILTER (WHERE ce.activity = 'P') AS t_p
  FROM claim_events ce
  JOIN claims cl ON cl.claim_id = ce.claim_id
  GROUP BY ce.claim_id, cl.claim_type
  HAVING MIN(ce.timestamp) FILTER (WHERE ce.activity = 'R') IS NOT NULL
     AND MIN(ce.timestamp) FILTER (WHERE ce.activity = 'P') IS NOT NULL
)
SELECT 
  claim_id,
  claim_type,
  EXTRACT(EPOCH FROM (t_p - t_r)) AS r_p_seconds,
  ABS(EXTRACT(EPOCH FROM (t_p - t_r)) - 90000.0) / 3600.0 AS deviation_hours
FROM r_p_intervals
WHERE ABS(EXTRACT(EPOCH FROM (t_p - t_r)) - 90000.0) > 3 * 3600
ORDER BY deviation_hours DESC;
```
