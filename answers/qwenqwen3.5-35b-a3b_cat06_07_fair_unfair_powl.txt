import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- Model 1: POWL Model with Potential Unfairness ---
# This model includes the XOR branch where 'CommunityAffiliationCheck' provides a subtle advantage.

# 1. Define Activities (Transitions)
ReceiveApplication = Transition(label="ReceiveApplication")
DataCompletenessCheck = Transition(label="DataCompletenessCheck")
RequestMoreInfo = Transition(label="RequestMoreInfo")
SkillAssessment = Transition(label="SkillAssessment")
CulturalFitCheck = Transition(label="CulturalFitCheck")
CommunityAffiliationCheck = Transition(label="CommunityAffiliationCheck")
ManagerialReview = Transition(label="ManagerialReview")
FinalDecision = Transition(label="FinalDecision")

# 2. Define Control Flow Operators
# Loop: Execute DataCompletenessCheck, then either exit or execute RequestMoreInfo and loop back
data_loop = OperatorPOWL(
    operator=Operator.LOOP, 
    children=[DataCompletenessCheck, RequestMoreInfo]
)

# XOR: Choose between standard CulturalFitCheck OR CommunityAffiliationCheck (Bias Point)
unfair_cultural_branch = OperatorPOWL(
    operator=Operator.XOR, 
    children=[CulturalFitCheck, CommunityAffiliationCheck]
)

# 3. Define Partial Order (Sequence)
# Nodes: All major steps including operators
nodes_unfair = [
    ReceiveApplication, 
    data_loop, 
    SkillAssessment, 
    unfair_cultural_branch, 
    ManagerialReview, 
    FinalDecision
]

model_unfair = StrictPartialOrder(nodes=nodes_unfair)

# Add edges to enforce sequence
model_unfair.order.add_edge(ReceiveApplication, data_loop)
model_unfair.order.add_edge(data_loop, SkillAssessment)
model_unfair.order.add_edge(SkillAssessment, unfair_cultural_branch)
model_unfair.order.add_edge(unfair_cultural_branch, ManagerialReview)
model_unfair.order.add_edge(ManagerialReview, FinalDecision)


# --- Model 2: POWL Model Without Unfairness ---
# This model removes the selective XOR branch. All applicants undergo the same CulturalFitCheck.

# 1. Define Activities (Transitions)
# (Reusing labels for clarity, though logically they are new instances in this scope)
ReceiveApplication_Fair = Transition(label="ReceiveApplication")
DataCompletenessCheck_Fair = Transition(label="DataCompletenessCheck")
RequestMoreInfo_Fair = Transition(label="RequestMoreInfo")
SkillAssessment_Fair = Transition(label="SkillAssessment")
CulturalFitCheck_Fair = Transition(label="CulturalFitCheck")
ManagerialReview_Fair = Transition(label="ManagerialReview")
FinalDecision_Fair = Transition(label="FinalDecision")

# 2. Define Control Flow Operators
# Loop: Same logic as Model 1
data_loop_fair = OperatorPOWL(
    operator=Operator.LOOP, 
    children=[DataCompletenessCheck_Fair, RequestMoreInfo_Fair]
)

# No XOR operator here. CulturalFitCheck is a direct transition.
# (No unfair branch exists)

# 3. Define Partial Order (Sequence)
nodes_fair = [
    ReceiveApplication_Fair, 
    data_loop_fair, 
    SkillAssessment_Fair, 
    CulturalFitCheck_Fair, 
    ManagerialReview_Fair, 
    FinalDecision_Fair
]

model_fair = StrictPartialOrder(nodes=nodes_fair)

# Add edges to enforce sequence
model_fair.order.add_edge(ReceiveApplication_Fair, data_loop_fair)
model_fair.order.add_edge(data_loop_fair, SkillAssessment_Fair)
model_fair.order.add_edge(SkillAssessment_Fair, CulturalFitCheck_Fair)
model_fair.order.add_edge(CulturalFitCheck_Fair, ManagerialReview_Fair)
model_fair.order.add_edge(ManagerialReview_Fair, FinalDecision_Fair)

# Output Summary (Optional verification)
print("Model 1 (Unfair) Created Successfully.")

print(f"Nodes: {[str(n) for n in model_fair.nodes]}")
