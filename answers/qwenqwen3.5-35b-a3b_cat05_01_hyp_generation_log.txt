### Query 1: Sequence Validation (Detecting Out-of-Order Activities)
This query assigns a sequence number to each event per case and checks if the activities appear in the expected order. It flags cases where a later step in the process appears before an earlier step.

```sql
WITH event_sequence AS (
    SELECT 
        case_id,
        activity,
        timestamp,
        ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) as event_order,
        LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_activity
    FROM order_event_log
),
expected_flow AS (
    -- Define the expected sequence logic
    SELECT 
        case_id,
        activity,
        event_order,
        -- Check for specific known anomalies based on the schema
        CASE 
            WHEN activity = 'Receive Payment' AND prev_activity IN ('Register Order', 'Issue Invoice') THEN 'Payment Before Invoice'
            WHEN activity = 'Ship Goods' AND prev_activity NOT IN ('Validate Stock', 'Confirm Shipment') THEN 'Shipping Without Validation'
            WHEN activity = 'Confirm Shipment' AND prev_activity NOT IN ('Validate Stock', 'Perform Credit Check') THEN 'Confirmation Before Checks'
        END as anomaly_flag
    FROM event_sequence
)
SELECT 
    case_id,
    activity,
    timestamp,
    anomaly_flag
FROM expected_flow
WHERE anomaly_flag IS NOT NULL
ORDER BY case_id, timestamp;
```

### Query 2: Missing Activity Detection
This query checks if all expected activities (Register, Credit, Stock, Confirm, Ship, Invoice, Payment) exist for every `case_id`.

```sql
WITH expected_activities AS (
    SELECT 'Register Order' as activity UNION ALL
    SELECT 'Perform Credit Check' UNION ALL
    SELECT 'Validate Stock' UNION ALL
    SELECT 'Confirm Shipment' UNION ALL
    SELECT 'Ship Goods' UNION ALL
    SELECT 'Issue Invoice' UNION ALL
    SELECT 'Receive Payment'
),
case_activities AS (
    SELECT DISTINCT case_id, activity
    FROM order_event_log
),
missing_steps AS (
    SELECT 
        ea.activity,
        ca.case_id
    FROM expected_activities ea
    CROSS JOIN (SELECT DISTINCT case_id FROM order_event_log) ca
    LEFT JOIN case_activities cact ON ea.activity = cact.activity AND ca.case_id = cact.case_id
    WHERE cact.activity IS NULL
)
SELECT 
    case_id,
    STRING_AGG(activity, ', ') as missing_activities
FROM missing_steps
GROUP BY case_id
HAVING COUNT(*) > 0
ORDER BY case_id;
```

### Query 3: Cross-Table Analysis (Order Type vs. Anomalies)
This query joins the `orders` table to determine if `priority` orders are statistically more likely to exhibit process deviations compared to `standard` orders.

```sql
WITH process_order AS (
    SELECT 
        o.case_id,
        o.order_type,
        o.order_value,
        STRING_AGG(oe.activity, ' -> ' ORDER BY oe.timestamp) as process_flow
    FROM orders o
    JOIN order_event_log oe ON o.case_id = oe.case_id
    GROUP BY o.case_id, o.order_type, o.order_value
)
SELECT 
    order_type,
    COUNT(*) as total_orders,
    COUNT(CASE WHEN process_flow LIKE '%Confirm Shipment%' AND process_flow LIKE '%Perform Credit Check%' THEN 1 END) as bypass_checks,
    COUNT(CASE WHEN process_flow LIKE '%Receive Payment%' AND process_flow NOT LIKE '%Issue Invoice%' THEN 1 END) as payment_before_invoice
FROM process_order
GROUP BY order_type
ORDER BY total_orders DESC;
```

### Query 4: Resource Department Latency Analysis
This query investigates if specific departments are associated with longer delays or anomalies by joining the `resources` table.

```sql
WITH event_with_dept AS (
    SELECT 
        oel.case_id,
        oel.activity,
        oel.timestamp,
        r.department,
        LEAD(oel.timestamp) OVER (PARTITION BY oel.case_id ORDER BY oel.timestamp) as next_timestamp
    FROM order_event_log oel
    JOIN resources r ON oel.resource = r.resource_id
),
activity_durations AS (
    SELECT 
        case_id,
        department,
        activity,
        next_timestamp - timestamp as duration_seconds
    FROM event_with_dept
    WHERE next_timestamp IS NOT NULL
)
SELECT 
    department,
    activity,
    AVG(duration_seconds) as avg_duration_sec,
    MAX(duration_seconds) as max_duration_sec
FROM activity_durations
GROUP BY department, activity
ORDER BY avg_duration_sec DESC;
```
