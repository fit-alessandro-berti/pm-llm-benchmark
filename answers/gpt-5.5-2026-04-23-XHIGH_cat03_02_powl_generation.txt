```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# -----------------------------
# Activity transitions
# -----------------------------

SR = Transition(label="SR")       # Service Request verification
CS = Transition(label="CS")       # Contract Setup

DA = Transition(label="DA")       # Datacenter Allocation
ACI = Transition(label="ACI")     # Access Control Initialization

BNC = Transition(label="BNC")     # Base Network Configuration
DR = Transition(label="D&R")      # Debug & Reconfigure

CD = Transition(label="CD")       # Containerized Deployment
MD = Transition(label="MD")       # Monolithic Deployment

QA = Transition(label="QA")       # Quality Assurance
SA = Transition(label="SA")       # Security Audit
GLA = Transition(label="GLA")     # Go-Live Approval

# No silent transition is needed here:
# - loop exit is handled by Operator.LOOP
# - deployment is a mandatory XOR between CD and MD

# -----------------------------
# Looped configuration phase
# Semantics: execute BNC, then either exit if stable,
# or execute D&R and repeat BNC.
# -----------------------------

configuration_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[BNC, DR]
)

# -----------------------------
# Exclusive deployment choice
# -----------------------------

deployment_choice = OperatorPOWL(
    operator=Operator.XOR,
    children=[CD, MD]
)

# -----------------------------
# Root partial order
# Overall structure:
#
# SR -> CS -> {DA || ACI} -> configuration_loop
#    -> XOR(CD, MD) -> {QA || SA} -> GLA
# -----------------------------

root = StrictPartialOrder(
    nodes=[
        SR,
        CS,
        DA,
        ACI,
        configuration_loop,
        deployment_choice,
        QA,
        SA,
        GLA,
    ]
)

# Helper to add multiple ordering edges
def add_order_edges(source, targets):
    for target in targets:
        root.order.add_edge(source, target)

# Full strict partial order / transitive closure of the intended dependencies

add_order_edges(SR, [
    CS,
    DA,
    ACI,
    configuration_loop,
    deployment_choice,
    QA,
    SA,
    GLA,
])

add_order_edges(CS, [
    DA,
    ACI,
    configuration_loop,
    deployment_choice,
    QA,
    SA,
    GLA,
])

# DA and ACI are concurrent with each other,
# but both must complete before the configuration loop.
add_order_edges(DA, [
    configuration_loop,
    deployment_choice,
    QA,
    SA,
    GLA,
])

add_order_edges(ACI, [
    configuration_loop,
    deployment_choice,
    QA,
    SA,
    GLA,
])

add_order_edges(configuration_loop, [
    deployment_choice,
    QA,
    SA,
    GLA,
])

add_order_edges(deployment_choice, [
    QA,
    SA,
    GLA,
])

# QA and SA are concurrent with each other,
# but both must complete before Go-Live Approval.
add_order_edges(QA, [GLA])
add_order_edges(SA, [GLA])

# The complete POWL model
powl_model = root
```
