## Proposed SQL Investigation Queries

### Query 1: Identify All Process Sequence Violations
```sql
WITH normal_sequence AS (
    SELECT 'Register Order' AS activity, 1 AS expected_order
    UNION ALL SELECT 'Perform Credit Check', 2
    UNION ALL SELECT 'Validate Stock', 3
    UNION ALL SELECT 'Confirm Shipment', 4
    UNION ALL SELECT 'Ship Goods', 5
    UNION ALL SELECT 'Issue Invoice', 6
    UNION ALL SELECT 'Receive Payment', 7
),
case_sequences AS (
    SELECT 
        e.case_id,
        e.activity,
        e.timestamp,
        ROW_NUMBER() OVER (PARTITION BY e.case_id ORDER BY e.timestamp) AS actual_order,
        n.expected_order
    FROM order_event_log e
    JOIN normal_sequence n ON e.activity = n.activity
)
SELECT 
    case_id,
    activity,
    timestamp,
    actual_order,
    expected_order,
    CASE 
        WHEN actual_order != expected_order THEN 'SEQUENCE_VIOLATION'
        ELSE 'NORMAL'
    END AS status
FROM case_sequences
WHERE actual_order != expected_order
ORDER BY case_id, actual_order;
```

### Query 2: Find Cases with Missing Critical Steps
```sql
WITH required_steps AS (
    SELECT UNNEST(ARRAY[
        'Register Order',
        'Perform Credit Check', 
        'Validate Stock',
        'Confirm Shipment',
        'Ship Goods',
        'Issue Invoice',
        'Receive Payment'
    ]) AS required_activity
),
case_activities AS (
    SELECT DISTINCT case_id, activity
    FROM order_event_log
)
SELECT 
    r.required_activity,
    ARRAY_AGG(DISTINCT c.case_id ORDER BY c.case_id) AS missing_in_cases
FROM required_steps r
CROSS JOIN (SELECT DISTINCT case_id FROM order_event_log) all_cases
LEFT JOIN case_activities c ON all_cases.case_id = c.case_id 
    AND r.required_activity = c.activity
WHERE c.case_id IS NULL
GROUP BY r.required_activity
ORDER BY r.required_activity;
```

### Query 3: Analyze Payment Before Invoice Cases
```sql
WITH payment_invoices AS (
    SELECT 
        p.case_id,
        p.timestamp AS payment_time,
        i.timestamp AS invoice_time,
        EXTRACT(EPOCH FROM (i.timestamp - p.timestamp))/3600 AS hours_invoice_after_payment
    FROM order_event_log p
    JOIN order_event_log i ON p.case_id = i.case_id
    WHERE p.activity = 'Receive Payment'
        AND i.activity = 'Issue Invoice'
        AND p.timestamp < i.timestamp
)
SELECT 
    pi.*,
    o.order_type,
    o.order_value,
    o.customer_id
FROM payment_invoices pi
JOIN orders o ON pi.case_id = o.case_id
ORDER BY hours_invoice_after_payment DESC;
```

### Query 4: Resource Behavior Analysis
```sql
SELECT 
    r.resource_id,
    r.role,
    r.department,
    COUNT(DISTINCT e.case_id) AS cases_handled,
    COUNT(CASE WHEN e.activity = 'Ship Goods' THEN 1 END) AS shipments_made,
    COUNT(CASE WHEN e.activity = 'Validate Stock' THEN 1 END) AS stock_validations,
    COUNT(CASE WHEN e.activity = 'Perform Credit Check' THEN 1 END) AS credit_checks,
    ROUND(COUNT(CASE WHEN e.activity = 'Ship Goods' THEN 1 END) * 100.0 / 
          NULLIF(COUNT(CASE WHEN e.activity = 'Validate Stock' THEN 1 END), 0), 2) 
          AS shipment_to_validation_ratio
FROM resources r
JOIN order_event_log e ON r.resource_id = e.resource
GROUP BY r.resource_id, r.role, r.department
HAVING COUNT(DISTINCT e.case_id) > 0
ORDER BY shipment_to_validation_ratio DESC;
```

### Query 5: Priority vs Standard Order Process Compliance
```sql
WITH order_compliance AS (
    SELECT 
        o.case_id,
        o.order_type,
        COUNT(DISTINCT CASE 
            WHEN e.activity IN ('Perform Credit Check', 'Validate Stock', 'Confirm Shipment') 
            THEN e.activity 
        END) AS compliance_steps,
        MAX(CASE WHEN e.activity = 'Ship Goods' THEN 1 ELSE 0 END) AS was_shipped,
        MAX(CASE WHEN e.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) AS had_credit_check
    FROM orders o
    LEFT JOIN order_event_log e ON o.case_id = e.case_id
    GROUP BY o.case_id, o.order_type
)
SELECT 
    order_type,
    COUNT(*) AS total_orders,
    SUM(was_shipped) AS orders_shipped,
    SUM(had_credit_check) AS orders_with_credit_check,
    ROUND(AVG(compliance_steps) * 100.0 / 3, 2) AS avg_compliance_percentage,
    COUNT(CASE WHEN was_shipped = 1 AND had_credit_check = 0 THEN 1 END) 
        AS shipped_without_credit_check
FROM order_compliance
GROUP BY order_type
ORDER BY order_type;
```

### Query 6: Temporal Analysis of Process Violations
```sql
WITH process_timeline AS (
    SELECT 
        case_id,
        activity,
        timestamp,
        LEAD(activity) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_activity,
        LEAD(timestamp) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_timestamp
    FROM order_event_log
)
SELECT 
    case_id,
    activity,
    next_activity,
    timestamp AS current_time,
    next_timestamp AS next_time,
    EXTRACT(EPOCH FROM (next_timestamp - timestamp))/60 AS minutes_between,
    CASE 
        WHEN activity = 'Ship Goods' AND next_activity != 'Issue Invoice' 
            THEN 'SHIPMENT_WITHOUT_INVOICE'
        WHEN activity = 'Register Order' AND next_activity != 'Perform Credit Check' 
            THEN 'ORDER_WITHOUT_CREDIT_CHECK'
        ELSE 'NORMAL_SEQUENCE'
    END AS sequence_status
FROM process_timeline
WHERE next_activity IS NOT NULL
    AND (activity = 'Ship Goods' AND next_activity != 'Issue Invoice'
         OR activity = 'Register Order' AND next_activity != 'Perform Credit Check')
ORDER BY case_id, timestamp;
```
