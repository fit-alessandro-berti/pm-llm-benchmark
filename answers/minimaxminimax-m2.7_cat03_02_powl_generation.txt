```python
# --------------------------------------------------------------
# POWL model for the telecom “Digital Service Provisioning” process
# --------------------------------------------------------------
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# -----------------------------------------------------------------
# 1  Atomic activities (Transition objects)
# -----------------------------------------------------------------
verify_sr = Transition(label="Verify Service Request")       # SR verification
cs        = Transition(label="Contract Setup")                # Contract Setup
da        = Transition(label="Datacenter Allocation")          # DA – can run in parallel with ACI
aci       = Transition(label="Access Control Initialization")  # ACI – can run in parallel with DA
bnc       = Transition(label="Base Network Configuration")    # BNC inside the loop
dr        = Transition(label="Debug & Reconfigure")           # D&R inside the loop (optional)
cd        = Transition(label="Containerized Deployment")      # Deployment option 1
md        = Transition(label="Monolithic Deployment")         # Deployment option 2
qa        = Transition(label="Quality Assurance")              # QA – concurrent with SA
sa        = Transition(label="Security Audit")                # SA – concurrent with QA
gla       = Transition(label="GoLive Approval")              # Final signoff

# -----------------------------------------------------------------
# 2  Composite nodes (operators)
# -----------------------------------------------------------------
# Loop: * (BNC , D&R)    execute BNC, then optionally D&R and repeat until stable.
loop = OperatorPOWL(operator=Operator.LOOP,
                    children=[bnc, dr])

# Exclusive choice: X (CD , MD)  exactly one deployment method is performed.
xor = OperatorPOWL(operator=Operator.XOR,
                   children=[cd, md])

# -----------------------------------------------------------------
# 3  Root partial order – ties everything together
# -----------------------------------------------------------------
root = StrictPartialOrder(
    nodes=[verify_sr, cs, da, aci, loop, xor, qa, sa, gla]
)

# -----------------------------------------------------------------
# 4  Ordering constraints (edges in the partial order)
# -----------------------------------------------------------------
# Sequential start of the process
root.order.add_edge(verify_sr, cs)          # SR verification  Contract Setup

# After CS the two preparatory tasks can run **concurrently**
root.order.add_edge(cs, da)                 # CS  Datacenter Allocation
root.order.add_edge(cs, aci)                # CS  Access Control Init

# Both DA and ACI must finish before the configuration loop begins
root.order.add_edge(da, loop)               # DA  loop
root.order.add_edge(aci, loop)              # ACI  loop

# The loop finishes  exclusive deployment choice
root.order.add_edge(loop, xor)              # loop  deploymentchoice (xor)

# After the deployment choice QA and SA run (they may be concurrent)
root.order.add_edge(xor, qa)                # xor  Quality Assurance
root.order.add_edge(xor, sa)                # xor  Security Audit

# Both QA and SA must be done before the final GoLive Approval
root.order.add_edge(qa, gla)                # QA  GoLive Approval
root.order.add_edge(sa, gla)                # SA  GoLive Approval

# -----------------------------------------------------------------
# 5  Resulting model is stored in `root`.
#     It can now be used for analysis, visualization, conformance checking, etc.
# -----------------------------------------------------------------
# Example visualisation (requires pm4py’s visualisation module):
#   from pm4py.visualization.powl import view as powl_view
#   powl_view(root)   # opens a graphical view of the POWL graph
```
