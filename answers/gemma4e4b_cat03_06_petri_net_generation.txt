This scenario requires modeling sequential processing with two key non-linear paths (loops): one for document re-submission and one for in-depth investigation.

Here is the Python implementation for the accepting Petri net.

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# --- 1. Setup the Petri Net ---
net = PetriNet(name="Insurance Claim Process")

# --- 2. Define Places (States) ---
# Places represent states where the claim resides
P_START = PetriNet.Place(name="Claim Filed (Start)")
P_CR = PetriNet.Place(name="Claim Registered")
P_DV = PetriNet.Place(name="Documents Verified")
P_FC = PetriNet.Place(name="Fraud Check Performed")
P_CA = PetriNet.Place(name="Assessment Done")
P_AP = PetriNet.Place(name="Awaiting Approval")
P_P = PetriNet.Place(name="Ready for Payment")
P_N = PetriNet.Place(name="Notified Customer")
P_CL = PetriNet.Place(name="Claim Closed")
P_END = PetriNet.Place(name="Claim Process Complete") # Final sink

# Add all places to the network
net.places.add(P_START)
net.places.add(P_CR)
net.places.add(P_DV)
net.places.add(P_FC)
net.places.add(P_CA)
net.places.add(P_AP)
net.places.add(P_P)
net.places.add(P_N)
net.places.add(P_CL)
net.places.add(P_END)


# --- 3. Define Transitions (Activities) ---
T_CR = PetriNet.Transition(name="Claim Registration")
T_DV = PetriNet.Transition(name="Document Verification")
T_FC = PetriNet.Transition(name="Fraud Check")
T_CA = PetriNet.Transition(name="Claim Assessment")
T_AP = PetriNet.Transition(name="Manager Approval")
T_P = PetriNet.Transition(name="Process Payment")
T_N = PetriNet.Transition(name="Customer Notification")
T_RDS = PetriNet.Transition(name="Re-Document Submission") # Loop component
T_II = PetriNet.Transition(name="In-depth Investigation") # Loop component
T_CL = PetriNet.Transition(name="Close Claim")

# Add all transitions
net.transitions.add(T_CR)
net.transitions.add(T_DV)
net.transitions.add(T_FC)
net.transitions.add(T_CA)
net.transitions.add(T_AP)
net.transitions.add(T_P)
net.transitions.add(T_N)
net.transitions.add(T_RDS)
net.transitions.add(T_II)
net.transitions.add(T_CL)


# --- 4. Define Arcs (Flow Connections) ---

# 1. Start -> CR
petri_utils.add_arc_from_to(P_START, T_CR, net)
# 2. CR -> DV
petri_utils.add_arc_from_to(T_CR, P_CR, net)
petri_utils.add_arc_from_to(P_CR, T_DV, net)

# 3. DV Process and Loop (Re-Document Submission)
# Main path: DV -> FC
petri_utils.add_arc_from_to(T_DV, P_DV, net)
petri_utils.add_arc_from_to(P_DV, T_FC, net)

# Re-Submission Loop: If documents fail, loop back to P_DV
# T_RDS starts at P_DV and leads back to P_DV
petri_utils.add_arc_from_to(P_DV, T_RDS, net)
petri_utils.add_arc_from_to(T_RDS, P_DV, net)


# 4. FC Process and Loop (In-depth Investigation)
# Main path: FC -> CA
petri_utils.add_arc_from_to(T_FC, P_FC, net)
petri_utils.add_arc_from_to(P_FC, T_CA, net)

# Investigation Loop: If suspicious, loop back to P_FC
# T_II starts at P_FC and leads back to P_FC
petri_utils.add_arc_from_to(P_FC, T_II, net)
petri_utils.add_arc_from_to(T_II, P_FC, net)


# 5. Sequential steps
# CA -> AP
petri_utils.add_arc_from_to(P_CA, T_AP, net)
# AP -> P
petri_utils.add_arc_from_to(T_AP, P_AP, net)
petri_utils.add_arc_from_to(P_AP, T_P, net)
# P -> N
petri_utils.add_arc_from_to(T_P, P_P, net)
petri_utils.add_arc_from_to(P_P, T_N, net)
# N -> CL
petri_utils.add_arc_from_to(T_N, P_N, net)
petri_utils.add_arc_from_to(P_N, T_CL, net)
# CL -> END
petri_utils.add_arc_from_to(T_CL, P_CL, net)
petri_utils.add_arc_from_to(P_CL, P_END, net)


# --- 5. Define Markings ---

# Initial Marking (im): Token at the start place
im = Marking()
im[P_START] = 1

# Final Marking (fm): Token at the end sink place
fm = Marking()
# We must temporarily add P_END before defining the marking,
# and then remove it from the marking dictionary itself.
petri_utils.remove_place(net, P_END)
del fm[P_END]


# --- Output Results ---
print("--- Accepting Petri Net Construction Complete ---")
print(f"Network Name: {net.name}")
print("-" * 40)
print(f"Initial Marking (im): {im}")
print(f"Final Marking (fm): {fm}")
```
