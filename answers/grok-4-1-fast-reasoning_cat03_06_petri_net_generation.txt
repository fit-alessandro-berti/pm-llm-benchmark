```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

net = PetriNet("insurance_claim")

# Places
p_start = PetriNet.Place("p_start")
p_cr_ready = PetriNet.Place("p_cr_ready")
p_dv_ready = PetriNet.Place("p_dv_ready")
p_dv_done = PetriNet.Place("p_dv_done")
p_fc_ready = PetriNet.Place("p_fc_ready")
p_fc_done = PetriNet.Place("p_fc_done")
p_ca_ready = PetriNet.Place("p_ca_ready")
p_ca_done = PetriNet.Place("p_ca_done")
p_p_ready = PetriNet.Place("p_p_ready")
p_p_done = PetriNet.Place("p_p_done")
p_n_done = PetriNet.Place("p_n_done")
p_end = PetriNet.Place("p_end")

net.places.add(p_start)
net.places.add(p_cr_ready)
net.places.add(p_dv_ready)
net.places.add(p_dv_done)
net.places.add(p_fc_ready)
net.places.add(p_fc_done)
net.places.add(p_ca_ready)
net.places.add(p_ca_done)
net.places.add(p_p_ready)
net.places.add(p_p_done)
net.places.add(p_n_done)
net.places.add(p_end)

# Transitions (labeled)
t_C = PetriNet.Transition("C", label="C")
t_CR = PetriNet.Transition("CR", label="CR")
t_DV = PetriNet.Transition("DV", label="DV")
t_RDS = PetriNet.Transition("RDS", label="RDS")
t_FC = PetriNet.Transition("FC", label="FC")
t_II = PetriNet.Transition("II", label="II")
t_CA = PetriNet.Transition("CA", label="CA")
t_AP = PetriNet.Transition("AP", label="AP")
t_P = PetriNet.Transition("P", label="P")
t_N = PetriNet.Transition("N", label="N")
t_CL = PetriNet.Transition("CL", label="CL")

# Silent transitions
t_silent_dv_ok = PetriNet.Transition("silent_dv_ok", label=None)
t_silent_fc_ok = PetriNet.Transition("silent_fc_ok", label=None)
t_silent_skip_ap = PetriNet.Transition("silent_skip_ap", label=None)

net.transitions.add(t_C)
net.transitions.add(t_CR)
net.transitions.add(t_DV)
net.transitions.add(t_RDS)
net.transitions.add(t_FC)
net.transitions.add(t_II)
net.transitions.add(t_CA)
net.transitions.add(t_AP)
net.transitions.add(t_P)
net.transitions.add(t_N)
net.transitions.add(t_CL)
net.transitions.add(t_silent_dv_ok)
net.transitions.add(t_silent_fc_ok)
net.transitions.add(t_silent_skip_ap)

# Arcs for C -> CR
petri_utils.add_arc_from_to(p_start, t_C, net)
petri_utils.add_arc_from_to(t_C, p_cr_ready, net)
petri_utils.add_arc_from_to(p_cr_ready, t_CR, net)
petri_utils.add_arc_from_to(t_CR, p_dv_ready, net)

# DV-RDS loop
petri_utils.add_arc_from_to(p_dv_ready, t_DV, net)
petri_utils.add_arc_from_to(t_DV, p_dv_done, net)
petri_utils.add_arc_from_to(p_dv_done, t_RDS, net)
petri_utils.add_arc_from_to(t_RDS, p_dv_ready, net)

# Silent from DV done to FC ready
petri_utils.add_arc_from_to(p_dv_done, t_silent_dv_ok, net)
petri_utils.add_arc_from_to(t_silent_dv_ok, p_fc_ready, net)

# FC-II loop
petri_utils.add_arc_from_to(p_fc_ready, t_FC, net)
petri_utils.add_arc_from_to(t_FC, p_fc_done, net)
petri_utils.add_arc_from_to(p_fc_done, t_II, net)
petri_utils.add_arc_from_to(t_II, p_fc_ready, net)

# Silent from FC done to CA ready
petri_utils.add_arc_from_to(p_fc_done, t_silent_fc_ok, net)
petri_utils.add_arc_from_to(t_silent_fc_ok, p_ca_ready, net)

# CA
petri_utils.add_arc_from_to(p_ca_ready, t_CA, net)
petri_utils.add_arc_from_to(t_CA, p_ca_done, net)

# Conditional AP or skip to P ready
petri_utils.add_arc_from_to(p_ca_done, t_AP, net)
petri_utils.add_arc_from_to(t_AP, p_p_ready, net)
petri_utils.add_arc_from_to(p_ca_done, t_silent_skip_ap, net)
petri_utils.add_arc_from_to(t_silent_skip_ap, p_p_ready, net)

# P -> N -> CL -> end
petri_utils.add_arc_from_to(p_p_ready, t_P, net)
petri_utils.add_arc_from_to(t_P, p_p_done, net)
petri_utils.add_arc_from_to(p_p_done, t_N, net)
petri_utils.add_arc_from_to(t_N, p_n_done, net)
petri_utils.add_arc_from_to(p_n_done, t_CL, net)
petri_utils.add_arc_from_to(t_CL, p_end, net)

im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_end] = 1
```

## Explanation
This accepting Petri net models the insurance claim process as follows:
- **Initial marking**: Token in `p_start` (new claim arriving).
- **Sequence**: `C`  `CR`  (`DV`  (`RDS`  `DV`)*  silent  `FC`  (`II`  `FC`)*  silent  `CA`  (`AP` | silent)  `P`  `N`  `CL`  `p_end`.
- **Loops**: Multiple `RDS` after `DV` (re-submission), multiple `II` after `FC` (investigation), modeled via cycles enabled by choices at `p_dv_done` and `p_fc_done`.
- **Conditional AP**: Optional via choice at `p_ca_done` (fire `AP` or silent skip).
- **Final marking**: Token in `p_end` (closed claim).
- Silent transitions (label=`None`) handle choices/proceeds without visible activities, ensuring behavioral fidelity for process mining/conformance. All paths reach the final marking.
