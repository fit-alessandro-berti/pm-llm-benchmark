```sql
-- Replace <K> with the number of variants you want, e.g. 10

WITH case_sequences AS (
    SELECT
        case_id,
        list(activity ORDER BY timestamp) AS activity_sequence
    FROM event_log
    GROUP BY case_id
),

variant_counts AS (
    SELECT
        activity_sequence,
        COUNT(*) AS case_count
    FROM case_sequences
    GROUP BY activity_sequence
),

top_variants AS (
    SELECT
        activity_sequence
    FROM variant_counts
    ORDER BY case_count DESC
    LIMIT <K>
),

top_cases AS (
    SELECT
        cs.case_id
    FROM case_sequences cs
    JOIN top_variants tv
      ON cs.activity_sequence IS NOT DISTINCT FROM tv.activity_sequence
)

SELECT
    e.*
FROM event_log e
JOIN top_cases tc
  ON e.case_id = tc.case_id
ORDER BY
    e.case_id,
    e.timestamp;
```

This query:

1. Builds the ordered activity sequence per `case_id` using `list(activity ORDER BY timestamp)`.
2. Groups identical sequences to count variant frequencies.
3. Selects the top `K` most frequent variants.
4. Returns only events whose `case_id` belongs to one of those top variants.
