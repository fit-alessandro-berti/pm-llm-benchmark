```python
from pm4py.objects.process_tree.obj import ProcessTree, Operator

# Root is sequence: arrival/registration/intake -> triage -> initial diagnostics -> loop -> advanced diagnostics -> final admin
root = ProcessTree(operator=Operator.SEQUENCE)

# 1. Arrival -> Registration -> Intake Assessment
arrival_reg_intake = ProcessTree(operator=Operator.SEQUENCE, parent=root)
A = ProcessTree(label="A", parent=arrival_reg_intake)
RG = ProcessTree(label="RG", parent=arrival_reg_intake)
IA = ProcessTree(label="IA", parent=arrival_reg_intake)
arrival_reg_intake.children = [A, RG, IA]

# 2. Triage Interview -> Triage Classification  
triage = ProcessTree(operator=Operator.SEQUENCE, parent=root)
TI = ProcessTree(label="TI", parent=triage)
TC = ProcessTree(label="TC", parent=triage)
triage.children = [TI, TC]

# 3. Initial diagnostics: exclusive choice CE xor FBP
initial_diag = ProcessTree(operator=Operator.XOR, parent=root)
CE = ProcessTree(label="CE", parent=initial_diag)
FBP = ProcessTree(label="FBP", parent=initial_diag)
initial_diag.children = [CE, FBP]

# 4. Loop structure: SC -> (TM xor SPT) -> RDT(X(FCE,FBA))
loop = ProcessTree(operator=Operator.LOOP, parent=root)

# Loop 'do' part (always executed first): SC -> (TM xor SPT) -> RDT
loop_do = ProcessTree(operator=Operator.SEQUENCE, parent=loop)
SC = ProcessTree(label="SC", parent=loop_do)

treat_choice = ProcessTree(operator=Operator.XOR, parent=loop_do)
TM = ProcessTree(label="TM", parent=treat_choice)
SPT = ProcessTree(label="SPT", parent=treat_choice)
treat_choice.children = [TM, SPT]

rdt_choice = ProcessTree(operator=Operator.XOR, parent=loop_do)
RDT = ProcessTree(label="RDT", parent=rdt_choice)  # RDT as label for the choice gateway
FCE = ProcessTree(label="FCE", parent=rdt_choice)
FBA = ProcessTree(label="FBA", parent=rdt_choice)
rdt_choice.children = [FCE, FBA]

loop_do.children = [SC, treat_choice, rdt_choice]

# Loop 'redo' part: empty (represented by tau or implicit)
tau = ProcessTree(label="tau", parent=loop)
loop.children = [loop_do, tau]

# 5. Advanced diagnostics: AI + NC (parallel)
adv_diag = ProcessTree(operator=Operator.PARALLEL, parent=root)
AI = ProcessTree(label="AI", parent=adv_diag)
NC = ProcessTree(label="NC", parent=adv_diag)
adv_diag.children = [AI, NC]

# 6. Final sequence: FAS -> DDI
final_seq = ProcessTree(operator=Operator.SEQUENCE, parent=root)
FAS = ProcessTree(label="FAS", parent=final_seq)
DDI = ProcessTree(label="DDI", parent=final_seq)
final_seq.children = [FAS, DDI]

# 7. Final concurrency: BL + ICP + MRF
final_conc = ProcessTree(operator=Operator.PARALLEL, parent=root)
BL = ProcessTree(label="BL", parent=final_conc)
ICP = ProcessTree(label="ICP", parent=final_conc)
MRF = ProcessTree(label="MRF", parent=final_conc)
final_conc.children = [BL, ICP, MRF]

# Connect all children to root
root.children = [
    arrival_reg_intake, 
    triage, 
    initial_diag, 
    loop, 
    adv_diag, 
    final_seq, 
    final_conc
]
```

