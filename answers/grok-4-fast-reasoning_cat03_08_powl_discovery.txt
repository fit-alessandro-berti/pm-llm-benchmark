```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Define transitions for activities
OR = Transition(label="OR")
MS = Transition(label="MS")
QS = Transition(label="QS")
AS = Transition(label="AS")
IQC = Transition(label="IQC")
TST = Transition(label="TST")
RT = Transition(label="RT")
PK = Transition(label="PK")
LB = Transition(label="LB")
DP = Transition(label="DP")
D = Transition(label="D")
tau = SilentTransition()

# One or more IQC: *(IQC, tau)
one_or_more_IQC = OperatorPOWL(operator=Operator.LOOP, children=[IQC, tau])

# Inner block E: one_or_more_IQC --> AS (sequence)
inner_block = StrictPartialOrder(nodes=[one_or_more_IQC, AS])
inner_block.order.add_edge(one_or_more_IQC, AS)

# One or more inner blocks: *(inner_block, tau)
one_or_more_inner_block = OperatorPOWL(operator=Operator.LOOP, children=[inner_block, tau])

# Choice after initial AS: X(tau, one_or_more_inner_block)
after_AS_choice = OperatorPOWL(operator=Operator.XOR, children=[tau, one_or_more_inner_block])

# Assembly phase: AS --> after_AS_choice (sequence)
assembly_phase = StrictPartialOrder(nodes=[AS, after_AS_choice])
assembly_phase.order.add_edge(AS, after_AS_choice)

# Testing phase: *(TST, RT)
testing_phase = OperatorPOWL(operator=Operator.LOOP, children=[TST, RT])

# Labeling choice: X(LB, tau)
label_choice = OperatorPOWL(operator=Operator.XOR, children=[LB, tau])

# Root partial order for the main sequence
root = StrictPartialOrder(nodes=[OR, MS, QS, assembly_phase, testing_phase, PK, label_choice, DP, D])

# Add sequential order edges
root.order.add_edge(OR, MS)
root.order.add_edge(MS, QS)
root.order.add_edge(QS, assembly_phase)
root.order.add_edge(assembly_phase, testing_phase)
root.order.add_edge(testing_phase, PK)
root.order.add_edge(PK, label_choice)
root.order.add_edge(label_choice, DP)
root.order.add_edge(DP, D)
```
