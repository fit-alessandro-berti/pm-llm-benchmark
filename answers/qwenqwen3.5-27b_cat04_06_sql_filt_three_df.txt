WITH ordered_events AS (
    -- Step 1: Order events within each case and look ahead to the next two events
    SELECT
        case_id,
        activity,
        timestamp,
        LEAD(activity, 1) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_activity,
        LEAD(timestamp, 1) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_timestamp,
        LEAD(activity, 2) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_next_activity,
        LEAD(timestamp, 2) OVER (PARTITION BY case_id ORDER BY timestamp) AS next_next_timestamp
    FROM event_log
),
bad_cases AS (
    -- Step 2 & 3: Identify case_ids that contain the specific sequence AND the time delay condition
    -- Sequence: 'Approve Purchase Requisition' -> 'Create Order' -> 'Confirm Order'
    -- Time Condition: 'Confirm Order' timestamp - 'Create Order' timestamp > 5 days
    SELECT DISTINCT case_id
    FROM ordered_events
    WHERE activity = 'Approve Purchase Requisition'
