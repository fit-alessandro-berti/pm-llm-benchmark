#### Query 1: Identify Cases with Sequence Violations (Shipment before Credit/Stock)
*Goal: Verify if Cases like 1002 are systemic issues, particularly for Priority orders.*

```sql
WITH shipment_cases AS (
    SELECT case_id, MIN(timestamp) as ship_start_time
    FROM order_event_log
    WHERE activity IN ('Confirm Shipment', 'Ship Goods')
    GROUP BY case_id
),
credit_stock_cases AS (
    SELECT case_id, MAX(timestamp) as check_end_time
    FROM order_event_log
    WHERE activity IN ('Perform Credit Check', 'Validate Stock')
    GROUP BY case_id
)
SELECT 
    s.case_id,
    o.order_type,
    s.ship_start_time,
    c.check_end_time
FROM shipment_cases s
JOIN credit_stock_cases c ON s.case_id = c.case_id
JOIN orders o ON s.case_id = o.case_id
WHERE s.ship_start_time < c.check_end_time;
```

#### Query 2: Identify Cases Missing Mandatory Activities
*Goal: Find cases like 1003 and 1004 that skip 'Validate Stock' or 'Credit Check'.*

```sql
WITH all_cases AS (
    SELECT DISTINCT case_id FROM orders
),
required_activities AS (
    SELECT 'Perform Credit Check' as activity UNION ALL
    SELECT 'Validate Stock'
),
missing_steps AS (
    SELECT 
        ac.case_id,
        ra.activity
    FROM all_cases ac
    CROSS JOIN required_activities ra
    WHERE NOT EXISTS (
        SELECT 1 
        FROM order_event_log oel 
        WHERE oel.case_id = ac.case_id 
        AND oel.activity = ra.activity
    )
)
SELECT case_id, string_agg(activity, ', ') as missing_activities
FROM missing_steps
GROUP BY case_id
ORDER BY case_id;
```

```sql
WITH case_flow_summary AS (
    SELECT 
        o.case_id,
        o.order_type,
        COUNT(CASE WHEN oel.activity = 'Perform Credit Check' THEN 1 END) as has_credit_check,
        COUNT(CASE WHEN oel.activity = 'Validate Stock' THEN 1 END) as has_stock_check,
        COUNT(CASE WHEN oel.activity = 'Issue Invoice' THEN 1 END) as has_invoice
    FROM orders o
    LEFT JOIN order_event_log oel ON o.case_id = oel.case_id
    GROUP BY o.case_id, o.order_type
)

    order_type,
    COUNT(*) as total_orders,
    SUM(has_credit_check) as credit_checks_performed,
    SUM(has_stock_check) as stock_checks_performed
FROM case_flow_summary
GROUP BY order_type;
```
