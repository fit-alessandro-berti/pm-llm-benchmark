```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# -------------------------------------------------------------------
# Places: states between activities
# -------------------------------------------------------------------

# The initial token here represents a new Insurance Claim (C) arriving
p_start = PetriNet.Place(name="p_start_new_claim")

p_registered = PetriNet.Place(name="p_claim_registered")
p_documents_checked = PetriNet.Place(name="p_documents_checked")
p_ready_for_fraud_check = PetriNet.Place(name="p_ready_for_fraud_check")
p_fraud_checked = PetriNet.Place(name="p_fraud_checked")
p_claim_assessed = PetriNet.Place(name="p_claim_assessed")
p_ready_for_payment = PetriNet.Place(name="p_ready_for_payment")
p_paid = PetriNet.Place(name="p_claim_paid")
p_notified = PetriNet.Place(name="p_customer_notified")
p_closed = PetriNet.Place(name="p_claim_closed")

for place in [
    p_start,
    p_registered,
    p_documents_checked,
    p_ready_for_fraud_check,
    p_fraud_checked,
    p_claim_assessed,
    p_ready_for_payment,
    p_paid,
    p_notified,
    p_closed,
]:
    net.places.add(place)

# -------------------------------------------------------------------
# Transitions: activities
# -------------------------------------------------------------------

t_cr = PetriNet.Transition(name="CR", label="Claim Registration")
t_dv = PetriNet.Transition(name="DV", label="Document Verification")
t_rds = PetriNet.Transition(name="RDS", label="Re-Document Submission")
t_fc = PetriNet.Transition(name="FC", label="Fraud Check")
t_ii = PetriNet.Transition(name="II", label="In-depth Investigation")
t_ca = PetriNet.Transition(name="CA", label="Claim Assessment")
t_ap = PetriNet.Transition(name="AP", label="Approval")
t_skip_ap = PetriNet.Transition(name="SKIP_AP", label=None)  # silent transition: approval not required
t_p = PetriNet.Transition(name="P", label="Payment")
t_n = PetriNet.Transition(name="N", label="Notification")
t_cl = PetriNet.Transition(name="CL", label="Closure")

for transition in [
    t_cr,
    t_dv,
    t_rds,
    t_fc,
    t_ii,
    t_ca,
    t_ap,
    t_skip_ap,
    t_p,
    t_n,
    t_cl,
]:
    net.transitions.add(transition)

# -------------------------------------------------------------------
# Arcs: process flow
# -------------------------------------------------------------------

# New claim -> Claim Registration
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_registered, net)

# Claim Registration -> Document Verification
petri_utils.add_arc_from_to(p_registered, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_documents_checked, net)

# If documents are incomplete: Re-Document Submission loops back to verification
petri_utils.add_arc_from_to(p_documents_checked, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_registered, net)

# If documents are complete: proceed to Fraud Check
petri_utils.add_arc_from_to(p_documents_checked, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_fraud_checked, net)

# If Fraud Check raises doubts: In-depth Investigation loops back to Fraud Check
petri_utils.add_arc_from_to(p_fraud_checked, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_ready_for_fraud_check, net)

petri_utils.add_arc_from_to(p_ready_for_fraud_check, t_fc, net)

# If fraud concerns are cleared: proceed to Claim Assessment
petri_utils.add_arc_from_to(p_fraud_checked, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_claim_assessed, net)

# Optional Approval:
# - if threshold exceeded, Approval is performed
# - otherwise, silent transition skips approval
petri_utils.add_arc_from_to(p_claim_assessed, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_ready_for_payment, net)

petri_utils.add_arc_from_to(p_claim_assessed, t_skip_ap, net)
petri_utils.add_arc_from_to(t_skip_ap, p_ready_for_payment, net)

# Payment
petri_utils.add_arc_from_to(p_ready_for_payment, t_p, net)
petri_utils.add_arc_from_to(t_p, p_paid, net)

# Notification
petri_utils.add_arc_from_to(p_paid, t_n, net)
petri_utils.add_arc_from_to(t_n, p_notified, net)

# Closure
petri_utils.add_arc_from_to(p_notified, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# -------------------------------------------------------------------
# Initial and final markings
# -------------------------------------------------------------------

im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1
```

This constructs an accepting Petri net where the initial marking represents a newly arrived insurance claim, and the final marking represents a closed and archived claim. The loops allow repeated document re-submission and repeated fraud investigation before the process continues.
