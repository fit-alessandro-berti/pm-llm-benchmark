### SQL Verification Queries (PostgreSQL)

**1. Verify 'R' to 'P' Rigidity and Identify Patterns**  
This query isolates claims where the time between Receive and Approve falls within the suspiciously narrow 24–26 hour window and groups them by claim type and resource to detect batch processing or specific user behavior.

```sql
WITH r_p_intervals AS (
    SELECT 
        e1.claim_id,
        c.claim_type,
        e1.resource AS r_resource,
        e2.resource AS p_resource,
        EXTRACT(EPOCH FROM (e2.timestamp - e1.timestamp)) AS time_diff_seconds
    FROM claim_events e1
    JOIN claim_events e2 ON e1.claim_id = e2.claim_id
    JOIN claims c ON e1.claim_id = c.claim_id
    WHERE e1.activity = 'R' AND e2.activity = 'P'
)
SELECT 
    claim_type,
    p_resource,
    COUNT(*) AS claim_count,
    ROUND(AVG(time_diff_seconds) / 3600.0, 2) AS avg_hours,
    ROUND(STDDEV(time_diff_seconds) / 3600.0, 2) AS stddev_hours
FROM r_p_intervals
WHERE time_diff_seconds BETWEEN 82800 AND 93600 -- 23 to 26 hours
GROUP BY claim_type, p_resource
HAVING STDDEV(time_diff_seconds) < 3600 -- Standard deviation less than 1 hour
ORDER BY claim_count DESC;
```

**2. Investigate 'P' to 'N' Delays and Resource Bottlenecks**  
This query identifies claims where the notification took longer than 5 days (432,000 seconds) after approval, helping to pinpoint if specific resources or claim types are driving the backlog.

```sql
WITH p_n_intervals AS (
    SELECT 
        e1.claim_id,
        c.claim_type,
        e1.resource AS p_resource,
        e2.resource AS n_resource,
        EXTRACT(EPOCH FROM (e2.timestamp - e1.timestamp)) AS time_diff_seconds
    FROM claim_events e1
    JOIN claim_events e2 ON e1.claim_id = e2.claim_id
    JOIN claims c ON e1.claim_id = c.claim_id
    WHERE e1.activity = 'P' AND e2.activity = 'N'
)
SELECT 
    claim_type,
    p_resource,
    COUNT(*) AS delayed_claims_count,
    ROUND(AVG(time_diff_seconds) / 86400.0, 2) AS avg_delay_days,
    ROUND(MAX(time_diff_seconds) / 86400.0, 2) AS max_delay_days
FROM p_n_intervals
WHERE time_diff_seconds > 432000 -- Greater than 5 days
GROUP BY claim_type, p_resource
ORDER BY avg_delay_days DESC;
```

**3. Detect Premature Closures ('A' to 'C') with Missing Intermediate Steps**  
This query finds claims that were closed within 4 hours of being assigned and checks whether the mandatory 'E' (Evaluate) or 'P' (Approve) steps are entirely missing from their event history.

```sql
WITH a_c_intervals AS (
    SELECT 
        e1.claim_id,
        EXTRACT(EPOCH FROM (e2.timestamp - e1.timestamp)) AS time_diff_seconds
    FROM claim_events e1
    JOIN claim_events e2 ON e1.claim_id = e2.claim_id
    WHERE e1.activity = 'A' AND e2.activity = 'C'
),
step_validation AS (
    SELECT 
        claim_id,
        COUNT(CASE WHEN activity = 'E' THEN 1 END) AS has_evaluate,
        COUNT(CASE WHEN activity = 'P' THEN 1 END) AS has_approve
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    ac.claim_id,
    c.claim_type,
    c.claim_amount,
    ac.time_diff_seconds,
    sv.has_evaluate,
    sv.has_approve
FROM a_c_intervals ac
JOIN step_validation sv ON ac.claim_id = sv.claim_id
JOIN claims c ON ac.claim_id = c.claim_id
WHERE ac.time_diff_seconds < 14400 -- Less than 4 hours
  AND (sv.has_evaluate = 0 OR sv.has_approve = 0)
ORDER BY ac.time_diff_seconds ASC;
```

**4. Flag 'E' to 'N' Speed and Verify Missing Approvals**  
This query isolates claims where the transition from Evaluate to Notify happened in under 10 minutes (600 seconds) and verifies if the Approval ('P') step is missing or occurred out of sequence.

```sql
WITH e_n_intervals AS (
    SELECT 
        e1.claim_id,
        e1.timestamp AS e_timestamp,
        e2.timestamp AS n_timestamp,
        EXTRACT(EPOCH FROM (e2.timestamp - e1.timestamp)) AS time_diff_seconds
    FROM claim_events e1
    JOIN claim_events e2 ON e1.claim_id = e2.claim_id
    WHERE e1.activity = 'E' AND e2.activity = 'N'
),
approval_check AS (
    SELECT 
        claim_id,
        COUNT(CASE WHEN activity = 'P' THEN 1 END) AS has_approve,
        MIN(CASE WHEN activity = 'P' THEN timestamp END) AS p_timestamp
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    en.claim_id,
    c.claim_type,
    en.time_diff_seconds,
    ac.has_approve,
    ac.p_timestamp
FROM e_n_intervals en
JOIN approval_check ac ON en.claim_id = ac.claim_id
JOIN claims c ON en.claim_id = c.claim_id
WHERE en.time_diff_seconds < 600 -- Less than 10 minutes
  AND (ac.has_approve = 0 OR ac.p_timestamp > en.n_timestamp) -- Missing P, or P happened after N
ORDER BY en.time_diff_seconds ASC;
```
