```python
from pm4py.objects.process_tree.obj import ProcessTree, Operator

# Helper functions
def add_leaf(label, parent):
    n = ProcessTree(label=label, parent=parent)
    parent.children.append(n)
    return n

def add_node(op, parent):
    n = ProcessTree(operator=op, parent=parent)
    if parent is not None:
        parent.children.append(n)
    return n

def add_tau(parent):
    n = ProcessTree(parent=parent)
    parent.children.append(n)
    return n

# ---- ROOT: top-level sequence ----
root = ProcessTree(operator=Operator.SEQUENCE)

# 1. Arrival -> Registration -> Intake Assessment
add_leaf('A',  root)
add_leaf('RG', root)
add_leaf('IA', root)

# 2. Triage Interview -> Triage Classification
add_leaf('TI', root)
add_leaf('TC', root)

# 3. Initial diagnostic: XOR(CE, FBP)
xor_initial = add_node(Operator.XOR, root)
add_leaf('CE',  xor_initial)
add_leaf('FBP', xor_initial)

# 4. Optional loop (executed only if abnormal results):
#    XOR( tau , LOOP( SC -> XOR(TM,SPT) -> XOR(FCE,FBA) , tau ) )
optional = add_node(Operator.XOR, root)
add_tau(optional)                      # skip the loop entirely if results are normal

loop = add_node(Operator.LOOP, optional)

loop_body = add_node(Operator.SEQUENCE, loop)
add_leaf('SC', loop_body)

xor_treatment = add_node(Operator.XOR, loop_body)
add_leaf('TM',  xor_treatment)
add_leaf('SPT', xor_treatment)

xor_rdt = add_node(Operator.XOR, loop_body)   # Re-Diagnostic Test
add_leaf('FCE', xor_rdt)
add_leaf('FBA', xor_rdt)

add_tau(loop)   # silent "redo" connector of the loop

# 5. Parallel: Advanced Imaging || Nutritional Counseling
par_imaging = add_node(Operator.PARALLEL, root)
add_leaf('AI', par_imaging)
add_leaf('NC', par_imaging)

# 6. Follow-up Appointment Scheduling -> Detailed Discharge Instructions
add_leaf('FAS', root)
add_leaf('DDI', root)

# 7. Parallel: Billing || Insurance Claims Processing || Medical Records Finalization
par_admin = add_node(Operator.PARALLEL, root)
add_leaf('BL',  par_admin)
add_leaf('ICP', par_admin)
add_leaf('MRF', par_admin)
```

**Symbolic representation of the resulting process tree:**

```
->(
   'A', 'RG', 'IA', 'TI', 'TC',
   X('CE', 'FBP'),
   X( tau,
      *( ->( 'SC', X('TM','SPT'), X('FCE','FBA') ), tau )
   ),
   +('AI', 'NC'),
   'FAS', 'DDI',
   +('BL', 'ICP', 'MRF')
)
```

Explanation of the key construct:
- The initial visit consists of a strict sequence A  RG  IA  TI  TC.
- An exclusive choice picks the initial diagnostic test (CE or FBP).
- The abnormal-results loop is wrapped in an XOR with `tau` so that, if the very first results are normal, the whole loop block can be skipped. Inside the loop, each iteration performs SC  XOR(TM,SPT)  XOR(FCE,FBA); the loop's "redo" branch is a `tau`, so after the re-diagnostic test the loop can either exit (results normal) or restart at SC (results still abnormal).
- AI and NC are placed under a PARALLEL operator (true concurrency).
- FAS and DDI follow strictly in sequence.
- Finally BL, ICP and MRF are executed concurrently before the case ends.
