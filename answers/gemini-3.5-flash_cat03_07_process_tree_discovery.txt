from pm4py.objects.process_tree.obj import ProcessTree, Operator

# Root sequence
root = ProcessTree(operator=Operator.SEQUENCE)

# Main sequence activities
ra = ProcessTree(label="RA", parent=root)
vf1 = ProcessTree(label="VF", parent=root)
ae = ProcessTree(label="AE", parent=root)

# Optional block for loop (Exclusive Choice between doing nothing or requesting info + re-validating)
opt_block = ProcessTree(operator=Operator.XOR, parent=root)
tau_opt = ProcessTree(label=None, parent=opt_block) # Silent step (tau)

# Loop sequence: Request Info (one or more times) then Validate Form
loop_seq = ProcessTree(operator=Operator.SEQUENCE, parent=opt_block)

# The RMI Loop: *(RMI, tau) ensures RMI can occur 1, 2, or more times
rmi_loop = ProcessTree(operator=Operator.LOOP, parent=loop_seq)
rmi = ProcessTree(label="RMI", parent=rmi_loop)
tau_loop = ProcessTree(label=None, parent=rmi_loop)
rmi_loop.children.extend([rmi, tau_loop])

vf2 = ProcessTree(label="VF", parent=loop_seq)
loop_seq.children.extend([rmi_loop, vf2])

opt_block.children.extend([tau_opt, loop_seq])

# Remaining main sequence activities
ad = ProcessTree(label="AD", parent=root)
sc = ProcessTree(label="SC", parent=root)
ac = ProcessTree(label="AC", parent=root)

# Assemble root children
root.children.extend([ra, vf1, ae, opt_block, ad, sc, ac])
