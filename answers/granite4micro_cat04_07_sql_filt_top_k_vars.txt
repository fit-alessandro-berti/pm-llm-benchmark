```sql
WITH activity_sequences AS (
    SELECT
        case_id,
        GROUP_CONCAT(activity ORDERED BY timestamp) AS sequence
    FROM event_log
    GROUP BY case_id, timestamp
),
sequence_counts AS (
    SELECT
        sequence,
        COUNT(*) AS frequency
    FROM activity_sequences
    GROUP BY sequence
),
top_k_variants AS (
    SELECT
        sequence,
        frequency
    FROM sequence_counts
    ORDER BY frequency DESC
    LIMIT K
)
SELECT e.*
FROM event_log e
JOIN top_k_variants t ON e.case_id = t.sequence;
```
