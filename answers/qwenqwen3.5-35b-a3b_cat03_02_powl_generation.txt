import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Define all activities as transitions
SR = Transition(label="SR")
CS = Transition(label="CS")
DA = Transition(label="DA")
ACI = Transition(label="ACI")
BNC = Transition(label="BNC")
DnR = Transition(label="D&R")
CD = Transition(label="CD")
MD = Transition(label="MD")
QA = Transition(label="QA")
SA = Transition(label="SA")
GLA = Transition(label="GLA")

# Loop structure: BNC followed by optional D&R
# The LOOP operator means: execute BNC, then either exit OR execute D&R and repeat
loop = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DnR])

# Exclusive choice: Containerized Deployment OR Monolithic Deployment
choice = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# Build the partial order with all main components as nodes
root = StrictPartialOrder(nodes=[SR, CS, DA, ACI, loop, choice, QA, SA, GLA])

# Add edges based on the process dependencies
# 1. SR must be completed before CS can start
root.order.add_edge(SR, CS)

# 2. CS must be completed before DA and ACI can start
root.order.add_edge(CS, DA)
root.order.add_edge(CS, ACI)

# 3. Both DA and ACI must be completed before the loop (BNC/D&R) can start
root.order.add_edge(DA, loop)
root.order.add_edge(ACI, loop)

# 4. Loop must complete before the deployment choice can start
root.order.add_edge(loop, choice)

# 5. Deployment (choice) must complete before QA and SA can start
root.order.add_edge(choice, QA)
root.order.add_edge(choice, SA)

# 6. Both QA and SA must be completed before Go-Live Approval can start
root.order.add_edge(QA, GLA)
root.order.add_edge(SA, GLA)
