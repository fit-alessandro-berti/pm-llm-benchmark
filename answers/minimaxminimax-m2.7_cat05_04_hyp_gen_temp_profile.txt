## 3. SQL Verification Queries

### Query A: Detect Claims with Suspiciously Consistent RP Timing
```sql
WITH claim_timing AS (
    SELECT 
        ce.claim_id,
        c.claim_type,
        c.claim_amount,
        MAX(CASE WHEN activity = 'R' THEN timestamp END) AS receive_ts,
        MAX(CASE WHEN activity = 'P' THEN timestamp END) AS approve_ts,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN activity = 'P' THEN timestamp END) - 
            MAX(CASE WHEN activity = 'R' THEN timestamp END)
        )) AS time_to_approve_seconds
    FROM claim_events ce
    JOIN claims c ON ce.claim_id = c.claim_id
    GROUP BY ce.claim_id, c.claim_type, c.claim_amount
    HAVING MAX(CASE WHEN activity = 'P' THEN timestamp END) IS NOT NULL
)
SELECT 
    claim_id,
    claim_type,
    claim_amount,
    time_to_approve_seconds,
    ROUND(time_to_approve_seconds / 3600.0, 2) AS hours_to_approve
FROM claim_timing
WHERE time_to_approve_seconds BETWEEN 82800 AND 97200  -- Within 1 STDEV of 90,000s
ORDER BY claim_amount DESC;
```

### Query B: Identify Claims with Unrealistically Fast EN Transitions
```sql
WITH eval_to_notify AS (
    SELECT 
        ce.claim_id,
        c.claim_type,
        c.region,
        MAX(CASE WHEN activity = 'E' THEN timestamp END) AS evaluate_ts,
        MAX(CASE WHEN activity = 'N' THEN timestamp END) AS notify_ts,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN activity = 'N' THEN timestamp END) - 
            MAX(CASE WHEN activity = 'E' THEN timestamp END)
        )) AS eval_to_notify_seconds
    FROM claim_events ce
    JOIN claims c ON ce.claim_id = c.claim_id
    GROUP BY ce.claim_id, c.claim_type, c.region
    HAVING MAX(CASE WHEN activity = 'E' THEN timestamp END) IS NOT NULL
        AND MAX(CASE WHEN activity = 'N' THEN timestamp END) IS NOT NULL
)
SELECT 
    claim_id,
    claim_type,
    region,
    eval_to_notify_seconds,
    CASE 
        WHEN eval_to_notify_seconds < 300 THEN 'CRITICALLY_FAST'
        WHEN eval_to_notify_seconds < 600 THEN 'SUSPICIOUSLY_FAST'
        ELSE 'NORMAL'
    END AS anomaly_flag
FROM eval_to_notify
WHERE eval_to_notify_seconds < 600
ORDER BY eval_to_notify_seconds ASC;
```

### Query C: Investigate Extended PN Delays with Correlation Analysis
```sql
WITH approve_to_notify AS (
    SELECT 
        ce.claim_id,
        c.claim_type,
        c.region,
        c.claim_amount,
        adj.specialization,
        adj.adjuster_id,
        MAX(CASE WHEN activity = 'P' THEN timestamp END) AS approve_ts,
        MAX(CASE WHEN activity = 'N' THEN timestamp END) AS notify_ts,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN activity = 'N' THEN timestamp END) - 
            MAX(CASE WHEN activity = 'P' THEN timestamp END)
        )) / 86400.0 AS days_to_notify
    FROM claim_events ce
    JOIN claims c ON ce.claim_id = c.claim_id
    LEFT JOIN adjusters adj ON ce.resource = adj.adjuster_id::VARCHAR
    GROUP BY ce.claim_id, c.claim_type, c.region, c.claim_amount, 
             adj.specialization, adj.adjuster_id
    HAVING MAX(CASE WHEN activity = 'P' THEN timestamp END) IS NOT NULL
        AND MAX(CASE WHEN activity = 'N' THEN timestamp END) IS NOT NULL
)
SELECT 
    claim_type,
    region,
    COUNT(*) AS total_claims,
    AVG(days_to_notify) AS avg_days,
    MAX(days_to_notify) AS max_days,
    MIN(days_to_notify) AS min_days,
    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY days_to_notify) AS median_days
FROM approve_to_notify
WHERE days_to_notify > 7  -- Exceeds expected 7-day average
GROUP BY claim_type, region
ORDER BY avg_days DESC;
```

### Query D: Detect Premature Closure (Claims Closed Without Required Steps)
```sql
WITH claim_step_analysis AS (
    SELECT 
        claim_id,
        MAX(CASE WHEN activity = 'A' THEN 1 ELSE 0 END) AS has_assign,
        MAX(CASE WHEN activity = 'E' THEN 1 ELSE 0 END) AS has_evaluate,
        MAX(CASE WHEN activity = 'P' THEN 1 ELSE 0 END) AS has_approve,
        MAX(CASE WHEN activity = 'N' THEN 1 ELSE 0 END) AS has_notify,
        MAX(CASE WHEN activity = 'C' THEN 1 ELSE 0 END) AS has_close,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN activity = 'C' THEN timestamp END) - 
            MAX(CASE WHEN activity = 'A' THEN timestamp END)
        )) / 3600.0 AS hours_assign_to_close
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    claim_id,
    has_assign,
    has_evaluate,
    has_approve,
    has_notify,
    has_close,
    hours_assign_to_close,
    CASE 
        WHEN has_evaluate = 0 AND has_approve = 0 AND hours_assign_to_close < 2 
            THEN 'PREMATURE_CLOSE_NO_REVIEW'
        WHEN has_notify = 0 AND hours_assign_to_close < 1 
            THEN 'PREMATURE_CLOSE_NO_NOTIFICATION'
        ELSE 'NORMAL'
    END AS closure_status
FROM claim_step_analysis
WHERE has_close = 1 
    AND (
        (has_evaluate = 0 AND has_approve = 0 AND hours_assign_to_close < 2)
        OR (has_notify = 0 AND hours_assign_to_close < 1)
    )
ORDER BY hours_assign_to_close ASC;
```

### Query E: Comprehensive Anomaly Summary by Adjuster and Claim Type
```sql
WITH all_claim_metrics AS (
    SELECT 
        c.claim_id,
        c.claim_type,
        c.region,
        adj.adjuster_id,
        adj.name AS adjuster_name,
        adj.specialization,
        -- R to P metrics
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN ce.activity = 'P' THEN ce.timestamp END) -
            MAX(CASE WHEN ce.activity = 'R' THEN ce.timestamp END)
        )) AS r_to_p_seconds,
        -- P to N metrics  
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN ce.activity = 'N' THEN ce.timestamp END) -
            MAX(CASE WHEN ce.activity = 'P' THEN ce.timestamp END)
        )) AS p_to_n_seconds,
        -- E to N metrics
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN ce.activity = 'N' THEN ce.timestamp END) -
            MAX(CASE WHEN ce.activity = 'E' THEN ce.timestamp END)
        )) AS e_to_n_seconds
    FROM claims c
    JOIN claim_events ce ON c.claim_id = ce.claim_id
    LEFT JOIN adjusters adj ON ce.resource = adj.adjuster_id::VARCHAR
    GROUP BY c.claim_id, c.claim_type, c.region, adj.adjuster_id, adj.name, adj.specialization
)
SELECT 
    claim_type,
    specialization,
    COUNT(*) AS total_claims,
    -- Flag claims with anomalies
    SUM(CASE WHEN r_to_p_seconds IS NOT NULL 
             AND r_to_p_seconds BETWEEN 82800 AND 97200 THEN 1 ELSE 0 END) 
        AS suspicious_consistent_approval,
    SUM(CASE WHEN p_to_n_seconds > 864000 THEN 1 ELSE 0 END) 
        AS excessive_notify_delay,  -- > 10 days
    SUM(CASE WHEN e_to_n_seconds < 300 THEN 1 ELSE 0 END) 
        AS rapid_eval_to_notify
FROM all_claim_metrics
GROUP BY claim_type, specialization
HAVING 
    SUM(CASE WHEN r_to_p_seconds IS NOT NULL 
             AND r_to_p_seconds BETWEEN 82800 AND 97200 THEN 1 ELSE 0 END) > 0
    OR SUM(CASE WHEN p_to_n_seconds > 864000 THEN 1 ELSE 0 END) > 0
    OR SUM(CASE WHEN e_to_n_seconds < 300 THEN 1 ELSE 0 END) > 0
ORDER BY excessive_notify_delay DESC, rapid_eval_to_notify DESC;
```
