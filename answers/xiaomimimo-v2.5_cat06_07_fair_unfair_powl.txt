```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Common activities for both models
ReceiveApplication = Transition(label="ReceiveApplication")
DataCompletenessCheck = Transition(label="DataCompletenessCheck")
RequestMoreInfo = Transition(label="RequestMoreInfo")
SkillAssessment = Transition(label="SkillAssessment")
CulturalFitCheck = Transition(label="CulturalFitCheck")
ManagerialReview = Transition(label="ManagerialReview")
FinalDecision = Transition(label="FinalDecision")

# Activity unique to unfair model
CommunityAffiliationCheck = Transition(label="CommunityAffiliationCheck")

# Loop for data completeness (common to both models)
data_loop = OperatorPOWL(operator=Operator.LOOP, children=[DataCompletenessCheck, RequestMoreInfo])

# MODEL 1: With potential unfairness (includes XOR branch)
# Create XOR branch between CulturalFitCheck and CommunityAffiliationCheck
cultural_fit_xor = OperatorPOWL(operator=Operator.XOR, children=[CulturalFitCheck, CommunityAffiliationCheck])

# Build partial order for unfair model
unfair_model = StrictPartialOrder(nodes=[
    ReceiveApplication,
    data_loop,
    SkillAssessment,
    cultural_fit_xor,
    ManagerialReview,
    FinalDecision
])

# Define execution order
unfair_model.order.add_edge(ReceiveApplication, data_loop)
unfair_model.order.add_edge(data_loop, SkillAssessment)
unfair_model.order.add_edge(SkillAssessment, cultural_fit_xor)
unfair_model.order.add_edge(cultural_fit_xor, ManagerialReview)
unfair_model.order.add_edge(ManagerialReview, FinalDecision)

# MODEL 2: Without unfairness (no XOR branch, all applicants get same evaluation)
# Build partial order for fair model
fair_model = StrictPartialOrder(nodes=[
    ReceiveApplication,
    data_loop,
    SkillAssessment,
    CulturalFitCheck,
    ManagerialReview,
    FinalDecision
])

# Define execution order
fair_model.order.add_edge(ReceiveApplication, data_loop)
fair_model.order.add_edge(data_loop, SkillAssessment)
fair_model.order.add_edge(SkillAssessment, CulturalFitCheck)
fair_model.order.add_edge(CulturalFitCheck, ManagerialReview)
fair_model.order.add_edge(ManagerialReview, FinalDecision)

# Verify both models
print("Model 1 (with potential unfairness) nodes:", [n.label if hasattr(n, 'label') else type(n).__name__ for n in unfair_model.nodes])
print("Model 1 edges:", [(src.label if hasattr(src, 'label') else type(src).__name__, 
                        tgt.label if hasattr(tgt, 'label') else type(tgt).__name__) 
                       for src, tgt in unfair_model.order.edges])

print("\nModel 2 (without unfairness) nodes:", [n.label if hasattr(n, 'label') else type(n).__name__ for n in fair_model.nodes])
print("Model 2 edges:", [(src.label if hasattr(src, 'label') else type(src).__name__, 
                        tgt.label if hasattr(tgt, 'label') else type(tgt).__name__) 
                       for src, tgt in fair_model.order.edges])
```
