import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Initial tasks
sr = Transition(label="SR")
cs = Transition(label="CS")

# 2. Parallel preparation tasks
da = Transition(label="DA")
aci = Transition(label="ACI")

# 3. Looped configuration phase
bnc = Transition(label="BNC")
d_r = Transition(label="D&R")

# Create the loop operator: execute BNC, then optionally D&R, repeating until stable
loop = OperatorPOWL(operator=Operator.LOOP, children=[bnc, d_r])

# 4. Exclusive choice for deployment methods
cd = Transition(label="CD")
md = Transition(label="MD")

# Create the XOR operator: choose exactly one deployment method
xor = OperatorPOWL(operator=Operator.XOR, children=[cd, md])

# 5. Final concurrent tasks
qa = Transition(label="QA")
sa = Transition(label="SA")

# 6. Final sign-off
gla = Transition(label="GLA")

# Construct the root StrictPartialOrder containing all nodes
root = StrictPartialOrder(nodes=[sr, cs, da, aci, loop, xor, qa, sa, gla])

# Define the ordering constraints (edges)
# Step 1: SR must be completed before CS
root.order.add_edge(sr, cs)

# Step 2: CS must be completed before DA and ACI (DA and ACI have no order between them -> concurrent)
root.order.add_edge(cs, da)
root.order.add_edge(cs, aci)

# Step 3: Both DA and ACI must be completed before the looped configuration starts
root.order.add_edge(da, loop)
root.order.add_edge(aci, loop)

# Step 4: The loop must complete before the exclusive choice deployment starts
root.order.add_edge(loop, xor)

# Step 5: The exclusive choice must be completed before QA and SA (QA and SA have no order -> concurrent)
root.order.add_edge(xor, qa)
root.order.add_edge(xor, sa)

# Step 6: Both QA and SA must be completed before the final Go-Live Approval
