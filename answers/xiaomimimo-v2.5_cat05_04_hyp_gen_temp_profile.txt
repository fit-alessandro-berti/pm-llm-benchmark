#### **Query 1: Investigate R  P Rigidity**  
*Goal: Identify claims approved within a narrow time window after receipt, and check for adjuster or claim-type patterns.*

```sql
WITH rp_timing AS (
    SELECT 
        claim_id,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN activity = 'P' THEN timestamp END) - 
            MAX(CASE WHEN activity = 'R' THEN timestamp END)
        )) AS rp_seconds
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    claim_id,
    rp_seconds,
    ROUND(rp_seconds / 3600, 2) AS rp_hours
FROM rp_timing
WHERE rp_seconds BETWEEN 86400 AND 93600  -- 24–26 hours
ORDER BY rp_seconds;
```

#### **Query 2: Investigate P  N Delays**  
*Goal: Find claims where approval-to-notification took over 5 days, and correlate with adjusters or regions.*

```sql
WITH pn_timing AS (
    SELECT 
        ce.claim_id,
        EXTRACT(EPOCH FROM (
            MAX(CASE WHEN ce.activity = 'N' THEN ce.timestamp END) - 
            MAX(CASE WHEN ce.activity = 'P' THEN ce.timestamp END)
        )) AS pn_seconds,
        a.adjuster_id,
        a.region
    FROM claim_events ce
    JOIN claims c ON ce.claim_id = c.claim_id
    LEFT JOIN adjusters a ON ce.resource = a.name  -- assuming resource stores adjuster name
    GROUP BY ce.claim_id, a.adjuster_id, a.region
)
SELECT 
    claim_id,
    ROUND(pn_seconds / 86400, 2) AS pn_days,
    adjuster_id,
    region
FROM pn_timing
WHERE pn_seconds > 432000  -- 5 days
ORDER BY pn_seconds DESC;
```

#### **Query 3: Investigate A  C Rapid Closures**  
*Goal: Identify claims closed within 3 hours of assignment, and check if they lacked intermediate steps (E, P).*

```sql
WITH ac_timing AS (
    SELECT 
        claim_id,
        MIN(CASE WHEN activity = 'A' THEN timestamp END) AS assigned_at,
        MAX(CASE WHEN activity = 'C' THEN timestamp END) AS closed_at
    FROM claim_events
    GROUP BY claim_id
),
activity_flags AS (
    SELECT 
        claim_id,
        MAX(CASE WHEN activity = 'E' THEN 1 ELSE 0 END) AS has_evaluate,
        MAX(CASE WHEN activity = 'P' THEN 1 ELSE 0 END) AS has_approve
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    t.claim_id,
    EXTRACT(EPOCH FROM (t.closed_at - t.assigned_at)) / 3600 AS hours_to_close,
    af.has_evaluate,
    af.has_approve
FROM ac_timing t
JOIN activity_flags af ON t.claim_id = af.claim_id
WHERE EXTRACT(EPOCH FROM (t.closed_at - t.assigned_at)) < 10800  -- less than 3 hours
ORDER BY hours_to_close;
```

#### **Query 4: Investigate E  N Speed**  
*Goal: Detect claims where notification occurred within 10 minutes of evaluation, and verify if approval was skipped.*

```sql
WITH en_timing AS (
    SELECT 
        claim_id,
        MAX(CASE WHEN activity = 'E' THEN timestamp END) AS evaluated_at,
        MAX(CASE WHEN activity = 'N' THEN timestamp END) AS notified_at
    FROM claim_events
    GROUP BY claim_id
),
activity_sequence AS (
    SELECT 
        claim_id,
        ARRAY_AGG(activity ORDER BY timestamp) AS sequence
    FROM claim_events
    GROUP BY claim_id
)
SELECT 
    t.claim_id,
    EXTRACT(EPOCH FROM (t.notified_at - t.evaluated_at)) / 60 AS minutes_to_notify,
    s.sequence
FROM en_timing t
JOIN activity_sequence s ON t.claim_id = s.claim_id
WHERE EXTRACT(EPOCH FROM (t.notified_at - t.evaluated_at)) < 600  -- less than 10 minutes
  AND NOT (s.sequence @> ARRAY['P'])  -- check if approval step is missing
ORDER BY minutes_to_notify;
```

#### **Query 5: Broader Correlation Analysis**  
*Goal: Aggregate timing anomalies by adjuster region or claim type to identify systemic patterns.*

```sql
WITH activity_pairs AS (
    SELECT 
        ce.claim_id,
        ce.activity AS act1,
        ce2.activity AS act2,
        EXTRACT(EPOCH FROM (ce2.timestamp - ce.timestamp)) AS time_diff
    FROM claim_events ce
    JOIN claim_events ce2 ON ce.claim_id = ce2.claim_id 
        AND ce.timestamp < ce2.timestamp
    WHERE (ce.activity, ce2.activity) IN (('R','P'), ('P','N'), ('A','C'), ('E','N'))
)
SELECT 
    ap.act1 || '  ' || ap.act2 AS pair,
    c.claim_type,
    a.region,
    AVG(ap.time_diff) AS avg_seconds,
    COUNT(*) AS sample_size
FROM activity_pairs ap
JOIN claims c ON ap.claim_id = c.claim_id
LEFT JOIN claim_events ce ON ap.claim_id = ce.claim_id AND ce.activity = 'A'
LEFT JOIN adjusters a ON ce.resource = a.name
GROUP BY ap.act1, ap.act2, c.claim_type, a.region
ORDER BY pair, sample_size DESC;
```
