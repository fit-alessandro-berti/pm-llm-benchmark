```sql
-- Find claims closed shortly after assignment
WITH AssignToClose AS (
    SELECT a.claim_id,
           a.resource AS assigned_resource,
           a.timestamp AS assign_time,
           c.timestamp AS close_time,
           EXTRACT(EPOCH FROM (c.timestamp - a.timestamp)) AS duration_sec
    FROM claim_events a
    JOIN claim_events c ON a.claim_id = c.claim_id
    WHERE a.activity = 'A' AND c.activity = 'C'
)
SELECT ac.assigned_resource,
       COUNT(ac.claim_id) as fast_closes,
       AVG(ac.duration_sec) as avg_closing_time
FROM AssignToClose ac
WHERE ac.duration_sec < 10800 -- under 3 hours
GROUP BY ac.assigned_resource
ORDER BY fast_closes DESC;
```

**Verification 1: Investigating the Rigid, Automated Approvals (R to P)**
This query checks if the claims being approved exactly around the 25-hour mark correlate with specific conditions, such as low `claim_amount` thresholds, which would confirm automated processing rules.

```sql
WITH ProcessTimes AS (
    SELECT r.claim_id,
           EXTRACT(EPOCH FROM (p.timestamp - r.timestamp)) AS seconds_to_approve
    FROM claim_events r
    JOIN claim_events p ON r.claim_id = p.claim_id
    WHERE r.activity = 'R' AND p.activity = 'P'
)
SELECT c.claim_type,
       c.claim_amount,
       COUNT(pt.claim_id) AS claim_count
FROM ProcessTimes pt
JOIN claims c ON pt.claim_id = c.claim_id
WHERE pt.seconds_to_approve BETWEEN 86400 AND 93600 -- Between 24 and 26 hours
GROUP BY c.claim_type, c.claim_amount
ORDER BY claim_count DESC;
```

**Verification 2: Isolating the Notification Bottleneck (P to N)**
This query identifies which customer segments or regions are dealing with the 7-day notification delay, which might reveal if the bottleneck is localized to a specific geographic region (e.g., postal delays) or a global system issue.

```sql
WITH ApprovalToNotify AS (
    SELECT p.claim_id,
           EXTRACT(EPOCH FROM (n.timestamp - p.timestamp)) AS delay_seconds
    FROM claim_events p
    JOIN claim_events n ON p.claim_id = n.claim_id
    WHERE p.activity = 'P' AND n.activity = 'N'
)
SELECT a.region,
       AVG(an.delay_seconds) AS average_delay_sec,
       COUNT(an.claim_id) AS affected_claims
FROM ApprovalToNotify an
JOIN claim_events e ON an.claim_id = e.claim_id AND e.activity = 'A'
JOIN adjusters a ON e.resource = a.adjuster_id::varchar -- Assuming resource holds adjuster_id
WHERE an.delay_seconds > 172800 -- Delays greater than 2 days
GROUP BY a.region
ORDER BY average_delay_sec DESC;
```

**Verification 3: Identifying Premature Closures (A to C)**
This query finds claims closed within 3 hours directly after assignment. It counts the occurrences by adjuster to see if a specific employee or resource pool is responsible for bypassing standard operating procedures.

```sql
WITH AssignToClose AS (
    SELECT a_event.claim_id,
           a_event.resource AS assigned_resource,
           a_event.timestamp AS assign_time,
           c_event.timestamp AS close_time,
           EXTRACT(EPOCH FROM (c_event.timestamp - a_event.timestamp)) AS duration_sec
    FROM claim_events a_event
    JOIN claim_events c_event ON a_event.claim_id = c_event.claim_id

       COUNT(claim_id) AS fast_closures,
       AVG(duration_sec) AS avg_duration
FROM AssignToClose
WHERE duration_sec <= 10800 -- 3 hours or less
GROUP BY assigned_resource
ORDER BY fast_closures DESC;
```
