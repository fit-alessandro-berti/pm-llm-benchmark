### **Query 3: Analyze Notification Skipping Patterns**
```sql
WITH claim_status AS (
    SELECT 
        c.claim_id,
        c.claim_type,
        c.claim_amount,
        c.submission_date,
        MAX(CASE WHEN ce.activity = 'C' THEN 1 ELSE 0 END) as has_close,
        MAX(CASE WHEN ce.activity = 'N' THEN 1 ELSE 0 END) as has_notify,
        MAX(CASE WHEN ce.activity = 'P' THEN 1 ELSE 0 END) as has_approve,
        MIN(CASE WHEN ce.activity = 'C' THEN ce.timestamp END) as close_time
    FROM 
        claims c
    LEFT JOIN 
        claim_events ce ON c.claim_id = ce.claim_id
    GROUP BY 
        c.claim_id, c.claim_type, c.claim_amount, c.submission_date
)
SELECT 
    claim_type,
    COUNT(*) as total_closed_claims,
    SUM(has_notify) as claims_with_notification,
    COUNT(*) - SUM(has_notify) as claims_without_notification,
    ROUND(100.0 * SUM(has_notify) / COUNT(*), 2) as notification_percentage,
    AVG(claim_amount) as avg_claim_amount
FROM 
    claim_status
WHERE 
    has_close = 1
GROUP BY 
    claim_type
ORDER BY 
    notification_percentage ASC;
```

### **Query 4: Detect Premature Closure Patterns**
```sql
WITH event_sequence AS (
    SELECT 
        ce.claim_id,
        ce.activity,
        ce.timestamp,
        c.claim_type,
        c.claim_amount,
        ROW_NUMBER() OVER (PARTITION BY ce.claim_id ORDER BY ce.timestamp) as event_order
    FROM 
        claim_events ce
    JOIN 
        claims c ON ce.claim_id = c.claim_id
    WHERE 
        ce.activity IN ('R', 'A', 'E', 'P', 'N', 'C')
),
claim_paths AS (
    SELECT 
        claim_id,
        claim_type,
        claim_amount,
        STRING_AGG(activity, ' -> ' ORDER BY event_order) as process_path,
        MIN(CASE WHEN activity = 'A' THEN timestamp END) as assign_time,
        MIN(CASE WHEN activity = 'C' THEN timestamp END) as close_time,
        MIN(CASE WHEN activity = 'E' THEN timestamp END) as evaluate_time
    FROM 
        event_sequence
    GROUP BY 
        claim_id, claim_type, claim_amount
)
SELECT 
    process_path,
    COUNT(*) as occurrence_count,
    AVG(claim_amount) as avg_amount,
    AVG(EXTRACT(EPOCH FROM (close_time - assign_time))/3600) as avg_hours_to_close
FROM 
    claim_paths
WHERE 
    close_time IS NOT NULL
GROUP BY 
    process_path
ORDER BY 
    occurrence_count DESC;
```

```sql
SELECT
```
