```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# Define places
source = PetriNet.Place(name="source")
p_claim_filed = PetriNet.Place(name="claim_filed")
p_registered = PetriNet.Place(name="registered")
p_verified = PetriNet.Place(name="documents_verified")
p_fraud_checked = PetriNet.Place(name="fraud_checked")
p_assessed = PetriNet.Place(name="claim_assessed")
p_approved = PetriNet.Place(name="approved")
p_paid = PetriNet.Place(name="paid")
p_notified = PetriNet.Place(name="notified")
p_closed = PetriNet.Place(name="closed")

# Add places to the net
net.places.add(source)
net.places.add(p_claim_filed)
net.places.add(p_registered)
net.places.add(p_verified)
net.places.add(p_fraud_checked)
net.places.add(p_assessed)
net.places.add(p_approved)
net.places.add(p_paid)
net.places.add(p_notified)
net.places.add(p_closed)

# Define transitions
t_c = PetriNet.Transition(name="C", label="Insurance Claim")
t_cr = PetriNet.Transition(name="CR", label="Claim Registration")
t_dv = PetriNet.Transition(name="DV", label="Document Verification")
t_rds = PetriNet.Transition(name="RDS", label="Re-Document Submission")
t_fc = PetriNet.Transition(name="FC", label="Fraud Check")
t_ii = PetriNet.Transition(name="II", label="In-depth Investigation")
t_ca = PetriNet.Transition(name="CA", label="Claim Assessment")
t_ap = PetriNet.Transition(name="AP", label="Approval")
t_p = PetriNet.Transition(name="P", label="Payment")
t_p2 = PetriNet.Transition(name="P2", label="Payment")
t_n = PetriNet.Transition(name="N", label="Notification")
t_cl = PetriNet.Transition(name="CL", label="Closure")

# Add transitions to the net
net.transitions.add(t_c)
net.transitions.add(t_cr)
net.transitions.add(t_dv)
net.transitions.add(t_rds)
net.transitions.add(t_fc)
net.transitions.add(t_ii)
net.transitions.add(t_ca)
net.transitions.add(t_ap)
net.transitions.add(t_p)
net.transitions.add(t_p2)
net.transitions.add(t_n)
net.transitions.add(t_cl)

# Add arcs (main flow)
petri_utils.add_arc_from_to(source, t_c, net)
petri_utils.add_arc_from_to(t_c, p_claim_filed, net)
petri_utils.add_arc_from_to(p_claim_filed, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_registered, net)
petri_utils.add_arc_from_to(p_registered, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_verified, net)

# Document re-submission loop
petri_utils.add_arc_from_to(p_verified, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_registered, net)

# Fraud check
petri_utils.add_arc_from_to(p_verified, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_fraud_checked, net)

# Fraud investigation loop (multiple times possible)
petri_utils.add_arc_from_to(p_fraud_checked, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_fraud_checked, net)

# Proceed to assessment
petri_utils.add_arc_from_to(p_fraud_checked, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_assessed, net)

# Approval choice (conditional)
petri_utils.add_arc_from_to(p_assessed, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_approved, net)

# Payment (direct path - no approval needed)
petri_utils.add_arc_from_to(p_assessed, t_p, net)
petri_utils.add_arc_from_to(t_p, p_paid, net)

# Payment (after approval)
petri_utils.add_arc_from_to(p_approved, t_p2, net)
petri_utils.add_arc_from_to(t_p2, p_paid, net)

# Notification and closure
petri_utils.add_arc_from_to(p_paid, t_n, net)
petri_utils.add_arc_from_to(t_n, p_notified, net)
petri_utils.add_arc_from_to(p_notified, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# Initial marking (token at source - new claim arrives)
im = Marking()
im[source] = 1

# Final marking (token at closed - claim is closed and archived)
fm = Marking()
fm[p_closed] = 1
```
