from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Atomic activities
sr = Transition(label="SR")      # Service Request verification
cs = Transition(label="CS")      # Contract Setup
da = Transition(label="DA")      # Datacenter Allocation
aci = Transition(label="ACI")    # Access Control Initialization
bnc = Transition(label="BNC")    # Base Network Configuration
dr = Transition(label="D&R")     # Debug & Reconfigure
cd = Transition(label="CD")      # Containerized Deployment
md = Transition(label="MD")      # Monolithic Deployment
qa = Transition(label="QA")      # Quality Assurance
sa = Transition(label="SA")      # Security Audit
gla = Transition(label="GLA")    # Go-Live Approval

# SilentTransition is imported for completeness, but no explicit silent step is needed here.

# Parallel preparation phase: DA || ACI
prep_phase = StrictPartialOrder(nodes=[da, aci])

# Looped configuration phase: BNC, then either exit or do D&R and BNC again
config_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[bnc, dr]
)

# Exclusive deployment choice: CD xor MD
deployment_choice = OperatorPOWL(
    operator=Operator.XOR,
    children=[cd, md]
)

# Parallel final checks: QA || SA
final_checks = StrictPartialOrder(nodes=[qa, sa])

# Overall process:
# SR -> CS -> (DA || ACI) -> *(BNC, D&R) -> X(CD, MD) -> (QA || SA) -> GLA
root = StrictPartialOrder(
    nodes=[sr, cs, prep_phase, config_loop, deployment_choice, final_checks, gla]
)

# Add the precedence relation between the main phases.
# Add all implied edges so the relation is transitive.
main_phases = [sr, cs, prep_phase, config_loop, deployment_choice, final_checks, gla]
for i, src in enumerate(main_phases):
    for dst in main_phases[i + 1:]:
        root.order.add_edge(src, dst)

# Final POWL model
powl_model = root
