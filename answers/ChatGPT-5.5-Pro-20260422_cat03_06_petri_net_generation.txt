```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils


# -------------------------------------------------------------------
# Create the Petri net
# -------------------------------------------------------------------

net = PetriNet(name="insurance_claim_management")


# -------------------------------------------------------------------
# Places: states between activities
# -------------------------------------------------------------------

p_start = PetriNet.Place("p_start")
p_after_cr = PetriNet.Place("p_after_claim_registration")
p_after_dv = PetriNet.Place("p_after_document_verification")
p_before_rds = PetriNet.Place("p_before_re_document_submission")
p_before_fc = PetriNet.Place("p_before_fraud_check")
p_after_fc = PetriNet.Place("p_after_fraud_check")
p_before_ii = PetriNet.Place("p_before_in_depth_investigation")
p_before_ca = PetriNet.Place("p_before_claim_assessment")
p_after_ca = PetriNet.Place("p_after_claim_assessment")
p_before_ap = PetriNet.Place("p_before_approval")
p_after_ap = PetriNet.Place("p_after_approval_or_skip")
p_after_payment = PetriNet.Place("p_after_payment")
p_after_notification = PetriNet.Place("p_after_notification")
p_closed = PetriNet.Place("p_closed_claim")


for place in [
    p_start,
    p_after_cr,
    p_after_dv,
    p_before_rds,
    p_before_fc,
    p_after_fc,
    p_before_ii,
    p_before_ca,
    p_after_ca,
    p_before_ap,
    p_after_ap,
    p_after_payment,
    p_after_notification,
    p_closed,
]:
    net.places.add(place)


# -------------------------------------------------------------------
# Visible transitions: business activities
# -------------------------------------------------------------------

t_cr = PetriNet.Transition("CR", "Claim Registration")
t_dv = PetriNet.Transition("DV", "Document Verification")
t_rds = PetriNet.Transition("RDS", "Re-Document Submission")
t_fc = PetriNet.Transition("FC", "Fraud Check")
t_ii = PetriNet.Transition("II", "In-depth Investigation")
t_ca = PetriNet.Transition("CA", "Claim Assessment")
t_ap = PetriNet.Transition("AP", "Approval")
t_p = PetriNet.Transition("P", "Payment")
t_n = PetriNet.Transition("N", "Notification")
t_cl = PetriNet.Transition("CL", "Closure")


# -------------------------------------------------------------------
# Silent transitions: routing / decision points
# label=None makes them invisible transitions
# -------------------------------------------------------------------

t_docs_complete = PetriNet.Transition("tau_docs_complete", None)
t_docs_incomplete = PetriNet.Transition("tau_docs_incomplete", None)

t_fraud_clear = PetriNet.Transition("tau_fraud_clear", None)
t_fraud_doubt = PetriNet.Transition("tau_fraud_doubt", None)

t_approval_required = PetriNet.Transition("tau_approval_required", None)
t_approval_not_required = PetriNet.Transition("tau_approval_not_required", None)


for transition in [
    t_cr,
    t_dv,
    t_rds,
    t_fc,
    t_ii,
    t_ca,
    t_ap,
    t_p,
    t_n,
    t_cl,
    t_docs_complete,
    t_docs_incomplete,
    t_fraud_clear,
    t_fraud_doubt,
    t_approval_required,
    t_approval_not_required,
]:
    net.transitions.add(transition)


# -------------------------------------------------------------------
# Arcs: main process flow
# -------------------------------------------------------------------

# Start -> Claim Registration -> Document Verification
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_after_cr, net)

petri_utils.add_arc_from_to(p_after_cr, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)


# -------------------------------------------------------------------
# Document Verification loop
#
# If documents are complete:
#   DV -> Fraud Check
#
# If documents are incomplete:
#   DV -> Re-Document Submission -> DV again
# -------------------------------------------------------------------

petri_utils.add_arc_from_to(p_after_dv, t_docs_complete, net)
petri_utils.add_arc_from_to(t_docs_complete, p_before_fc, net)

petri_utils.add_arc_from_to(p_after_dv, t_docs_incomplete, net)
petri_utils.add_arc_from_to(t_docs_incomplete, p_before_rds, net)

petri_utils.add_arc_from_to(p_before_rds, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_after_cr, net)


# -------------------------------------------------------------------
# Fraud Check loop
#
# If fraud check is clear:
#   FC -> Claim Assessment
#
# If doubts remain:
#   FC -> In-depth Investigation -> FC again
# -------------------------------------------------------------------

petri_utils.add_arc_from_to(p_before_fc, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

petri_utils.add_arc_from_to(p_after_fc, t_fraud_clear, net)
petri_utils.add_arc_from_to(t_fraud_clear, p_before_ca, net)

petri_utils.add_arc_from_to(p_after_fc, t_fraud_doubt, net)
petri_utils.add_arc_from_to(t_fraud_doubt, p_before_ii, net)

petri_utils.add_arc_from_to(p_before_ii, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_before_fc, net)


# -------------------------------------------------------------------
# Claim Assessment and optional Approval
#
# If the claim amount is above the threshold:
#   CA -> Approval -> Payment
#
# If approval is not required:
#   CA -> Payment
# -------------------------------------------------------------------

petri_utils.add_arc_from_to(p_before_ca, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

petri_utils.add_arc_from_to(p_after_ca, t_approval_required, net)
petri_utils.add_arc_from_to(t_approval_required, p_before_ap, net)

petri_utils.add_arc_from_to(p_before_ap, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_after_ap, net)

petri_utils.add_arc_from_to(p_after_ca, t_approval_not_required, net)
petri_utils.add_arc_from_to(t_approval_not_required, p_after_ap, net)


# -------------------------------------------------------------------
# Payment -> Notification -> Closure
# -------------------------------------------------------------------

petri_utils.add_arc_from_to(p_after_ap, t_p, net)
petri_utils.add_arc_from_to(t_p, p_after_payment, net)

petri_utils.add_arc_from_to(p_after_payment, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_notification, net)

petri_utils.add_arc_from_to(p_after_notification, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)


# -------------------------------------------------------------------
# Initial and final markings
# -------------------------------------------------------------------

im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1


# The accepting Petri net is represented by:
#   net: the Petri net structure
#   im:  initial marking
#   fm:  final marking
```

This accepting Petri net supports:

* the normal path: `CR -> DV -> FC -> CA -> AP/P -> N -> CL`
* repeated document re-submission: `DV -> RDS -> DV`
* repeated fraud investigation: `FC -> II -> FC`
* optional approval using silent transitions, depending on whether the claim amount exceeds the threshold.
