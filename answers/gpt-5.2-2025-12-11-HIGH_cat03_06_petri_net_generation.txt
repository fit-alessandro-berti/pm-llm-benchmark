```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Petri net
net = PetriNet(name="insurance_claim_management")

# -----------------------
# Places (states)
# -----------------------
p_start = PetriNet.Place("start")                 # new claim arrives
p_after_c = PetriNet.Place("after_C")             # claim filed

p_before_dv = PetriNet.Place("before_DV")         # ready for document verification
p_before_rds = PetriNet.Place("before_RDS")       # documents incomplete -> resubmission needed

p_before_fc = PetriNet.Place("before_FC")         # ready for fraud check
p_before_ii = PetriNet.Place("before_II")         # fraud doubts -> in-depth investigation needed

p_before_ca = PetriNet.Place("before_CA")         # ready for claim assessment
p_after_ca = PetriNet.Place("after_CA")           # assessed (amount decided)

p_after_ap = PetriNet.Place("after_AP")           # approved (if needed)

p_after_p = PetriNet.Place("after_P")             # paid
p_after_n = PetriNet.Place("after_N")             # notified
p_end = PetriNet.Place("end")                     # closed (final state)

net.places.update({
    p_start, p_after_c,
    p_before_dv, p_before_rds,
    p_before_fc, p_before_ii,
    p_before_ca, p_after_ca, p_after_ap,
    p_after_p, p_after_n, p_end
})

# -----------------------
# Transitions (activities)
# -----------------------
t_c = PetriNet.Transition("t_C", label="Insurance Claim (C)")
t_cr = PetriNet.Transition("t_CR", label="Claim Registration (CR)")

# DV can either succeed (docs complete) or fail (docs incomplete) -> both are "DV" executions
t_dv_ok = PetriNet.Transition("t_DV_ok", label="Document Verification (DV)")
t_dv_inc = PetriNet.Transition("t_DV_incomplete", label="Document Verification (DV)")
t_rds = PetriNet.Transition("t_RDS", label="Re-Document Submission (RDS)")

# FC can either clear the claim or raise doubts -> both are "FC" executions
t_fc_ok = PetriNet.Transition("t_FC_ok", label="Fraud Check (FC)")
t_fc_doubt = PetriNet.Transition("t_FC_doubt", label="Fraud Check (FC)")
t_ii = PetriNet.Transition("t_II", label="In-depth Investigation (II)")

t_ca = PetriNet.Transition("t_CA", label="Claim Assessment (CA)")
t_ap = PetriNet.Transition("t_AP", label="Approval (AP)")

# Payment can happen either directly after CA (no approval needed) or after AP
t_p_noap = PetriNet.Transition("t_P_noAP", label="Payment (P)")
t_p_ap = PetriNet.Transition("t_P_afterAP", label="Payment (P)")

t_n = PetriNet.Transition("t_N", label="Notification (N)")
t_cl = PetriNet.Transition("t_CL", label="Closure (CL)")

net.transitions.update({
    t_c, t_cr,
    t_dv_ok, t_dv_inc, t_rds,
    t_fc_ok, t_fc_doubt, t_ii,
    t_ca, t_ap,
    t_p_noap, t_p_ap,
    t_n, t_cl
})

# -----------------------
# Arcs (flow)
# -----------------------
# Claim arrives -> filed -> registered
petri_utils.add_arc_from_to(p_start, t_c, net)
petri_utils.add_arc_from_to(t_c, p_after_c, net)

petri_utils.add_arc_from_to(p_after_c, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_before_dv, net)

# Document verification with possible loop via RDS:
# DV complete -> proceed to FC
petri_utils.add_arc_from_to(p_before_dv, t_dv_ok, net)
petri_utils.add_arc_from_to(t_dv_ok, p_before_fc, net)

# DV incomplete -> go to RDS -> back to DV (loop)
petri_utils.add_arc_from_to(p_before_dv, t_dv_inc, net)
petri_utils.add_arc_from_to(t_dv_inc, p_before_rds, net)

petri_utils.add_arc_from_to(p_before_rds, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_before_dv, net)

# Fraud check with possible loop via II:
# FC ok -> proceed to CA
petri_utils.add_arc_from_to(p_before_fc, t_fc_ok, net)
petri_utils.add_arc_from_to(t_fc_ok, p_before_ca, net)

# FC doubts -> II -> back to FC (loop, can repeat multiple times)
petri_utils.add_arc_from_to(p_before_fc, t_fc_doubt, net)
petri_utils.add_arc_from_to(t_fc_doubt, p_before_ii, net)

petri_utils.add_arc_from_to(p_before_ii, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_before_fc, net)

# Claim assessment
petri_utils.add_arc_from_to(p_before_ca, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

# Approval optional (only when needed)
petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_after_ap, net)

# Payment either directly after CA or after AP
petri_utils.add_arc_from_to(p_after_ca, t_p_noap, net)
petri_utils.add_arc_from_to(t_p_noap, p_after_p, net)

petri_utils.add_arc_from_to(p_after_ap, t_p_ap, net)
petri_utils.add_arc_from_to(t_p_ap, p_after_p, net)

# Notification and closure
petri_utils.add_arc_from_to(p_after_p, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_n, net)

petri_utils.add_arc_from_to(p_after_n, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_end, net)

# -----------------------
# Initial and final markings
# -----------------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_end] = 1
```
