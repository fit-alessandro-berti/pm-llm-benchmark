### **Query Set 1: Premature Claim Closure Detection**
```sql
-- Find claims closed without proper evaluation
WITH claim_timeline AS (
  SELECT 
    ce.claim_id,
    MIN(CASE WHEN ce.activity = 'R' THEN ce.timestamp END) AS received_time,
    MIN(CASE WHEN ce.activity = 'A' THEN ce.timestamp END) AS assigned_time,
    MIN(CASE WHEN ce.activity = 'E' THEN ce.timestamp END) AS evaluated_time,
    MIN(CASE WHEN ce.activity = 'P' THEN ce.timestamp END) AS approved_time,
    MIN(CASE WHEN ce.activity = 'C' THEN ce.timestamp END) AS closed_time,
    MIN(CASE WHEN ce.activity = 'N' THEN ce.timestamp END) AS notified_time
  FROM claim_events ce
  WHERE ce.claim_id IN (SELECT claim_id FROM claims)
  GROUP BY ce.claim_id
)
SELECT 
  claim_id,
  closed_time - assigned_time AS time_to_close_from_assignment,
  CASE 
    WHEN evaluated_time IS NULL THEN 'No Evaluation'
    WHEN approved_time IS NULL THEN 'No Approval'
    WHEN notified_time IS NULL THEN 'No Notification'
    ELSE 'Complete'
  END AS closure_status
FROM claim_timeline
WHERE closed_time IS NOT NULL
  AND (evaluated_time IS NULL OR approved_time IS NULL);
```

### **Query Set 2: Multiple Approvals Detection**
```sql
-- Identify claims with multiple approval events (loop anomaly)
SELECT

  STRING_AGG(timestamp::text, '; ') as approval_timestamps
```

### **Query Set 3: Skipped Customer Notifications**
```sql
-- Find claims where notification was skipped
SELECT 
  c.claim_id,
  c.customer_id,
  c.claim_type,
  ce_assigned.timestamp AS assignment_time,
  ce_closed.timestamp AS closure_time,
  EXTRACT(EPOCH FROM (ce_closed.timestamp - ce_assigned.timestamp))/3600 AS hours_to_close
FROM claims c
JOIN claim_events ce_assigned ON c.claim_id = ce_assigned.claim_id AND ce_assigned.activity = 'A'
JOIN claim_events ce_closed ON c.claim_id = ce_closed.claim_id AND ce_closed.activity = 'C'
WHERE NOT EXISTS (
  SELECT 1 FROM claim_events ce_notify 
  WHERE ce_notify.claim_id = c.claim_id AND ce_notify.activity = 'N'
)
ORDER BY ce_closed.timestamp DESC
LIMIT 100;
```

### **Query Set 4: Process Flow Violation Analysis**
```sql
-- Comprehensive violation detection
SELECT 
  ce.claim_id,
  c.customer_id,
  c.claim_amount,
  c.claim_type,
  STRING_AGG(ce.activity || ':' || ce.timestamp::date, '  ' ORDER BY ce.timestamp) as actual_flow,
  CASE 
    WHEN bool_or(ce.activity = 'C') AND NOT bool_or(ce.activity = 'E') THEN 'Closed without Evaluation'
    WHEN bool_or(ce.activity = 'C') AND NOT bool_or(ce.activity = 'P') THEN 'Closed without Approval'
    WHEN bool_or(ce.activity = 'C') AND NOT bool_or(ce.activity = 'N') THEN 'Closed without Notification'
    ELSE 'Other Violation'
  END as violation_type
FROM claims c
JOIN claim_events ce ON c.claim_id = ce.claim_id
GROUP BY ce.claim_id, c.customer_id, c.claim_amount, c.claim_type
HAVING (bool_or(ce.activity = 'C') AND 
        (NOT bool_or(ce.activity = 'E') OR 
         NOT bool_or(ce.activity = 'P') OR 
         NOT bool_or(ce.activity = 'N')))
ORDER BY c.claim_id;
```

### **Query Set 5: Temporal Anomaly Detection**
```sql
-- Find claims where closure happened before expected process completion
SELECT 
  c.claim_id,
  c.claim_amount,
  c.submission_date,
  MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'A') as assigned_time,
  MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'E') as evaluated_time,
  MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'P') as approved_time,
  MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'C') as closed_time,
  CASE 
    WHEN MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'C') < 
         MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'P') 
    THEN 'Closed before approval'
    WHEN MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'C') < 
         MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'E') 
    THEN 'Closed before evaluation'
    ELSE 'Normal sequence'
  END as temporal_violation
FROM claims c
JOIN claim_events ce_all ON c.claim_id = ce_all.claim_id
WHERE c.submission_date >= CURRENT_DATE - INTERVAL '90 days'
GROUP BY c.claim_id, c.claim_amount, c.submission_date
HAVING MIN(ce_all.timestamp) FILTER (WHERE ce_all.activity = 'C') IS NOT NULL
ORDER BY temporal_violation DESC, c.submission_date DESC;
```

```sql
SELECT 
  COUNT(DISTINCT c.claim_id) as total_claims,

  COUNT(DISTINCT CASE WHEN ce_notify.claim_id IS NOT NULL THEN c.claim_id END) as notified_claims,

    COUNT(DISTINCT CASE WHEN ce_notify.claim_id IS NOT NULL THEN c.claim_id END)::DECIMAL /
```
