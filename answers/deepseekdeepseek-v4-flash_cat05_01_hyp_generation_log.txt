### 1. Detect orders where *Confirm Shipment* occurs before *Perform Credit Check*

```sql
SELECT 
    a.case_id,
    a.timestamp AS credit_check_time,
    b.timestamp AS confirm_shipment_time
FROM order_event_log a
JOIN order_event_log b 
    ON a.case_id = b.case_id
WHERE a.activity = 'Perform Credit Check'
  AND b.activity = 'Confirm Shipment'
  AND b.timestamp < a.timestamp;
```

### 2. Find cases missing required activities (e.g., *Validate Stock*)

```sql
SELECT

    o.order_value

    ON o.case_id = log.case_id AND log.activity = 'Validate Stock'
WHERE log.event_id IS NULL;
```

### 3. Identify orders where *Receive Payment* occurs before *Issue Invoice*

### 4. List all activities that appear in an outofsequence order (general pattern)

```sql
-- Define expected order using a helper CTE
WITH expected_order AS (
    SELECT 'Register Order' AS activity, 1 AS seq
    UNION ALL SELECT 'Perform Credit Check', 2
    UNION ALL SELECT 'Validate Stock', 3
    UNION ALL SELECT 'Confirm Shipment', 4
    UNION ALL SELECT 'Ship Goods', 5
    UNION ALL SELECT 'Issue Invoice', 6
    UNION ALL SELECT 'Receive Payment', 7
),
actual_sequence AS (
    SELECT 
        case_id,
        activity,
        ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS seq
    FROM order_event_log
)
SELECT 
    a.case_id,
    a.activity,
    a.seq AS actual_seq,
    e.seq AS expected_seq
FROM actual_sequence a
JOIN expected_order e ON a.activity = e.activity
WHERE a.seq != e.seq
ORDER BY a.case_id, a.seq;
```

### 5. Find resources that are repeatedly involved in anomalies

```sql
-- Combine sequence violations and missing activities
WITH anomalies AS (
    SELECT case_id, resource 
    FROM order_event_log
    WHERE case_id IN (
        -- cases with early Confirm Shipment
        SELECT a.case_id
        FROM order_event_log a
        JOIN order_event_log b ON a.case_id = b.case_id
        WHERE a.activity = 'Perform Credit Check'
          AND b.activity = 'Confirm Shipment'
          AND b.timestamp < a.timestamp
        UNION
        -- cases missing Validate Stock
        SELECT case_id FROM orders
        WHERE case_id NOT IN (SELECT case_id FROM order_event_log WHERE activity = 'Validate Stock')
        UNION
        -- cases with early payment
        SELECT a.case_id
        FROM order_event_log a
        JOIN order_event_log b ON a.case_id = b.case_id
        WHERE a.activity = 'Issue Invoice'
          AND b.activity = 'Receive Payment'
          AND b.timestamp < a.timestamp
    )
)
SELECT 
    r.resource_id,
    r.role,
    r.department,
    COUNT(*) AS anomaly_involvement
FROM anomalies a
JOIN resources r ON a.resource = r.resource_id
GROUP BY r.resource_id, r.role, r.department
ORDER BY anomaly_involvement DESC;
```

### 6. Check if *priority* orders have different activity patterns

```sql
SELECT 
    o.case_id,
    o.order_type,
    COUNT(DISTINCT log.activity) AS activities_count,
    BOOL_OR(log.activity = 'Perform Credit Check') AS has_credit_check,
    BOOL_OR(log.activity = 'Validate Stock') AS has_stock_validation
FROM orders o
LEFT JOIN order_event_log log ON o.case_id = log.case_id
GROUP BY o.case_id, o.order_type
ORDER BY o.order_type, o.case_id;
```

### 7. Temporal gaps ñ find unusually fast transitions (e.g., *Receive Payment* minutes after *Register Order*)

```sql
SELECT 
    a.case_id,
    a.activity AS first_activity,
    a.timestamp AS first_time,
    b.activity AS second_activity,
    b.timestamp AS second_time,
    EXTRACT(EPOCH FROM (b.timestamp - a.timestamp)) / 60 AS gap_minutes
FROM order_event_log a
JOIN order_event_log b 
    ON a.case_id = b.case_id 
    AND a.event_id < b.event_id
WHERE a.activity = 'Register Order'
  AND b.activity = 'Receive Payment'
  AND EXTRACT(EPOCH FROM (b.timestamp - a.timestamp)) < 3600  -- less than 1 hour
ORDER BY gap_minutes;
```
