```sql
WITH RankedEvents AS (
    -- Assign a rank to each event within its case, ordered by timestamp.
    -- This is crucial for identifying directly consecutive events.
    SELECT
        case_id,
        activity,
        timestamp,
        ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) as rn
    FROM event_log
),
ActivitySequence AS (
    -- Join RankedEvents to itself to find events that are directly consecutive (rn, rn+1, rn+2).
    -- This identifies potential occurrences of the target 3-activity sequence:
    -- 'Approve Purchase Requisition' -> 'Create Order' -> 'Confirm Order'
    SELECT
        re1.case_id,
        re1.timestamp AS seq_ts_appr,  -- Timestamp of 'Approve Purchase Requisition'
        re2.timestamp AS seq_ts_create, -- Timestamp of 'Create Order'
        re3.timestamp AS seq_ts_confirm -- Timestamp of 'Confirm Order'
    FROM RankedEvents re1
    JOIN RankedEvents re2 ON re1.case_id = re2.case_id AND re1.rn + 1 = re2.rn
    JOIN RankedEvents re3 ON re1.case_id = re3.case_id AND re1.rn + 2 = re3.rn
    WHERE re1.activity = 'Approve Purchase Requisition'
      AND re2.activity = 'Create Order'
      AND re3.activity = 'Confirm Order'
),
ProblematicCases AS (
    -- From the identified sequences, filter for those where the time elapsed between
    -- the 'Create Order' event and the 'Confirm Order' event of that specific sequence
    -- is more than 5 days.
    -- Then, collect the unique case_ids that meet this condition.
    SELECT DISTINCT case_id
    FROM ActivitySequence
    WHERE seq_ts_confirm - seq_ts_create > INTERVAL '5 days'
)
-- Select all events from the original event_log table.
-- Exclude any events that belong to a case_id found in the ProblematicCases CTE.
SELECT
    e.case_id,
    e.activity,
    e.timestamp
FROM
    event_log e
WHERE NOT EXISTS (
    SELECT 1
    FROM ProblematicCases pc
    WHERE pc.case_id = e.case_id
)
ORDER BY
    e.case_id, e.timestamp;
```
