```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Activities
SR = Transition(label="Service Request Verification")
CS = Transition(label="Contract Setup")

DA = Transition(label="Datacenter Allocation")
ACI = Transition(label="Access Control Initialization")

BNC = Transition(label="Base Network Configuration")
DR = Transition(label="Debug & Reconfigure")

CD = Transition(label="Containerized Deployment")
MD = Transition(label="Monolithic Deployment")

QA = Transition(label="Quality Assurance")
SA = Transition(label="Security Audit")
GLA = Transition(label="Go-Live Approval")

# Silent transition import is available if needed, but this scenario does not require one
# skip = SilentTransition()

# 1) Parallel preparation after contract setup:
#    DA and ACI can be executed concurrently.
prep_phase = StrictPartialOrder(nodes=[DA, ACI])
# no edges added -> DA and ACI are concurrent

# 2) Looped configuration phase:
#    Execute BNC, then either exit, or execute DR and BNC again.
config_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[BNC, DR]
)

# 3) Exclusive deployment choice:
#    Either CD or MD.
deployment_choice = OperatorPOWL(
    operator=Operator.XOR,
    children=[CD, MD]
)

# 4) Final checks partial order:
#    QA and SA can happen in any order / concurrently,
#    but both must complete before GLA.
final_phase = StrictPartialOrder(nodes=[QA, SA, GLA])
final_phase.order.add_edge(QA, GLA)
final_phase.order.add_edge(SA, GLA)

# 5) Root process:
#    SR -> CS -> prep_phase -> config_loop -> deployment_choice -> final_phase
root = StrictPartialOrder(
    nodes=[SR, CS, prep_phase, config_loop, deployment_choice, final_phase]
)

root.order.add_edge(SR, CS)
root.order.add_edge(CS, prep_phase)
root.order.add_edge(prep_phase, config_loop)
root.order.add_edge(config_loop, deployment_choice)
root.order.add_edge(deployment_choice, final_phase)
```
