```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- Model 1: POWL with Potential Unfairness ---
# This model includes an XOR branch for the Cultural Fit stage where 
# one path (CommunityAffiliationCheck) provides a subtle advantage.

# Define Activities (Transitions)
receive_app = Transition(label="ReceiveApplication")
data_check = Transition(label="DataCompletenessCheck")
request_info = Transition(label="RequestMoreInfo")
skill_assess = Transition(label="SkillAssessment")
cultural_fit_standard = Transition(label="CulturalFitCheck")
community_affiliation = Transition(label="CommunityAffiliationCheck") # The biased branch
managerial_review = Transition(label="ManagerialReview")
final_decision = Transition(label="FinalDecision")

# Define Loop: Data Completeness Check vs Request More Info
# Logic: Execute data check, then either exit (handled by loop op) or request info -> back to check
loop_data_check = OperatorPOWL(operator=Operator.LOOP, children=[data_check, request_info])

# Define XOR Choice: Standard Cultural Fit vs Community Affiliation Check
xor_cultural = OperatorPOWL(operator=Operator.XOR, children=[cultural_fit_standard, community_affiliation])

# Define Root Partial Order to connect the operators
root_unfair = StrictPartialOrder(
    nodes=[receive_app, loop_data_check, skill_assess, xor_cultural, managerial_review, final_decision]
)

# Add Edges for Unfair Model
# 1. Receive Application -> Data Check Loop
root_unfair.order.add_edge(receive_app, loop_data_check)
# 2. Loop Completes -> Skill Assessment
root_unfair.order.add_edge(loop_data_check, skill_assess)
# 3. Skill Assessment -> XOR Choice (Cultural Fit / Community Affiliation)
root_unfair.order.add_edge(skill_assess, xor_cultural)
# 4. XOR Choice -> Managerial Review (Both branches merge here implicitly via the XOR operator logic)
root_unfair.order.add_edge(xor_cultural, managerial_review)
# 5. Managerial Review -> Final Decision
root_unfair.order.add_edge(managerial_review, final_decision)

print("Model 1 (Unfair) Construction Complete.")
print(f"Nodes: {len(root_unfair.nodes)}")
print(f"Edges: {len(root_unfair.order.edges)}")


# --- Model 2: POWL Without Unfairness ---
# This model removes the biased branch. All applicants go through the same 
# standard Cultural Fit evaluation. The XOR is removed, and it becomes sequential.

# Define Activities (Transitions) - Note: CommunityAffiliationCheck is removed/merged
receive_app = Transition(label="ReceiveApplication")
data_check = Transition(label="DataCompletenessCheck")
request_info = Transition(label="RequestMoreInfo")
skill_assess = Transition(label="SkillAssessment")
cultural_fit_standard = Transition(label="CulturalFitCheck") # Only one path now
managerial_review = Transition(label="ManagerialReview")
final_decision = Transition(label="FinalDecision")

# Define Loop: Data Completeness Check vs Request More Info (Same logic as Model 1)
loop_data_check = OperatorPOWL(operator=Operator.LOOP, children=[data_check, request_info])

# No XOR needed for Cultural Fit. It is a direct sequential step.
# However, to maintain structure consistency with the previous model's root definition style:
# We define the sequence directly.

# Define Root Partial Order to connect the operators
root_fair = StrictPartialOrder(
    nodes=[receive_app, loop_data_check, skill_assess, cultural_fit_standard, managerial_review, final_decision]
)

# Add Edges for Fair Model
# 1. Receive Application -> Data Check Loop
root_fair.order.add_edge(receive_app, loop_data_check)
# 2. Loop Completes -> Skill Assessment
root_fair.order.add_edge(loop_data_check, skill_assess)
# 3. Skill Assessment -> Cultural Fit Check (Sequential, No XOR)
root_fair.order.add_edge(skill_assess, cultural_fit_standard)
# 4. Cultural Fit Check -> Managerial Review
root_fair.order.add_edge(cultural_fit_standard, managerial_review)
# 5. Managerial Review -> Final Decision
root_fair.order.add_edge(managerial_review, final_decision)

print("Model 2 (Fair) Construction Complete.")
print(f"Nodes: {len(root_fair.nodes)}")
print(f"Edges: {len(root_fair.order.edges)}")
```
