#### Query 4: Resource Behavior Analysis (Who is involved in the most deviations?)
This query joins the event log with the `resources` table to identify if specific roles, departments, or individual resources are disproportionately involved in cases that exhibit anomalous behavior (like skipping credit checks).

```sql
WITH DeviatingCases AS (
    SELECT case_id
    FROM order_event_log
    GROUP BY case_id
    HAVING MAX(CASE WHEN activity = 'Perform Credit Check' THEN timestamp END) IS NULL
       OR MAX(CASE WHEN activity = 'Validate Stock' THEN timestamp END) IS NULL
       OR MAX(CASE WHEN activity = 'Ship Goods' THEN timestamp END) < MAX(CASE WHEN activity = 'Confirm Shipment' THEN timestamp END)
),
ResourceInvolvement AS (
    SELECT 
        e.resource,
        r.role,
        r.department,
        COUNT(DISTINCT e.case_id) AS total_cases_worked
    FROM order_event_log e
    JOIN resources r ON e.resource = r.resource_id
    GROUP BY e.resource, r.role, r.department
)
SELECT 
    ri.resource,
    ri.role,
    ri.department,
    COUNT(DISTINCT d.case_id) AS deviating_cases_involved,
    ROUND((COUNT(DISTINCT d.case_id)::numeric / ri.total_cases_worked) * 100, 2) AS deviation_rate_pct
FROM ResourceInvolvement ri
JOIN order_event_log e ON ri.resource = e.resource
JOIN DeviatingCases d ON e.case_id = d.case_id
GROUP BY ri.resource, ri.role, ri.department, ri.total_cases_worked
ORDER BY deviation_rate_pct DESC;
```
