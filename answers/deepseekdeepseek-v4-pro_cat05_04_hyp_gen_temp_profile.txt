    WHERE activity = 'P'

SELECT r.claim_id,
       EXTRACT(EPOCH FROM (p.p_time - r.r_time)) / 3600 AS hours_r2p

#### 5. Correlate anomalous pairs with adjusters, claim types, and regions
```sql
-- Example: high AdjusttoClose variance per adjuster
WITH a AS (
    SELECT claim_id, resource AS adjuster_id_str, timestamp AS a_time
    FROM claim_events
    WHERE activity = 'A'
), c AS (
    SELECT claim_id, timestamp AS c_time
    FROM claim_events
    WHERE activity = 'C'
), intervals AS (
    SELECT a.adjuster_id_str,
           EXTRACT(EPOCH FROM (c.c_time - a.a_time)) / 3600 AS hours
    FROM a JOIN c ON a.claim_id = c.claim_id
)
SELECT adj.name, adj.region,
       AVG(i.hours) AS avg_hours,
       STDDEV(i.hours) AS stddev_hours,
       COUNT(*) AS cnt
FROM intervals i
JOIN adjusters adj ON i.adjuster_id_str = adj.adjuster_id::VARCHAR
GROUP BY adj.name, adj.region
HAVING COUNT(*) > 5
ORDER BY stddev_hours DESC;
-- Identify adjusters with abnormally low/high standard deviation, then drill down by claim_type.
```
