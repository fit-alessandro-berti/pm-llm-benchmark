from pm4py.objects.process_tree.obj import ProcessTree, Operator

# --- small helpers ------------------------------------------------------------
def leaf(label: str) -> ProcessTree:
    return ProcessTree(label=label)  # parent will be set when attaching

def add(parent: ProcessTree, child: ProcessTree) -> None:
    child.parent = parent
    parent.children.append(child)

def seq_chain(nodes):
    """Builds a (binary) left-right nested SEQUENCE chain from a list of nodes."""
    if len(nodes) == 1:
        return nodes[0]
    sub = nodes[-1]
    for n in reversed(nodes[:-1]):
        parent = ProcessTree(operator=Operator.SEQUENCE)
        add(parent, n)
        add(parent, sub)
        sub = parent
    return sub

# --- leaves ------------------------------------------------------------------
A   = leaf("A")
RG  = leaf("RG")
IA  = leaf("IA")
TI  = leaf("TI")
TC  = leaf("TC")
CE  = leaf("CE")
FBP = leaf("FBP")

SC   = leaf("SC")
TM   = leaf("TM")
SPT  = leaf("SPT")

RDT_tau = leaf("tau")  # (loop redo part) silent step
FCE  = leaf("FCE")
FBA  = leaf("FBA")

AI = leaf("AI")
NC = leaf("NC")

FAS = leaf("FAS")
DDI = leaf("DDI")

BL  = leaf("BL")
ICP = leaf("ICP")
MRF = leaf("MRF")

tau_normal = leaf("tau")  # (XOR branch) silent step meaning "results already normal"

# --- triage sequence: TI -> TC ----------------------------------------------
triage_seq = ProcessTree(operator=Operator.SEQUENCE)
add(triage_seq, TI)
add(triage_seq, TC)

# --- initial diagnostic XOR: CE X FBP ----------------------------------------
initial_test = ProcessTree(operator=Operator.XOR)
add(initial_test, CE)
add(initial_test, FBP)

# --- treatment XOR: TM X SPT -----------------------------------------------
treatment_choice = ProcessTree(operator=Operator.XOR)
add(treatment_choice, TM)
add(treatment_choice, SPT)

# --- re-diagnostic XOR: FCE X FBA ------------------------------------------
rdt_choice = ProcessTree(operator=Operator.XOR)
add(rdt_choice, FCE)
add(rdt_choice, FBA)

# --- loop body: SC -> (TM X SPT) -> (FCE X FBA) ----------------------------
loop_body = ProcessTree(operator=Operator.SEQUENCE)
add(loop_body, SC)
add(loop_body, treatment_choice)
add(loop_body, rdt_choice)

# --- loop: *(loop_body, tau)  (repeat loop_body until normal) ----------------
abnormal_loop = ProcessTree(operator=Operator.LOOP)
add(abnormal_loop, loop_body)
add(abnormal_loop, RDT_tau)

# --- after initial test: tau (normal) X abnormal_loop ------------------------
post_initial_choice = ProcessTree(operator=Operator.XOR)
add(post_initial_choice, tau_normal)
add(post_initial_choice, abnormal_loop)

# --- concurrent advanced diagnostics: AI + NC --------------------------------
advanced_parallel = ProcessTree(operator=Operator.PARALLEL)
add(advanced_parallel, AI)
add(advanced_parallel, NC)

# --- follow-up sequence: FAS -> DDI ------------------------------------------
followup_seq = ProcessTree(operator=Operator.SEQUENCE)
add(followup_seq, FAS)
add(followup_seq, DDI)

# --- final parallel clearances: BL + (ICP + MRF) ------------------------------
parallel_icp_mrf = ProcessTree(operator=Operator.PARALLEL)
add(parallel_icp_mrf, ICP)
add(parallel_icp_mrf, MRF)

final_parallel = ProcessTree(operator=Operator.PARALLEL)
add(final_parallel, BL)
add(final_parallel, parallel_icp_mrf)

# --- entire patient stay: sequence of phases -------------------------------
root = seq_chain([
    A,
    RG,
    IA,
    triage_seq,
    initial_test,
    post_initial_choice,
    advanced_parallel,
    followup_seq,
    final_parallel
])

# `root` is the constructed process tree
