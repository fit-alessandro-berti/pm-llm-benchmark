```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# 1. Create the Petri Net object
net = PetriNet(name="Insurance_Claim_Process")

# 2. Define Places (States)
# We define places to represent the state of the claim between activities.
# Using descriptive names for clarity.
places = [
    PetriNet.Place(name="start_claim"),           # Initial State
    PetriNet.Place(name="claim_registered"),      # After Claim Registration (CR)
    PetriNet.Place(name="doc_ready_for_dv"),     # Input to Document Verification (DV)
    PetriNet.Place(name="doc_verified"),          # Output of Document Verification (DV)
    PetriNet.Place(name="fraud_check_passed"),    # Output of Fraud Check (FC)
    PetriNet.Place(name="claim_assessed"),        # After Claim Assessment (CA)
    PetriNet.Place(name="approved"),              # After Approval (AP)
    PetriNet.Place(name="paid"),                  # After Payment (P)
    PetriNet.Place(name="notified"),              # After Notification (N)
    PetriNet.Place(name="closed_claim")           # Final State (Closure)
]

for place in places:
    net.places.add(place)

# 3. Define Transitions (Activities)
transitions = [
    PetriNet.Transition(name="CR", label="Claim Registration"),
    PetriNet.Transition(name="DV", label="Document Verification"),
    PetriNet.Transition(name="RDS", label="Re-Document Submission"), # Loop back to DV
    PetriNet.Transition(name="FC", label="Fraud Check"),
    PetriNet.Transition(name="II", label="In-depth Investigation"), # Loop back to FC
    PetriNet.Transition(name="CA", label="Claim Assessment"),
    PetriNet.Transition(name="AP", label="Approval"),
    PetriNet.Transition(name="P", label="Payment"),
    PetriNet.Transition(name="N", label="Notification"),
    PetriNet.Transition(name="CL", label="Closure")
]

for transition in transitions:
    net.transitions.add(transition)

# 4. Define Arcs (Flow Logic)
# --- Main Linear Path ---
petri_utils.add_arc_from_to(start_claim, CR, net)
petri_utils.add_arc_from_to(CR, claim_registered, net)

petri_utils.add_arc_from_to(claim_registered, DV, net)
petri_utils.add_arc_from_to(DV, doc_ready_for_dv, net)

# Note: In a strict Petri Net flow, we usually define the input place for the next transition.
# However, to allow loops cleanly, we can treat 'doc_verified' as the state after DV.
# If DV is passed, token goes to FC. 
# To model the loop logic clearly using standard PM4Py structure:
# We will route tokens explicitly between states.

# Let's refine the flow for loops based on standard Petri Net construction in PM4Py:
# Place A -> Transition X -> Place B
# Loop: Place B -> Transition Y -> Place A

# Re-defining arcs with specific logic to ensure connectivity:
# 1. Start -> CR -> claim_registered (Output of CR)
petri_utils.add_arc_from_to(start_claim, CR, net)
petri_utils.add_arc_from_to(CR, claim_registered, net)

# 2. claim_registered -> DV -> doc_ready_for_dv
petri_utils.add_arc_from_to(claim_registered, DV, net)
petri_utils.add_arc_from_to(DV, doc_ready_for_dv, net)

# 3. Loop DV: If documents incomplete (RDS), go back to claim_registered (Input of DV)
# Note: In the example above, 'doc_ready_for_dv' is the output of DV. 
# The input for DV is 'claim_registered'. RDS should feed into 'claim_registered'.
petri_utils.add_arc_from_to(doc_ready_for_dv, RDS, net)
petri_utils.add_arc_from_to(RDS, claim_registered, net)

# 4. doc_ready_for_dv -> FC (Proceed if DV passed)
# Wait, to make the loop work cleanly, 'doc_ready_for_dv' is the state after DV.
# The next step is FC. So we need an arc from doc_ready_for_dv to FC.
petri_utils.add_arc_from_to(doc_ready_for_dv, FC, net)

# 5. FC -> fraud_check_passed (Output of FC)
petri_utils.add_arc_from_to(FC, fraud_check_passed, net)

# 6. Loop FC: If doubts (II), go back to doc_ready_for_dv (Input of FC)
# Wait, if II is performed, we need to re-run FC. So II output -> Input of FC.
petri_utils.add_arc_from_to(fraud_check_passed, II, net)
petri_utils.add_arc_from_to(II, doc_ready_for_dv, net)

# 7. fraud_check_passed -> CA (Proceed if FC passed)
petri_utils.add_arc_from_to(fraud_check_passed, CA, net)

# 8. CA -> claim_assessed (Output of CA)
petri_utils.add_arc_from_to(CA, claim_assessed, net)

# 9. claim_assessed -> AP (Proceed to Assessment/Approval logic)
# Note: In this model, we assume successful assessment leads to approval transition for the accepting net.
petri_utils.add_arc_from_to(claim_assessed, AP, net)

# 10. AP -> approved (Output of AP)
petri_utils.add_arc_from_to(AP, approved, net)

# 11. approved -> P (Proceed to Payment)
petri_utils.add_arc_from_to(approved, P, net)

# 12. P -> paid (Output of P)
petri_utils.add_arc_from_to(P, paid, net)

# 13. paid -> N (Proceed to Notification)
petri_utils.add_arc_from_to(paid, N, net)

# 14. N -> notified (Output of N)
petri_utils.add_arc_from_to(N, notified, net)

# 15. notified -> CL (Proceed to Closure)
petri_utils.add_arc_from_to(notified, CL, net)

# 16. CL -> closed_claim (Final State)
petri_utils.add_arc_from_to(CL, closed_claim, net)

# --- Optional: Add a sink place if you want to strictly follow the prompt's "remove sink" style 
# But for an Accepting Petri Net, the 'closed_claim' place serves as the final state.
# We will not add/remove extra places unless necessary to match the example exactly, 
# but we must define the Final Marking on 'closed_claim'.

# 5. Define Initial Marking (im)
# Token starts at 'start_claim'
im = Marking()
im[places[0]] = 1  # start_claim

# 6. Define Final Marking (fm)
# Token ends at 'closed_claim'
fm = Marking()
fm[places[-1]] = 1  # closed_claim

# Output the constructed Petri Net and Markings
print(f"Constructed Petri Net: {net.name}")
print(f"Number of Places: {len(net.places)}")
print(f"Number of Transitions: {len(net.transitions)}")
print(f"Initial Marking Tokens: {im}")
print(f"Final Marking Tokens: {fm}")
```
