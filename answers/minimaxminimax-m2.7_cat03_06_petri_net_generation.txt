```python
    t_fc_flag = PetriNet.Transition(name="t_fc_flag", label="FC Flag")
    t_ii = PetriNet.Transition(name="t_ii", label="In-depth Investigation")
```

```python
# --------------------------------------------------------------
# Accepting Petri net for an insuranceclaim management process
# --------------------------------------------------------------
# The net contains:
#   • a start place  – a token is placed here when a new claim arrives
#   • a sink place   – a token ends up here when the claim is closed
#   • a linear flow that follows the normal case
#   • two loops:
#        1. Document Verification    ReDocument Submission    Document Verification (again)
#        2. Fraud Check             Indepth Investigation     Fraud Check (again)
#
# The initial and final markings are:
#       im : { start : 1 }
#       fm : { sink  : 1 }
# --------------------------------------------------------------

from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

def build_insurance_claim_net():
    """
    Constructs the accepting Petri net described in the scenario.
    Returns a tuple (net, im, fm).
    """
    # ------------------------------------------------------------------
    # 1. Create an empty Petri net
    # ------------------------------------------------------------------
    net = PetriNet(name="InsuranceClaimAcceptingPetriNet")

    # ------------------------------------------------------------------
    # 2. Places
    # ------------------------------------------------------------------
    #   start                     – new claim arrives
    #   after_ClaimRegistration   – after the claim has been logged
    #   after_DocumentVerification – after the documents have been checked
    #   after_ReDocumentSubmission – while the customer is resubmitting documents
    #   before_FraudCheck          – just before the fraudcheck activity
    #   after_FraudCheck           – after the fraudcheck activity
    #   after_InDepthInvestigation – while an indepth investigation is performed
    #   after_ClaimAssessment      – after the adjuster has assessed the claim
    #   after_Approval             – after a senior manager has approved (if needed)
    #   after_Payment              – after the money has been transferred
    #   after_Notification        – after the customer has been notified
    #   sink                       – final place (closed claim)

    p_start = PetriNet.Place(name="start")
    p_cr = PetriNet.Place(name="after_ClaimRegistration")
    p_dv = PetriNet.Place(name="after_DocumentVerification")
    p_rds = PetriNet.Place(name="after_ReDocumentSubmission")
    p_fc_in = PetriNet.Place(name="before_FraudCheck")
    p_fc_res = PetriNet.Place(name="after_FraudCheck")
    p_ii = PetriNet.Place(name="after_InDepthInvestigation")
    p_ca = PetriNet.Place(name="after_ClaimAssessment")
    p_ap = PetriNet.Place(name="after_Approval")
    p_pay = PetriNet.Place(name="after_Payment")
    p_not = PetriNet.Place(name="after_Notification")
    p_sink = PetriNet.Place(name="sink")

    for pl in [p_start, p_cr, p_dv, p_rds,
               p_fc_in, p_fc_res, p_ii,
               p_ca, p_ap, p_pay, p_not, p_sink]:
        net.places.add(pl)

    # ------------------------------------------------------------------
    # 3. Transitions  (label = the activity name that appears in the log)
    # ------------------------------------------------------------------
    #   Claim Registration
    t_cr = PetriNet.Transition(name="t_ClaimRegistration",
                               label="Claim Registration")

    #   Document Verification (first occurrence)
    t_dv_init = PetriNet.Transition(name="t_DocumentVerification",
                                    label="Document Verification")

    #   “DV OK” – documents are fine, go on to fraud check
    t_dv_ok = PetriNet.Transition(name="t_DV_OK",
                                  label="DV OK")

    #   “ReDocument Submission” – documents were incomplete
    t_dv_resub = PetriNet.Transition(name="t_ReDocumentSubmission",
                                     label="ReDocument Submission")

    #   “Document Verification” after the customer has resubmitted docs
    t_dv_repeat = PetriNet.Transition(name="t_DocumentVerificationAfterResub",
                                      label="Document Verification")

    #   Fraud Check (first occurrence)
    t_fc_init = PetriNet.Transition(name="t_FraudCheck",
                                    label="Fraud Check")

    #   “FC OK” – no fraud suspicion, continue with assessment
    t_fc_ok = PetriNet.Transition(name="t_FC_OK",
                                  label="FC OK")

    #   “FC Flag” – fraud suspicion, go to indepth investigation
    t_fc_flag = PetriNet.Transition(name="t_FC_Flag",
                                    label="FC Flag")

    #   “Fraud Check” after an indepth investigation (loop back)
    t_fc_repeat = PetriNet.Transition(name="t_FraudCheckAfterInvestigation",
                                      label="Fraud Check")

    #   Indepth Investigation
    t_ii = PetriNet.Transition(name="t_InDepthInvestigation",
                               label="Indepth Investigation")

    #   Claim Assessment
    t_ca = PetriNet.Transition(name="t_ClaimAssessment",
                               label="Claim Assessment")

    #   Approval (senior manager)
    t_ap = PetriNet.Transition(name="t_Approval",
                               label="Approval")

    #   Payment
    t_pay = PetriNet.Transition(name="t_Payment",
                                label="Payment")

    #   Notification
    t_not = PetriNet.Transition(name="t_Notification",
                               label="Notification")

    #   Closure (optional explicit step – we keep the sink as the final place)
    #   If you prefer an explicit “Closure” transition, you can add it later.

    all_transitions = [
        t_cr,
        t_dv_init, t_dv_ok, t_dv_resub, t_dv_repeat,
        t_fc_init, t_fc_ok, t_fc_flag, t_fc_repeat,
        t_ii,
        t_ca,
        t_ap,
        t_pay,
        t_not,
        # t_cl   # <-- you could add a “Closure” transition here if desired
    ]
    for tr in all_transitions:
        net.transitions.add(tr)

    # ------------------------------------------------------------------
    # 4. Arcs – connect places  transitions
    # ------------------------------------------------------------------
    # 4.1. Start  Claim Registration  after_ClaimRegistration
    petri_utils.add_arc_from_to(p_start, t_cr, net)
    petri_utils.add_arc_from_to(t_cr, p_cr, net)

    # 4.2. Document Verification (first pass)
    petri_utils.add_arc_from_to(p_cr, t_dv_init, net)
    petri_utils.add_arc_from_to(t_dv_init, p_dv, net)

    # 4.3. Two possible outcomes of DV:
    #      a) OK  go to fraud check
    petri_utils.add_arc_from_to(p_dv, t_dv_ok, net)
    petri_utils.add_arc_from_to(t_dv_ok, p_fc_in, net)

    #      b) Incomplete  ReDocument Submission  (repeat DV)
    petri_utils.add_arc_from_to(p_dv, t_dv_resub, net)       # incomplete docs
    petri_utils.add_arc_from_to(t_dv_resub, p_rds, net)      # place while waiting
    petri_utils.add_arc_from_to(p_rds, t_dv_repeat, net)     # reverify docs
    petri_utils.add_arc_from_to(t_dv_repeat, p_dv, net)      # back to DV

    # 4.4. Fraud Check (first pass)
    petri_utils.add_arc_from_to(p_fc_in, t_fc_init, net)
    petri_utils.add_arc_from_to(t_fc_init, p_fc_res, net)

    # 4.5. Two possible outcomes of FC:
    #      a) OK  Claim Assessment
    petri_utils.add_arc_from_to(p_fc_res, t_fc_ok, net)
    petri_utils.add_arc_from_to(t_fc_ok, p_ca, net)

    #      b) Flag  Indepth Investigation  (repeat FC)
    petri_utils.add_arc_from_to(p_fc_res, t_fc_flag, net)
    petri_utils.add_arc_from_to(t_fc_flag, p_ii, net)        # start investigation
    petri_utils.add_arc_from_to(p_ii, t_fc_repeat, net)      # rerun fraud check
    petri_utils.add_arc_from_to(t_fc_repeat, p_fc_in, net)   # back to FC

    # 4.6. Claim Assessment  Approval  Payment  Notification  sink
    petri_utils.add_arc_from_to(p_ca, t_ca, net)
    petri_utils.add_arc_from_to(t_ca, p_ap, net)

    petri_utils.add_arc_from_to(p_ap, t_ap, net)
    petri_utils.add_arc_from_to(t_ap, p_pay, net)

    petri_utils.add_arc_from_to(p_pay, t_pay, net)
    petri_utils.add_arc_from_to(t_pay, p_not, net)

    petri_utils.add_arc_from_to(p_not, t_not, net)
    petri_utils.add_arc_from_to(t_not, p_sink, net)

    # ------------------------------------------------------------------
    # 5. Initial and final markings
    # ------------------------------------------------------------------
    im = Marking()
    im[p_start] = 1                         # one token when a new claim arrives

    fm = Marking()
    fm[p_sink] = 1                          # one token when the claim is closed

    return net, im, fm


# ------------------------------------------------------------------
# Example usage (you can run this block directly)
# ------------------------------------------------------------------
if __name__ == "__main__":
    net, initial_marking, final_marking = build_insurance_claim_net()

    print("Petri net created successfully!")
    print(f"Number of places: {len(net.places)}")
    print(f"Number of transitions: {len(net.transitions)}")
    print(f"Initial marking: {initial_marking}")
    print(f"Final marking: {final_marking}")
```

### What the model contains

| Place / Transition | Meaning in the business process |
|--------------------|----------------------------------|
| **start** | A new claim arrives in the system |
| **t_ClaimRegistration**  `after_ClaimRegistration` | The claim is logged |
| **t_DocumentVerification**  `after_DocumentVerification` | Documents are verified |
| `after_DocumentVerification` has two outgoing arcs:<br>• `t_DV_OK`  `before_FraudCheck` (documents OK)<br>• `t_ReDocumentSubmission`  `after_ReDocumentSubmission` (incomplete) |
| `after_ReDocumentSubmission`  `t_DocumentVerificationAfterResub`  `after_DocumentVerification` | Loop: the customer resubmits docs and verification is repeated |
| **t_FraudCheck**  `after_FraudCheck` | Fraud check is executed |
| `after_FraudCheck` has two outgoing arcs:<br>• `t_FC_OK`  `after_ClaimAssessment` (no suspicion)<br>• `t_FC_Flag`  `after_InDepthInvestigation` (suspicion) |
| `after_InDepthInvestigation`  `t_FraudCheckAfterInvestigation`  `before_FraudCheck` | Loop: after investigation a new fraud check is performed |
| **t_ClaimAssessment**  `after_Approval` | Adjuster decides compensation |
| **t_Approval**  `after_Payment` | Senior manager approves (if needed) |
| **t_Payment**  `after_Notification` | Money is transferred |
| **t_Notification**  `sink` | Customer is notified and the claim is closed |

The **initial marking** (`im`) puts a single token in the `start` place, and the **final marking** (`fm`) expects a single token in the `sink` place, exactly matching the definition of an *accepting Petri net*.
