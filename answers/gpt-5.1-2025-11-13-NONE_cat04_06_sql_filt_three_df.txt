```sql
WITH ordered AS (
    SELECT
        case_id,
        activity,
        "timestamp",
        ROW_NUMBER() OVER (
            PARTITION BY case_id
            ORDER BY "timestamp", activity, rowid
        ) AS rn
    FROM event_log
),

-- Find all triplets of directly successive events
triplets AS (
    SELECT
        o1.case_id,
        o1.rn      AS rn1,
        o2.rn      AS rn2,
        o3.rn      AS rn3,
        o1.activity AS act1,
        o2.activity AS act2,
        o3.activity AS act3,
        o2."timestamp" AS ts_create_order,
        o3."timestamp" AS ts_confirm_order
    FROM ordered o1
    JOIN ordered o2
        ON o2.case_id = o1.case_id
       AND o2.rn = o1.rn + 1
    JOIN ordered o3
        ON o3.case_id = o1.case_id
       AND o3.rn = o1.rn + 2
),

-- Cases that have the specific sequence in direct succession
-- and where Create  Confirm takes more than 5 days
blocked_cases AS (
    SELECT DISTINCT case_id
    FROM triplets
    WHERE
        act1 = 'Approve Purchase Requisition'
        AND act2 = 'Create Order'
        AND act3 = 'Confirm Order'
        AND ts_confirm_order - ts_create_order > INTERVAL '5 days'
)

-- Final result: all events except those from blocked cases
SELECT *
FROM event_log
WHERE case_id NOT IN (SELECT case_id FROM blocked_cases)
ORDER BY case_id, "timestamp";
```
