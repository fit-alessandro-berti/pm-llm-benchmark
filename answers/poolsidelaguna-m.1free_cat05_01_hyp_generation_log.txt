### 1. Identify Out-of-Order Activity Sequences

```sql
-- Find cases where activities occurred in the wrong sequence
WITH activity_sequence AS (
  SELECT 
    case_id,
    activity,
    timestamp,
    LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp) as previous_activity
  FROM order_event_log
),
sequence_violations AS (
  SELECT 
    case_id,
    activity,
    previous_activity,
    timestamp
  FROM activity_sequence
  WHERE 
    -- Goods shipped before Validate Stock
    (activity = 'Ship Goods' AND previous_activity IS DISTINCT FROM 'Validate Stock') OR
    -- Payment received before Invoice issued
    (activity = 'Receive Payment' AND previous_activity IS DISTINCT FROM 'Issue Invoice') OR
    -- Credit check after Shipment confirmed
    (activity = 'Perform Credit Check' AND previous_activity IN ('Confirm Shipment', 'Ship Goods')) OR
    -- Shipment confirmed after goods shipped
    (activity = 'Confirm Shipment' AND previous_activity = 'Ship Goods')
)
SELECT * FROM sequence_violations ORDER BY case_id, timestamp;
```

### 2. Detect Missing Activities in Process Flow

```sql
-- Identify cases that skipped critical steps
WITH case_activities AS (
  SELECT 
    case_id,
    COUNT(*) FILTER (WHERE activity = 'Validate Stock') as has_stock_validation,
    COUNT(*) FILTER (WHERE activity = 'Perform Credit Check') as has_credit_check,
    COUNT(*) FILTER (WHERE activity = 'Confirm Shipment') as has_shipment_confirmation,
    COUNT(*) FILTER (WHERE activity = 'Issue Invoice') as has_invoice
  FROM order_event_log
  GROUP BY case_id
)
SELECT 
  case_id,
  CASE WHEN has_stock_validation = 0 THEN 'Missing Validate Stock' END as missing_stock,
  CASE WHEN has_credit_check = 0 THEN 'Missing Credit Check' END as missing_credit,
  CASE WHEN has_shipment_confirmation = 0 THEN 'Missing Shipment Confirmation' END as missing_confirmation,
  CASE WHEN has_invoice = 0 THEN 'Missing Invoice' END as missing_invoice
FROM case_activities
WHERE has_stock_validation = 0 
   OR has_credit_check = 0 
   OR has_shipment_confirmation = 0 
   OR has_invoice = 0;
```

### 3. Analyze Timing Anomalies Between Critical Steps

```sql
-- Calculate time gaps between activities to find unnatural sequences
WITH activity_dates AS (
  SELECT 
    o1.case_id,
    o1.activity as early_activity,
    o2.activity as later_activity,
    o1.timestamp as early_time,
    o2.timestamp as later_time,
    EXTRACT(EPOCH FROM (o2.timestamp - o1.timestamp))/60 as minutes_difference
  FROM order_event_log o1
  JOIN order_event_log o2 ON o1.case_id = o2.case_id
  WHERE o1.timestamp < o2.timestamp  -- Ensure proper chronological ordering
    AND (
      (o1.activity = 'Register Order' AND o2.activity IN ('Receive Payment', 'Ship Goods')) OR
      (o1.activity = 'Issue Invoice' AND o2.activity = 'Perform Credit Check') OR
      (o1.activity = 'Ship Goods' AND o2.activity = 'Validate Stock')
    )
)
SELECT 
  case_id,
  early_activity,
  later_activity,
  minutes_difference,
  CASE 
    WHEN minutes_difference < 5 THEN 'Highly Suspicious - Same Process Instance'
    ELSE 'Requires Investigation'
  END as risk_level
FROM activity_dates
WHERE minutes_difference < 60  -- Flag if these activities happened within an hour
ORDER BY minutes_difference ASC;
```

### 5. Root Cause Analysis - Department-Level Pattern Analysis

```sql
-- Analyze which departments/resources are more prone to process violations
WITH process_order AS (
  SELECT 
    case_id,
    activity,
    timestamp,
    ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) as step_number
  FROM order_event_log
),
anomaly_flags AS (
  SELECT 
    po.case_id,
    r.role,
    r.department,
    po.activity,
    po.step_number
  FROM process_order po
  JOIN order_event_log oel ON po.case_id = oel.case_id AND po.activity = oel.activity AND po.timestamp = oel.timestamp
  JOIN resources r ON oel.resource = r.resource_id
  WHERE po.activity IN ('Ship Goods', 'Receive Payment', 'Issue Invoice')
    AND po.step_number < 4  -- These activities should never be in steps 1-3
)
SELECT 
  department,
  role,
  COUNT(*) as anomaly_count,
  STRING_AGG(DISTINCT activity, ', ') as activities_involved
FROM anomaly_flags
GROUP BY department, role
ORDER BY anomaly_count DESC;
```
