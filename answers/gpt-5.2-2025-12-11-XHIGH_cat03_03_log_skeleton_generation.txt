acts = ["RO", "CA", "RI", "PI", "QC", "PO", "GSL", "D", "IC", "RP"]

log_skeleton = {
    # all activities occur exactly once in a case -> same occurrence count for every pair
    "equivalence": {(a, b) for a in acts for b in acts if a != b},

    # if a later activity occurs, all earlier activities must have occurred before
    "always_before": {(acts[i], acts[j]) for i in range(1, len(acts)) for j in range(0, i)},

    # if an activity occurs, all later activities must occur after it
    "always_after": {(acts[i], acts[j]) for i in range(0, len(acts) - 1) for j in range(i + 1, len(acts))},

    # no mutually-exclusive activities are stated in the scenario
    "never_together": set(),

    # each activity occurs exactly once per order
    "activ_freq": {a: {1} for a in acts},

    # strict step-by-step flow
    "directly_follows": set(zip(acts, acts[1:])),
}
