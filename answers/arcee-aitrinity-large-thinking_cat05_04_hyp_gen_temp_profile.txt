### SQL Verification Queries

#### 1. Identify Claims with Anomalous RP Timing
```sql
WITH claim_timing AS (
  SELECT 
    ce1.claim_id,
    ce1.activity AS act1,
    ce1.timestamp AS ts1,
    ce2.activity AS act2,
    ce2.timestamp AS ts2,
    EXTRACT(EPOCH FROM (ce2.timestamp - ce1.timestamp)) AS interval_seconds
  FROM claim_events ce1
  JOIN claim_events ce2 
    ON ce1.claim_id = ce2.claim_id 
    AND ce1.activity = 'R' 
    AND ce2.activity = 'P'
    AND ce1.timestamp < ce2.timestamp
),
anomalies AS (
  SELECT 
    claim_id,
    interval_seconds,
    CASE 
      WHEN interval_seconds < (3600 - 2*600) THEN 'Too Fast'
      WHEN interval_seconds > (3600 + 2*600) THEN 'Too Slow'
      ELSE 'Normal'
    END AS anomaly_type
  FROM claim_timing
)
SELECT * FROM anomalies WHERE anomaly_type != 'Normal';
```

#### 2. Detect PN Delays by Adjuster and Region
```sql
WITH p_to_n AS (
  SELECT 
    ce1.claim_id,
    ce1.resource AS adjuster,
    c.claim_type,
    a.region,
    EXTRACT(EPOCH FROM (ce2.timestamp - ce1.timestamp)) AS hours_diff
  FROM claim_events ce1
  JOIN claim_events ce2 
    ON ce1.claim_id = ce2.claim_id 
    AND ce1.activity = 'P' 
    AND ce2.activity = 'N'
    AND ce1.timestamp < ce2.timestamp
  JOIN claims c ON ce1.claim_id = c.claim_id
  JOIN adjusters a ON ce1.resource = a.name
)
SELECT 
  adjuster,
  region,
  AVG(hours_diff)/3600 AS avg_days,
  COUNT(*) AS claim_count,
  STDDEV(hours_diff)/3600 AS std_dev_days
FROM p_to_n
GROUP BY adjuster, region, claim_type
HAVING AVG(hours_diff)/3600 > 5  -- Filter for unusually long delays
ORDER BY avg_days DESC;
```

#### 3. Find Claims with AC but Missing E/P Steps
```sql
WITH claim_events_ordered AS (
  SELECT 
    claim_id,
    activity,
    timestamp,
    ROW_NUMBER() OVER (PARTITION BY claim_id ORDER BY timestamp) AS seq_num
  FROM claim_events
),
assign_close AS (
  SELECT claim_id
  FROM claim_events_ordered
  WHERE activity IN ('A', 'C')
  GROUP BY claim_id
  HAVING COUNT(DISTINCT activity) = 2  -- Has both A and C
),
missing_steps AS (
  SELECT ac.claim_id
  FROM assign_close ac
  LEFT JOIN claim_events ce 
    ON ac.claim_id = ce.claim_id 
    AND ce.activity IN ('E', 'P')
  GROUP BY ac.claim_id
  HAVING COUNT(ce.activity) = 0  -- No E or P events
)
SELECT * FROM missing_steps;
```

#### 5. Correlate Anomalies with Claim Type and Region
```sql
WITH anomalies_flagged AS (
  SELECT 
    ce1.claim_id,
    c.claim_type,
    a.region,

      WHEN EXTRACT(EPOCH FROM (ce2.timestamp - ce1.timestamp)) < 300 THEN 'EN Too Fast'
      WHEN EXTRACT(EPOCH FROM (ce2.timestamp - ce1.timestamp)) > 86400 THEN 'RE Too Slow'

    AND ce1.activity = 'E' 
    AND ce2.activity = 'N'
    AND ce1.timestamp < ce2.timestamp

  JOIN adjusters a ON ce1.resource = a.name

SELECT 
  claim_type,
  region,
  anomaly_category,
  COUNT(*) AS anomaly_count
FROM anomalies_flagged
WHERE anomaly_category != 'Normal'
GROUP BY claim_type, region, anomaly_category
ORDER BY anomaly_count DESC;
```
