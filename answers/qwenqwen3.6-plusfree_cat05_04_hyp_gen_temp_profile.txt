### SQL Verification Queries

The following PostgreSQL queries are designed to isolate anomalous claims, calculate actual timing distributions, and correlate deviations with business dimensions (claim type, region, adjuster, resource).

#### 1. Base Temporal Pivot (Reusable CTE)
```sql
WITH claim_timings AS (
    SELECT 
        claim_id,
        MAX(CASE WHEN activity = 'R' THEN timestamp END) AS ts_R,
        MAX(CASE WHEN activity = 'A' THEN timestamp END) AS ts_A,
        MAX(CASE WHEN activity = 'E' THEN timestamp END) AS ts_E,
        MAX(CASE WHEN activity = 'P' THEN timestamp END) AS ts_P,
        MAX(CASE WHEN activity = 'N' THEN timestamp END) AS ts_N,
        MAX(CASE WHEN activity = 'C' THEN timestamp END) AS ts_C

    GROUP BY claim_id
)
SELECT * FROM claim_timings; -- Base for subsequent queries
```

#### 2. Verify R  P Low Variance & Identify Patterns
```sql
WITH claim_timings AS (
    SELECT claim_id,
           MAX(CASE WHEN activity = 'R' THEN timestamp END) AS ts_R,
           MAX(CASE WHEN activity = 'P' THEN timestamp END) AS ts_P
    FROM claim_events GROUP BY claim_id
)
SELECT 
    c.claim_type,
    ce.resource,
    COUNT(*) AS claim_count,
    AVG(EXTRACT(EPOCH FROM (ts_P - ts_R))) AS avg_duration_sec,
    STDDEV(EXTRACT(EPOCH FROM (ts_P - ts_R))) AS stdev_duration_sec
FROM claim_timings ct
JOIN claims c ON ct.claim_id = c.claim_id
JOIN claim_events ce ON ct.claim_id = ce.claim_id AND ce.activity = 'P'
WHERE ct.ts_R IS NOT NULL AND ct.ts_P IS NOT NULL
GROUP BY c.claim_type, ce.resource
HAVING STDDEV(EXTRACT(EPOCH FROM (ts_P - ts_R))) < 3600; -- STDEV < 1 hour
```

#### 3. Identify P  N Excessive Delays & Regional/Adjuster Correlation
```sql
WITH claim_timings AS (
    SELECT claim_id,
           MAX(CASE WHEN activity = 'P' THEN timestamp END) AS ts_P,
           MAX(CASE WHEN activity = 'N' THEN timestamp END) AS ts_N
    FROM claim_events GROUP BY claim_id
)
SELECT 
    c.claim_id,
    c.customer_id,
    a.region,
    a.name AS adjuster_name,
    EXTRACT(EPOCH FROM (ct.ts_N - ct.ts_P)) AS p_to_n_seconds,
    EXTRACT(DAY FROM (ct.ts_N - ct.ts_P)) AS delay_days
FROM claim_timings ct
JOIN claims c ON ct.claim_id = c.claim_id
LEFT JOIN claim_events ce ON ct.claim_id = ce.claim_id AND ce.activity = 'P'
LEFT JOIN adjusters a ON ce.resource::INTEGER = a.adjuster_id
WHERE ct.ts_P IS NOT NULL AND ct.ts_N IS NOT NULL
  AND EXTRACT(EPOCH FROM (ct.ts_N - ct.ts_P)) > 864000 -- > 10 days (threshold for outlier)
ORDER BY delay_days DESC;
```

#### 4. Detect A  C Premature Closures & Missing Steps
```sql
WITH claim_timings AS (
    SELECT claim_id,
           MAX(CASE WHEN activity = 'A' THEN timestamp END) AS ts_A,
           MAX(CASE WHEN activity = 'E' THEN timestamp END) AS ts_E,
           MAX(CASE WHEN activity = 'P' THEN timestamp END) AS ts_P,
           MAX(CASE WHEN activity = 'C' THEN timestamp END) AS ts_C
    FROM claim_events GROUP BY claim_id
)
SELECT 
    ct.claim_id,
    c.claim_type,
    EXTRACT(EPOCH FROM (ct.ts_C - ct.ts_A)) AS a_to_c_seconds,
    CASE WHEN ct.ts_E IS NULL THEN 'Missing Evaluate' ELSE 'Present' END AS eval_status,
    CASE WHEN ct.ts_P IS NULL THEN 'Missing Approve' ELSE 'Present' END AS approve_status
FROM claim_timings ct
JOIN claims c ON ct.claim_id = c.claim_id
WHERE ct.ts_A IS NOT NULL AND ct.ts_C IS NOT NULL
  AND EXTRACT(EPOCH FROM (ct.ts_C - ct.ts_A)) <= 7200 --  2 hours
ORDER BY a_to_c_seconds ASC;
```

#### 5. Analyze E  N Instantaneous Transitions & Resource Type
```sql
WITH claim_timings AS (
    SELECT claim_id,
           MAX(CASE WHEN activity = 'E' THEN timestamp END) AS ts_E,
           MAX(CASE WHEN activity = 'N' THEN timestamp END) AS ts_N
    FROM claim_events GROUP BY claim_id
)
SELECT 
    ct.claim_id,
    c.claim_type,
    EXTRACT(EPOCH FROM (ct.ts_N - ct.ts_E)) AS e_to_n_seconds,
    ce.resource,
    CASE WHEN ce.resource ILIKE '%system%' OR ce.resource ILIKE '%bot%' OR ce.resource ILIKE '%auto%' 
         THEN 'Automated' ELSE 'Human/Manual' END AS resource_type
FROM claim_timings ct
JOIN claims c ON ct.claim_id = c.claim_id
JOIN claim_events ce ON ct.claim_id = ce.claim_id AND ce.activity = 'E'
WHERE ct.ts_E IS NOT NULL AND ct.ts_N IS NOT NULL
  AND EXTRACT(EPOCH FROM (ct.ts_N - ct.ts_E)) <= 300 --  5 minutes
ORDER BY e_to_n_seconds ASC;
```
