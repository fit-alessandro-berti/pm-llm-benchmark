#### 1. Identify Sequence Violations (Logistics before Finance)
**Purpose:** To test Hypotheses A and B by finding all cases where physical logistics steps happen before financial/warehouse safeguards.
```sql
WITH EventTimestamps AS (
    SELECT 
        case_id,
        MIN(timestamp) FILTER (WHERE activity = 'Perform Credit Check') AS credit_check_ts,
        MIN(timestamp) FILTER (WHERE activity = 'Ship Goods') AS ship_goods_ts,
        MIN(timestamp) FILTER (WHERE activity = 'Confirm Shipment') AS confirm_shipment_ts
    FROM order_event_log
    GROUP BY case_id
)
SELECT 
    case_id, 
    credit_check_ts, 
    confirm_shipment_ts, 
    ship_goods_ts
FROM EventTimestamps
WHERE ship_goods_ts < credit_check_ts 
   OR ship_goods_ts < confirm_shipment_ts;
```

#### 2. Correlate Anomalies with Order Type and Value
**Purpose:** To test Hypothesis A (Fast-tracking). This checks if out-of-sequence events are statistically linked to `priority` orders or high `order_value`.
```sql
WITH Anomalies AS (
    SELECT DISTINCT case_id
    FROM (
        SELECT case_id,
               LAG(activity) OVER(PARTITION BY case_id ORDER BY timestamp) as prev_activity,
               activity
        FROM order_event_log
    ) t
    WHERE activity IN ('Ship Goods', 'Confirm Shipment')
      AND prev_activity NOT IN ('Perform Credit Check', 'Validate Stock')
      AND prev_activity IS NOT NULL
)
SELECT 
    o.order_type, 
    COUNT(a.case_id) as anomaly_count,
    AVG(o.order_value) as avg_order_value,
    COUNT(o.case_id) as total_orders_in_type
FROM orders o
LEFT JOIN Anomalies a ON o.case_id = a.case_id
GROUP BY o.order_type;
```

#### 3. Find Cases with Missing Mandatory Activities
**Purpose:** To test Hypothesis C. Identifies processes that completely skipped critical control points (like Stock Validation).
```sql
SELECT 
    o.case_id, 
    o.order_type,
    COUNT(e.activity) as total_events_recorded,
    BOOL_OR(e.activity = 'Validate Stock') as stock_checked,
    BOOL_OR(e.activity = 'Perform Credit Check') as credit_checked
FROM orders o
JOIN order_event_log e ON o.case_id = e.case_id
GROUP BY o.case_id, o.order_type
HAVING NOT BOOL_OR(e.activity = 'Validate Stock') 
   OR NOT BOOL_OR(e.activity = 'Perform Credit Check');
```

#### 4. Analyze Resource Propensity for Out-of-Sequence Events
**Purpose:** To test Hypothesis D (Training issues). Identifies which specific users or roles are most frequently associated with triggering events out of the standard sequence.
```sql
WITH SequenceChecks AS (
    SELECT 
        case_id,
        activity,
        resource,
        timestamp,
        LAG(activity) OVER(PARTITION BY case_id ORDER BY timestamp) as prev_activity
    FROM order_event_log
),
OutOFSequence AS (
    SELECT 
        resource,
        activity
    FROM SequenceChecks
    WHERE (activity = 'Ship Goods' AND prev_activity != 'Confirm Shipment')
       OR (activity = 'Confirm Shipment' AND prev_activity NOT IN ('Validate Stock', 'Perform Credit Check'))
)
SELECT 
    r.resource_id,
    r.role,
    r.department,
    oos.activity,
    COUNT(*) as out_of_sequence_count
FROM OutOFSequence oos
JOIN resources r ON oos.resource = r.resource_id
GROUP BY r.resource_id, r.role, r.department, oos.activity
ORDER BY out_of_sequence_count DESC;
```
