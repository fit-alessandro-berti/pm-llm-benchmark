```sql
WITH case_sequences AS (
    SELECT 
        case_id,
        LIST_AGG(activity ORDER BY timestamp) AS sequence
    FROM event_log
    GROUP BY case_id
),
variant_counts AS (
    SELECT 
        sequence,
        COUNT(*) AS frequency
    FROM case_sequences
    GROUP BY sequence
),
top_variants AS (
    SELECT sequence
    FROM variant_counts
    ORDER BY frequency DESC
    LIMIT K
)
SELECT 
    el.*
FROM event_log el
INNER JOIN case_sequences cs 
    ON el.case_id = cs.case_id
INNER JOIN top_variants tv 
    ON cs.sequence = tv.sequence
ORDER BY el.case_id, el.timestamp;
```

**Notes:**
- Replace `K` with the desired integer (e.g., `LIMIT 5`).
- `LIST_AGG` returns an ordered list per case; DuckDB can group by and join on lists.
- The final result contains only the original events belonging to cases that follow one of the top-K most frequent variants.
