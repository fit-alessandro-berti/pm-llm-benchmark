from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Define activities as transitions
OR = Transition(label="Order Received")
MS = Transition(label="Material Sourcing")
QS = Transition(label="Quality Screening")
AS = Transition(label="Assembly")
IQC = Transition(label="In-line Quality Check")
TST = Transition(label="Testing")
RT = Transition(label="Re-Testing")
PK = Transition(label="Packaging")
LB = Transition(label="Labeling")
DP = Transition(label="Distribution Planning")
D = Transition(label="Dispatch")

# Define silent transition
END = SilentTransition()  # Represents the end of the process

# Create the quality screening loop
quality_screening_loop = OperatorPOWL(operator=Operator.LOOP, 
                                     children=[QS, AS])

# Create the re-testing loop structure
re_test_decision = OperatorPOWL(operator=Operator.XOR, 
                               children=[IQC, END])
re_test_loop = OperatorPOWL(operator=Operator.LOOP, 
                           children=[TST, re_test_decision])

# Main process structure
# Order -> Material Sourcing -> Quality Screening (may repeat) -> Assembly
# Assembly leads to multiple possible paths
main_path = OperatorPOWL(operator=Operator.XOR, 
                        children=[quality_screening_loop, AS])

# Testing can have a re-testing loop
test_section = OperatorPOWL(operator=Operator.XOR, 
                           children=[TST, re_test_loop])

# Packaging can go directly or through labeling (optional)
pkg_section = OperatorPOWL(operator=Operator.XOR, 
                          children=[PK, LB])

# Create the strict partial order
root = StrictPartialOrder(nodes=[OR, MS, main_path, test_section, 
                              pkg_section, DP, D])

# Define the partial order relations
# Main process order
root.order.add_edge(OR, MS)
root.order.add_edge(MS, main_path)

# Quality screening loop path
root.order.add_edge(main_path, test_section)

# Critical path with re-testing
root.order.add_edge(test_section, TST)
root.order.add_edge(TST, RT)  # Re-testing can happen after initial test
root.order.add_edge(RT, TST)  # Loop back to testing
root.order.add_edge(TST, re_test_decision)

# After all test stages, go to packaging
root.order.add_edge(re_test_decision, PK)  # Bypass labeling if no loop
root.order.add_edge(pkg_section, DP)
root.order.add_edge(DP, D)

# Handle alternative case where LB is skipped (without quality screening loop)

The POWL model captures:
- **Loops**: The quality screening loop (`* (QS, AS)`) and re-testing loop (`* (TST, re_test_decision)`)
- **Choice structures**: Multiple XOR operators to model different paths based on process requirements

- **Partial ordering**: The sequence of activities with appropriate dependencies
