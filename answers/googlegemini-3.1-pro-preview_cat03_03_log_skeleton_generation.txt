# The chronological strict sequence of activities in the fulfillment process
activities = ['RO', 'CA', 'RI', 'PI', 'QC', 'PO', 'GSL', 'D', 'IC', 'RP']

log_skeleton = {
    # EQUIVALENCE: All activities occur exactly once, therefore every activity
    # has the same number of occurrences as any other activity in the sequence.
    'equivalence': {
        (a, b) for a in activities for b in activities if a != b
    },

    # ALWAYS BEFORE: If activity A occurs, B must have occurred before it.
    # We map (A, B) such that B's index is strictly less than A's index.
    'always_before': {
        (activities[i], activities[j]) 
        for i in range(1, len(activities)) 
        for j in range(i)
    },

    # ALWAYS AFTER: If activity A occurs, B must occur after it.
    # We map (A, B) such that B's index is strictly greater than A's index.
    'always_after': {
        (activities[i], activities[j]) 
        for i in range(len(activities) - 1) 
        for j in range(i + 1, len(activities))
    },

    # NEVER TOGETHER: In this straight-through scenario, all listed activities 
    # harmonize and coexist. No negative conditions (e.g., Reject vs. Dispatch) exist.
    'never_together': set(),

    # ACTIVITY OCCURRENCES: Every activity executes exactly once {1}.
    'activ_freq': {
        'RO': {1}, 'CA': {1}, 'RI': {1}, 'PI': {1}, 'QC': {1}, 
        'PO': {1}, 'GSL': {1}, 'D': {1}, 'IC': {1}, 'RP': {1}
    },

    # DIRECTLY-FOLLOWS: If the first activity occurs, the second occurs immediately.
    # Represents the 9 direct sequence edges of the process flow.
    'directly_follows': {
        ('RO', 'CA'),
        ('CA', 'RI'),
        ('RI', 'PI'),

        ('PO', 'GSL'),
        ('GSL', 'D'),
        ('D', 'IC'),
        ('IC', 'RP')
    }
}
