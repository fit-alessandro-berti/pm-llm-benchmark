## 3. Verification Queries

Here are SQL queries to detect these anomalies in production data:

### Query 1: Claims Closed Without Evaluation/Approval
```sql
SELECT c.claim_id, c.claim_type, c.submission_date
FROM claims c
WHERE NOT EXISTS (
    -- No evaluation event
    SELECT 1 FROM claim_events ce 
    WHERE ce.claim_id = c.claim_id AND ce.activity = 'E'
) AND NOT EXISTS (
    -- No approval event  
    SELECT 1 FROM claim_events ce 
    WHERE ce.claim_id = c.claim_id AND ce.activity = 'P'
);
```

### Query 2: Multiple Approvals Per Claim
```sql
SELECT claim_id, COUNT(*) as approval_count
FROM claim_events 
WHERE activity = 'P'
GROUP BY claim_id
HAVING COUNT(*) > 1
ORDER BY approval_count DESC;
```

### Query 3: Prematurely Closed Claims (AC without E/P)
```sql
WITH claim_sequence AS (
    SELECT 
        claim_id,
        activity,
        ROW_NUMBER() OVER (PARTITION BY claim_id ORDER BY timestamp) as seq
    FROM claim_events ce
    JOIN claims c ON ce.claim_id = c.claim_id
    WHERE activity IN ('R', 'A', 'E', 'P', 'N', 'C')
)
SELECT cs.claim_id
FROM claim_sequence cs
WHERE cs.activity = 'C' 
  AND cs.seq < (
    SELECT MIN(seq) 
    FROM claim_sequence cs2 
    WHERE cs2.claim_id = cs.claim_id 
      AND cs2.activity IN ('E', 'P')
  );
```

### Query 4: Claims Missing Customer Notification
```sql
SELECT c.claim_id, c.claim_type
FROM claims c
WHERE c.claim_id NOT IN (
    SELECT DISTINCT claim_id 
    FROM claim_events 
    WHERE activity = 'N'
) AND c.claim_id IN (
    SELECT DISTINCT claim_id 
    FROM claim_events 
    WHERE activity = 'C'  -- Only closed claims
);
```

### Query 5: Process Sequence Anomaly Detection
```sql
SELECT 
    c.claim_id,
    STRING_AGG(ce.activity, ',' ORDER BY ce.timestamp) as actual_sequence
FROM claims c
JOIN claim_events ce ON c.claim_id = ce.claim_id
WHERE ce.activity IN ('R', 'A', 'E', 'P', 'N', 'C')
GROUP BY c.claim_id
HAVING 
    -- Closure before evaluation/approval

    AND NOT STRING_AGG(ce.activity, ',' ORDER BY ce.timestamp) LIKE '%,E,P%'
    OR
    -- Multiple E/P cycles
    (LENGTH(STRING_AGG(ce.activity, ',' ORDER BY ce.timestamp)) - 
     LENGTH(REPLACE(STRING_AGG(ce.activity, ',' ORDER BY ce.timestamp), 'E', ''))) > 2;
```
