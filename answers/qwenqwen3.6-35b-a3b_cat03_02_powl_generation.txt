import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Define all activities (Transitions)
SR = Transition(label="SR")
CS = Transition(label="CS")
DA = Transition(label="DA")
ACI = Transition(label="ACI")
BNC = Transition(label="BNC")
DR = Transition(label="D&R")  # Debug & Reconfigure
CD = Transition(label="CD")
MD = Transition(label="MD")
QA = Transition(label="QA")
SA = Transition(label="SA")
GLA = Transition(label="GLA")

# 2. Define Loop and Choice operators
# Loop: BNC is executed; if issues, D&R is executed, then BNC again.
# OperatorPOWL with LOOP takes [A, B] where A is the body and B is the loop continuation.
loop_config = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DR])

# Choice: Either Containerized Deployment or Monolithic Deployment.
# OperatorPOWL with XOR takes [A, B] for exclusive choice.
xor_deploy = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# 3. Construct the root Partial Order model
# The root contains all top-level nodes (activities and operator nodes).
# Children inside operators (BNC, DR, CD, MD) are encapsulated and not listed here.
root = StrictPartialOrder(nodes=[SR, CS, DA, ACI, loop_config, xor_deploy, QA, SA, GLA])

# 4. Define the execution order (edges)

# Step 1 & 2: SR -> CS -> {DA, ACI}
root.order.add_edge(SR, CS)
root.order.add_edge(CS, DA)
root.order.add_edge(CS, ACI)

# Step 3: DA and ACI must complete before the loop starts
root.order.add_edge(DA, loop_config)
root.order.add_edge(ACI, loop_config)

# Step 4: Loop completes before the deployment choice
root.order.add_edge(loop_config, xor_deploy)

# Step 5: Deployment choice completes before QA and SA
root.order.add_edge(xor_deploy, QA)
root.order.add_edge(xor_deploy, SA)

# Step 6: QA and SA must complete before Go-Live Approval
root.order.add_edge(QA, GLA)
root.order.add_edge(SA, GLA)

# The POWL model 'root' is now fully constructed.
# It represents:
# SR -> CS -> {DA || ACI} -> Loop(BNC, D&R) -> XOR(CD, MD) -> {QA || SA} -> GLA
