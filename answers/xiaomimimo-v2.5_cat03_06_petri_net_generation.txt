from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_process")

# Create places
places = {}
place_names = [
    "new_claim",           # Start: new claim arrives
    "after_cr",            # After Claim Registration
    "after_dv",            # After Document Verification
    "after_rds",           # After Re-Document Submission
    "after_fc",            # After Fraud Check
    "after_ii",            # After In-depth Investigation
    "after_ca",            # After Claim Assessment
    "after_ap",            # After Approval
    "after_p",             # After Payment
    "after_n",             # After Notification
    "claim_closed"         # Final: claim is closed
]

for name in place_names:
    places[name] = PetriNet.Place(name=name)
    net.places.add(places[name])

# Create transitions
transitions = {}
transition_data = {
    "CR": ("Claim Registration", "CR"),
    "DV": ("Document Verification", "DV"),
    "RDS": ("Re-Document Submission", "RDS"),
    "FC": ("Fraud Check", "FC"),
    "II": ("In-depth Investigation", "II"),
    "CA": ("Claim Assessment", "CA"),
    "AP": ("Approval", "AP"),
    "P": ("Payment", "P"),
    "N": ("Notification", "N"),
    "CL": ("Closure", "CL")
}

for key, (label, name) in transition_data.items():
    transitions[key] = PetriNet.Transition(name=name, label=label)
    net.transitions.add(transitions[key])

# Create arcs
arc_connections = [
    # Main flow
    (places["new_claim"], transitions["CR"]),
    (transitions["CR"], places["after_cr"]),
    (places["after_cr"], transitions["DV"]),
    (transitions["DV"], places["after_dv"]),
    (places["after_dv"], transitions["FC"]),
    (transitions["FC"], places["after_fc"]),
    (places["after_fc"], transitions["CA"]),
    (transitions["CA"], places["after_ca"]),
    (places["after_ca"], transitions["AP"]),
    (transitions["AP"], places["after_ap"]),
    (places["after_ap"], transitions["P"]),
    (transitions["P"], places["after_p"]),
    (places["after_p"], transitions["N"]),
    (transitions["N"], places["after_n"]),
    (places["after_n"], transitions["CL"]),
    (transitions["CL"], places["claim_closed"]),
    
    # Loop for Document Re-submission (DV -> RDS -> DV)
    (places["after_dv"], transitions["RDS"]),  # From DV place to RDS transition
    (transitions["RDS"], places["after_rds"]), # From RDS to after_rds place
    (places["after_rds"], transitions["DV"]),  # From after_rds back to DV transition
    (transitions["DV"], places["after_dv"]),   # From DV back to after_dv (this creates the loop)
    
    # Loop for In-depth Investigation (FC -> II -> FC)
    (places["after_fc"], transitions["II"]),   # From FC place to II transition
    (transitions["II"], places["after_ii"]),   # From II to after_ii place
    (places["after_ii"], transitions["FC"]),   # From after_ii back to FC transition
    (transitions["FC"], places["after_fc"]),   # From FC back to after_fc (this creates the loop)
]

for from_obj, to_obj in arc_connections:
    petri_utils.add_arc_from_to(from_obj, to_obj, net)

# Create initial marking (token at new_claim)
im = Marking()
im[places["new_claim"]] = 1

# Create final marking (token at claim_closed)
fm = Marking()
fm[places["claim_closed"]] = 1

# Print net details for verification
print("Petri net created successfully!")
print(f"Places: {len(net.places)}")
print(f"Transitions: {len(net.transitions)}")
print(f"Arcs: {len(net.arcs)}")
print(f"Initial marking: {im}")
print(f"Final marking: {fm}")

This Petri net models the insurance claim management process with the following key features:

1. **Sequential Flow**: The main path follows the standard process: CR  DV  FC  CA  AP  P  N  CL

2. **Document Re-submission Loop**: If documents are incomplete after Document Verification (DV), the process can loop back through Re-Document Submission (RDS) and return to Document Verification (DV)

3. **In-depth Investigation Loop**: If Fraud Check (FC) raises doubts, the process can loop through In-depth Investigation (II) and return to Fraud Check (FC), allowing multiple investigations before proceeding

4. **Initial Marking**: Token starts in the "new_claim" place, representing a new claim arriving

5. **Final Marking**: Token ends in the "claim_closed" place, representing the claim has been closed and archived

The loops are implemented using intermediate places ("after_rds" and "after_ii") to allow multiple iterations of the respective activities while maintaining the Petri net structure.
