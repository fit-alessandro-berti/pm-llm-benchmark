Below is one reasonable accepting Petri net for the insurance-claim scenario.

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# -------------------------
# Places
# -------------------------
source = PetriNet.Place("source")              # new claim arrived
p_after_cr = PetriNet.Place("p_after_cr")      # after Claim Registration
p_after_dv = PetriNet.Place("p_after_dv")      # after Document Verification
p_ready_fc = PetriNet.Place("p_ready_fc")      # ready for Fraud Check
p_after_fc = PetriNet.Place("p_after_fc")      # after Fraud Check
p_after_ca = PetriNet.Place("p_after_ca")      # after Claim Assessment
p_after_ap = PetriNet.Place("p_after_ap")      # after Approval / approval not needed
p_after_p = PetriNet.Place("p_after_p")        # after Payment
p_after_n = PetriNet.Place("p_after_n")        # after Notification
sink = PetriNet.Place("sink")                  # claim closed

for p in [
    source, p_after_cr, p_after_dv, p_ready_fc, p_after_fc,
    p_after_ca, p_after_ap, p_after_p, p_after_n, sink
]:
    net.places.add(p)

# -------------------------
# Transitions
# -------------------------
t_cr = PetriNet.Transition("t_cr", "Claim Registration (CR)")
t_dv = PetriNet.Transition("t_dv", "Document Verification (DV)")
t_rds = PetriNet.Transition("t_rds", "Re-Document Submission (RDS)")
t_docs_ok = PetriNet.Transition("t_docs_ok", None)   # invisible routing transition
t_fc = PetriNet.Transition("t_fc", "Fraud Check (FC)")
t_ii = PetriNet.Transition("t_ii", "In-depth Investigation (II)")
t_ca = PetriNet.Transition("t_ca", "Claim Assessment (CA)")
t_ap = PetriNet.Transition("t_ap", "Approval (AP)")
t_skip_ap = PetriNet.Transition("t_skip_ap", None)   # invisible skip if approval not needed
t_p = PetriNet.Transition("t_p", "Payment (P)")
t_n = PetriNet.Transition("t_n", "Notification (N)")
t_cl = PetriNet.Transition("t_cl", "Closure (CL)")

for t in [
    t_cr, t_dv, t_rds, t_docs_ok, t_fc, t_ii,
    t_ca, t_ap, t_skip_ap, t_p, t_n, t_cl
]:
    net.transitions.add(t)

# -------------------------
# Arcs
# -------------------------

# Start -> Claim Registration
petri_utils.add_arc_from_to(source, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_after_cr, net)

# Claim Registration -> Document Verification
petri_utils.add_arc_from_to(p_after_cr, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

# If documents are incomplete: re-submit docs, then verify again
petri_utils.add_arc_from_to(p_after_dv, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_after_cr, net)

# If documents are okay: continue to fraud-check stage
petri_utils.add_arc_from_to(p_after_dv, t_docs_ok, net)
petri_utils.add_arc_from_to(t_docs_ok, p_ready_fc, net)

# Fraud Check
petri_utils.add_arc_from_to(p_ready_fc, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

# If doubts remain: in-depth investigation, then fraud check again
petri_utils.add_arc_from_to(p_after_fc, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_ready_fc, net)

# If cleared: move to claim assessment
petri_utils.add_arc_from_to(p_after_fc, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

# Approval may be required for large claims
petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_after_ap, net)

# Otherwise skip approval
petri_utils.add_arc_from_to(p_after_ca, t_skip_ap, net)
petri_utils.add_arc_from_to(t_skip_ap, p_after_ap, net)

# Payment -> Notification -> Closure
petri_utils.add_arc_from_to(p_after_ap, t_p, net)
petri_utils.add_arc_from_to(t_p, p_after_p, net)

petri_utils.add_arc_from_to(p_after_p, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_n, net)

petri_utils.add_arc_from_to(p_after_n, t_cl, net)
petri_utils.add_arc_from_to(t_cl, sink, net)

# -------------------------
# Initial and final markings
# -------------------------
im = Marking()
im[source] = 1

fm = Marking()
fm[sink] = 1

# The accepting Petri net is (net, im, fm)
```

Notes:
- The token starts in `source`, representing a newly arrived claim.
- The loop `DV -> RDS -> DV` is modeled by sending `RDS` back to `p_after_cr`.
- The loop `FC -> II -> FC` is modeled by sending `II` back to `p_ready_fc`.
- `Approval (AP)` is optional here, so I used an invisible transition `t_skip_ap` to bypass it when not needed.
