```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create a Petri net for the insurance claim process
net = PetriNet(name="insurance_claim_process")

# Create places (states)
start = PetriNet.Place(name="start")  # New claim arrives
after_cr = PetriNet.Place(name="after_cr")  # After claim registration
after_dv = PetriNet.Place(name="after_dv")  # After document verification
after_fc = PetriNet.Place(name="after_fc")  # After fraud check
after_ca = PetriNet.Place(name="after_ca")  # After claim assessment
after_ap = PetriNet.Place(name="after_ap")  # After approval
after_p = PetriNet.Place(name="after_p")  # After payment
after_n = PetriNet.Place(name="after_n")  # After notification
end = PetriNet.Place(name="end")  # Claim closed

# Add places to the net
for place in [start, after_cr, after_dv, after_fc, after_ca, after_ap, after_p, after_n, end]:
    net.places.add(place)

# Create transitions (activities)
cr = PetriNet.Transition(name="cr", label="Claim Registration")
dv = PetriNet.Transition(name="dv", label="Document Verification")
rds = PetriNet.Transition(name="rds", label="Re-Document Submission")
fc = PetriNet.Transition(name="fc", label="Fraud Check")
ii = PetriNet.Transition(name="ii", label="In-depth Investigation")
ca = PetriNet.Transition(name="ca", label="Claim Assessment")
ap = PetriNet.Transition(name="ap", label="Approval")
p = PetriNet.Transition(name="p", label="Payment")
n = PetriNet.Transition(name="n", label="Notification")
cl = PetriNet.Transition(name="cl", label="Closure")

# Add transitions to the net
for transition in [cr, dv, rds, fc, ii, ca, ap, p, n, cl]:
    net.transitions.add(transition)

# Build the process flow with arcs
# Main linear flow
petri_utils.add_arc_from_to(start, cr, net)
petri_utils.add_arc_from_to(cr, after_cr, net)
petri_utils.add_arc_from_to(after_cr, dv, net)
petri_utils.add_arc_from_to(dv, after_dv, net)

# Document re-submission loop (DV  RDS)
petri_utils.add_arc_from_to(after_dv, rds, net)  # If documents incomplete
petri_utils.add_arc_from_to(rds, dv, net)  # Go back to DV after resubmission
petri_utils.add_arc_from_to(after_dv, fc, net)  # If documents complete

# Fraud check with investigation loop (FC  II)
petri_utils.add_arc_from_to(fc, after_fc, net)
petri_utils.add_arc_from_to(after_fc, ii, net)  # If doubts raised
petri_utils.add_arc_from_to(ii, fc, net)  # Go back to FC after investigation
petri_utils.add_arc_from_to(after_fc, ca, net)  # If no doubts

# Continue linear flow
petri_utils.add_arc_from_to(ca, after_ca, net)
petri_utils.add_arc_from_to(after_ca, ap, net)
petri_utils.add_arc_from_to(ap, after_ap, net)
petri_utils.add_arc_from_to(after_ap, p, net)
petri_utils.add_arc_from_to(p, after_p, net)
petri_utils.add_arc_from_to(after_p, n, net)
petri_utils.add_arc_from_to(n, after_n, net)
petri_utils.add_arc_from_to(after_n, cl, net)
petri_utils.add_arc_from_to(cl, end, net)

# Define initial marking (one token in start place)
im = Marking()
im[start] = 1

# Define final marking (one token in end place)
fm = Marking()
fm[end] = 1
```

This Petri net models the insurance claim process with the following features:

1. **Initial State**: Starts with a token in the `start` place (new claim arrives)

2. **Main Process Flow**: Sequential activities: Claim Registration  Document Verification  Fraud Check  Claim Assessment  Approval  Payment  Notification  Closure

3. **Document Resubmission Loop**: 
   - After Document Verification (`after_dv`), there are two possible paths:
     - If documents are incomplete: Goes to Re-Document Submission (`rds`)  back to Document Verification (`dv`)
     - If documents are complete: Continues to Fraud Check (`fc`)

4. **Investigation Loop**:
   - After Fraud Check (`after_fc`), there are two possible paths:
     - If doubts are raised: Goes to In-depth Investigation (`ii`)  back to Fraud Check (`fc`)
     - If no doubts: Continues to Claim Assessment (`ca`)

5. **Final State**: Ends with a token in the `end` place (claim closed)

The Petri net captures the complexity of the process including the iterative document verification and fraud investigation phases, while maintaining a clear linear progression through the main activities. The initial marking represents a new claim entering the system, and the final marking represents a successfully processed and closed claim.
