SA = Transition(label="Security Audit")

GLA = Transition(label="Go-Live Approval")

workflow.order.add_edge(CS, DA)

# Step 3: Both DA and ACI must complete before the configuration loop starts
workflow.order.add_edge(DA, loop)
workflow.order.add_edge(ACI, loop)

# Step 4: Loop must complete before choosing deployment method
workflow.order.add_edge(loop, xor)

# Step 5: Deployment choice must complete before QA and SA (which run in parallel)
workflow.order.add_edge(xor, QA)
workflow.order.add_edge(xor, SA)
