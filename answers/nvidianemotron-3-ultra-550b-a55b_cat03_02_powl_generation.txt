import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Define all Transitions (Activities)
sr = Transition(label="Service Request (SR)")
cs = Transition(label="Contract Setup (CS)")

da = Transition(label="Datacenter Allocation (DA)")
aci = Transition(label="Access Control Initialization (ACI)")

bnc = Transition(label="Base Network Configuration (BNC)")
dr = Transition(label="Debug & Reconfigure (D&R)")

cd = Transition(label="Containerized Deployment (CD)")
md = Transition(label="Monolithic Deployment (MD)")

qa = Transition(label="Quality Assurance (QA)")
sa = Transition(label="Security Audit (SA)")

gla = Transition(label="Go-Live Approval (GLA)")

# 2. Construct the Inner Partial Order for Initial Parallel Setup (Step 2)
# DA and ACI are concurrent (no order between them)
po_init = StrictPartialOrder(nodes=[da, aci])
# No edges added to po_init.order, meaning they are unordered/concurrent

# 3. Construct the Loop for Configuration Phase (Step 3)
# Loop semantics: Execute BNC, then either exit or execute D&R and repeat BNC.
config_loop = OperatorPOWL(operator=Operator.LOOP, children=[bnc, dr])

# 4. Construct the Exclusive Choice for Deployment (Step 4)
# XOR semantics: Choose exactly one of CD or MD.
deployment_choice = OperatorPOWL(operator=Operator.XOR, children=[cd, md])

# 5. Construct the Inner Partial Order for Final Checks (Step 5)
# QA and SA are concurrent (no order between them), both must finish before GLA.
po_final = StrictPartialOrder(nodes=[qa, sa])
# No edges added to po_final.order, meaning they are unordered/concurrent

# 6. Construct the Root Partial Order (Top-level Process Flow)
# Nodes represent the major sequential blocks/steps of the process.
root = StrictPartialOrder(nodes=[
    sr,              # Step 1 Start
    cs,              # Step 1 End / Prereq
    po_init,         # Step 2 (Parallel Block)
    config_loop,     # Step 3 (Loop Block)
    deployment_choice, # Step 4 (Choice Block)
    po_final,        # Step 5 (Parallel Block)
    gla              # Step 6 (Final Step)
])

# 7. Define Execution Order (Edges) in the Root Partial Order
# Step 1: SR -> CS
root.order.add_edge(sr, cs)

# Step 2: CS -> (DA || ACI) represented by CS -> po_init
root.order.add_edge(cs, po_init)

# Step 3: (DA && ACI) -> Loop. Since po_init contains DA and ACI with no internal order,
# an edge from po_init to config_loop implies both must complete before loop starts.
root.order.add_edge(po_init, config_loop)

# Step 4: Loop -> Deployment Choice
root.order.add_edge(config_loop, deployment_choice)

# Step 5: Deployment Choice -> (QA || SA) represented by deployment_choice -> po_final
root.order.add_edge(deployment_choice, po_final)

# Step 6: (QA && SA) -> GLA. Edge from po_final to gla implies both QA and SA must complete.
root.order.add_edge(po_final, gla)

# The 'root' variable now holds the complete POWL model for the scenario.
