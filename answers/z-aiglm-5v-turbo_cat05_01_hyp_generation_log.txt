#### Query 1: Detect "Dangerous" Transitions (Sequence Checking)
This query identifies instances where a high-risk activity (like Shipping or Payment) occurs immediately after Registration without the prerequisite safety checks (Credit/Stock).

```sql
WITH OrderedEvents AS (
    SELECT 
        case_id,
        activity,
        timestamp,
        LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp, event_id) as prev_activity
    FROM 
        order_event_log
),
CriticalViolations AS (
    SELECT 
        *,
        CASE 
            -- Scenario: Shipping/Payment happens right after Registration
            WHEN activity IN ('Ship Goods', 'Receive Payment') AND prev_activity = 'Register Order' THEN 'CRITICAL: Action taken before validation'
            -- Scenario: Shipping happens before Stock Validation (Generic check)
            WHEN activity = 'Ship Goods' AND prev_activity NOT IN ('Validate Stock', 'Confirm Shipment') THEN 'WARNING: Premature Shipping'
            -- Scenario: Payment before Invoice
            WHEN activity = 'Receive Payment' AND prev_activity != 'Issue Invoice' THEN 'ANOMALY: Payment before Invoice'
            ELSE NULL
        END as violation_type
    FROM 
        OrderedEvents
)
SELECT 
    case_id,
    timestamp,
    prev_activity as step_before,
    activity as current_step,
    violation_type
FROM 
    CriticalViolations
WHERE 
    violation_type IS NOT NULL
ORDER BY 
    timestamp;
```

```sql
WITH ShipDates AS (

    FROM order_event_log WHERE activity = 'Ship Goods'

CreditCheckDates AS (
    SELECT case_id, timestamp AS credit_time

    o.order_value,
```
