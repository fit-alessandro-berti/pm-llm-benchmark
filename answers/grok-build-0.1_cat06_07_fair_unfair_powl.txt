```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# ============================================
# POWL Model with Potential Unfairness
# ============================================

receive = Transition(label="ReceiveApplication")
data_check = Transition(label="DataCompletenessCheck")
request_info = Transition(label="RequestMoreInfo")
skill = Transition(label="SkillAssessment")
cultural = Transition(label="CulturalFitCheck")
community = Transition(label="CommunityAffiliationCheck")
review = Transition(label="ManagerialReview")
final = Transition(label="FinalDecision")

# Loop for data completeness (do check, then either exit or request more info and repeat)
data_loop = OperatorPOWL(operator=Operator.LOOP, children=[data_check, request_info])

# XOR branch after skill assessment: standard cultural fit OR community affiliation (potential bias point)
cultural_xor = OperatorPOWL(operator=Operator.XOR, children=[cultural, community])

# Main partial order
unfair_model = StrictPartialOrder(nodes=[receive, data_loop, skill, cultural_xor, review, final])
unfair_model.order.add_edge(receive, data_loop)
unfair_model.order.add_edge(data_loop, skill)
unfair_model.order.add_edge(skill, cultural_xor)
unfair_model.order.add_edge(cultural_xor, review)
unfair_model.order.add_edge(review, final)


# ============================================
# POWL Model Without Unfairness
# ============================================

receive_f = Transition(label="ReceiveApplication")
data_check_f = Transition(label="DataCompletenessCheck")
request_info_f = Transition(label="RequestMoreInfo")
skill_f = Transition(label="SkillAssessment")
cultural_f = Transition(label="CulturalFitCheck")
review_f = Transition(label="ManagerialReview")
final_f = Transition(label="FinalDecision")

# Loop for data completeness (same as above)
data_loop_f = OperatorPOWL(operator=Operator.LOOP, children=[data_check_f, request_info_f])

# Fair model: no XOR, all applicants go through the same cultural fit check
fair_model = StrictPartialOrder(nodes=[receive_f, data_loop_f, skill_f, cultural_f, review_f, final_f])
fair_model.order.add_edge(receive_f, data_loop_f)
fair_model.order.add_edge(data_loop_f, skill_f)
fair_model.order.add_edge(skill_f, cultural_f)
fair_model.order.add_edge(cultural_f, review_f)
fair_model.order.add_edge(review_f, final_f)
```
