def node(op, parent=None):
    n = ProcessTree(operator=op, parent=parent)
    if parent is not None:
        parent.children.append(n)
    return n

def leaf(label, parent):
    n = ProcessTree(label=label, parent=parent)
    parent.children.append(n)
    return n

root = node(Operator.SEQUENCE)

# 1. Arrival/Registration/Intake
arrival = node(Operator.SEQUENCE, root)
leaf("A", arrival); leaf("RG", arrival); leaf("IA", arrival)

# 2. Triage
triage = node(Operator.SEQUENCE, root)
leaf("TI", triage); leaf("TC", triage)

# 3. Initial diagnostic test
initial = node(Operator.XOR, root)
leaf("CE", initial); leaf("FBP", initial)

# 4. Optional abnormal-results loop
post_test = node(Operator.XOR, root)
leaf(None, post_test)  # normal: skip loop
loop = node(Operator.LOOP, post_test)
body = node(Operator.SEQUENCE, loop)
leaf("SC", body)
treat = node(Operator.XOR, body)
leaf("TM", treat); leaf("SPT", treat)
rdt = node(Operator.XOR, body)
leaf("FCE", rdt); leaf("FBA", rdt)

leaf("FAS", final); leaf("DDI", final)

# 7. Final clearances (parallel)

Below is the process tree for the described hospital patient-management scenario, first in compact notation and then as a concrete `pm4py` construction.

### Process-tree notation

```text
->(
  ->('A', 'RG', 'IA'),
  ->('TI', 'TC'),
  X('CE', 'FBP'),
  X(
    tau,                                          # initial results normal: skip loop
    *(
      ->('SC', X('TM', 'SPT'), X('FCE', 'FBA')),  # one loop iteration
      tau                                          # redo (abnormal) or exit (normal)
    )
  ),
  +('AI', 'NC'),
  ->('FAS', 'DDI'),
  +('BL', 'ICP', 'MRF')
)
```

* `X('CE','FBP')` is the initial diagnostic test.
* The outer `X(tau, *(...))` models the evaluation of those results: either nothing happens (`tau`, normal results) or the abnormal-results loop is entered.
* Inside the loop, `*(`body`, tau)` means: execute the body at least once, then either exit or repeat the body again.
* The body is `->('SC', X('TM','SPT'), X('FCE','FBA'))`, i.e. specialist consultation, an exclusive treatment, and a re-diagnostic test.
* `RDT` is represented by the inner `X('FCE','FBA')` node.

### Python / pm4py construction

```python
from pm4py.objects.process_tree.obj import ProcessTree, Operator

def new_node(op, parent=None):
    n = ProcessTree(operator=op, parent=parent)
    if parent is not None:
        parent.children.append(n)
    return n

def new_leaf(label, parent):
    n = ProcessTree(label=label, parent=parent)
    parent.children.append(n)
    return n

# Root: the whole process is one big sequence
root = new_node(Operator.SEQUENCE)

# 1. Arrival -> Registration -> Intake Assessment
arrival = new_node(Operator.SEQUENCE, root)
new_leaf("A", arrival)
new_leaf("RG", arrival)
new_leaf("IA", arrival)

# 2. Triage Interview -> Triage Classification
triage = new_node(Operator.SEQUENCE, root)
new_leaf("TI", triage)
new_leaf("TC", triage)

# 3. Initial diagnostic test: CE xor FBP
initial_test = new_node(Operator.XOR, root)
new_leaf("CE", initial_test)
new_leaf("FBP", initial_test)

# 4. Result evaluation: either skip (normal) or enter the abnormal-results loop
result_eval = new_node(Operator.XOR, root)
new_leaf(None, result_eval)          # tau: normal results, skip loop

loop = new_node(Operator.LOOP, result_eval)

# Loop body: SC -> (TM xor SPT) -> (FCE xor FBA)
loop_body = new_node(Operator.SEQUENCE, loop)
new_leaf("SC", loop_body)

treatment = new_node(Operator.XOR, loop_body)
new_leaf("TM", treatment)
new_leaf("SPT", treatment)

rdt = new_node(Operator.XOR, loop_body)   # RDT = FCE xor FBA
new_leaf("FCE", rdt)
new_leaf("FBA", rdt)

new_leaf(None, loop)                 # tau redo step (repeat body or exit)

# 5. Advanced Imaging || Nutritional Counseling
advanced = new_node(Operator.PARALLEL, root)
new_leaf("AI", advanced)
new_leaf("NC", advanced)

# 6. Follow-up Appointment Scheduling -> Detailed Discharge Instructions
final_admin = new_node(Operator.SEQUENCE, root)
new_leaf("FAS", final_admin)
new_leaf("DDI", final_admin)

# 7. Final clearances: Billing || Insurance Claims || Medical Records
clearance = new_node(Operator.PARALLEL, root)
new_leaf("BL", clearance)
new_leaf("ICP", clearance)
new_leaf("MRF", clearance)

# `root` is now the complete process tree
```

Every child has its `parent` explicitly set, and every node is appended to its parent’s `children` list, so the tree is fully connected.
