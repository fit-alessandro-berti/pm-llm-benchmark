### 1) POWL model **with potential unfairness** (XOR branch introduces community-affiliation uplift)

```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition
from pm4py.objects.process_tree.obj import Operator

# Activities
ReceiveApplication = Transition(label="ReceiveApplication")

DataCompletenessCheck = Transition(label="DataCompletenessCheck")
RequestMoreInfo = Transition(label="RequestMoreInfo")
completeness_loop = OperatorPOWL(operator=Operator.LOOP, children=[DataCompletenessCheck, RequestMoreInfo])

SkillAssessment = Transition(label="SkillAssessment")

# --- Potential bias point: XOR branching after the skill assessment ---
# Branch 1: standard cultural fit evaluation
CulturalFitCheck_standard = Transition(label="CulturalFitCheck")

# Branch 2: community-affiliation-based path that can lead to an implicit uplift
CommunityAffiliationCheck = Transition(label="CommunityAffiliationCheck")
CulturalFitCheck_after_uplift = Transition(label="CulturalFitCheck")

affiliation_uplift_path = StrictPartialOrder(nodes=[CommunityAffiliationCheck, CulturalFitCheck_after_uplift])
affiliation_uplift_path.order.add_edge(CommunityAffiliationCheck, CulturalFitCheck_after_uplift)

cultural_xor_with_bias = OperatorPOWL(
    operator=Operator.XOR,
    children=[CulturalFitCheck_standard, affiliation_uplift_path]
)

ManagerialReview = Transition(label="ManagerialReview")
FinalDecision = Transition(label="FinalDecision")

# Main flow (sequence)
root_unfair = StrictPartialOrder(nodes=[
    ReceiveApplication,
    completeness_loop,
    SkillAssessment,
    cultural_xor_with_bias,   # <- unfair tilt can occur here
    ManagerialReview,
    FinalDecision
])

root_unfair.order.add_edge(ReceiveApplication, completeness_loop)
root_unfair.order.add_edge(completeness_loop, SkillAssessment)
root_unfair.order.add_edge(SkillAssessment, cultural_xor_with_bias)
root_unfair.order.add_edge(cultural_xor_with_bias, ManagerialReview)
root_unfair.order.add_edge(ManagerialReview, FinalDecision)
```

### 2) POWL model **without unfairness** (no community-based XOR branch; everyone gets the same cultural fit check)

```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition
from pm4py.objects.process_tree.obj import Operator

# Activities
ReceiveApplication = Transition(label="ReceiveApplication")

DataCompletenessCheck = Transition(label="DataCompletenessCheck")
RequestMoreInfo = Transition(label="RequestMoreInfo")
completeness_loop = OperatorPOWL(operator=Operator.LOOP, children=[DataCompletenessCheck, RequestMoreInfo])

SkillAssessment = Transition(label="SkillAssessment")

# No special community-affiliation branch: same check for all applicants
CulturalFitCheck = Transition(label="CulturalFitCheck")

ManagerialReview = Transition(label="ManagerialReview")
FinalDecision = Transition(label="FinalDecision")

# Main flow (sequence)
root_fair = StrictPartialOrder(nodes=[
    ReceiveApplication,
    completeness_loop,
    SkillAssessment,
    CulturalFitCheck,         # <- uniform evaluation for everyone
    ManagerialReview,
    FinalDecision
])

root_fair.order.add_edge(ReceiveApplication, completeness_loop)
root_fair.order.add_edge(completeness_loop, SkillAssessment)
root_fair.order.add_edge(SkillAssessment, CulturalFitCheck)
root_fair.order.add_edge(CulturalFitCheck, ManagerialReview)
root_fair.order.add_edge(ManagerialReview, FinalDecision)
```
