# High-level activity grouping
high_level_steps = [
    {"start": "Retrieve raw metal sheet", "end": "Preheat metal sheet", "name": "Material Preparation"},
    {"start": "Pick up welding tool", "end": "Weld corner B", "name": "Assembly"},
    {"start": "Measure weld integrity", "end": "Dry coating", "name": "Quality Inspection"},
    {"start": "Apply protective coating", "end": "Visual check", "name": "Finishing and Quality Assurance"}
]

# Extract high-level activities
def extract_high_level_activities(df, high_level_steps):
    activities = []
    for _, row in df.iterrows():
        for step in high_level_steps:
            if step["start"] in row["Activity"]:
                activities.append({"CaseID": row["CaseID"], "Timestamp": row["Timestamp"], "Activity": step["name"]})
        if row["Activity"] == high_level_steps[-1]["end"]:
            for step in high_level_steps:
                if step["end"] == row["Activity"]:
                    activities.append({"CaseID": row["CaseID"], "Timestamp": row["Timestamp"], "Activity": step["name"]})
    return pd.DataFrame(activities).drop_duplicates().sort_values(by=["CaseID", "Timestamp"])

result = extract_high_level_activities(df, high_level_steps)
