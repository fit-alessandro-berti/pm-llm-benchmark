Looking at this problem, I need to:

1. Identify cases that have the exact sequence 'Approve Purchase Requisition'  'Create Order'  'Confirm Order' in direct succession
2. Filter those cases where the time between 'Create Order' and 'Confirm Order' is more than 5 days
3. Exclude all events from those filtered cases

Here's the SQL query:

```sql
WITH qualifying_cases AS (
    -- Find cases that have the exact sequence with the timing condition
    SELECT DISTINCT case_id
    FROM (
        -- Find all instances of the sequence in direct succession
        SELECT 
            case_id,
            LEAD(activity, 0) OVER (PARTITION BY case_id ORDER BY timestamp) as next_1,
            LEAD(activity, 1) OVER (PARTITION BY case_id ORDER BY timestamp) as next_2,
            LEAD(timestamp, 0) OVER (PARTITION BY case_id ORDER BY timestamp) as next_1_time,
            LEAD(timestamp, 1) OVER (PARTITION BY case_id ORDER BY timestamp) as next_2_time,
            LAG(activity, 0) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_1,
            LAG(activity, 1) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_2
        FROM event_log
    ) seq_candidates
    WHERE 
        prev_2 = 'Approve Purchase Requisition'
        AND prev_1 = 'Create Order'
        AND activity = 'Confirm Order'
        AND next_1 = 'Approve Purchase Requisition'  -- ensuring no other events in between
        AND QUARTER(next_2_time - next_1_time) > 130  -- More than 5 days (quarter calculation for days)
), excluded_cases AS (
    -- Get all case_ids that need to be excluded
    SELECT case_id
    FROM qualifying_cases
)
-- Final result: all events except those from excluded cases
SELECT *
FROM event_log
WHERE case_id NOT IN (SELECT case_id FROM excluded_cases)
ORDER BY case_id, timestamp;
```
