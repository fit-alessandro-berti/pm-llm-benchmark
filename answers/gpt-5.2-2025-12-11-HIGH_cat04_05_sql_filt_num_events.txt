```sql
-- Keep only cases (case_id) with <= 6 events, then return all their events
WITH case_sizes AS (
  SELECT
    case_id,
    COUNT(*) AS event_count
  FROM event_log
  GROUP BY case_id
),
kept_cases AS (
  SELECT case_id
  FROM case_sizes
  WHERE event_count <= 6
)
SELECT e.*
FROM event_log e
JOIN kept_cases k
  ON e.case_id = k.case_id
ORDER BY e.case_id, e.timestamp;
```

Alternative (more compact) using a window function:

```sql
SELECT *
FROM (
  SELECT
    e.*,
    COUNT(*) OVER (PARTITION BY case_id) AS event_count
  FROM event_log e
) t
WHERE event_count <= 6
ORDER BY case_id, timestamp;
```
