log_skeleton = {
    # Activities:
    # RO = Receive Order
    # CA = Check Availability
    # RI = Reserve Items
    # PI = Pick Items
    # QC = Quality Check
    # PO = Pack Order
    # GSL = Generate Shipping Label
    # D  = Dispatch
    # IC = Invoice Customer
    # RP = Record Payment

    'equivalence': {
        # These pairs always occur equally often in a case
        ('RO', 'IC'),   # every received order is invoiced
        ('PO', 'GSL'),  # every packed order has a shipping label
        ('GSL', 'D'),   # every shipping label leads to a dispatch
        ('IC', 'RP'),   # every invoice leads to a recorded payment
    },

    'always_before': {
        # (A, B): if A occurs, B must have occurred earlier in the case
        ('CA', 'RO'),   # must receive order before checking availability
        ('RI', 'CA'),   # must check availability before reserving items
        ('PI', 'RI'),   # must reserve before picking
        ('QC', 'PI'),   # must pick before quality check
        ('PO', 'PI'),   # must pick before packing
        ('GSL', 'PO'),  # must pack before generating label
        ('D',  'GSL'),  # must generate label before dispatch
        ('IC', 'D'),    # must dispatch before invoicing
        ('RP', 'IC'),   # must invoice before recording payment
    },

    'always_after': {
        # (A, B): if A occurs, B must occur later in the case
        ('RO', 'CA'),   # every order received is later checked for availability
        ('CA', 'RI'),   # every availability check is followed by reservation
        ('RI', 'PI'),   # every reservation is followed by picking
        ('PI', 'PO'),   # every pick is eventually packed
        ('QC', 'PO'),   # every quality check is followed by packing
        ('PO', 'GSL'),  # every packed order gets a shipping label
        ('GSL', 'D'),   # every label leads to dispatch
        ('D',  'IC'),   # every dispatch is followed by invoicing
        ('IC', 'RP'),   # every invoice is eventually paid/recorded
    },

    'never_together': set(),  # no mutually exclusive activities defined

    'activ_freq': {
        # Allowed occurrence counts per case
        'RO':  {1},      # every case has exactly one "Receive Order"
        'CA':  {1},
        'RI':  {1},
        'PI':  {1},
        'QC':  {0, 1},   # quality check is optional
        'PO':  {1},
        'GSL': {1},
        'D':   {1},
        'IC':  {1},
        'RP':  {1},
    },

    'directly_follows': {
        # (A, B): if A occurs, B must occur immediately afterwards
        ('RO', 'CA'),   # after receiving an order, check availability
        ('CA', 'RI'),   # after checking availability, reserve items
        ('RI', 'PI'),   # after reserving, pick items
        ('QC', 'PO'),   # if QC happens, packing follows immediately
        ('PO', 'GSL'),  # after packing, generate label
        ('GSL', 'D'),   # after label, dispatch
        ('D',  'IC'),   # after dispatch, invoice
        ('IC', 'RP'),   # after invoicing, record payment
    }
}
