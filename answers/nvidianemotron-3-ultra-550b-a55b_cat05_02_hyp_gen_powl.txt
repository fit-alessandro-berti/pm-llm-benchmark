-- Query 1: Claims closed without evaluation (E) or approval (P)
WITH claim_activities AS (
    SELECT
        claim_id,
        bool_or(activity = 'E') AS has_evaluation,
        bool_or(activity = 'P') AS has_approval,
        bool_or(activity = 'C') AS has_close
    FROM claim_events
    GROUP BY claim_id
)
SELECT
    c.claim_id,
    c.customer_id,
    c.claim_amount,
    c.claim_type,
    c.submission_date
FROM claims c
JOIN claim_activities ca ON c.claim_id = ca.claim_id
WHERE ca.has_close
  AND (NOT ca.has_evaluation OR NOT ca.has_approval);

-- Query 2: Claims with multiple approval events (P) indicating loop between E and P
SELECT
    claim_id,
    count(*) AS approval_count
FROM claim_events
WHERE activity = 'P'
GROUP BY claim_id
HAVING count(*) > 1;

-- Query 3: Claims closed without notification (N)
WITH claim_activities AS (
    SELECT
        claim_id,
        bool_or(activity = 'N') AS has_notification,
        bool_or(activity = 'C') AS has_close
    FROM claim_events
    GROUP BY claim_id
)
SELECT
    c.claim_id,
    c.customer_id,
    c.claim_amount,
    c.claim_type,
    c.submission_date
FROM claims c
JOIN claim_activities ca ON c.claim_id = ca.claim_id
WHERE ca.has_close
  AND NOT ca.has_notification;

-- Query 4: Claims where close event (C) occurs before evaluation (E) or approval (P)
WITH event_timestamps AS (
    SELECT
        claim_id,
        min(timestamp) FILTER (WHERE activity = 'E') AS first_evaluation,
        min(timestamp) FILTER (WHERE activity = 'P') AS first_approval,
        min(timestamp) FILTER (WHERE activity = 'C') AS first_close
    FROM claim_events
    GROUP BY claim_id
)
SELECT
    c.claim_id,
    c.customer_id,
    c.claim_amount,
    c.claim_type,
    c.submission_date,
    et.first_evaluation,
    et.first_approval,
    et.first_close
FROM claims c
JOIN event_timestamps et ON c.claim_id = et.claim_id
WHERE et.first_close IS NOT NULL
  AND (
        (et.first_evaluation IS NOT NULL AND et.first_close < et.first_evaluation)
     OR (et.first_approval IS NOT NULL AND et.first_close < et.first_approval)
      );
