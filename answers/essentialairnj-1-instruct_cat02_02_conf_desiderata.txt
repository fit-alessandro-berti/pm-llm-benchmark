# Normative Process Behavior
process_steps = [
    "Application Received",
    "Credit Check",
    "Risk Assessment",
    "Loan Decision",
    "Inform Applicant",
    "Finalize Documentation",
    "Close Case"
]

# Desiderata timing constraints
credit_check_timelimit = 1  # Days
inform_applicant_timelimit = 2  # Days
follow_up_call_deadline = 10  # Days

# Event Log Table
event_log = [
    (1, "2024-01-10 09:00", "Application Received"),
    (1, "2024-01-10 09:30", "Credit Check"),
    (1, "2024-01-10 10:00", "Risk Assessment"),
    (1, "2024-01-10 11:00", "Loan Decision (Approved)"),
    (1, "2024-01-10 13:00", "Inform Applicant"),
    (1, "2024-01-10 13:30", "Finalize Documentation"),
    (1, "2024-01-10 14:00", "Close Case"),
    (2, "2024-01-11 08:00", "Application Received"),
    (2, "2024-01-11 08:20", "Risk Assessment"),
    (2, "2024-01-11 09:00", "Credit Check"),
    (2, "2024-01-11 09:30", "Loan Decision (Rejected)"),
    (2, "2024-01-11 10:00", "Finalize Documentation"),
    (2, "2024-01-11 10:30", "Close Case"),
    (3, "2024-01-12 09:00", "Application Received"),
    (3, "2024-01-14 10:00", "Credit Check"),
    (3, "2024-01-14 11:00", "Risk Assessment"),
    (3, "2024-01-14 12:00", "Loan Decision (Approved)"),
    (3, "2024-01-16 15:00", "Inform Applicant"),
    (3, "2024-01-17 09:00", "Finalize Documentation"),
    (3, "2024-01-17 10:00", "Close Case")
]

# Data processing functions
def analyze_event_log(event_log):
    case_violations = {}
    desiderata_violations = {}

    for case_id, timestamp, activity in event_log:
        if case_id not in case_violations:
            case_violations[case_id] = []
        if case_id not in desiderata_violations:
            desiderata_violations[case_id] = []
        
        if activity not in process_steps:
            case_violations[case_id].append(f"Unknown activity: {activity}")
        
        prev_activity = None
        for prev_case_id, prev_timestamp, prev_activity in event_log:
            if prev_case_id == case_id and prev_activity != activity:
                expected_steps = process_steps[process_steps.index(prev_activity):]
                if activity in expected_steps:
                    idx = expected_steps.index(activity)
                    if idx != 0:
                        case_violations[case_id].append(f"Out of order: {activity} should come after {expected_steps[idx-1]}")
                    break
        
        if "Loan Decision" in activity and "Finalize Documentation" in [a for c, t, a in event_log if c == case_id and t > timestamp]:
            if "Rejected" in activity:
                case_violations[case_id].append("Finalize Documentation after rejection")
            else:
                desiderata_violations[case_id].append("Finalize Documentation after loan approval")
        
        if "Inform Applicant" in activity:
            credit_check_time = min(abs((t - timestamp).days) for t, a in zip(event_log, event_log[event_log.index((case_id, timestamp, activity))+1:] if a[0] == case_id and a[2] == "Credit Check"))
            if credit_check_time > credit_check_timelimit:
                desiderata_violations[case_id].append("Credit Check more than {} days after application".format(credit_check_timelimit))
            
            loan_decision_time = min(abs((t - timestamp).days) for t, a in zip(event_log, event_log[event_log.index((case_id, timestamp, activity))+1:] if a[0] == case_id and "Loan Decision" in a[2]))
            if loan_decision_time > inform_applicant_timelimit:
                desiderata_violations[case_id].append("Inform Applicant more than {} days after loan decision".format(inform_applicant_timelimit))
