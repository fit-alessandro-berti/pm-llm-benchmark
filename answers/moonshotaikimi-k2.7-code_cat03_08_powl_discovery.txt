from pm4py.objects.process_tree.obj import Operator

QS  = Transition(label="QS")
AS  = Transition(label="AS")
IQC = Transition(label="IQC")
TST = Transition(label="TST")

DP  = Transition(label="DP")

assembly_iqc_loop = OperatorPOWL(operator=Operator.LOOP, children=[AS, iqc_repeat])

testing_loop = OperatorPOWL(operator=Operator.LOOP, children=[TST, RT])

# Labeling may be performed or skipped
labeling_choice = OperatorPOWL(operator=Operator.XOR, children=[LB, skip])

# Global partial order (sequential pipeline)
root = StrictPartialOrder(nodes=[

root.order.add_edge(MS, QS)
