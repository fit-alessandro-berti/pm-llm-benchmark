Here is a parameterized DuckDB SQL query that satisfies all four requirements.  
Replace `K` with the desired integer (e.g., `5` for top 5 variants).

```sql
-- Define how many top variants to keep
WITH params AS (
  SELECT 5 AS K  -- replace 5 with the desired K
),

-- 1) Determine the ordered sequence of activities per case
case_variants AS (
  SELECT
    case_id,
    LIST(activity ORDER BY timestamp) AS variant
  FROM event_log
  GROUP BY case_id
),

-- 2) Group cases by variant and count frequencies
variant_counts AS (
  SELECT
    variant,
    COUNT(*) AS case_count
  FROM case_variants
  GROUP BY variant
),

-- 3) Identify top K variants by frequency
top_variants AS (
  SELECT variant
  FROM variant_counts
  CROSS JOIN params
  ORDER BY case_count DESC
  LIMIT (SELECT K FROM params)
)

-- 4) Return all events from cases belonging to top K variants
SELECT e.*
FROM event_log e
INNER JOIN case_variants cv
  ON e.case_id = cv.case_id
WHERE cv.variant IN (SELECT variant FROM top_variants)
ORDER BY e.case_id, e.timestamp;
```
