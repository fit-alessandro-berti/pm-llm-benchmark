**Query 1: Missing activities per case**

WITH required(activity) AS (
  VALUES ('Register Order'),('Perform Credit Check'),('Validate Stock'),('Confirm Shipment'),('Ship Goods'),('Issue Invoice'),('Receive Payment')
)
SELECT DISTINCT c.case_id, r.activity AS missing_activity
FROM (SELECT DISTINCT case_id FROM order_event_log) c
CROSS JOIN required r
LEFT JOIN order_event_log o
  ON o.case_id = c.case_id AND o.activity = r.activity
WHERE o.activity IS NULL
ORDER BY c.case_id, r.activity;

**Query 2: Out-of-order events using normal sequence**

WITH normal(activity, seq) AS (
  VALUES ('Register Order',1),('Perform Credit Check',2),('Validate Stock',3),('Confirm Shipment',4),('Ship Goods',5),('Issue Invoice',6),('Receive Payment',7)
)
SELECT e1.case_id,
       e1.activity AS earlier_activity,
       e2.activity AS later_activity,
       e1.timestamp AS earlier_time,
       e2.timestamp AS later_time
FROM order_event_log e1
JOIN order_event_log e2
  ON e1.case_id = e2.case_id
JOIN normal n1 ON n1.activity = e1.activity
JOIN normal n2 ON n2.activity = e2.activity
WHERE e1.timestamp < e2.timestamp
  AND n1.seq > n2.seq
ORDER BY e1.case_id, e1.timestamp;

**Query 8: Credit score vs shipment timing**

WITH credit AS (
  SELECT case_id,
         timestamp AS credit_time,
         (substring(additional_info FROM 'credit_score=([0-9]+)'))::int AS credit_score
  FROM order_event_log
  WHERE activity = 'Perform Credit Check'
),
first_ship AS (
  SELECT case_id, MIN(timestamp) AS first_shipment_time
  FROM order_event_log
  WHERE activity IN ('Confirm Shipment','Ship Goods')
  GROUP BY case_id
)
SELECT c.case_id,
       c.credit_score,
       c.credit_time,
       s.first_shipment_time,
       s.first_shipment_time < c.credit_time AS shipped_before_credit
FROM credit c
JOIN first_ship s ON c.case_id = s.case_id
ORDER BY c.case_id;

**Query 9: Process duration and event counts (fast-track detection)**

SELECT case_id,
       COUNT(*) AS event_count,
       MIN(timestamp) AS process_start,
       MAX(timestamp) AS process_end,
       MAX(timestamp) - MIN(timestamp) AS duration
FROM order_event_log
GROUP BY case_id
HAVING COUNT(*) < 7
   OR MAX(timestamp) - MIN(timestamp) < INTERVAL '1 hour'
ORDER BY duration;

### 3.9 Credit score vs. first shipment/confirmation time

```sql
WITH credit AS (
  SELECT
    case_id,
    timestamp AS credit_time,
    (substring(additional_info FROM 'credit_score=([0-9]+)'))::int AS credit_score
  FROM order_event_log
  WHERE activity = 'Perform Credit Check'
),
first_ship AS (
  SELECT
    case_id,
    MIN(timestamp) AS first_shipment_time
  FROM order_event_log
  WHERE activity IN ('Confirm Shipment','Ship Goods')
  GROUP BY case_id
)
SELECT
  c.case_id,
  c.credit_score,
  c.credit_time,
  s.first_shipment_time,
  s.first_shipment_time < c.credit_time AS shipped_before_credit_check
FROM credit c
JOIN first_ship s ON c.case_id = s.case_id
ORDER BY c.case_id;
```

### 3.10 Resources involved in “ship before approval” cases

```sql
WITH ship_before_ctrl AS (
  SELECT DISTINCT s.case_id
  FROM order_event_log s
  JOIN order_event_log c ON s.case_id = c.case_id
  WHERE s.activity IN ('Confirm Shipment','Ship Goods')
    AND c.activity IN ('Perform Credit Check','Validate Stock')
    AND s.timestamp < c.timestamp
)
SELECT
  oel.resource,
  r.role,
  r.department,
  COUNT(DISTINCT oel.case_id) AS cases_with_ship_before_approval
FROM order_event_log oel
JOIN resources r ON r.resource_id = oel.resource
JOIN ship_before_ctrl a ON a.case_id = oel.case_id
WHERE oel.activity IN ('Confirm Shipment','Ship Goods')
GROUP BY oel.resource, r.role, r.department
ORDER BY cases_with_ship_before_approval DESC;
```

### 3.11 Very fast process paths (possible automation / timestamp errors)

```sql
SELECT
  case_id,
  COUNT(*) AS event_count,
  MIN(timestamp) AS first_event,
  MAX(timestamp) AS last_event,
  MAX(timestamp) - MIN(timestamp) AS duration
FROM order_event_log
GROUP BY case_id
HAVING COUNT(*) < 7
    OR MAX(timestamp) - MIN(timestamp) < INTERVAL '1 hour'
ORDER BY duration;
```

### 3.12 Check for activity/department mismatches

```sql
WITH expected(activity, department) AS (
  VALUES
    ('Register Order',      'Sales'),
    ('Perform Credit Check','Finance'),
    ('Validate Stock',      'Logistics'),
    ('Confirm Shipment',    'Logistics'),
    ('Ship Goods',          'Logistics'),
    ('Issue Invoice',       'Finance'),
    ('Receive Payment',     'Finance')
)
SELECT
  oel.case_id,
  oel.event_id,
  oel.activity,
  oel.resource,
  r.role,
  r.department AS actual_department,
  e.department AS expected_department
FROM order_event_log oel
JOIN resources r ON r.resource_id = oel.resource
JOIN expected e  ON e.activity = oel.activity
WHERE r.department <> e.department
ORDER BY oel.case_id, oel.event_id;
```

### 3.13 Inspect explicit override flags in `additional_info`

```sql
SELECT
  case_id,
  event_id,
  activity,
  resource,
  additional_info
FROM order_event_log
WHERE additional_info ILIKE '%attempted_early%'
   OR additional_info ILIKE '%late_confirmation%'
   OR additional_info ILIKE '%shipment_scheduled=N%'
ORDER BY case_id, event_id;
```
