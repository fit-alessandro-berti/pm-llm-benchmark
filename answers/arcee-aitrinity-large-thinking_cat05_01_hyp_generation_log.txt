## **Proposed SQL Queries for Investigation**

### **1. Identify All Out-of-Order Event Sequences**
```sql
WITH case_activities AS (
  SELECT 
    case_id,
    activity,
    timestamp,
    ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) as seq_order
  FROM order_event_log
),
expected_order AS (
  SELECT 
    case_id,
    activity,
    LAG(activity) OVER (PARTITION BY case_id ORDER BY timestamp) as prev_activity
  FROM case_activities
)
SELECT 
  case_id,
  activity,
  timestamp,
  prev_activity
FROM expected_order
WHERE 
  (activity = 'Perform Credit Check' AND prev_activity NOT IN ('Register Order', NULL))
  OR (activity = 'Validate Stock' AND prev_activity NOT IN ('Perform Credit Check', NULL))
  OR (activity = 'Confirm Shipment' AND prev_activity NOT IN ('Validate Stock', NULL))
  OR (activity = 'Ship Goods' AND prev_activity NOT IN ('Confirm Shipment', NULL))
  OR (activity = 'Issue Invoice' AND prev_activity NOT IN ('Ship Goods', NULL))
  OR (activity = 'Receive Payment' AND prev_activity NOT IN ('Issue Invoice', NULL));
```

### **2. Find Cases Missing Mandatory Activities**
```sql
SELECT 
  o.case_id,
  o.order_type,
  o.order_value,
  COUNT(DISTINCT el.activity) as completed_activities,
  CASE WHEN COUNT(DISTINCT CASE WHEN el.activity = 'Perform Credit Check' THEN 1 END) = 0 THEN 'Missing Credit Check' END as credit_check_missing,
  CASE WHEN COUNT(DISTINCT CASE WHEN el.activity = 'Validate Stock' THEN 1 END) = 0 THEN 'Missing Stock Validation' END as stock_validation_missing
FROM orders o
LEFT JOIN order_event_log el ON o.case_id = el.case_id
GROUP BY o.case_id, o.order_type, o.order_value
HAVING 
  COUNT(DISTINCT CASE WHEN el.activity = 'Perform Credit Check' THEN 1 END) = 0
  OR COUNT(DISTINCT CASE WHEN el.activity = 'Validate Stock' THEN 1 END) = 0;
```

### **3. Detect Illogical Timing (Payment Before Invoice)**
```sql
WITH payment_invoice_timing AS (
  SELECT 
    case_id,
    MIN(CASE WHEN activity = 'Receive Payment' THEN timestamp END) as payment_time,
    MIN(CASE WHEN activity = 'Issue Invoice' THEN timestamp END) as invoice_time
  FROM order_event_log
  GROUP BY case_id
)
SELECT 
  case_id,
  payment_time,
  invoice_time,
  payment_time < invoice_time as is_payment_before_invoice
FROM payment_invoice_timing
WHERE payment_time IS NOT NULL AND invoice_time IS NOT NULL
  AND payment_time < invoice_time;
```

### **4. Identify Resources Frequently Involved in Anomalies**
```sql
WITH anomaly_cases AS (
  SELECT DISTINCT case_id
  FROM order_event_log
  WHERE activity IN ('Perform Credit Check', 'Validate Stock', 'Confirm Shipment', 'Ship Goods', 'Issue Invoice', 'Receive Payment')
    AND case_id IN (1002, 1003, 1004) -- Known anomaly cases
)
SELECT 
  r.resource_id,
  r.role,
  r.department,
  COUNT(DISTINCT el.case_id) as anomaly_cases_involved
FROM resources r
JOIN order_event_log el ON r.resource_id = el.resource
JOIN anomaly_cases ac ON el.case_id = ac.case_id
GROUP BY r.resource_id, r.role, r.department
HAVING COUNT(DISTINCT el.case_id) > 0
ORDER BY anomaly_cases_involved DESC;
```

### **5. Analyze Order Value vs. Anomaly Frequency**
```sql
SELECT 
  o.order_type,
  o.order_value,
  COUNT(DISTINCT CASE WHEN el.activity = 'Perform Credit Check' THEN o.case_id END) as credit_check_count,
  COUNT(DISTINCT CASE WHEN el.activity = 'Validate Stock' THEN o.case_id END) as stock_check_count,
  COUNT(DISTINCT CASE WHEN el.activity = 'Confirm Shipment' THEN o.case_id END) as confirm_shipment_count
FROM orders o
LEFT JOIN order_event_log el ON o.case_id = el.case_id
GROUP BY o.order_type, o.order_value
ORDER BY o.order_value DESC;
```

### **6. Check for Backdated Event Logging (Same-Day Anomalies)**
```sql
SELECT 
  case_id,
  activity,
  timestamp,
  DATE(timestamp) as event_date,
  COUNT(*) OVER (PARTITION BY case_id, DATE(timestamp)) as events_same_day
FROM order_event_log
WHERE case_id IN (1002, 1003, 1004)
ORDER BY case_id, timestamp;
```
