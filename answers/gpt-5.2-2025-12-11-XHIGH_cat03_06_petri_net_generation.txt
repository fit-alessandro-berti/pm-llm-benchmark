```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# --- Petri net ---
net = PetriNet(name="insurance_claim_management")

# --- Places (states between activities) ---
p_start = PetriNet.Place("start")                     # a new claim arrives
p_registered = PetriNet.Place("registered")           # after Claim Registration, waiting for DV

p_docs_incomplete = PetriNet.Place("docs_incomplete") # DV found missing/incomplete documents
p_docs_ok = PetriNet.Place("docs_ok")                 # documents complete, ready for Fraud Check

p_fraud_doubt = PetriNet.Place("fraud_doubt")         # FC raised doubts, needs investigation
p_fraud_clear = PetriNet.Place("fraud_clear")         # FC cleared, ready for Claim Assessment

p_assessed = PetriNet.Place("assessed")               # after Claim Assessment
p_ready_to_pay = PetriNet.Place("ready_to_pay")       # after Approval (or approval skipped)

p_paid = PetriNet.Place("paid")                       # after Payment
p_notified = PetriNet.Place("notified")               # after Notification
p_closed = PetriNet.Place("closed")                   # final place

for p in [
    p_start, p_registered,
    p_docs_incomplete, p_docs_ok,
    p_fraud_doubt, p_fraud_clear,
    p_assessed, p_ready_to_pay,
    p_paid, p_notified, p_closed
]:
    net.places.add(p)

# --- Transitions (activities) ---
t_CR = PetriNet.Transition("t_CR", "Claim Registration")

# Split DV into two transitions (same label) to model the “complete vs incomplete docs” choice
t_DV_ok = PetriNet.Transition("t_DV_ok", "Document Verification")
t_DV_bad = PetriNet.Transition("t_DV_bad", "Document Verification")

t_RDS = PetriNet.Transition("t_RDS", "Re-Document Submission")

# Split FC into two transitions (same label) to model “clear vs doubts” choice
t_FC_clear = PetriNet.Transition("t_FC_clear", "Fraud Check")
t_FC_doubt = PetriNet.Transition("t_FC_doubt", "Fraud Check")

t_II = PetriNet.Transition("t_II", "In-depth Investigation")
t_CA = PetriNet.Transition("t_CA", "Claim Assessment")

t_AP = PetriNet.Transition("t_AP", "Approval")
t_skip_AP = PetriNet.Transition("t_skip_AP", None)  # silent: approval not needed (below threshold)

t_P = PetriNet.Transition("t_P", "Payment")
t_N = PetriNet.Transition("t_N", "Notification")
t_CL = PetriNet.Transition("t_CL", "Closure")

for t in [
    t_CR, t_DV_ok, t_DV_bad, t_RDS,
    t_FC_clear, t_FC_doubt, t_II,
    t_CA, t_AP, t_skip_AP,
    t_P, t_N, t_CL
]:
    net.transitions.add(t)

# --- Arcs (control-flow) ---
# Start -> Claim Registration
petri_utils.add_arc_from_to(p_start, t_CR, net)
petri_utils.add_arc_from_to(t_CR, p_registered, net)

# Document Verification with possible loop via Re-Document Submission
petri_utils.add_arc_from_to(p_registered, t_DV_ok, net)
petri_utils.add_arc_from_to(t_DV_ok, p_docs_ok, net)

petri_utils.add_arc_from_to(p_registered, t_DV_bad, net)
petri_utils.add_arc_from_to(t_DV_bad, p_docs_incomplete, net)

petri_utils.add_arc_from_to(p_docs_incomplete, t_RDS, net)
petri_utils.add_arc_from_to(t_RDS, p_registered, net)  # loop back to DV

# Fraud Check with possible loop via In-depth Investigation
petri_utils.add_arc_from_to(p_docs_ok, t_FC_clear, net)
petri_utils.add_arc_from_to(t_FC_clear, p_fraud_clear, net)

petri_utils.add_arc_from_to(p_docs_ok, t_FC_doubt, net)
petri_utils.add_arc_from_to(t_FC_doubt, p_fraud_doubt, net)

petri_utils.add_arc_from_to(p_fraud_doubt, t_II, net)
petri_utils.add_arc_from_to(t_II, p_docs_ok, net)      # loop back to FC (can repeat multiple times)

# Claim Assessment
petri_utils.add_arc_from_to(p_fraud_clear, t_CA, net)
petri_utils.add_arc_from_to(t_CA, p_assessed, net)

# Approval (only if needed) OR skip (silent)
petri_utils.add_arc_from_to(p_assessed, t_AP, net)
petri_utils.add_arc_from_to(t_AP, p_ready_to_pay, net)

petri_utils.add_arc_from_to(p_assessed, t_skip_AP, net)
petri_utils.add_arc_from_to(t_skip_AP, p_ready_to_pay, net)

# Payment -> Notification -> Closure
petri_utils.add_arc_from_to(p_ready_to_pay, t_P, net)
petri_utils.add_arc_from_to(t_P, p_paid, net)

petri_utils.add_arc_from_to(p_paid, t_N, net)
petri_utils.add_arc_from_to(t_N, p_notified, net)

petri_utils.add_arc_from_to(p_notified, t_CL, net)
petri_utils.add_arc_from_to(t_CL, p_closed, net)

# --- Initial and final markings (accepting Petri net) ---
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1

# The accepting Petri net is (net, im, fm)
```
