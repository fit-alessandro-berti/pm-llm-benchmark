## 1 POWL Model **with** Potential Unfairness  

```python
# --------------------------------------------------------------
# POWL model that contains a biased XOR branch (CommunityAffiliationCheck)
# --------------------------------------------------------------
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# ---------- 1. Define the atomic activities ----------
receive_app          = Transition(label="ReceiveApplication")
data_check           = Transition(label="DataCompletenessCheck")
request_info         = Transition(label="RequestMoreInfo")
skill_assessment     = Transition(label="SkillAssessment")
cultural_fit         = Transition(label="CulturalFitCheck")          # standard culturalfit
community_affil_chk  = Transition(label="CommunityAffiliationCheck") # biased branch
managerial_review    = Transition(label="ManagerialReview")
final_decision       = Transition(label="FinalDecision")

# ---------- 2. Build the datacompleteness loop ----------
#   * (DataCompletenessCheck, RequestMoreInfo)  means:
#       ñ execute DataCompletenessCheck
#       ñ either exit the loop (data OK) or execute RequestMoreInfo and then DataCompletenessCheck again
loop_data = OperatorPOWL(operator=Operator.LOOP,
                         children=[data_check, request_info])

# ---------- 3. Build the XOR that introduces the bias ----------
xor_cultural = OperatorPOWL(operator=Operator.XOR,
                            children=[cultural_fit,    # standard path
                                      community_affil_chk])   # privileged path

# ---------- 4. Assemble the whole process as a StrictPartialOrder ----------
root_unfair = StrictPartialOrder(
    nodes=[receive_app, loop_data, skill_assessment,
           xor_cultural, managerial_review, final_decision]
)

# Add the mandatory ordering constraints (edges)
root_unfair.order.add_edge(receive_app,    loop_data)          # first receive, then check data
root_unfair.order.add_edge(loop_data,     skill_assessment)   # data OK  skill test
root_unfair.order.add_edge(skill_assessment, xor_cultural)    # skill test  culturalfit XOR
root_unfair.order.add_edge(xor_cultural,  managerial_review) # either branch  manager review
root_unfair.order.add_edge(managerial_review, final_decision) # manager  final decision
```

## 2 POWL Model **without** Unfairness  

```python
# --------------------------------------------------------------
# POWL model that treats all candidates identically in the culturalfit stage
# --------------------------------------------------------------
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# ---------- 1. Define the atomic activities ----------
receive_app          = Transition(label="ReceiveApplication")
data_check           = Transition(label="DataCompletenessCheck")
request_info         = Transition(label="RequestMoreInfo")
skill_assessment     = Transition(label="SkillAssessment")
cultural_fit_fair    = Transition(label="CulturalFitCheck")   # only one activity ñ no XOR
managerial_review    = Transition(label="ManagerialReview")
final_decision       = Transition(label="FinalDecision")

# ---------- 2. Build the datacompleteness loop ----------
loop_data = OperatorPOWL(operator=Operator.LOOP,
                         children=[data_check, request_info])

# ---------- 3. No XOR ñ we simply have a single culturalfit activity ----------
# (the variable name reflects the purpose, but the POWL node is just a Transition)

# ---------- 4. Assemble the whole process as a StrictPartialOrder ----------
root_fair = StrictPartialOrder(
    nodes=[receive_app, loop_data, skill_assessment,
           cultural_fit_fair, managerial_review, final_decision]
)

# Add the mandatory ordering constraints (edges)
root_fair.order.add_edge(receive_app,       loop_data)
root_fair.order.add_edge(loop_data,        skill_assessment)
root_fair.order.add_edge(skill_assessment, cultural_fit_fair)   # straight line to culturalfit
root_fair.order.add_edge(cultural_fit_fair, managerial_review)
root_fair.order.add_edge(managerial_review, final_decision)

#   pm4py.visualization.powl.visualize(root_fair)
```

**Why this eliminates the bias:**

The loop that guarantees data completeness remains, the meritbased skill test remains, and the manager review follows the uniform culturalfit evaluation. Consequently there is no XORdriven ìshortcutî that could give any group an unfair advantage.
