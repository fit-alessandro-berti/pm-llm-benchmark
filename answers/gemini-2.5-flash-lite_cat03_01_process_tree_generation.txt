# Helper function to create a sequence of activities
def create_sequence(activities_labels, parent_node=None):
    if not activities_labels:
        return None
    if len(activities_labels) == 1:
        return ProcessTree(label=activities_labels[0], parent=parent_node)

    seq_node = ProcessTree(operator=Operator.SEQUENCE, parent=parent_node)
    first_activity = ProcessTree(label=activities_labels[0], parent=seq_node)
    seq_node.children.append(first_activity)

    current_child = first_activity
    for i in range(1, len(activities_labels)):
        next_activity_label = activities_labels[i]
        next_activity_node = ProcessTree(label=next_activity_label, parent=seq_node)
        # Link current_child to next_activity_node in sequence
        # This requires modifying the parent's children list, which is handled by append.
        # The parent of next_activity_node is seq_node.
        seq_node.children.append(next_activity_node)

    return seq_node
