```sql
WITH case_counts AS (
    SELECT
        case_id,
        COUNT(*) AS event_count
    FROM event_log
    GROUP BY case_id
    HAVING COUNT(*) <= 6
)
SELECT e.*
FROM event_log e
JOIN case_counts c
  ON e.case_id = c.case_id;
```
