```python
# Activity labels:
# IG  = Idea Generation
# DD  = Design Draft
# TFC = Technical Feasibility Check
# CE  = Cost Evaluation
# PC  = Prototype Creation
# LT  = Laboratory Testing
# UT  = User Testing
# AG  = Approval Gate
# MP  = Marketing Plan
# FL  = Final Launch

def rule():
    return {"support": 1.0, "confidence": 1.0}


activities = ["IG", "DD", "TFC", "CE", "PC", "LT", "UT", "AG", "MP", "FL"]

# Main ordering/dependency relations
core_flow = [
    ("IG", "DD"),
    ("DD", "TFC"),
    ("DD", "CE"),
    ("TFC", "PC"),
    ("CE", "PC"),
    ("PC", "LT"),
    ("PC", "UT"),
    ("LT", "AG"),
    ("UT", "AG"),
    ("AG", "MP"),
    ("MP", "FL"),
]

# Relations assumed to be immediate
strict_chain_flow = [
    ("IG", "DD"),
    ("AG", "MP"),
    ("MP", "FL"),
]

declare_model = {
    "existence": {
        a: rule() for a in activities
    },

    # No activity is declared globally absent in this end-to-end process.
    "absence": {},

    # Each product idea goes through each major step exactly once.
    "exactly_one": {
        a: rule() for a in activities
    },

    # The process starts with idea generation.
    "init": {
        "IG": rule()
    },

    "responded_existence": {
        pair: rule() for pair in core_flow
    },

    "coexistence": {
        ("IG", "DD"): rule(),
        ("DD", "TFC"): rule(),
        ("DD", "CE"): rule(),
        ("TFC", "CE"): rule(),
        ("TFC", "PC"): rule(),
        ("CE", "PC"): rule(),
        ("PC", "LT"): rule(),
        ("PC", "UT"): rule(),
        ("LT", "UT"): rule(),
        ("LT", "AG"): rule(),
        ("UT", "AG"): rule(),
        ("AG", "MP"): rule(),
        ("MP", "FL"): rule(),
    },

    "response": {
        pair: rule() for pair in core_flow
    },

    "precedence": {
        pair: rule() for pair in core_flow
    },

    "succession": {
        pair: rule() for pair in core_flow
    },

    "altresponse": {
        pair: rule() for pair in core_flow
    },

    "altprecedence": {
        pair: rule() for pair in core_flow
    },

    "altsuccession": {
        pair: rule() for pair in core_flow
    },

    "chainresponse": {
        pair: rule() for pair in strict_chain_flow
    },

    "chainprecedence": {
        pair: rule() for pair in strict_chain_flow
    },

    "chainsuccession": {
        pair: rule() for pair in strict_chain_flow
    },

    # No mutually exclusive activities are modeled in this scenario.
    "noncoexistence": {},

    # Prevent going back to earlier phases after later phases.
    "nonsuccession": {
        ("DD", "IG"): rule(),
        ("TFC", "DD"): rule(),
        ("CE", "DD"): rule(),
        ("PC", "TFC"): rule(),
        ("PC", "CE"): rule(),
        ("LT", "PC"): rule(),
        ("UT", "PC"): rule(),
        ("AG", "LT"): rule(),
        ("AG", "UT"): rule(),
        ("MP", "AG"): rule(),
        ("FL", "MP"): rule(),
    },

    # Prevent direct skipping of required intermediate phases.
    "nonchainsuccession": {
        ("DD", "PC"): rule(),   # prototype cannot immediately follow design draft
        ("TFC", "AG"): rule(),  # approval cannot immediately follow feasibility check
        ("CE", "AG"): rule(),   # approval cannot immediately follow cost evaluation
        ("PC", "AG"): rule(),   # approval cannot immediately follow prototype creation
        ("AG", "FL"): rule(),   # launch cannot immediately follow approval without marketing plan
    },
}
```
