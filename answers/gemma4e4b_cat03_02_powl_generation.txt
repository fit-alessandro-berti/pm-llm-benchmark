import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# --- 1. Define Atomic Activities (Transitions) ---

# Initial Steps
SR = Transition(label="Service Request (SR) Verification")
CS = Transition(label="Contract Setup (CS)")

# Parallel Setup Tasks
DA = Transition(label="Datacenter Allocation (DA)")
ACI = Transition(label="Access Control Initialization (ACI)")

# Loop Components
BNC = Transition(label="Base Network Configuration (BNC)")
DR = Transition(label="Debug & Reconfigure (D&R)")

# Choice Components
CD = Transition(label="Containerized Deployment (CD)")
MD = Transition(label="Monolithic Deployment (MD)")

# Final Tasks
QA = Transition(label="Quality Assurance (QA)")
SA = Transition(label="Security Audit (SA)")
GLA = Transition(label="Go-Live Approval (GLA)")


# ==================================================================
# 2. Construct the Sub-Processes and Flow Logic
# ==================================================================

# --- Phase 1: Initial Setup Sequence ---
# SR -> CS -> {DA || ACI} (Concurrent)
# DA and ACI run concurrently after CS.
# We will handle the mandatory sequence later in the Root.

# --- Phase 2: Looped Configuration (BNC -> loop of (D&R -> BNC)) ---
# The loop structure: BNC runs first, then the loop repeats (D&R followed by BNC).
# The general pattern for an iterative process is to initialize the loop
# or put the entry step outside and let the loop handle the repetition.
# Here, BNC starts it, and the loop handles the subsequent re-configurations.

# The loop structure: A=D&R, B=BNC. We want to execute D&R, and then BNC (which sets up the next iteration).
# Since the description implies the loop continues until stable, we structure the loop itself:
# Loop: Executes D&R, then BNC (which marks the start of the next potential loop iteration).
loop_children = [DR, BNC]
looped_config = OperatorPOWL(operator=Operator.LOOP, children=loop_children)


# --- Phase 3: Exclusive Choice (CD XOR MD) ---
# Only one deployment method is chosen.
deployment_children = [CD, MD]
deployment_choice = OperatorPOWL(operator=Operator.XOR, children=deployment_children)


# --- Phase 4: Final Partial Order (QA and SA run concurrently) ---
# This sub-process requires QA and SA to be independent but must precede GLA.
# We use StrictPartialOrder to enforce concurrency and ordering relative to the next step.
final_tasks_root = StrictPartialOrder(nodes=[QA, SA])

# Connect the end of the partial order to the final approval step
# The GLA waits until both QA and SA are done.
# We model this by making GLA the next sequential node, waiting for all inputs.
# Since StrictPartialOrder itself is a node, we use it as the parent node for GLA's dependency.

# --- 3. Construct the Overall Root Model ---
# The overall process is strictly sequential through these main stages:
# SR -> CS -> {DA, ACI} (Concurrent) -> BNC (Entry) -> Loop -> X (Choice) -> {QA, SA} (Concurrent) -> GLA

# To manage the complexity and dependencies:
# 1. Combine {DA, ACI} into a partial order node (PO_DA_ACI)
po_da_aci = StrictPartialOrder(nodes=[DA, ACI])

# 2. Create the main linear flow connecting all parts:
# Start with SR. SR must precede CS.
# CS must precede PO_DA_ACI.
# PO_DA_ACI must precede the Loop.
# The Loop must precede the Choice.
# The Choice must precede the final partial order.
# The final partial order must precede GLA.

root_nodes = [
    SR,
    CS,
    po_da_aci,
    looped_config,
    deployment_choice,
    final_tasks_root,
    GLA
]

# Build the root model using sequential dependency additions
root = StrictPartialOrder(nodes=root_nodes)

# Define the mandatory sequence:
# SR -> CS
root.order.add_edge(SR, CS)

# CS -> PO_DA_ACI
root.order.add_edge(CS, po_da_aci)

# PO_DA_ACI -> Loop
root.order.add_edge(po_da_aci, looped_config)

# Loop -> Choice
root.order.add_edge(looped_config, deployment_choice)

# Choice -> Final Partial Order
root.order.add_edge(deployment_choice, final_tasks_root)

# Final Partial Order -> GLA
# GLA can only start after both QA and SA are completed (enforced by the dependency on final_tasks_root)
root.order.add_edge(final_tasks_root, GLA)


# Print the resulting model representation (optional, for verification)
print("--- POWL Model Successfully Constructed ---")
print("Model Type:", type(root).__name__)
print(f"Total Nodes: {len(root.nodes)}")
print("\nExample Dependencies (Edges):")
