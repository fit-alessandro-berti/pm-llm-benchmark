### 3. Verification SQL Queries

#### 3.1. Baseline: Compute Actual Time Differences Between Activities per Claim
```sql
WITH ordered_events AS (
    SELECT
        ce.claim_id,
        ce.activity,
        ce.timestamp,
        ce.resource,
        ROW_NUMBER() OVER (PARTITION BY ce.claim_id ORDER BY ce.timestamp) AS seq
    FROM claim_events ce
),
pairwise AS (
    SELECT
        o1.claim_id,
        o1.activity AS from_activity,
        o2.activity AS to_activity,
        EXTRACT(EPOCH FROM (o2.timestamp - o1.timestamp)) AS diff_seconds,
        o1.resource AS from_resource,
        o2.resource AS to_resource
    FROM ordered_events o1
    JOIN ordered_events o2
        ON o1.claim_id = o2.claim_id
       AND o1.seq < o2.seq
    WHERE (o1.activity, o2.activity) IN (
        ('R','A'), ('R','E'), ('R','P'),
        ('A','C'), ('E','N'), ('E','C'),
        ('P','N'), ('N','C')
    )
)
SELECT * FROM pairwise;
```

#### 3.2. Identify Claims Where RP Time Deviates from Expected (using 3sigma rule)
```sql
WITH rp_times AS (
    SELECT
        claim_id,
        diff_seconds
    FROM pairwise
    WHERE from_activity = 'R' AND to_activity = 'P'
)
SELECT
    claim_id,
    diff_seconds,
    (diff_seconds - 90000) / 3600 AS z_score
FROM rp_times
WHERE ABS(diff_seconds - 90000) > 3 * 3600;  -- outside ±3 STDEV
```

#### 3.3. Find Claims with Excessive PN Delay (e.g., > 10 days) and Correlate with Adjuster/Region
```sql
WITH pn_times AS (
    SELECT
        p.claim_id,
        EXTRACT(EPOCH FROM (n.timestamp - p.timestamp)) AS diff_seconds,
        p.resource AS approver,
        n.resource AS notifier
    FROM claim_events p
    JOIN claim_events n
        ON p.claim_id = n.claim_id
       AND p.activity = 'P'
       AND n.activity = 'N'
       AND n.timestamp > p.timestamp
    WHERE NOT EXISTS (
        SELECT 1 FROM claim_events mid
        WHERE mid.claim_id = p.claim_id
          AND mid.activity = 'P'
          AND mid.timestamp > p.timestamp
          AND mid.timestamp < n.timestamp
    )
)
SELECT
    pn.claim_id,
    pn.diff_seconds / 86400 AS delay_days,
    pn.approver,
    pn.notifier,
    a.region,
    c.claim_type
FROM pn_times pn
JOIN claims c ON c.claim_id = pn.claim_id
LEFT JOIN adjusters a ON a.adjuster_id = pn.notifier::int
WHERE pn.diff_seconds > 10 * 86400  -- more than 10 days
ORDER BY pn.diff_seconds DESC;
```

#### 3.4. Detect Claims Closed Shortly After Assignment (AC < 1 hour) and Check for Missing E/P Steps
```sql
WITH ac_times AS (
    SELECT
        a.claim_id,
        EXTRACT(EPOCH FROM (c.timestamp - a.timestamp)) AS diff_seconds,
        a.resource AS assigner,
        c.resource AS closer
    FROM claim_events a
    JOIN claim_events c
        ON a.claim_id = c.claim_id
       AND a.activity = 'A'
       AND c.activity = 'C'
       AND c.timestamp > a.timestamp
    WHERE NOT EXISTS (
        SELECT 1 FROM claim_events mid
        WHERE mid.claim_id = a.claim_id
          AND mid.activity IN ('E','P')
          AND mid.timestamp > a.timestamp
          AND mid.timestamp < c.timestamp
    )
)
SELECT
    ac.claim_id,
    ac.diff_seconds / 60 AS delay_minutes,
    ac.assigner,
    ac.closer,
    c.claim_type,
    c.claim_amount,
    cu.region AS customer_region
FROM ac_times ac
JOIN claims c ON c.claim_id = ac.claim_id
JOIN customers cu ON cu.customer_id = c.customer_id  -- assuming a customers table exists; adjust if not
WHERE ac.diff_seconds < 3600  -- less than 1 hour
ORDER BY ac.diff_seconds;
```

#### 3.5. Examine EN Transitions Under 2 Minutes and Verify if Approval (P) Is Present
```sql
WITH en_times AS (
    SELECT
        e.claim_id,
        EXTRACT(EPOCH FROM (n.timestamp - e.timestamp)) AS diff_seconds,
        e.resource AS evaluator,
        n.resource AS notifier
    FROM claim_events e
    JOIN claim_events n
        ON e.claim_id = n.claim_id
       AND e.activity = 'E'
       AND n.activity = 'N'
       AND n.timestamp > e.timestamp
    WHERE NOT EXISTS (
        SELECT 1 FROM claim_events mid
        WHERE mid.claim_id = e.claim_id
          AND mid.activity = 'E'
          AND mid.timestamp > e.timestamp
          AND mid.timestamp < n.timestamp
    )
)
SELECT
    en.claim_id,
    en.diff_seconds,
    en.evaluator,
    en.notifier,
    CASE WHEN EXISTS (
        SELECT 1 FROM claim_events p
        WHERE p.claim_id = en.claim_id
          AND p.activity = 'P'
          AND p.timestamp > e.timestamp
          AND p.timestamp < n.timestamp
    ) THEN 'Approval present' ELSE 'Approval skipped' END AS approval_status
FROM en_times en
JOIN claim_events e ON e.claim_id = en.claim_id AND e.activity = 'E' AND e.timestamp = (
    SELECT MAX(timestamp) FROM claim_events WHERE claim_id = en.claim_id AND activity = 'E' AND timestamp < en.diff_seconds + EXTRACT(EPOCH FROM e.timestamp)
)
WHERE en.diff_seconds < 120  -- less than 2 minutes
ORDER BY en.diff_seconds;
```

#### 3.6. Analyze RE Variability by Adjuster Specialization and Claim Type
```sql
WITH re_times AS (
    SELECT
        r.claim_id,
        EXTRACT(EPOCH FROM (e.timestamp - r.timestamp)) AS diff_seconds,
        r.resource AS receiver,
        e.resource AS evaluator
    FROM claim_events r
    JOIN claim_events e
        ON r.claim_id = e.claim_id
       AND r.activity = 'R'
       AND e.activity = 'E'
       AND e.timestamp > r.timestamp
    WHERE NOT EXISTS (
        SELECT 1 FROM claim_events mid
        WHERE mid.claim_id = r.claim_id
          AND mid.activity = 'R'
          AND mid.timestamp > r.timestamp
          AND mid.timestamp < e.timestamp
    )
)
SELECT
    re.claim_id,
    re.diff_seconds / 3600 AS delay_hours,
    c.claim_type,
    a.specialization,
    a.region
FROM re_times re
JOIN claims c ON c.claim_id = re.claim_id
LEFT JOIN adjusters a ON a.adjuster_id = re.evaluator::int
ORDER BY re.diff_seconds DESC;
```
