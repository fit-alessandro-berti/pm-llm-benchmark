```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

net = PetriNet(name="insurance_claim_process")

# ---------------- Places ----------------
p_start    = PetriNet.Place(name="p_start")      # new claim arrives
p_after_CR = PetriNet.Place(name="p_after_CR")   # after Claim Registration
p_after_DV = PetriNet.Place(name="p_after_DV")   # after Document Verification
p_after_FC = PetriNet.Place(name="p_after_FC")   # after Fraud Check
p_after_CA = PetriNet.Place(name="p_after_CA")   # after Claim Assessment
p_after_AP = PetriNet.Place(name="p_after_AP")   # after Approval
p_after_P  = PetriNet.Place(name="p_after_P")    # after Payment
p_after_N  = PetriNet.Place(name="p_after_N")    # after Notification
p_end      = PetriNet.Place(name="p_end")        # claim closed

for p in [p_start, p_after_CR, p_after_DV, p_after_FC,
          p_after_CA, p_after_AP, p_after_P, p_after_N, p_end]:
    net.places.add(p)

# ---------------- Transitions ----------------
t_CR  = PetriNet.Transition(name="CR",  label="Claim Registration")
t_DV  = PetriNet.Transition(name="DV",  label="Document Verification")
t_RDS = PetriNet.Transition(name="RDS", label="Re-Document Submission")
t_FC  = PetriNet.Transition(name="FC",  label="Fraud Check")
t_II  = PetriNet.Transition(name="II",  label="In-depth Investigation")
t_CA  = PetriNet.Transition(name="CA",  label="Claim Assessment")
t_AP  = PetriNet.Transition(name="AP",  label="Approval")
t_P   = PetriNet.Transition(name="P",   label="Payment")
t_N   = PetriNet.Transition(name="N",   label="Notification")
t_CL  = PetriNet.Transition(name="CL",  label="Closure")

for t in [t_CR, t_DV, t_RDS, t_FC, t_II, t_CA, t_AP, t_P, t_N, t_CL]:
    net.transitions.add(t)

# ---------------- Arcs ----------------
# Start -> CR -> after_CR
petri_utils.add_arc_from_to(p_start, t_CR, net)
petri_utils.add_arc_from_to(t_CR, p_after_CR, net)

# after_CR -> DV -> after_DV
petri_utils.add_arc_from_to(p_after_CR, t_DV, net)
petri_utils.add_arc_from_to(t_DV, p_after_DV, net)

# Loop: after_DV -> RDS -> back to after_CR (re-submit docs, then verify again)
petri_utils.add_arc_from_to(p_after_DV, t_RDS, net)
petri_utils.add_arc_from_to(t_RDS, p_after_CR, net)

# after_DV -> FC -> after_FC
petri_utils.add_arc_from_to(p_after_DV, t_FC, net)
petri_utils.add_arc_from_to(t_FC, p_after_FC, net)

# Loop: after_FC -> II -> back to a state that allows FC again (multiple times)
# Model: II returns token to p_after_DV so FC can be executed again
petri_utils.add_arc_from_to(p_after_FC, t_II, net)
petri_utils.add_arc_from_to(t_II, p_after_DV, net)

# after_FC -> CA -> after_CA
petri_utils.add_arc_from_to(p_after_FC, t_CA, net)
petri_utils.add_arc_from_to(t_CA, p_after_CA, net)

# after_CA -> AP -> after_AP
petri_utils.add_arc_from_to(p_after_CA, t_AP, net)
petri_utils.add_arc_from_to(t_AP, p_after_AP, net)

# after_AP -> P -> after_P
petri_utils.add_arc_from_to(p_after_AP, t_P, net)
petri_utils.add_arc_from_to(t_P, p_after_P, net)

# after_P -> N -> after_N
petri_utils.add_arc_from_to(p_after_P, t_N, net)
petri_utils.add_arc_from_to(t_N, p_after_N, net)

# after_N -> CL -> end
petri_utils.add_arc_from_to(p_after_N, t_CL, net)
petri_utils.add_arc_from_to(t_CL, p_end, net)

# ---------------- Markings ----------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_end] = 1
```

**Explanation of the model:**

- **Sequential backbone:** `CR  DV  FC  CA  AP  P  N  CL`, connected through intermediate places (`p_after_*`).
- **DV / RDS loop:** From `p_after_DV`, instead of proceeding to `FC`, the transition `RDS` can fire, returning the token to `p_after_CR`, so `DV` is executed again. This models incomplete documents requiring re-submission and re-verification, and can occur any number of times.
- **FC / II loop:** From `p_after_FC`, the `II` transition returns the token to `p_after_DV`, allowing `FC` to be re-executed after the in-depth investigation. This can repeat multiple times.
- **Initial marking `im`:** one token in `p_start` (a new claim arrives).
- **Final marking `fm`:** one token in `p_end` (the claim has been closed and archived).
