```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# -------------------------------------------------------------------
# Places: states between activities
# -------------------------------------------------------------------
p_start = PetriNet.Place("p_start_new_claim")
p_registered = PetriNet.Place("p_claim_registered")
p_after_dv = PetriNet.Place("p_after_document_verification")
p_ready_for_fc = PetriNet.Place("p_ready_for_fraud_check")
p_after_fc = PetriNet.Place("p_after_fraud_check")
p_ready_for_ca = PetriNet.Place("p_ready_for_claim_assessment")
p_after_ca = PetriNet.Place("p_after_claim_assessment")
p_approved = PetriNet.Place("p_claim_approved")
p_paid = PetriNet.Place("p_claim_paid")
p_notified = PetriNet.Place("p_customer_notified")
p_closed = PetriNet.Place("p_claim_closed")

places = [
    p_start,
    p_registered,
    p_after_dv,
    p_ready_for_fc,
    p_after_fc,
    p_ready_for_ca,
    p_after_ca,
    p_approved,
    p_paid,
    p_notified,
    p_closed,
]

for p in places:
    net.places.add(p)

# -------------------------------------------------------------------
# Transitions: visible activities
# -------------------------------------------------------------------
t_cr = PetriNet.Transition("t_cr", "Claim Registration")
t_dv = PetriNet.Transition("t_dv", "Document Verification")
t_rds = PetriNet.Transition("t_rds", "Re-Document Submission")
t_fc = PetriNet.Transition("t_fc", "Fraud Check")
t_ii = PetriNet.Transition("t_ii", "In-depth Investigation")
t_ca = PetriNet.Transition("t_ca", "Claim Assessment")
t_ap = PetriNet.Transition("t_ap", "Approval")
t_p = PetriNet.Transition("t_p", "Payment")
t_n = PetriNet.Transition("t_n", "Notification")
t_cl = PetriNet.Transition("t_cl", "Closure")

# Invisible transitions for routing/choices
t_docs_ok = PetriNet.Transition("t_docs_ok", None)          # documents complete
t_no_fraud_doubt = PetriNet.Transition("t_no_fraud_doubt", None)
t_skip_approval = PetriNet.Transition("t_skip_approval", None)

transitions = [
    t_cr,
    t_dv,
    t_rds,
    t_fc,
    t_ii,
    t_ca,
    t_ap,
    t_p,
    t_n,
    t_cl,
    t_docs_ok,
    t_no_fraud_doubt,
    t_skip_approval,
]

for t in transitions:
    net.transitions.add(t)

# -------------------------------------------------------------------
# Arcs: main sequence and loops
# -------------------------------------------------------------------

# New claim -> Claim Registration
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_registered, net)

# Claim Registration -> Document Verification
petri_utils.add_arc_from_to(p_registered, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

# If documents are incomplete: Re-Document Submission, then verify again
petri_utils.add_arc_from_to(p_after_dv, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_registered, net)

# If documents are complete: continue to Fraud Check
petri_utils.add_arc_from_to(p_after_dv, t_docs_ok, net)
petri_utils.add_arc_from_to(t_docs_ok, p_ready_for_fc, net)

# Fraud Check
petri_utils.add_arc_from_to(p_ready_for_fc, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

# If fraud doubts are raised: perform In-depth Investigation and then check again
petri_utils.add_arc_from_to(p_after_fc, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_ready_for_fc, net)

# If no fraud doubts remain: proceed to Claim Assessment
petri_utils.add_arc_from_to(p_after_fc, t_no_fraud_doubt, net)
petri_utils.add_arc_from_to(t_no_fraud_doubt, p_ready_for_ca, net)

# Claim Assessment
petri_utils.add_arc_from_to(p_ready_for_ca, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

# Optional Approval:
# - If amount exceeds threshold, Approval is required
# - Otherwise, skip approval through an invisible transition
petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_approved, net)

petri_utils.add_arc_from_to(p_after_ca, t_skip_approval, net)
petri_utils.add_arc_from_to(t_skip_approval, p_approved, net)

# Payment -> Notification -> Closure
petri_utils.add_arc_from_to(p_approved, t_p, net)
petri_utils.add_arc_from_to(t_p, p_paid, net)

petri_utils.add_arc_from_to(p_paid, t_n, net)
petri_utils.add_arc_from_to(t_n, p_notified, net)

petri_utils.add_arc_from_to(p_notified, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# -------------------------------------------------------------------
# Initial and final markings
# -------------------------------------------------------------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1

# The accepting Petri net is represented by:
# net, im, fm
```
