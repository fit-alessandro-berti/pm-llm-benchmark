```sql
-- Return all events except those from cases that contain the direct sequence
-- Approve Purchase Requisition -> Create Order -> Confirm Order
-- where (Confirm Order - Create Order) > 5 days for that sequence.

WITH ordered AS (
  SELECT
    case_id,
    activity,
    "timestamp",
    row_number() OVER (
      PARTITION BY case_id
      ORDER BY "timestamp"
    ) AS rn
  FROM event_log
),
bad_cases AS (
  SELECT DISTINCT o1.case_id
  FROM ordered o1
  JOIN ordered o2
    ON o2.case_id = o1.case_id AND o2.rn = o1.rn + 1
  JOIN ordered o3
    ON o3.case_id = o1.case_id AND o3.rn = o1.rn + 2
  WHERE o1.activity = 'Approve Purchase Requisition'
    AND o2.activity = 'Create Order'
    AND o3.activity = 'Confirm Order'
    AND (o3."timestamp" - o2."timestamp") > INTERVAL '5 days'
)
SELECT e.*
FROM event_log e
WHERE e.case_id NOT IN (SELECT case_id FROM bad_cases)
ORDER BY e.case_id, e."timestamp";
```
