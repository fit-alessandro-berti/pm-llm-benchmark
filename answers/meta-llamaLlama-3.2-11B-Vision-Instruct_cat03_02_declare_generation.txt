from pm4py.objects.dfg.importer import importer as dfg_importer
from pm4py.objects.dfg.exporter import exporter as dfg_exporter
from pm4py.objects.log.importer.xes import importer as xes_importer
from pm4py.objects.log.exporter.xes import exporter as xes_exporter
import pm4py
import os

# Assuming that we have a log file with xes extension
log_file = "purchase_to_pay.xes"
log = xes_importer.import_log(log_file)

# Create the DECLARE model
declare_model = {}

# Existence, absence, exactly one
declare_model['existence'] = {
    'Place Order': {'support': 1.0, 'confidence': 1.0},
    'Invoice': {'support': 1.0, 'confidence': 1.0},
    'Payment': {'support': 1.0, 'confidence': 1.0},
}

declare_model['absence'] = {
    'Cancel Order': {'support': 1.0, 'confidence': 1.0},
    'Return Goods': {'support': 1.0, 'confidence': 1.0},
}

    'Place Order': {'support': 1.0, 'confidence': 1.0},
}

# Initialization
declare_model['init'] = {
    'Place Order': {'support': 1.0, 'confidence': 1.0},
}

# Responded existence
declare_model['responded_existence'] = {
    ('Place Order', 'Invoice'): {'support': 1.0, 'confidence': 1.0},
    ('Invoice', 'Payment'): {'support': 1.0, 'confidence': 1.0},
}

# Coexistence

    ('Place Order', 'Invoice'): {'support': 1.0, 'confidence': 1.0},

}
