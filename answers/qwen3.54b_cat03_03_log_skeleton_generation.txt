log_skeleton = {
    'equivalence': {
        ('RO', 'D')  # The order must have exactly one entry (Receive Order) for every one exit (Dispatch)
    },
    
    'always_before': {
        ('RI', 'CA'),   # If Reserve Items occurs, Check Availability must have happened previously
        ('PI', 'RI'),   # If Pick Items occurs, Reserve Items must have happened previously
        ('QC', 'PI'),   # If Quality Check occurs, Pick Items must have happened previously
        ('PO', 'QC'),   # If Pack Order occurs, Quality Check must have happened previously
        ('GSL', 'PO'),  # If Generate Shipping Label occurs, Pack Order must have happened previously
        ('D', 'GSL'),   # If Dispatch occurs, Generate Shipping Label must have happened previously
        ('IC', 'D'),    # If Invoice Customer occurs, Dispatch must have happened previously
        ('RP', 'IC')    # If Record Payment occurs, Invoice Customer must have happened previously
    },
    
    'always_after': {
        ('CA', 'RI'),   # If Check Availability occurs, Reserve Items follows
        ('RI', 'PI'),   # If Reserve Items occurs, Pick Items follows
        ('PI', 'QC'),   # If Pick Items occurs, Quality Check follows
        ('QC', 'PO'),   # If Quality Check occurs, Pack Order follows
        ('PO', 'GSL'),  # If Pack Order occurs, Generate Shipping Label follows
        ('GSL', 'D'),   # If Generate Shipping Label occurs, Dispatch follows
        ('D', 'IC'),    # If Dispatch occurs, Invoice Customer follows (or parallel)
        ('IC', 'RP')    # If Invoice Customer occurs, Record Payment follows
    },
    
    'never_together': {
        # In a standard fulfillment flow, all activities co-exist within the same order case.
        # We assume no strict mutual exclusion defined by this specific scenario description.
        {} 
    },
    
    'activ_freq': {
        'RO': {1},          # One order always starts with one receive
        'CA': {1},          # Availability check is required
        'RI': {1},          # Reservation is required upon availability confirmation
        'PI': {1},          # Picking is required for fulfillment
        'QC': {0, 1},       # Quality check might be skipped in some scenarios (e.g. trusted suppliers)
        'PO': {1},          # Packing is required
        'GSL': {1},         # Label generation is required for shipping
        'D': {1},           # Dispatch is the end goal
        'IC': {0, 1},       # Invoice might be generated conditionally (e.g. credit vs cash)
        'RP': {0, 1}        # Payment record depends on successful payment capture
    },
    
    'directly_follows': {
        ('RO', 'CA'),      # Immediately after receiving, check availability
        ('CA', 'RI'),      # Immediately after checking, reserve items
        ('RI', 'PI'),      # Immediately after reserving, pick items
        ('PI', 'QC'),      # Immediately after picking, quality check
        ('QC', 'PO'),      # Immediately after QC, pack order
        ('PO', 'GSL'),     # Immediately after packing, generate label
        ('GSL', 'D'),      # Immediately after labeling, dispatch
        ('D', 'IC'),       # Immediately after dispatch (or parallel to), invoice customer
        ('IC', 'RP')       # Immediately after invoicing, record payment
    }
}
