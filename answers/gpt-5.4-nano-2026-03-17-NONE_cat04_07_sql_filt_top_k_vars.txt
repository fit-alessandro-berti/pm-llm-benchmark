```sql
-- Parameters:
--   :K = top K variants to keep
WITH ordered_events AS (
    SELECT
        case_id,
        activity,
        "timestamp",
        -- position within the case ordered by timestamp
        row_number() OVER (
            PARTITION BY case_id
            ORDER BY "timestamp", activity
        ) AS rn
    FROM event_log
),

-- Build the full activity sequence per case (variant key)
case_variants AS (
    SELECT
        case_id,
        string_agg(activity, ' -> ' ORDER BY rn) AS variant_seq
    FROM ordered_events
    GROUP BY case_id
),

-- Count cases per variant
variant_counts AS (
    SELECT
        variant_seq,
        count(*) AS case_count
    FROM case_variants
    GROUP BY variant_seq
),

-- Select top K variants by frequency
top_k_variants AS (
    SELECT
        variant_seq,
        case_count
    FROM variant_counts
    ORDER BY case_count DESC, variant_seq
    LIMIT :K
),

-- Keep only cases whose variant is in the top K
filtered_cases AS (
    SELECT cv.case_id
    FROM case_variants cv
    JOIN top_k_variants tk
      ON cv.variant_seq = tk.variant_seq
)

-- Return all events belonging to those filtered cases
SELECT
    el.*
FROM event_log el
JOIN filtered_cases fc
  ON el.case_id = fc.case_id
ORDER BY el.case_id, el."timestamp", el.activity;
```

**Notes**
- The variant key is the ordered activity sequence per `case_id`, created via `string_agg(... ORDER BY rn)`.
- If multiple events share the same `timestamp` within a case, the tie-breaker `activity` is used to make the ordering deterministic.
