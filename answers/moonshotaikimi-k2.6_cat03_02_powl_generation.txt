import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# ------------------------------------------------------------------------------
# 1. Define atomic activities
# ------------------------------------------------------------------------------
SR  = Transition(label="SR")      # Service Request verification
CS  = Transition(label="CS")      # Contract Setup
DA  = Transition(label="DA")      # Datacenter Allocation
ACI = Transition(label="ACI")     # Access Control Initialization
BNC = Transition(label="BNC")     # Base Network Configuration
DR  = Transition(label="D&R")     # Debug & Reconfigure
CD  = Transition(label="CD")      # Containerized Deployment
MD  = Transition(label="MD")      # Monolithic Deployment
QA  = Transition(label="QA")      # Quality Assurance
SA  = Transition(label="SA")      # Security Audit
GLA = Transition(label="GLA")     # Go-Live Approval

# ------------------------------------------------------------------------------
# 2. Loop node: * (BNC, D&R)
#    Execute BNC, then either exit or execute D&R and loop back to BNC again.
# ------------------------------------------------------------------------------
loop_config = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DR])

# ------------------------------------------------------------------------------
# 3. Exclusive choice node: X (CD, MD)
#    Choose exactly one of the two deployment methods.
# ------------------------------------------------------------------------------
xor_deploy = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# ------------------------------------------------------------------------------
# 4. Partial order for concurrent final tasks
#    QA and SA have no ordering constraints between them, so they can run
#    concurrently. They are grouped in their own StrictPartialOrder.
# ------------------------------------------------------------------------------
final_tasks = StrictPartialOrder(nodes=[QA, SA])
# (No edges added between QA and SA, meaning they are concurrent.)

# ------------------------------------------------------------------------------
# 5. Root partial order
#    Assembles all stages of the provisioning process.
# ------------------------------------------------------------------------------
root = StrictPartialOrder(nodes=[
    SR, CS, DA, ACI, loop_config, xor_deploy, final_tasks, GLA
])

# ------------------------------------------------------------------------------
# 6. Define execution dependencies
# ------------------------------------------------------------------------------
root.order.add_edge(SR, CS)               # SR must finish before CS
root.order.add_edge(CS, DA)               # CS must finish before DA
root.order.add_edge(CS, ACI)              # CS must finish before ACI
root.order.add_edge(DA, loop_config)      # DA must finish before loop
root.order.add_edge(ACI, loop_config)     # ACI must finish before loop
root.order.add_edge(loop_config, xor_deploy)  # Loop must finish before deployment choice
root.order.add_edge(xor_deploy, final_tasks)  # Deployment must finish before QA/SA
root.order.add_edge(final_tasks, GLA)     # QA & SA must finish before GLA
