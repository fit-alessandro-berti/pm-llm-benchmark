```sql
-- Benchmark Query for Process Variant Analysis in DuckDB

-- This query identifies the most frequent process variants in an event log
-- and returns all original events associated with those top variants.

-- Define the number of top variants to consider (K).
-- Replace '3' with your desired integer value for K.
-- For example, to get the top 5 variants, change LIMIT 3 to LIMIT 5.
-- The chosen value for K is embedded directly in the query below.

WITH CaseVariants AS (
    -- Step 1: For each case_id, construct its unique ordered sequence of activities.
    -- STRING_AGG concatenates activities, using '->' as a delimiter.
    -- ORDER BY timestamp ensures the activities are in chronological order within each case.
    SELECT
        case_id,
        STRING_AGG(activity, '->' ORDER BY timestamp) AS variant_sequence
    FROM
        event_log
    GROUP BY
        case_id
),
VariantFrequencies AS (
    -- Step 2 & 3 (part 1): Group the identified variants and count the number of distinct cases
    -- belonging to each unique variant sequence.
    SELECT
        variant_sequence,
        COUNT(DISTINCT case_id) AS frequency
    FROM
        CaseVariants
    GROUP BY
        variant_sequence
),
TopKVariants AS (
    -- Step 3 (part 2): Select the top K most frequent variant sequences.
    -- We order the variants by their frequency in descending order and limit the result to K.
    SELECT
        variant_sequence
    FROM
        VariantFrequencies
    ORDER BY
        frequency DESC
    LIMIT 3 -- <<< REPLACE '3' with your desired value for K >>>
)
-- Step 4: Retrieve all original events from the event_log table.
-- We filter these events to include only those belonging to cases
-- whose variant sequence is present in our 'TopKVariants' list.
-- Cases not matching any of the top K variants are excluded from the final result.
SELECT
    el.* -- Select all columns from the original event_log table.
FROM
    event_log el
JOIN
    CaseVariants cv ON el.case_id = cv.case_id -- Join to link original events to their derived case variant.
JOIN
    TopKVariants tkv ON cv.variant_sequence = tkv.variant_sequence; -- Filter to keep only events from cases whose variant is in the top K.
```
