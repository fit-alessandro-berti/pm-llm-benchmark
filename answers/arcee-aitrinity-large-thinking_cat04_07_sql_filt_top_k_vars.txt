  SELECT variant_seq, COUNT(*) AS case_count
  FROM case_variants
  GROUP BY variant_seq
),
ranked_variants AS (
  SELECT variant_seq,
         ROW_NUMBER() OVER (ORDER BY case_count DESC, variant_seq) AS rn
  FROM variant_counts
),
top_k_variants AS (
  SELECT variant_seq
  FROM ranked_variants
  WHERE rn <= $1
)
SELECT el.*
FROM event_log el
JOIN case_variants cv ON el.case_id = cv.case_id
JOIN top_k_variants tk ON cv.variant_seq = tk.variant_seq
ORDER BY el.case_id, el.timestamp;
