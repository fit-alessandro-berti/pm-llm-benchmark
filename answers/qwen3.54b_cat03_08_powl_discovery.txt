```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# Define Activity Transitions
OR = Transition(label="OR")
MS = Transition(label="MS")
QS = Transition(label="QS")
AS1 = Transition(label="AS")
IQC = Transition(label="IQC")
RT = Transition(label="RT")
TST1 = Transition(label="TST") # Initial Test
TST2 = Transition(label="TST") # Second Test (inside loop)
PK = Transition(label="PK")
LB = Transition(label="LB")
DP = Transition(label="DP")
D = Transition(label="D")

# Define Loop Operators
# \* (A, B) means execute A, then either exit or execute B and then A again.
# LQ: IQC -> AS (Repeats if needed). Matches Case 3 pattern (IQC->AS->IQC->AS).
LQ = OperatorPOWL(operator=Operator.LOOP, children=[IQC, AS1])

# LT: RT -> TST (Repeats if needed). Matches Case 1 pattern (TST->RT->TST).
LT = OperatorPOWL(operator=Operator.LOOP, children=[RT, TST2])

# Define Choice Operators
# CQ: XOR(Silent, LQ) - Skip Quality Loop or Perform IQC/AS Loop. Matches Case 4 vs Case 3.
CQ = OperatorPOWL(operator=Operator.XOR, children=[SilentTransition(), LQ])

# CT: XOR(Silent, LT) - Skip Retest Loop or Perform RT/TST Loop. Matches Case 4 vs Case 1.
CT = OperatorPOWL(operator=Operator.XOR, children=[SilentTransition(), LT])

# LBNode: XOR(Silent, LB) - Skip Labeling or Perform Labeling. Matches Case 2 vs Case 1/3/4/5.
LBNode = OperatorPOWL(operator=Operator.XOR, children=[SilentTransition(), LB])

# Define Root Partial Order
root = StrictPartialOrder(nodes=[OR, MS, QS, AS1, CQ, TST1, CT, PK, LBNode, DP, D])

# Set Execution Order (Dependencies)
root.order.add_edge(OR, MS)
root.order.add_edge(MS, QS)
root.order.add_edge(QS, AS1)
root.order.add_edge(AS1, CQ)
root.order.add_edge(CQ, TST1)
root.order.add_edge(TST1, CT)
root.order.add_edge(CT, PK)
root.order.add_edge(PK, LBNode)
root.order.add_edge(LBNode, DP)
root.order.add_edge(DP, D)
```
