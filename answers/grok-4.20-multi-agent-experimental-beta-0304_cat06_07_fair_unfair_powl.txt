```python
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition
from pm4py.objects.process_tree.obj import Operator

# POWL Model with Potential Unfairness (XOR branch after skill assessment introduces bias via CommunityAffiliationCheck)
receive = Transition(label="ReceiveApplication")
data_check = Transition(label="DataCompletenessCheck")
request_info = Transition(label="RequestMoreInfo")
skill = Transition(label="SkillAssessment")
cultural = Transition(label="CulturalFitCheck")
community = Transition(label="CommunityAffiliationCheck")
managerial = Transition(label="ManagerialReview")
decision = Transition(label="FinalDecision")

data_loop = OperatorPOWL(operator=Operator.LOOP, children=[data_check, request_info])
bias_xor = OperatorPOWL(operator=Operator.XOR, children=[cultural, community])

unfair_model = StrictPartialOrder(nodes=[receive, data_loop, skill, bias_xor, managerial, decision])
unfair_model.order.add_edge(receive, data_loop)
unfair_model.order.add_edge(data_loop, skill)
unfair_model.order.add_edge(skill, bias_xor)
unfair_model.order.add_edge(bias_xor, managerial)
unfair_model.order.add_edge(managerial, decision)

# POWL Model Without Unfairness (no XOR/community branch; uniform CulturalFitCheck for all applicants)
receive_f = Transition(label="ReceiveApplication")
data_check_f = Transition(label="DataCompletenessCheck")
request_info_f = Transition(label="RequestMoreInfo")
skill_f = Transition(label="SkillAssessment")
cultural_f = Transition(label="CulturalFitCheck")
managerial_f = Transition(label="ManagerialReview")
decision_f = Transition(label="FinalDecision")

data_loop_f = OperatorPOWL(operator=Operator.LOOP, children=[data_check_f, request_info_f])

fair_model = StrictPartialOrder(nodes=[receive_f, data_loop_f, skill_f, cultural_f, managerial_f, decision_f])
fair_model.order.add_edge(receive_f, data_loop_f)
fair_model.order.add_edge(data_loop_f, skill_f)
fair_model.order.add_edge(skill_f, cultural_f)
fair_model.order.add_edge(cultural_f, managerial_f)
fair_model.order.add_edge(managerial_f, decision_f)
```
(The unfair model isolates subtle bias in the XOR after `SkillAssessment`; the fair model enforces a single uniform path.)
