    },

    'existence': {
        'Design Draft (DD)': (1.0, 1.0),
        'Cost Evaluation (CE)': (1.0, 1.0),
        'Technical Feasibility Check (TFC)': (1.0, 1.0)
    },
    
    # Activities that should not happen at all (Empty as none are restricted here).
    'absence': {},
    
    # Specific milestone activities that should only occur exactly once.
    'exactly_one': {
        'Idea Generation (IG)': (1.0, 1.0),
        'Approval Gate (AG)': (1.0, 1.0),
        'Final Launch (FL)': (1.0, 1.0)
    },

    # ---------------- Binary Constraints ----------------
    
    # If a design is drafted, management must eventually be involved for approval.
    'responded_existence': {
        ('Design Draft (DD)', 'Approval Gate (AG)'): (1.0, 1.0)
    },
    
    # Certain parallel activities must mutually happen in the same process instance.

        ('Technical Feasibility Check (TFC)', 'Cost Evaluation (CE)'): (1.0, 1.0),

    'chainsuccession': {

    },
