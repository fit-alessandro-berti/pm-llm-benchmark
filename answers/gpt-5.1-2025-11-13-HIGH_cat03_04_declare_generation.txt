# DECLARE model for the product design and launch process
# Activities (labels):
# IG = Idea Generation
# DD = Design Draft
# TFC = Technical Feasibility Check
# CE = Cost Evaluation
# PC = Prototype Creation
# LT = Laboratory Testing
# UT = User Testing
# AG = Approval Gate
# MP = Marketing Plan
# FL = Final Launch
#
# Each constraint has value (support, confidence); here we set both to 1.0.

declare_model = {
    # Unary templates
    "existence": {
        "IG": (1.0, 1.0),
        "DD": (1.0, 1.0),
        "TFC": (1.0, 1.0),
        "CE": (1.0, 1.0),
        "PC": (1.0, 1.0),
        "LT": (1.0, 1.0),
        "UT": (1.0, 1.0),
        "AG": (1.0, 1.0),
        "MP": (1.0, 1.0),
        "FL": (1.0, 1.0),
    },
    "absence": {
        # no forbidden activities in this scenario
    },
    "exactly_one": {
        # start and end are unique per product idea
        "IG": (1.0, 1.0),
        "AG": (1.0, 1.0),
        "FL": (1.0, 1.0),
    },
    "init": {
        # every process instance starts with idea generation
        "IG": (1.0, 1.0),
    },

    # Binary templates: keys are (activity1, activity2)

    "responded_existence": {
        # not used explicitly in this model
    },

    "coexistence": {
        # feasibility and cost evaluation always occur together
        ("TFC", "CE"): (1.0, 1.0),
        # lab and user testing both occur in any project that tests a prototype
        ("LT", "UT"): (1.0, 1.0),
    },

    "response": {
        # after a design draft, both feasibility and cost evaluations eventually occur
        ("DD", "TFC"): (1.0, 1.0),
        ("DD", "CE"): (1.0, 1.0),
    },

    "precedence": {
        # prototype creation is only allowed after both feasibility and cost checks
        ("TFC", "PC"): (1.0, 1.0),
        ("CE", "PC"): (1.0, 1.0),
    },

    "succession": {
        # high-level process flow
        ("IG", "DD"): (1.0, 1.0),   # designs follow ideas
        ("PC", "LT"): (1.0, 1.0),   # lab tests follow prototype
        ("LT", "UT"): (1.0, 1.0),   # user tests follow lab tests
        ("UT", "AG"): (1.0, 1.0),   # approval follows user tests
        ("AG", "MP"): (1.0, 1.0),   # marketing plan follows approval
        ("MP", "FL"): (1.0, 1.0),   # launch follows marketing plan
    },

    "altresponse": {
        # not used
    },
    "altprecedence": {
        # not used
    },
    "altsuccession": {
        # not used
    },

    "chainresponse": {
        # no immediate-next-step constraints modeled
    },
    "chainprecedence": {
        # not used
    },
    "chainsuccession": {
        # not used
    },

    "noncoexistence": {
        # no mutually exclusive activities in this basic model
    },
    "nonsuccession": {
        # no forbidden orderings modeled
    },
    "nonchainsuccession": {
        # no forbidden immediate-successor patterns modeled
    },
}
