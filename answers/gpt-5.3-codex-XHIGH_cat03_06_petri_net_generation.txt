```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# -------------------------
# Accepting Petri net setup
# -------------------------
net = PetriNet("insurance_claim_management")

# Places (states)
p_start = PetriNet.Place("p_start")                 # new claim arrives
p_after_c = PetriNet.Place("p_after_c")             # after Insurance Claim filing (C)
p_before_dv = PetriNet.Place("p_before_dv")         # ready for Document Verification
p_after_dv = PetriNet.Place("p_after_dv")           # after DV
p_after_fc = PetriNet.Place("p_after_fc")           # after Fraud Check
p_recheck = PetriNet.Place("p_recheck")             # investigation requires another FC
p_after_ca = PetriNet.Place("p_after_ca")           # after Claim Assessment
p_after_ap = PetriNet.Place("p_after_ap")           # after Approval (or skipped)
p_after_p = PetriNet.Place("p_after_p")             # after Payment
p_after_n = PetriNet.Place("p_after_n")             # after Notification
p_end = PetriNet.Place("p_end")                     # closed claim (final)

for p in [
    p_start, p_after_c, p_before_dv, p_after_dv, p_after_fc,
    p_recheck, p_after_ca, p_after_ap, p_after_p, p_after_n, p_end
]:
    net.places.add(p)

# Transitions (activities)
t_c = PetriNet.Transition("t_c", "C")                       # Insurance Claim filing
t_cr = PetriNet.Transition("t_cr", "CR")                    # Claim Registration
t_dv = PetriNet.Transition("t_dv", "DV")                    # Document Verification
t_rds = PetriNet.Transition("t_rds", "RDS")                 # Re-Document Submission
t_fc_initial = PetriNet.Transition("t_fc_initial", "FC")    # Fraud Check (first time)
t_ii = PetriNet.Transition("t_ii", "II")                    # In-depth Investigation
t_fc_recheck = PetriNet.Transition("t_fc_recheck", "FC")    # Fraud Check (after II)
t_ca = PetriNet.Transition("t_ca", "CA")                    # Claim Assessment
t_ap = PetriNet.Transition("t_ap", "AP")                    # Approval
t_skip_ap = PetriNet.Transition("t_skip_ap", None)          # optional skip if no AP needed
t_p = PetriNet.Transition("t_p", "P")                       # Payment
t_n = PetriNet.Transition("t_n", "N")                       # Notification
t_cl = PetriNet.Transition("t_cl", "CL")                    # Closure

for t in [
    t_c, t_cr, t_dv, t_rds, t_fc_initial, t_ii, t_fc_recheck,
    t_ca, t_ap, t_skip_ap, t_p, t_n, t_cl
]:
    net.transitions.add(t)

# Arcs (flow)
petri_utils.add_arc_from_to(p_start, t_c, net)
petri_utils.add_arc_from_to(t_c, p_after_c, net)

petri_utils.add_arc_from_to(p_after_c, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_before_dv, net)

petri_utils.add_arc_from_to(p_before_dv, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

# DV <-> RDS loop (repeat until documents complete)
petri_utils.add_arc_from_to(p_after_dv, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_before_dv, net)

# Proceed to fraud check
petri_utils.add_arc_from_to(p_after_dv, t_fc_initial, net)
petri_utils.add_arc_from_to(t_fc_initial, p_after_fc, net)

# FC <-> II loop (can repeat multiple times)
petri_utils.add_arc_from_to(p_after_fc, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_recheck, net)
petri_utils.add_arc_from_to(p_recheck, t_fc_recheck, net)
petri_utils.add_arc_from_to(t_fc_recheck, p_after_fc, net)

# Continue after FC
petri_utils.add_arc_from_to(p_after_fc, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

# Approval branch (AP required for high amount; skip otherwise)
petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_after_ap, net)
petri_utils.add_arc_from_to(p_after_ca, t_skip_ap, net)
petri_utils.add_arc_from_to(t_skip_ap, p_after_ap, net)

petri_utils.add_arc_from_to(p_after_ap, t_p, net)
petri_utils.add_arc_from_to(t_p, p_after_p, net)

petri_utils.add_arc_from_to(p_after_p, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_n, net)

petri_utils.add_arc_from_to(p_after_n, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_end, net)

# Initial and final markings
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_end] = 1

# Accepting Petri net is (net, im, fm)
```

If you want AP to be **always mandatory**, just remove `t_skip_ap` and its arcs.
