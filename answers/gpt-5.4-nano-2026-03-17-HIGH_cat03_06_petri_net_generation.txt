```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# -----------------------------
# Build accepting Petri net
# -----------------------------
net = PetriNet(name="InsuranceClaimManagement")

# Places (states between activities)
p_start = PetriNet.Place("p_start")                 # new claim arrives
p_dv_ready = PetriNet.Place("p_dv_ready")         # ready for Document Verification
p_dv_done = PetriNet.Place("p_dv_done")           # DV completed (either OK or needs re-submission)
p_fc_done = PetriNet.Place("p_fc_done")           # Fraud Check completed (may loop to II)
p_ca_ready = PetriNet.Place("p_ca_ready")         # ready for Claim Assessment
p_ap_ready = PetriNet.Place("p_ap_ready")         # ready for Approval
p_paid = PetriNet.Place("p_paid")                 # payment done
p_notified = PetriNet.Place("p_notified")         # customer notified
p_closed = PetriNet.Place("p_closed")             # archived/closed (final)

for p in [p_start, p_dv_ready, p_dv_done, p_fc_done, p_ca_ready, p_ap_ready, p_paid, p_notified, p_closed]:
    net.places.add(p)

# Transitions (activities)
t_cr = PetriNet.Transition("t_cr", label="Claim Registration (CR)")
t_dv = PetriNet.Transition("t_dv", label="Document Verification (DV)")
t_rds = PetriNet.Transition("t_rds", label="Re-Document Submission (RDS)")
t_fc = PetriNet.Transition("t_fc", label="Fraud Check (FC)")
t_ii = PetriNet.Transition("t_ii", label="In-depth Investigation (II)")
t_ca = PetriNet.Transition("t_ca", label="Claim Assessment (CA)")
t_ap = PetriNet.Transition("t_ap", label="Approval (AP)")
t_p = PetriNet.Transition("t_p", label="Payment (P)")
t_n = PetriNet.Transition("t_n", label="Notification (N)")
t_cl = PetriNet.Transition("t_cl", label="Closure (CL)")

for t in [t_cr, t_dv, t_rds, t_fc, t_ii, t_ca, t_ap, t_p, t_n, t_cl]:
    net.transitions.add(t)

# -----------------------------
# Arcs (logical sequence + loops)
# -----------------------------

# Start -> CR -> DV
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_dv_ready, net)

# DV step
petri_utils.add_arc_from_to(p_dv_ready, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_dv_done, net)

# If documents incomplete after DV: RDS, then go back to DV (loop)
petri_utils.add_arc_from_to(p_dv_done, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_dv_ready, net)

# If documents OK after DV: proceed to FC
petri_utils.add_arc_from_to(p_dv_done, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_fc_done, net)

# If doubts after FC: In-depth Investigation (II) multiple times, staying in FC outcome
petri_utils.add_arc_from_to(p_fc_done, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_fc_done, net)

# When fraud check is satisfactory: proceed to CA
petri_utils.add_arc_from_to(p_fc_done, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_ap_ready, net)

# Approval -> Payment -> Notification -> Closure
petri_utils.add_arc_from_to(p_ap_ready, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_paid, net)

petri_utils.add_arc_from_to(p_paid, t_p, net)
petri_utils.add_arc_from_to(t_p, p_notified, net)

petri_utils.add_arc_from_to(p_notified, t_n, net)
petri_utils.add_arc_from_to(t_n, p_closed, net)

petri_utils.add_arc_from_to(p_closed, t_cl, net)
# Note: CL transition fires from p_closed to final marking; simplest is to consume token and
# keep final token in p_closed by producing itself:
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# -----------------------------
# Initial and final markings
# -----------------------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1
```

This net reaches the final marking (`p_closed`) after `Payment (P)`  `Notification (N)`  `Closure (CL)`, and supports loops for:
- `Document Verification (DV)`  `Re-Document Submission (RDS)`
- `Fraud Check (FC)`  `In-depth Investigation (II)` (via repeated `II` firings before continuing to `Claim Assessment (CA)`).
