```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# Places (states between activities)
start = PetriNet.Place("start")                  # Initial place: new claim arrives
after_C = PetriNet.Place("after_C")
after_CR = PetriNet.Place("after_CR")
after_DV = PetriNet.Place("after_DV")
after_RDS = PetriNet.Place("after_RDS")
after_FC = PetriNet.Place("after_FC")
after_II = PetriNet.Place("after_II")
after_CA = PetriNet.Place("after_CA")
after_AP = PetriNet.Place("after_AP")
after_P = PetriNet.Place("after_P")
after_N = PetriNet.Place("after_N")
end = PetriNet.Place("end")                      # Final place: claim closed

# Add places to the net
net.places.add(start)
net.places.add(after_C)
net.places.add(after_CR)
net.places.add(after_DV)
net.places.add(after_RDS)
net.places.add(after_FC)
net.places.add(after_II)
net.places.add(after_CA)
net.places.add(after_AP)
net.places.add(after_P)
net.places.add(after_N)
net.places.add(end)

# Transitions (labeled with activity names)
t_C   = PetriNet.Transition("C",   label="Insurance Claim (C)")
t_CR  = PetriNet.Transition("CR",  label="Claim Registration (CR)")
t_DV  = PetriNet.Transition("DV",  label="Document Verification (DV)")
t_RDS = PetriNet.Transition("RDS", label="Re-Document Submission (RDS)")
t_FC  = PetriNet.Transition("FC",  label="Fraud Check (FC)")
t_II  = PetriNet.Transition("II",  label="In-depth Investigation (II)")
t_CA  = PetriNet.Transition("CA",  label="Claim Assessment (CA)")
t_AP  = PetriNet.Transition("AP",  label="Approval (AP)")
t_P   = PetriNet.Transition("P",   label="Payment (P)")
t_N   = PetriNet.Transition("N",   label="Notification (N)")
t_CL  = PetriNet.Transition("CL",  label="Closure (CL)")

# Add transitions to the net
net.transitions.add(t_C)
net.transitions.add(t_CR)
net.transitions.add(t_DV)
net.transitions.add(t_RDS)
net.transitions.add(t_FC)
net.transitions.add(t_II)
net.transitions.add(t_CA)
net.transitions.add(t_AP)
net.transitions.add(t_P)
net.transitions.add(t_N)
net.transitions.add(t_CL)

# Arcs for the main sequence and loops
petri_utils.add_arc_from_to(start, t_C, net)
petri_utils.add_arc_from_to(t_C, after_C, net)

petri_utils.add_arc_from_to(after_C, t_CR, net)
petri_utils.add_arc_from_to(t_CR, after_CR, net)

petri_utils.add_arc_from_to(after_CR, t_DV, net)
petri_utils.add_arc_from_to(t_DV, after_DV, net)

# Loop for incomplete documents: DV -> RDS -> DV
petri_utils.add_arc_from_to(after_DV, t_RDS, net)   # incomplete -> request re-submission
petri_utils.add_arc_from_to(t_RDS, after_RDS, net)
petri_utils.add_arc_from_to(after_RDS, t_DV, net)   # back to verification

# Normal continuation after DV (complete documents)
petri_utils.add_arc_from_to(after_DV, t_FC, net)    # complete -> proceed to fraud check

petri_utils.add_arc_from_to(t_FC, after_FC, net)

# Loop for fraud doubts: FC -> II -> FC (can repeat multiple times)
petri_utils.add_arc_from_to(after_FC, t_II, net)    # suspicion -> in-depth investigation
petri_utils.add_arc_from_to(t_II, after_II, net)
petri_utils.add_arc_from_to(after_II, t_FC, net)    # back to fraud check

# Normal continuation after FC (no suspicion)
petri_utils.add_arc_from_to(after_FC, t_CA, net)    # no suspicion -> assessment

petri_utils.add_arc_from_to(t_CA, after_CA, net)

# Approval is only needed if amount > threshold; we model it as always required for simplicity
# (non-deterministic choice could be added with silent transitions, but not required here)
petri_utils.add_arc_from_to(after_CA, t_AP, net)
petri_utils.add_arc_from_to(t_AP, after_AP, net)

petri_utils.add_arc_from_to(after_AP, t_P, net)
petri_utils.add_arc_from_to(t_P, after_P, net)

petri_utils.add_arc_from_to(after_P, t_N, net)
petri_utils.add_arc_from_to(t_N, after_N, net)

petri_utils.add_arc_from_to(after_N, t_CL, net)
petri_utils.add_arc_from_to(t_CL, end, net)

# Initial and final markings
im = Marking()
im[start] = 1

fm = Marking()
fm[end] = 1
```

This Petri net models the described insurance claim process:

- Starts with a token in `start` (new claim arrives).
- Allows multiple loops:
  - After Document Verification, if documents are incomplete, the process can go through Re-Document Submission any number of times.
  - After Fraud Check, if doubts arise, In-depth Investigation can be repeated arbitrarily many times.
- The main path proceeds sequentially through all activities when no loops are taken.
- Ends with a token in `end` after Closure, representing a successfully closed claim.

The model uses explicit places between activities to clearly separate states and enable the required loops while keeping the structure readable.
