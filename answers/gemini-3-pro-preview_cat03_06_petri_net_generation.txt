Here is the Python code to construct the Accepting Petri Net for the Insurance Claim Management process using the `pm4py` library.

This solution models the sequential process and includes the two requested loops:
1.  **Re-Document Submission (RDS):** A loop back from the post-verification state to the pre-verification state.
2.  **In-depth Investigation (II):** A loop back from the post-fraud-check state to the pre-fraud-check state.

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# 1. Create the Petri Net container
net = PetriNet(name="Insurance Claim Management Process")

# 2. Create Places (States)
# p_source: The start of the process (Claim Arrives)
# p_sink: The end of the process (Claim Closed)
# Intermediate places represent the state of the claim between activities
p_source = PetriNet.Place("source")
p_reg_complete = PetriNet.Place("p_reg_complete")  # Registered, ready for verification
p_ver_complete = PetriNet.Place("p_ver_complete")  # Verified, ready for fraud check
p_chk_complete = PetriNet.Place("p_chk_complete")  # Checked, ready for assessment
p_ass_complete = PetriNet.Place("p_ass_complete")  # Assessed, ready for approval
p_app_complete = PetriNet.Place("p_app_complete")  # Approved, ready for payment
p_paid = PetriNet.Place("p_paid")                  # Paid, ready for notification
p_notified = PetriNet.Place("p_notified")          # Notified, ready for closure
p_sink = PetriNet.Place("sink")

# Add places to the net
net.places.add(p_source)
net.places.add(p_reg_complete)
net.places.add(p_ver_complete)
net.places.add(p_chk_complete)
net.places.add(p_ass_complete)
net.places.add(p_app_complete)
net.places.add(p_paid)
net.places.add(p_notified)
net.places.add(p_sink)

# 3. Create Transitions (Activities)
t_cr = PetriNet.Transition("t_cr", "Claim Registration")
t_dv = PetriNet.Transition("t_dv", "Document Verification")
t_rds = PetriNet.Transition("t_rds", "Re-Document Submission")  # Loop transition
t_fc = PetriNet.Transition("t_fc", "Fraud Check")
t_ii = PetriNet.Transition("t_ii", "In-depth Investigation")    # Loop transition
t_ca = PetriNet.Transition("t_ca", "Claim Assessment")
t_ap = PetriNet.Transition("t_ap", "Approval")
t_p = PetriNet.Transition("t_p", "Payment")
t_n = PetriNet.Transition("t_n", "Notification")
t_cl = PetriNet.Transition("t_cl", "Closure")

# Add transitions to the net
net.transitions.add(t_cr)
net.transitions.add(t_dv)
net.transitions.add(t_rds)
net.transitions.add(t_fc)
net.transitions.add(t_ii)
net.transitions.add(t_ca)
net.transitions.add(t_ap)
net.transitions.add(t_p)
net.transitions.add(t_n)
net.transitions.add(t_cl)

# 4. Create Arcs (Flow Logic)

# --- Step 1: Registration ---
petri_utils.add_arc_from_to(p_source, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_reg_complete, net)

# --- Step 2: Document Verification (with RDS Loop) ---
# Normal flow: p_reg_complete -> DV -> p_ver_complete
petri_utils.add_arc_from_to(p_reg_complete, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_ver_complete, net)

# Loop flow: If docs incomplete, go from p_ver_complete -> RDS -> p_reg_complete (to be verified again)
petri_utils.add_arc_from_to(p_ver_complete, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_reg_complete, net)

# --- Step 3: Fraud Check (with Investigation Loop) ---
# Normal flow: p_ver_complete -> FC -> p_chk_complete
petri_utils.add_arc_from_to(p_ver_complete, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_chk_complete, net)

# Loop flow: If doubts, go from p_chk_complete -> II -> p_ver_complete (to be fraud checked again)
petri_utils.add_arc_from_to(p_chk_complete, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_ver_complete, net)

# --- Step 4: Assessment ---
petri_utils.add_arc_from_to(p_chk_complete, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_ass_complete, net)

# --- Step 5: Approval ---
petri_utils.add_arc_from_to(p_ass_complete, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_app_complete, net)

# --- Step 6: Payment ---
petri_utils.add_arc_from_to(p_app_complete, t_p, net)
petri_utils.add_arc_from_to(t_p, p_paid, net)

# --- Step 7: Notification ---
petri_utils.add_arc_from_to(p_paid, t_n, net)
petri_utils.add_arc_from_to(t_n, p_notified, net)

# --- Step 8: Closure ---
petri_utils.add_arc_from_to(p_notified, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_sink, net)

# 5. Define Initial and Final Markings

# Initial Marking: One token in the Source place (New Claim Arrives)
im = Marking()
im[p_source] = 1

# Final Marking: One token in the Sink place (Claim Closed)
fm = Marking()
fm[p_sink] = 1

# Optional: Print debugging info to verify structure
print(f"Net Name: {net.name}")
print(f"Places: {len(net.places)}")
print(f"Transitions: {len(net.transitions)}")
print(f"Arcs: {len(net.arcs)}")
print(f"Initial Marking: {im}")
print(f"Final Marking: {fm}")
```
