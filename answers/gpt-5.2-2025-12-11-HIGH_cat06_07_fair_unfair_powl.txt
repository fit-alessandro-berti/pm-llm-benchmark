### 1) POWL model **with potential unfairness** (XOR branch introduces affiliation-based uplift)

```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition
from pm4py.objects.process_tree.obj import Operator

# --- Activities ---
ReceiveApplication      = Transition(label="ReceiveApplication")
FillQuestionnaire       = Transition(label="FillQuestionnaire")
ResumeParsing           = Transition(label="ResumeParsing")
DataCompletenessCheck   = Transition(label="DataCompletenessCheck")
RequestMoreInfo         = Transition(label="RequestMoreInfo")
SkillAssessment         = Transition(label="SkillAssessment")

CulturalFitCheck_std    = Transition(label="CulturalFitCheck")  # standard path
CommunityAffiliationChk = Transition(label="CommunityAffiliationCheck")
CulturalFitCheck_uplift = Transition(label="CulturalFitCheck (AffiliationAdjusted)")

ManagerialReview        = Transition(label="ManagerialReview")
FinalDecision           = Transition(label="FinalDecision")

# --- Loop for completeness: check, then either exit or request more info and check again ---
completeness_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[DataCompletenessCheck, RequestMoreInfo]
)

# --- Biased XOR after SkillAssessment ---
# Branch 1: standard cultural fit
branch_standard = CulturalFitCheck_std

# Branch 2: community affiliation check influences cultural fit scoring
branch_affiliation = StrictPartialOrder(nodes=[CommunityAffiliationChk, CulturalFitCheck_uplift])
branch_affiliation.order.add_edge(CommunityAffiliationChk, CulturalFitCheck_uplift)

cultural_xor = OperatorPOWL(
    operator=Operator.XOR,
    children=[branch_standard, branch_affiliation]
)

# --- Overall process as a (mostly) sequential partial order ---
root_unfair = StrictPartialOrder(nodes=[
    ReceiveApplication, FillQuestionnaire, ResumeParsing,
    completeness_loop, SkillAssessment, cultural_xor,
    ManagerialReview, FinalDecision
])

root_unfair.order.add_edge(ReceiveApplication, FillQuestionnaire)
root_unfair.order.add_edge(FillQuestionnaire, ResumeParsing)
root_unfair.order.add_edge(ResumeParsing, completeness_loop)
root_unfair.order.add_edge(completeness_loop, SkillAssessment)
root_unfair.order.add_edge(SkillAssessment, cultural_xor)
root_unfair.order.add_edge(cultural_xor, ManagerialReview)
root_unfair.order.add_edge(ManagerialReview, FinalDecision)
```

### 2) POWL model **without unfairness** (single cultural fit path for all applicants; no affiliation-based XOR)

```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition
from pm4py.objects.process_tree.obj import Operator

# --- Activities ---
ReceiveApplication      = Transition(label="ReceiveApplication")
FillQuestionnaire       = Transition(label="FillQuestionnaire")
ResumeParsing           = Transition(label="ResumeParsing")
DataCompletenessCheck   = Transition(label="DataCompletenessCheck")
RequestMoreInfo         = Transition(label="RequestMoreInfo")
SkillAssessment         = Transition(label="SkillAssessment")
CulturalFitCheck        = Transition(label="CulturalFitCheck")  # same for everyone
ManagerialReview        = Transition(label="ManagerialReview")
FinalDecision           = Transition(label="FinalDecision")

# --- Loop for completeness ---
completeness_loop = OperatorPOWL(
    operator=Operator.LOOP,
    children=[DataCompletenessCheck, RequestMoreInfo]
)

# --- Overall process (no XOR branch that privileges community affiliation) ---
root_fair = StrictPartialOrder(nodes=[
    ReceiveApplication, FillQuestionnaire, ResumeParsing,
    completeness_loop, SkillAssessment, CulturalFitCheck,
    ManagerialReview, FinalDecision
])

root_fair.order.add_edge(ReceiveApplication, FillQuestionnaire)
root_fair.order.add_edge(FillQuestionnaire, ResumeParsing)
root_fair.order.add_edge(ResumeParsing, completeness_loop)
root_fair.order.add_edge(completeness_loop, SkillAssessment)
root_fair.order.add_edge(SkillAssessment, CulturalFitCheck)
root_fair.order.add_edge(CulturalFitCheck, ManagerialReview)
root_fair.order.add_edge(ManagerialReview, FinalDecision)
```
