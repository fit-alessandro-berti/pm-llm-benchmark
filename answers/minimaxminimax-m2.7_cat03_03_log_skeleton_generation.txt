# LogSkeleton for a DistributionCenter Order Process
# -------------------------------------------------
log_skeleton = {

    # -----------------------------------------------------------------
    # 1  EQUIVALENCE
    #     If an activity from a pair occurs in a case, the two activities
    #     must appear the same number of times in that case.
    # -----------------------------------------------------------------
    "equivalence": {
        ("CA", "RI"),   # each availability check is matched by a reservation
        ("PO", "GSL")   # each packed order gets exactly one shipping label
    },

    # -----------------------------------------------------------------
    # 2  ALWAYSBEFORE
    #     Whenever the first activity of a pair occurs in a case,
    #     the second activity must have already occurred (i.e., earlier
    #     in the trace).
    # -----------------------------------------------------------------
    "always_before": {
        ("RO",  "CA"),   # Receive Order  Check Availability
        ("CA",  "RI"),   # Check Availability  Reserve Items
        ("RI",  "PI"),   # Reserve Items  Pick Items
        ("PI",  "QC"),   # Pick Items  Quality Check
        ("QC",  "PO"),   # Quality Check  Pack Order
        ("PO",  "GSL"),  # Pack Order  Generate Shipping Label
        ("GSL", "D"),    # Generate Shipping Label  Dispatch
        ("D",   "IC"),   # Dispatch  Invoice Customer
        ("IC",  "RP")    # Invoice Customer  Record Payment
    },

    # -----------------------------------------------------------------
    # 3  ALWAYSAFTER
    #     If the first activity of a pair occurs, the second one must
    #     occur later in the same case.  (Pairs here are not already
    #     covered by “always_before”.)
    # -----------------------------------------------------------------
    "always_after": {
        ("RO", "D"),    # after a order is received, dispatch will eventually happen
        ("RO", "IC"),   # after a order is received, an invoice will follow
        ("D",  "RP")    # after dispatch, the payment record will be created later
    },

    # -----------------------------------------------------------------
    # 4  NEVERTOGETHER
    #     The two activities of a pair can never appear together in the
    #     same case.  In this straightforward process we have no such
    #     exclusive pairs, so the set is empty.
    # -----------------------------------------------------------------
    "never_together": set(),   # no mutuallyexclusive activities in the described process

    # -----------------------------------------------------------------
    # 5  ACTIVITY OCCURRENCES (frequency bounds)
    #     For each activity we list the set of allowed occurrence counts
    #     within a single case.  Here every activity is mandatory and
    #     appears exactly once.
    # -----------------------------------------------------------------
    "activ_freq": {
        "RO":  {1},
        "CA":  {1},
        "RI":  {1},
        "PI":  {1},
        "QC":  {1},
        "PO":  {1},
        "GSL": {1},
        "D":   {1},
        "IC":  {1},
        "RP":  {1}
    },

    # -----------------------------------------------------------------
    # 6  DIRECTLYFOLLOWS
    #     Whenever the first activity occurs, the second one must be the
    #     *immediate* next activity in the trace.
    # -----------------------------------------------------------------
    "directly_follows": {
        ("RO",  "CA"),   # after receiving, the next step is availability check
        ("CA",  "RI"),   # check  reserve (no intermediate steps)
        ("RI",  "PI"),   # reserve  pick
        ("PI",  "QC"),   # pick  quality check
        ("QC",  "PO"),   # quality check  pack
        ("PO",  "GSL"),  # pack  generate label
        ("GSL", "D"),    # label  dispatch
        ("IC",  "RP")    # invoice  record payment (no steps in between)
    }
}
