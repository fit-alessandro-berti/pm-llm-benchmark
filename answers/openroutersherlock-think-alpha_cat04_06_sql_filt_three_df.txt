```sql
WITH ordered_events AS (
  SELECT *,
         ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS rn
  FROM event_log
),
bad_cases AS (
  SELECT DISTINCT oe1.case_id
  FROM ordered_events oe1
  JOIN ordered_events oe2 ON oe1.case_id = oe2.case_id AND oe2.rn = oe1.rn + 1
  JOIN ordered_events oe3 ON oe1.case_id = oe3.case_id AND oe3.rn = oe1.rn + 2
  WHERE oe1.activity = 'Approve Purchase Requisition'
    AND oe2.activity = 'Create Order'
    AND oe3.activity = 'Confirm Order'
    AND oe3.timestamp - oe2.timestamp > INTERVAL '5' DAY
)
SELECT *
FROM event_log
WHERE case_id NOT IN (SELECT case_id FROM bad_cases)
ORDER BY case_id, timestamp;
```
