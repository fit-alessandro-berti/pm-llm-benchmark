#### Query 1: Identify All Process Deviations Across the Full Dataset
This query flags all cases with out-of-order steps, missing mandatory activities, or both, to quantify how common anomalies are.
```sql
WITH mandatory_flow AS (
    SELECT 'Register Order' AS activity, 1 AS expected_seq UNION ALL
    SELECT 'Perform Credit Check', 2 UNION ALL
    SELECT 'Validate Stock', 3 UNION ALL
    SELECT 'Confirm Shipment', 4 UNION ALL
    SELECT 'Ship Goods', 5 UNION ALL
    SELECT 'Issue Invoice', 6 UNION ALL
    SELECT 'Receive Payment', 7
),
case_activity_sequence AS (
    SELECT 
        oel.case_id,
        oel.activity,
        oel.event_id,
        oel.timestamp,
        mf.expected_seq,
        ROW_NUMBER() OVER (PARTITION BY oel.case_id ORDER BY oel.event_id) AS actual_seq
    FROM order_event_log oel
    JOIN mandatory_flow mf ON oel.activity = mf.activity
),
case_deviation_summary AS (
    SELECT 
        case_id,
        -- List actual activity sequence
        STRING_AGG(activity, '  ' ORDER BY actual_seq) AS actual_flow,
        -- List missing mandatory activities
        ARRAY_AGG(mf.activity ORDER BY mf.expected_seq) FILTER (WHERE mf.activity NOT IN (SELECT activity FROM case_activity_sequence cas WHERE cas.case_id = c.case_id)) AS missing_activities,
        -- Flag out-of-order steps: check if expected sequence is not strictly increasing
        BOOL_OR(mf.expected_seq < LAG(mf.expected_seq) OVER (PARTITION BY case_id ORDER BY actual_seq)) AS has_out_of_order_steps
    FROM mandatory_flow mf
    LEFT JOIN case_activity_sequence c ON mf.activity = c.activity AND mf.expected_seq = c.expected_seq
    GROUP BY case_id
)
SELECT 
    case_id,
    actual_flow,
    missing_activities,
    has_out_of_order_steps,
    CASE 
        WHEN missing_activities IS NOT NULL AND has_out_of_order_steps THEN 'Missing steps + out-of-order steps'
        WHEN missing_activities IS NOT NULL THEN 'Missing mandatory steps'
        WHEN has_out_of_order_steps THEN 'Out-of-order mandatory steps'
    END AS deviation_type
FROM case_deviation_summary
WHERE missing_activities IS NOT NULL OR has_out_of_order_steps = TRUE
ORDER BY case_id;
```

#### Query 6: Investigate Unauthorized Shipment (Scheduled=N but Shipped)
This tests the hypothesis that unauthorized shipments are due to manual overrides or missing system guardrails.
```sql
SELECT 
    oel_confirm.case_id,
    o.order_type,
    o.order_value,
    oel_confirm.additional_info AS confirm_details,
    oel_confirm.timestamp AS confirm_ts,
    oel_ship.timestamp AS ship_ts,
    oel_ship.additional_info AS ship_details,
    -- Flag if manual override is noted in event logs
    CASE 
        WHEN oel_confirm.additional_info ILIKE '%override%' OR oel_confirm.additional_info ILIKE '%manual%' THEN 'Override in confirm event'
        WHEN oel_ship.additional_info ILIKE '%override%' OR oel_ship.additional_info ILIKE '%manual%' THEN 'Override in ship event'
        ELSE 'No override logged'
    END AS override_status
FROM order_event_log oel_confirm
JOIN order_event_log oel_ship 
    ON oel_confirm.case_id = oel_ship.case_id
    AND oel_ship.activity = 'Ship Goods'
    AND oel_ship.timestamp > oel_confirm.timestamp
JOIN orders o ON oel_confirm.case_id = o.case_id
WHERE oel_confirm.activity = 'Confirm Shipment'
AND oel_confirm.additional_info LIKE '%shipment_scheduled=N%'
ORDER BY confirm_ts;
```

#### Query 7: Identify High-Risk Resources Involved in Anomalies
This tests the hypothesis that anomalies are driven by untrained or non-compliant staff, by finding resources involved in the most deviations.
```sql
WITH anomalous_events AS (
    SELECT oel.case_id, oel.resource, oel.activity
    FROM order_event_log oel
    WHERE oel.case_id IN (
        -- All case IDs with deviations (from Query 1)
        SELECT case_id FROM (
            SELECT 
                oel1.case_id
            FROM order_event_log oel1
            JOIN order_event_log oel2 
                ON oel1.case_id = oel2.case_id AND oel1.event_id < oel2.event_id
            WHERE 
                (oel1.activity = 'Ship Goods' AND oel2.activity = 'Confirm Shipment')
                OR (oel1.activity = 'Receive Payment' AND oel2.activity = 'Issue Invoice')
                OR (oel1.activity = 'Ship Goods' AND oel2.activity = 'Perform Credit Check')
            UNION
            SELECT case_id FROM order_event_log
            EXCEPT
            SELECT oel.case_id
            FROM order_event_log oel
            JOIN mandatory_flow mf ON oel.activity = mf.activity
            GROUP BY oel.case_id
            HAVING COUNT(DISTINCT mf.activity) = 7
        ) dev_cases
    )
)
SELECT 
    r.resource_id,
    r.role,
    r.department,
    COUNT(*) AS total_anomalous_events,
    COUNT(DISTINCT ae.case_id) AS anomalous_cases_involved,
    STRING_AGG(DISTINCT ae.activity, ', ' ORDER BY ae.activity) AS anomaly_types
FROM anomalous_events ae
JOIN resources r ON ae.resource = r.resource_id
GROUP BY r.resource_id, r.role, r.department
ORDER BY total_anomalous_events DESC;
```
