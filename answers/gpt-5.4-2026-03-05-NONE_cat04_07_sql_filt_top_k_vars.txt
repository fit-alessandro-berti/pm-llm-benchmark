```sql
-- Replace ? with the desired K, or parameterize it as needed.
WITH case_variants AS (
    -- 1) Build the ordered activity sequence for each case
    SELECT
        case_id,
        string_agg(activity, ' -> ' ORDER BY timestamp) AS variant
    FROM event_log
    GROUP BY case_id
),
ranked_variants AS (
    -- 2) Count how many cases belong to each variant
    SELECT
        variant,
        COUNT(*) AS case_count
    FROM case_variants
    GROUP BY variant
    -- 3) Keep only the top K variants by frequency
    QUALIFY ROW_NUMBER() OVER (
        ORDER BY case_count DESC, variant
    ) <= ?
),
selected_cases AS (
    -- Cases whose variant is among the top K
    SELECT cv.case_id
    FROM case_variants cv
    INNER JOIN ranked_variants rv
        ON cv.variant = rv.variant
)
-- 4) Return all original events for cases in the top K variants
SELECT e.*
FROM event_log e
INNER JOIN selected_cases sc
    ON e.case_id = sc.case_id
ORDER BY e.case_id, e.timestamp;
```

If you want a concrete example with `K = 10`:

```sql
WITH case_variants AS (
    SELECT
        case_id,
        string_agg(activity, ' -> ' ORDER BY timestamp) AS variant
    FROM event_log
    GROUP BY case_id
),
ranked_variants AS (
    SELECT
        variant,
        COUNT(*) AS case_count
    FROM case_variants
    GROUP BY variant
    QUALIFY ROW_NUMBER() OVER (
        ORDER BY case_count DESC, variant
    ) <= 10
),
selected_cases AS (
    SELECT cv.case_id
    FROM case_variants cv
    INNER JOIN ranked_variants rv
        ON cv.variant = rv.variant
)
SELECT e.*
FROM event_log e
INNER JOIN selected_cases sc
    ON e.case_id = sc.case_id
ORDER BY e.case_id, e.timestamp;
```

Notes:
- `string_agg(activity, ' -> ' ORDER BY timestamp)` creates the ordered activity sequence per case.
- `COUNT(*)` over the grouped variants gives the number of cases following each variant.
- `ROW_NUMBER()` selects the top K variants by frequency.
- The final join brings back all events for only those cases in the selected variants.
