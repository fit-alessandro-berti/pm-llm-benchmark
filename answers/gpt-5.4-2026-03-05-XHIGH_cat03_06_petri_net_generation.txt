Below is one clean accepting Petri net for the insurance-claim scenario.

- The initial token in `start` represents a new claim arriving (`C`).
- The visible transitions are the business activities.
- Two silent transitions (`label=None`) are used for routing:
  - `tau_docs_ok`: documents are complete, so the process can move from DV to FC
  - `tau_skip_approval`: approval is not needed, so the process can move directly to payment

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="insurance_claim_management")

# -------------------------
# Places
# -------------------------
p_start = PetriNet.Place(name="start")  # new claim arrives here
p_wait_dv = PetriNet.Place(name="waiting_for_document_verification")
p_after_dv = PetriNet.Place(name="after_document_verification")
p_wait_fc = PetriNet.Place(name="waiting_for_fraud_check")
p_after_fc = PetriNet.Place(name="after_fraud_check")
p_after_ca = PetriNet.Place(name="after_claim_assessment")
p_wait_payment = PetriNet.Place(name="waiting_for_payment")
p_after_payment = PetriNet.Place(name="after_payment")
p_after_notification = PetriNet.Place(name="after_notification")
p_closed = PetriNet.Place(name="closed")

for p in [
    p_start, p_wait_dv, p_after_dv, p_wait_fc, p_after_fc,
    p_after_ca, p_wait_payment, p_after_payment,
    p_after_notification, p_closed
]:
    net.places.add(p)

# -------------------------
# Transitions (visible)
# -------------------------
t_cr = PetriNet.Transition(name="cr", label="Claim Registration")
t_dv = PetriNet.Transition(name="dv", label="Document Verification")
t_rds = PetriNet.Transition(name="rds", label="Re-Document Submission")
t_fc = PetriNet.Transition(name="fc", label="Fraud Check")
t_ii = PetriNet.Transition(name="ii", label="In-depth Investigation")
t_ca = PetriNet.Transition(name="ca", label="Claim Assessment")
t_ap = PetriNet.Transition(name="ap", label="Approval")
t_p = PetriNet.Transition(name="p", label="Payment")
t_n = PetriNet.Transition(name="n", label="Notification")
t_cl = PetriNet.Transition(name="cl", label="Closure")

# -------------------------
# Transitions (silent / routing)
# -------------------------
t_docs_ok = PetriNet.Transition(name="tau_docs_ok", label=None)
t_skip_approval = PetriNet.Transition(name="tau_skip_approval", label=None)

for t in [
    t_cr, t_dv, t_rds, t_fc, t_ii, t_ca, t_ap, t_p, t_n, t_cl,
    t_docs_ok, t_skip_approval
]:
    net.transitions.add(t)

# -------------------------
# Arcs
# -------------------------

# Start -> Claim Registration
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_wait_dv, net)

# Claim Registration -> Document Verification
petri_utils.add_arc_from_to(p_wait_dv, t_dv, net)
petri_utils.add_arc_from_to(t_dv, p_after_dv, net)

# If documents are incomplete: loop through Re-Document Submission
petri_utils.add_arc_from_to(p_after_dv, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_wait_dv, net)

# If documents are complete: continue to Fraud Check
petri_utils.add_arc_from_to(p_after_dv, t_docs_ok, net)
petri_utils.add_arc_from_to(t_docs_ok, p_wait_fc, net)

# Fraud Check
petri_utils.add_arc_from_to(p_wait_fc, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_after_fc, net)

# If doubts remain: In-depth Investigation loop
petri_utils.add_arc_from_to(p_after_fc, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_wait_fc, net)

# If fraud check is cleared: continue to Claim Assessment
petri_utils.add_arc_from_to(p_after_fc, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_after_ca, net)

# Optional Approval before payment
petri_utils.add_arc_from_to(p_after_ca, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_wait_payment, net)

# If approval is not needed
petri_utils.add_arc_from_to(p_after_ca, t_skip_approval, net)
petri_utils.add_arc_from_to(t_skip_approval, p_wait_payment, net)

# Payment -> Notification -> Closure
petri_utils.add_arc_from_to(p_wait_payment, t_p, net)
petri_utils.add_arc_from_to(t_p, p_after_payment, net)

petri_utils.add_arc_from_to(p_after_payment, t_n, net)
petri_utils.add_arc_from_to(t_n, p_after_notification, net)

petri_utils.add_arc_from_to(p_after_notification, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)

# -------------------------
# Initial and final markings
# -------------------------
im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1
```

This defines the accepting Petri net `(net, im, fm)`.
