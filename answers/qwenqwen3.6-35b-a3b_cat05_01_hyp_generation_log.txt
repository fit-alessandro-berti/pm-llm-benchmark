#### **Query 1: Identify Cases with Sequence Violations**
*Investigates if activities occur out of the expected order (e.g., Credit Check after Ship).*
```sql
WITH case_activities AS (
    SELECT 
        case_id, 
        activity, 
        timestamp,
        -- Assign expected order rank based on activity name
        CASE activity
            WHEN 'Register Order' THEN 1
            WHEN 'Perform Credit Check' THEN 2
            WHEN 'Validate Stock' THEN 3
            WHEN 'Confirm Shipment' THEN 4
            WHEN 'Ship Goods' THEN 5
            WHEN 'Issue Invoice' THEN 6
            WHEN 'Receive Payment' THEN 7
            ELSE 99
        END AS expected_rank
    FROM order_event_log
)
SELECT 
    ca1.case_id,
    ca1.activity AS activity_1,
    ca1.timestamp AS ts_1,
    ca2.activity AS activity_2,
    ca2.timestamp AS ts_2,
    ca1.expected_rank,
    ca2.expected_rank
FROM case_activities ca1
JOIN case_activities ca2 
    ON ca1.case_id = ca2.case_id
    AND ca1.expected_rank > ca2.expected_rank
    AND ca1.timestamp > ca2.timestamp
WHERE ca1.expected_rank != 99 AND ca2.expected_rank != 99
ORDER BY ca1.case_id, ca1.expected_rank;
```

#### **Query 6: Calculate Process Duration and Gaps**
*Identifies cases with unusual timing, such as instant payments or long gaps.*
```sql
WITH case_timeline AS (
    SELECT 
        case_id,
        MIN(CASE WHEN activity = 'Register Order' THEN timestamp END) AS start_time,
        MAX(CASE WHEN activity = 'Receive Payment' THEN timestamp END) AS end_time,
        MIN(CASE WHEN activity = 'Ship Goods' THEN timestamp END) AS ship_time,
        MIN(CASE WHEN activity = 'Issue Invoice' THEN timestamp END) AS invoice_time
    FROM order_event_log
    GROUP BY case_id
)
SELECT 
    case_id,
    start_time,
    ship_time,
    invoice_time,
    end_time,
    EXTRACT(EPOCH FROM (end_time - start_time)) / 3600.0 AS total_hours,
    EXTRACT(EPOCH FROM (invoice_time - start_time)) / 60.0 AS hours_to_invoice,
    EXTRACT(EPOCH FROM (end_time - invoice_time)) / 60.0 AS hours_to_payment_after_invoice
FROM case_timeline
WHERE start_time IS NOT NULL AND end_time IS NOT NULL
ORDER BY total_hours;
```
