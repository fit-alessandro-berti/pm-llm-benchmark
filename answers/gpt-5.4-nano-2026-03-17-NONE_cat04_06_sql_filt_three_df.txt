```sql
WITH seq_cases AS (
    SELECT
        e1.case_id
    FROM event_log e1
    JOIN event_log e2
      ON e2.case_id = e1.case_id
     AND e2.activity = 'Create Order'
     AND e2.timestamp = (
            SELECT MIN(e3.timestamp)
            FROM event_log e3
            WHERE e3.case_id = e1.case_id
              AND e3.timestamp > e1.timestamp
              AND e3.activity = 'Create Order'
        )
     AND NOT EXISTS (
            -- no other event between e1 (Approve PR) and the matched e2 (Create Order)
            SELECT 1
            FROM event_log x
            WHERE x.case_id = e1.case_id
              AND x.timestamp > e1.timestamp
              AND x.timestamp < e2.timestamp
        )
    JOIN event_log e3
      ON e3.case_id = e1.case_id
     AND e3.activity = 'Confirm Order'
     AND e3.timestamp = (
            SELECT MIN(e4.timestamp)
            FROM event_log e4
            WHERE e4.case_id = e1.case_id
              AND e4.timestamp > e2.timestamp
              AND e4.activity = 'Confirm Order'
        )
     AND NOT EXISTS (
            -- no other event between e2 (Create Order) and the matched e3 (Confirm Order)
            SELECT 1
            FROM event_log y
            WHERE y.case_id = e1.case_id
              AND y.timestamp > e2.timestamp
              AND y.timestamp < e3.timestamp
        )
    WHERE e1.activity = 'Approve Purchase Requisition'
),
filtered_cases AS (
    SELECT sc.case_id
    FROM seq_cases sc
    JOIN event_log eco
      ON eco.case_id = sc.case_id
     AND eco.activity = 'Create Order'
    JOIN event_log ecn
      ON ecn.case_id = sc.case_id
     AND ecn.activity = 'Confirm Order'
    -- keep only those cases where within the qualifying succession
    -- the elapsed time between Create Order and Confirm Order is > 5 days
    WHERE ecn.timestamp > eco.timestamp
      AND ecn.timestamp - eco.timestamp > INTERVAL '5 days'
    -- (If multiple occurrences exist, this approach may match any pair;
    -- the "direct succession" constraints above restrict candidates already.)
    GROUP BY sc.case_id
)
SELECT el.*
FROM event_log el
WHERE el.case_id NOT IN (SELECT case_id FROM filtered_cases)
ORDER BY el.case_id, el.timestamp, el.activity;
```

This returns all events except those belonging to cases where **(1)** the activities occur in direct succession in the order *Approve Purchase Requisition  Create Order  Confirm Order* with no intervening events, **(2)** the time between *Create Order* and *Confirm Order* is **more than 5 days**, and then **(3)** those cases are excluded from the result.
