Below is the Python code that constructs an accepting Petri net for the insurance claim management process. The net includes all described activities, the loops for document resubmission (`RDS`) and indepth investigation (`II`), and defines an initial marking (token in the start place) and a final marking (token in the sink place).

```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="Insurance Claim Process")

# ---------- Places ----------
# Source (initial) and sink (final)
p_source = PetriNet.Place(name="source")
p_sink = PetriNet.Place(name="sink")

# Intermediate places
p_after_C = PetriNet.Place(name="p_after_C")
p_after_CR = PetriNet.Place(name="p_after_CR")      # before DV
p_after_DV = PetriNet.Place(name="p_after_DV")      # after DV, before FC / RDS
p_after_FC = PetriNet.Place(name="p_after_FC")      # after FC, before CA / II
p_after_CA = PetriNet.Place(name="p_after_CA")      # before AP
p_after_AP = PetriNet.Place(name="p_after_AP")      # before P
p_after_P = PetriNet.Place(name="p_after_P")        # before N
p_after_N = PetriNet.Place(name="p_after_N")        # before CL

# Add all places to the net
places = [p_source, p_sink,
          p_after_C, p_after_CR, p_after_DV, p_after_FC,
          p_after_CA, p_after_AP, p_after_P, p_after_N]
for p in places:
    net.places.add(p)

# ---------- Transitions ----------
# Each transition corresponds to an activity (label = activity name)
t_C   = PetriNet.Transition(name="C",   label="Insurance Claim")
t_CR  = PetriNet.Transition(name="CR",  label="Claim Registration")
t_DV  = PetriNet.Transition(name="DV",  label="Document Verification")
t_RDS = PetriNet.Transition(name="RDS", label="Re-document Submission")
t_FC  = PetriNet.Transition(name="FC",  label="Fraud Check")
t_II  = PetriNet.Transition(name="II",  label="In-depth Investigation")
t_CA  = PetriNet.Transition(name="CA",  label="Claim Assessment")
t_AP  = PetriNet.Transition(name="AP",  label="Approval")
t_P   = PetriNet.Transition(name="P",   label="Payment")
t_N   = PetriNet.Transition(name="N",   label="Notification")
t_CL  = PetriNet.Transition(name="CL",  label="Closure")

transitions = [t_C, t_CR, t_DV, t_RDS, t_FC, t_II, t_CA, t_AP, t_P, t_N, t_CL]
for t in transitions:
    net.transitions.add(t)

# ---------- Arcs ----------
# Main sequential flow
petri_utils.add_arc_from_to(p_source,    t_C,     net)
petri_utils.add_arc_from_to(t_C,        p_after_C, net)

petri_utils.add_arc_from_to(p_after_C,  t_CR,    net)
petri_utils.add_arc_from_to(t_CR,       p_after_CR, net)

# DV: from p_after_CR to p_after_DV
petri_utils.add_arc_from_to(p_after_CR, t_DV,    net)
petri_utils.add_arc_from_to(t_DV,       p_after_DV, net)

# Normal continuation after DV: to FC
petri_utils.add_arc_from_to(p_after_DV, t_FC,    net)
petri_utils.add_arc_from_to(t_FC,       p_after_FC, net)

# Loop: if documents incomplete  RDS, then back to before DV
petri_utils.add_arc_from_to(p_after_DV, t_RDS,   net)
petri_utils.add_arc_from_to(t_RDS,      p_after_CR, net)

# Normal continuation after FC: to CA
petri_utils.add_arc_from_to(p_after_FC, t_CA,    net)
petri_utils.add_arc_from_to(t_CA,       p_after_CA, net)

# Loop: if fraud doubts  II, then back to after DV (redo FC)
petri_utils.add_arc_from_to(p_after_FC, t_II,    net)
petri_utils.add_arc_from_to(t_II,       p_after_DV, net)

# Remaining sequential steps
petri_utils.add_arc_from_to(p_after_CA, t_AP,    net)
petri_utils.add_arc_from_to(t_AP,       p_after_AP, net)

petri_utils.add_arc_from_to(p_after_AP, t_P,     net)
petri_utils.add_arc_from_to(t_P,        p_after_P, net)

petri_utils.add_arc_from_to(p_after_P,  t_N,     net)
petri_utils.add_arc_from_to(t_N,        p_after_N, net)

petri_utils.add_arc_from_to(p_after_N,  t_CL,    net)
petri_utils.add_arc_from_to(t_CL,       p_sink,   net)

# ---------- Markings ----------
im = Marking()
im[p_source] = 1          # initial token in the start place

fm = Marking()
fm[p_sink] = 1            # final token in the sink place
```
