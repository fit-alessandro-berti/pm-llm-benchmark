```python
import pm4py
from pm4py.objects.powl.obj import StrictPartialOrder, OperatorPOWL, Transition, SilentTransition
from pm4py.objects.process_tree.obj import Operator

# 1. Basic activities
SR  = Transition(label="SR")   # Service Request verification
CS  = Transition(label="CS")   # Contract Setup

DA  = Transition(label="DA")   # Datacenter Allocation
ACI = Transition(label="ACI")  # Access Control Initialization

BNC = Transition(label="BNC")  # Base Network Configuration
DR  = Transition(label="D&R")  # Debug & Reconfigure

CD  = Transition(label="CD")   # Containerized Deployment
MD  = Transition(label="MD")   # Monolithic Deployment

QA  = Transition(label="QA")   # Quality Assurance
SA  = Transition(label="SA")   # Security Audit

GLA = Transition(label="GLA")  # Go-Live Approval

# 2. Loop for configuration: (BNC, D&R)
#    Execute BNC, then either exit loop or execute D&R and then BNC again.
config_loop = OperatorPOWL(operator=Operator.LOOP, children=[BNC, DR])

# 3. Exclusive choice for deployment: CD XOR MD
deploy_choice = OperatorPOWL(operator=Operator.XOR, children=[CD, MD])

# 4. Partial order for QA and SA (can be concurrent, both before GLA)
qa_sa_gla = StrictPartialOrder(nodes=[QA, SA, GLA])
qa_sa_gla.order.add_edge(QA, GLA)
qa_sa_gla.order.add_edge(SA, GLA)

# 5. Top-level partial order of the whole process
#    Nodes:
#      SR -> CS -> {DA, ACI} in parallel -> config_loop -> deploy_choice -> qa_sa_gla
root = StrictPartialOrder(nodes=[SR, CS, DA, ACI, config_loop, deploy_choice, qa_sa_gla])

# Ordering constraints
root.order.add_edge(SR, CS)               # Request verification before Contract Setup
root.order.add_edge(CS, DA)               # Contract Setup before Datacenter Allocation
root.order.add_edge(CS, ACI)              # Contract Setup before Access Control Initialization

# Both DA and ACI must complete before the configuration loop
root.order.add_edge(DA, config_loop)
root.order.add_edge(ACI, config_loop)

# After successful configuration, choose deployment method
root.order.add_edge(config_loop, deploy_choice)

# After deployment, run QA/SA (possibly in parallel) and then GLA
root.order.add_edge(deploy_choice, qa_sa_gla)
```
