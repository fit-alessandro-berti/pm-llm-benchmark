####  Query 1: Detect Out-of-Sequence Milestones
Checks if any expected predecessor activity occurred **after** its successor.
```sql
WITH case_pivot AS (
  SELECT case_id,
         MAX(CASE WHEN activity = 'Register Order' THEN timestamp END) AS ts_reg,
         MAX(CASE WHEN activity = 'Perform Credit Check' THEN timestamp END) AS ts_credit,
         MAX(CASE WHEN activity = 'Validate Stock' THEN timestamp END) AS ts_stock,
         MAX(CASE WHEN activity = 'Confirm Shipment' THEN timestamp END) AS ts_confirm,
         MAX(CASE WHEN activity = 'Ship Goods' THEN timestamp END) AS ts_ship,
         MAX(CASE WHEN activity = 'Issue Invoice' THEN timestamp END) AS ts_invoice,
         MAX(CASE WHEN activity = 'Receive Payment' THEN timestamp END) AS ts_pay
  FROM order_event_log
  GROUP BY case_id
)
SELECT case_id,
       CASE WHEN ts_credit > ts_ship THEN 'Credit Check after Shipment' END AS credit_ship_violation,
       CASE WHEN ts_stock > ts_ship THEN 'Stock Validation after Shipment' END AS stock_ship_violation,
       CASE WHEN ts_confirm > ts_ship THEN 'Confirmation after Shipment' END AS confirm_ship_violation,
       CASE WHEN ts_pay < ts_invoice THEN 'Payment before Invoice' END AS pay_invoice_violation
FROM case_pivot
WHERE (ts_credit > ts_ship) OR (ts_stock > ts_ship) OR (ts_confirm > ts_ship) OR (ts_pay < ts_invoice);
```

####  Query 4: Extract & Analyze Process Override Flags
Parses `additional_info` for manual/system override indicators and groups by case.
```sql
SELECT case_id, activity, timestamp, resource,
       CASE WHEN additional_info LIKE '%attempted_early=Y%' THEN 'Early Override'
            WHEN additional_info LIKE '%late_confirmation=Y%' THEN 'Late Confirmation'
            WHEN additional_info LIKE '%shipment_scheduled=N%' THEN 'Shipment Blocked Flag'
            ELSE 'Other' END AS override_type,
       additional_info
FROM order_event_log
```

####  Query 5: Detect State Inconsistencies (Blocked but Executed)
Finds cases where `shipment_scheduled=N` was logged, but `Ship Goods` still occurred.
```sql
WITH blocked_shipments AS (
  SELECT case_id, timestamp AS block_ts
  FROM order_event_log
  WHERE additional_info LIKE '%shipment_scheduled=N%'
),
actual_shipments AS (
  SELECT case_id, timestamp AS ship_ts
  FROM order_event_log
  WHERE activity = 'Ship Goods'
)
SELECT b.case_id, b.block_ts, a.ship_ts,
       a.ship_ts - b.block_ts AS time_between_block_and_ship
FROM blocked_shipments b
JOIN actual_shipments a ON b.case_id = a.case_id
WHERE a.ship_ts > b.block_ts;
```
