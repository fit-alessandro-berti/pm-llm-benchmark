```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils


# Create the Petri net
net = PetriNet(name="insurance_claim_management")


# -----------------------
# Places
# -----------------------

# Start / end places
p_start = PetriNet.Place("p_start")                 # New insurance claim arrives
p_registered = PetriNet.Place("p_registered")       # Claim registered
p_docs_verified = PetriNet.Place("p_docs_verified") # Documents verified / ready for fraud check
p_fraud_checked = PetriNet.Place("p_fraud_checked") # Fraud check completed
p_assessed = PetriNet.Place("p_assessed")           # Claim assessed
p_approved = PetriNet.Place("p_approved")           # Claim approved / no approval needed
p_paid = PetriNet.Place("p_paid")                   # Claim paid
p_notified = PetriNet.Place("p_notified")           # Customer notified
p_closed = PetriNet.Place("p_closed")               # Claim closed


# Places for loop handling / decisions
p_docs_incomplete = PetriNet.Place("p_docs_incomplete")
p_investigation_needed = PetriNet.Place("p_investigation_needed")


# Add places to the net
net.places.add(p_start)
net.places.add(p_registered)
net.places.add(p_docs_verified)
net.places.add(p_fraud_checked)
net.places.add(p_assessed)
net.places.add(p_approved)
net.places.add(p_paid)
net.places.add(p_notified)
net.places.add(p_closed)

net.places.add(p_docs_incomplete)
net.places.add(p_investigation_needed)


# -----------------------
# Transitions
# -----------------------

t_cr = PetriNet.Transition("t_cr", "Claim Registration")
t_dv = PetriNet.Transition("t_dv", "Document Verification")
t_rds = PetriNet.Transition("t_rds", "Re-Document Submission")
t_fc = PetriNet.Transition("t_fc", "Fraud Check")
t_ii = PetriNet.Transition("t_ii", "In-depth Investigation")
t_ca = PetriNet.Transition("t_ca", "Claim Assessment")
t_ap = PetriNet.Transition("t_ap", "Approval")
t_p = PetriNet.Transition("t_p", "Payment")
t_n = PetriNet.Transition("t_n", "Notification")
t_cl = PetriNet.Transition("t_cl", "Closure")


# Silent transitions for routing/decisions
# After Document Verification:
# - either documents are complete and process continues to Fraud Check
# - or documents are incomplete and Re-Document Submission is required
t_docs_ok = PetriNet.Transition("t_docs_ok", None)
t_docs_incomplete = PetriNet.Transition("t_docs_incomplete", None)

# After Fraud Check:
# - either no doubts and process continues to Claim Assessment
# - or doubts are raised and In-depth Investigation is required
t_fraud_ok = PetriNet.Transition("t_fraud_ok", None)
t_fraud_doubt = PetriNet.Transition("t_fraud_doubt", None)

# After Claim Assessment:
# - either approval is required
# - or approval is not required and process continues directly to payment
t_approval_required = PetriNet.Transition("t_approval_required", None)
t_no_approval_required = PetriNet.Transition("t_no_approval_required", None)


# Add transitions to the net
net.transitions.add(t_cr)
net.transitions.add(t_dv)
net.transitions.add(t_rds)
net.transitions.add(t_fc)
net.transitions.add(t_ii)
net.transitions.add(t_ca)
net.transitions.add(t_ap)
net.transitions.add(t_p)
net.transitions.add(t_n)
net.transitions.add(t_cl)

net.transitions.add(t_docs_ok)
net.transitions.add(t_docs_incomplete)
net.transitions.add(t_fraud_ok)
net.transitions.add(t_fraud_doubt)
net.transitions.add(t_approval_required)
net.transitions.add(t_no_approval_required)


# -----------------------
# Arcs
# -----------------------

# Main sequence
petri_utils.add_arc_from_to(p_start, t_cr, net)
petri_utils.add_arc_from_to(t_cr, p_registered, net)

petri_utils.add_arc_from_to(p_registered, t_dv, net)

# After Document Verification, either proceed or loop through Re-Document Submission
petri_utils.add_arc_from_to(t_dv, p_docs_verified, net)
petri_utils.add_arc_from_to(p_docs_verified, t_docs_ok, net)
petri_utils.add_arc_from_to(t_docs_ok, p_fraud_checked, net)

petri_utils.add_arc_from_to(p_docs_verified, t_docs_incomplete, net)
petri_utils.add_arc_from_to(t_docs_incomplete, p_docs_incomplete, net)
petri_utils.add_arc_from_to(p_docs_incomplete, t_rds, net)
petri_utils.add_arc_from_to(t_rds, p_registered, net)
# This sends the claim back to Document Verification after re-submission.


# Fraud Check
petri_utils.add_arc_from_to(p_fraud_checked, t_fc, net)
petri_utils.add_arc_from_to(t_fc, p_fraud_checked, net)

# After Fraud Check, either continue or perform in-depth investigation.
# The loop allows In-depth Investigation to occur multiple times before assessment.
petri_utils.add_arc_from_to(p_fraud_checked, t_fraud_ok, net)
petri_utils.add_arc_from_to(t_fraud_ok, p_assessed, net)

petri_utils.add_arc_from_to(p_fraud_checked, t_fraud_doubt, net)
petri_utils.add_arc_from_to(t_fraud_doubt, p_investigation_needed, net)
petri_utils.add_arc_from_to(p_investigation_needed, t_ii, net)
petri_utils.add_arc_from_to(t_ii, p_fraud_checked, net)


# Claim Assessment
petri_utils.add_arc_from_to(p_assessed, t_ca, net)
petri_utils.add_arc_from_to(t_ca, p_assessed, net)

# Approval routing
petri_utils.add_arc_from_to(p_assessed, t_approval_required, net)
petri_utils.add_arc_from_to(t_approval_required, p_approved, net)

petri_utils.add_arc_from_to(p_assessed, t_no_approval_required, net)
petri_utils.add_arc_from_to(t_no_approval_required, p_approved, net)

# Approval if required
petri_utils.add_arc_from_to(p_approved, t_ap, net)
petri_utils.add_arc_from_to(t_ap, p_approved, net)

# Payment
petri_utils.add_arc_from_to(p_approved, t_p, net)
petri_utils.add_arc_from_to(t_p, p_paid, net)

# Notification
petri_utils.add_arc_from_to(p_paid, t_n, net)
petri_utils.add_arc_from_to(t_n, p_notified, net)

# Closure
petri_utils.add_arc_from_to(p_notified, t_cl, net)
petri_utils.add_arc_from_to(t_cl, p_closed, net)


# -----------------------
# Initial and Final Marking
# -----------------------

im = Marking()
im[p_start] = 1

fm = Marking()
fm[p_closed] = 1
```

This constructs an accepting Petri net with:

- an initial token in `p_start`, representing a newly arrived insurance claim;
- a final marking with one token in `p_closed`, representing a closed and archived claim;
- a loop from `Document Verification` through `Re-Document Submission` when documents are incomplete;
- a loop from `Fraud Check` through `In-depth Investigation` when fraud doubts remain;
- an optional approval path before payment.
