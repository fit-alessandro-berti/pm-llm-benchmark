-- Identify claims where time from Receive to Approve is within 1 hour of the average (24-26 hours)

    ce2.timestamp AS approve_time,
    EXTRACT(EPOCH FROM (ce2.timestamp - ce1.timestamp)) AS time_to_approve_seconds
FROM
    claims c
JOIN
    claim_events ce1 ON c.claim_id = ce1.claim_id AND ce1.activity = 'R'

ORDER BY
    time_to_approve_seconds;
