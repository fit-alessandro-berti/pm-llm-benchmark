```sql
WITH numbered AS (
  SELECT *,
         ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS rn
  FROM event_log
),
bad_cases AS (
  SELECT DISTINCT n1.case_id
  FROM numbered n1
  JOIN numbered n2 ON n1.case_id = n2.case_id AND n2.rn = n1.rn + 1
  JOIN numbered n3 ON n1.case_id = n3.case_id AND n3.rn = n1.rn + 2
  WHERE n1.activity = 'Approve Purchase Requisition'
    AND n2.activity = 'Create Order'
    AND n3.activity = 'Confirm Order'
    AND n3.timestamp - n2.timestamp > INTERVAL '5' DAY
)
SELECT *
FROM event_log
WHERE case_id NOT IN (SELECT case_id FROM bad_cases)
ORDER BY case_id, timestamp;
```
