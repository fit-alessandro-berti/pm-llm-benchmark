```python
def rule():
    return {"support": 1.0, "confidence": 1.0}


activities = ["IG", "DD", "TFC", "CE", "PC", "LT", "UT", "AG", "MP", "FL"]

# Core ordering dependencies.
# A pair ("A", "B") means that activity A constrains activity B.
mandatory_dependencies = [
    ("IG", "DD"),      # Idea Generation before Design Draft
    ("DD", "TFC"),     # Design Draft before Technical Feasibility Check
    ("DD", "CE"),      # Design Draft before Cost Evaluation
    ("TFC", "PC"),     # Technical Feasibility Check before Prototype Creation
    ("CE", "PC"),      # Cost Evaluation before Prototype Creation
    ("PC", "LT"),      # Prototype Creation before Laboratory Testing
    ("PC", "UT"),      # Prototype Creation before User Testing
    ("LT", "AG"),      # Laboratory Testing before Approval Gate
    ("UT", "AG"),      # User Testing before Approval Gate
    ("AG", "MP"),      # Approval Gate before Marketing Plan
    ("MP", "FL"),      # Marketing Plan before Final Launch
]

# Activities that should occur together in a successful launch case.
coexistence_pairs = [
    ("TFC", "CE"),
    ("LT", "UT"),
    ("AG", "MP"),
    ("MP", "FL"),
]

# Strict immediate-successor constraints.
direct_dependencies = [
    ("IG", "DD"),
    ("AG", "MP"),
    ("MP", "FL"),
]

# After final launch, no earlier process activity should happen.
no_after_launch = [
    ("FL", a) for a in activities if a != "FL"
]

# Prevent important stage-skipping.
no_direct_skip = [
    ("IG", "TFC"),
    ("IG", "CE"),
    ("DD", "PC"),
    ("TFC", "LT"),
    ("TFC", "UT"),
    ("CE", "LT"),
    ("CE", "UT"),
    ("PC", "AG"),
    ("LT", "MP"),
    ("UT", "MP"),
    ("AG", "FL"),
]

declare_model = {
    "existence": {
        a: rule() for a in activities
    },

    "absence": {},

    "exactly_one": {
        a: rule() for a in activities
    },

    "init": {
        "IG": rule()
    },

    "responded_existence": {
        pair: rule() for pair in mandatory_dependencies
    },

    "coexistence": {
        pair: rule() for pair in coexistence_pairs
    },

    "response": {
        pair: rule() for pair in mandatory_dependencies
    },

    "precedence": {
        pair: rule() for pair in mandatory_dependencies
    },

    "succession": {
        pair: rule() for pair in mandatory_dependencies
    },

    "altresponse": {
        pair: rule() for pair in mandatory_dependencies
    },

    "altprecedence": {
        pair: rule() for pair in mandatory_dependencies
    },

    "altsuccession": {
        pair: rule() for pair in mandatory_dependencies
    },

    "chainresponse": {
        pair: rule() for pair in direct_dependencies
    },

    "chainprecedence": {
        pair: rule() for pair in direct_dependencies
    },

    "chainsuccession": {
        pair: rule() for pair in direct_dependencies
    },

    "noncoexistence": {},

    "nonsuccession": {
        pair: rule() for pair in no_after_launch
    },

    "nonchainsuccession": {
        pair: rule() for pair in no_direct_skip
    },
}
```

This model assumes a successful “happy-path” launch case where each listed activity occurs exactly once. The technical feasibility check and cost evaluation are both required before prototype creation, but their relative order is left flexible. Likewise, laboratory testing and user testing are both required before the approval gate.
