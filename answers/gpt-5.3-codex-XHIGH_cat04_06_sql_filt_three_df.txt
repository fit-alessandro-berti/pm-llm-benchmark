```sql
WITH ordered_events AS (
    SELECT
        case_id,
        activity,
        "timestamp" AS ts,
        LEAD(activity, 1)   OVER (PARTITION BY case_id ORDER BY "timestamp") AS next_act_1,
        LEAD(activity, 2)   OVER (PARTITION BY case_id ORDER BY "timestamp") AS next_act_2,
        LEAD("timestamp",1) OVER (PARTITION BY case_id ORDER BY "timestamp") AS next_ts_1, -- Create Order ts
        LEAD("timestamp",2) OVER (PARTITION BY case_id ORDER BY "timestamp") AS next_ts_2  -- Confirm Order ts
    FROM event_log
),
cases_to_exclude AS (
    SELECT DISTINCT case_id
    FROM ordered_events
    WHERE activity   = 'Approve Purchase Requisition'
      AND next_act_1 = 'Create Order'
      AND next_act_2 = 'Confirm Order'
      AND (next_ts_2 - next_ts_1) > INTERVAL '5 days'
)
SELECT e.*
FROM event_log e
WHERE NOT EXISTS (
    SELECT 1
    FROM cases_to_exclude c
    WHERE c.case_id = e.case_id
)
ORDER BY e.case_id, e."timestamp";
```

If your log can contain multiple events with the exact same timestamp in a case, add a tie-breaker column (e.g., `event_id`) to every `ORDER BY` in the window functions and final sort.
