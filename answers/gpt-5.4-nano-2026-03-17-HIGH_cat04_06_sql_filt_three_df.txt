```sql
WITH ordered AS (
  SELECT
    e.*,
    ROW_NUMBER() OVER (
      PARTITION BY case_id
      ORDER BY timestamp
    ) AS rn
  FROM event_log e
),
cases_to_exclude AS (
  SELECT DISTINCT o1.case_id
  FROM ordered o1
  JOIN ordered o2
    ON o2.case_id = o1.case_id AND o2.rn = o1.rn + 1
  JOIN ordered o3
    ON o3.case_id = o1.case_id AND o3.rn = o1.rn + 2
  WHERE o1.activity = 'Approve Purchase Requisition'
    AND o2.activity = 'Create Order'
    AND o3.activity = 'Confirm Order'
    AND (
      EXTRACT(EPOCH FROM (o3.timestamp - o2.timestamp)) / 86400.0
    ) > 5
)
SELECT el.*
FROM event_log el
WHERE NOT EXISTS (
  SELECT 1
  FROM cases_to_exclude c
  WHERE c.case_id = el.case_id
)
ORDER BY el.case_id, el.timestamp;
```
