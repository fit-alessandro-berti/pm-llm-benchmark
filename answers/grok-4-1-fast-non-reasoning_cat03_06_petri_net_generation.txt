```python
from pm4py.objects.petri_net.obj import PetriNet, Marking
from pm4py.objects.petri_net.utils import petri_utils

# Create the Petri net
net = PetriNet(name="Insurance_Claim_Management")

# Define places (states between activities)
start = PetriNet.Place(name="start")
p_cr = PetriNet.Place(name="p_cr")  # after Claim Registration
p_dv = PetriNet.Place(name="p_dv")  # after Document Verification
p_rds = PetriNet.Place(name="p_rds")  # after Re-Document Submission
p_fc = PetriNet.Place(name="p_fc")  # after Fraud Check
p_ii = PetriNet.Place(name="p_ii")  # after In-depth Investigation
p_ca = PetriNet.Place(name="p_ca")  # after Claim Assessment
p_ap = PetriNet.Place(name="p_ap")  # after Approval
p_p = PetriNet.Place(name="p_p")    # after Payment
p_n = PetriNet.Place(name="p_n")    # after Notification
end = PetriNet.Place(name="end")    # final state

# Add all places to the net
net.places.add(start)
net.places.add(p_cr)
net.places.add(p_dv)
net.places.add(p_rds)
net.places.add(p_fc)
net.places.add(p_ii)
net.places.add(p_ca)
net.places.add(p_ap)
net.places.add(p_p)
net.places.add(p_n)
net.places.add(end)

# Define transitions with activity labels
cr = PetriNet.Transition(name="CR", label="Claim Registration")
dv = PetriNet.Transition(name="DV", label="Document Verification")
rds = PetriNet.Transition(name="RDS", label="Re-Document Submission")
fc = PetriNet.Transition(name="FC", label="Fraud Check")
ii = PetriNet.Transition(name="II", label="In-depth Investigation")
ca = PetriNet.Transition(name="CA", label="Claim Assessment")
ap = PetriNet.Transition(name="AP", label="Approval")
p = PetriNet.Transition(name="P", label="Payment")
n = PetriNet.Transition(name="N", label="Notification")
cl = PetriNet.Transition(name="CL", label="Closure")

# Add all transitions to the net
net.transitions.add(cr)
net.transitions.add(dv)
net.transitions.add(rds)
net.transitions.add(fc)
net.transitions.add(ii)
net.transitions.add(ca)
net.transitions.add(ap)
net.transitions.add(p)
net.transitions.add(n)
net.transitions.add(cl)

# Add arcs to define the flow
# Start -> Claim Registration -> Document Verification
petri_utils.add_arc_from_to(start, cr, net)
petri_utils.add_arc_from_to(cr, p_cr, net)
petri_utils.add_arc_from_to(p_cr, dv, net)

# Document Verification: normal flow to Fraud Check or loop to Re-Document Submission
petri_utils.add_arc_from_to(p_dv, fc, net)
petri_utils.add_arc_from_to(dv, p_dv, net)

# Re-Document Submission loop: back to Document Verification
petri_utils.add_arc_from_to(p_rds, dv, net)
petri_utils.add_arc_from_to(rds, p_rds, net)

# From Document Verification, option to go to Re-Document Submission
petri_utils.add_arc_from_to(p_dv, rds, net)

# Fraud Check: normal flow to Claim Assessment or loop to In-depth Investigation
petri_utils.add_arc_from_to(p_fc, ca, net)
petri_utils.add_arc_from_to(fc, p_fc, net)

# In-depth Investigation loop: back to Fraud Check
petri_utils.add_arc_from_to(p_ii, fc, net)
petri_utils.add_arc_from_to(ii, p_ii, net)

# From Fraud Check, option to go to In-depth Investigation
petri_utils.add_arc_from_to(p_fc, ii, net)

# Claim Assessment -> Approval -> Payment
petri_utils.add_arc_from_to(p_ca, ap, net)
petri_utils.add_arc_from_to(ca, p_ca, net)
petri_utils.add_arc_from_to(p_ap, p, net)
petri_utils.add_arc_from_to(ap, p_ap, net)

# Payment -> Notification -> Closure -> End
petri_utils.add_arc_from_to(p_p, n, net)
petri_utils.add_arc_from_to(p, p_p, net)
petri_utils.add_arc_from_to(p_n, cl, net)
petri_utils.add_arc_from_to(n, p_n, net)
petri_utils.add_arc_from_to(cl, end, net)

# Define initial and final markings
im = Marking()
im[start] = 1

fm = Marking()
fm[end] = 1
```

This accepting Petri net correctly models the insurance claim management process with:

1. **Main sequence**: start  CR  DV  FC  CA  AP  P  N  CL  end
2. **Document verification loop**: DV  RDS  DV (can repeat if documents incomplete)
3. **Fraud check loop**: FC  II  FC (can repeat if in-depth investigation needed)
4. **Proper place-token management**: Each transition consumes from input places and produces to output places
5. **Initial marking**: 1 token in `start` place (new claim arrives)
6. **Final marking**: 1 token in `end` place (claim closed and archived)

The net allows for multiple iterations of RDS and II as specified, while ensuring the process always ends with Payment  Notification  Closure when all conditions are satisfied.
