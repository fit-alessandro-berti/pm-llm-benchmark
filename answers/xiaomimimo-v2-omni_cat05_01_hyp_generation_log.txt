```sql
WITH normal_flow AS (
    SELECT unnest(ARRAY[
        'Register Order',
        'Perform Credit Check',
        'Validate Stock',
        'Confirm Shipment',
        'Ship Goods',
        'Issue Invoice',
        'Receive Payment'
    ]) AS activity,
    generate_subscripts(ARRAY[
        'Register Order',
        'Perform Credit Check',
        'Validate Stock',
        'Confirm Shipment',
        'Ship Goods',
        'Issue Invoice',
        'Receive Payment'
    ], 1) AS normal_sequence
),
activity_sequence AS (
    SELECT
        case_id,
        activity,
        timestamp,
        ROW_NUMBER() OVER (PARTITION BY case_id ORDER BY timestamp) AS actual_sequence
    FROM order_event_log
)
SELECT
    a.case_id,
    a.activity AS actual_activity,
    a.actual_sequence,
    n.activity AS expected_activity_at_this_position,
    n.normal_sequence AS expected_sequence
FROM activity_sequence a
LEFT JOIN normal_flow n ON a.actual_sequence = n.normal_sequence
WHERE n.activity IS NULL OR a.activity != n.activity
ORDER BY a.case_id, a.actual_sequence;
```

```sql
SELECT
    o.case_id,
    o.order_type,
    o.order_value,
    COUNT(DISTINCT e.activity) AS recorded_activity_count,
    MAX(CASE WHEN e.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) AS has_credit_check,
    MAX(CASE WHEN e.activity = 'Validate Stock' THEN 1 ELSE 0 END) AS has_stock_validation
FROM orders o
JOIN order_event_log e ON o.case_id = e.case_id
GROUP BY o.case_id, o.order_type, o.order_value
HAVING
    MAX(CASE WHEN e.activity = 'Perform Credit Check' THEN 1 ELSE 0 END) = 0
    OR
    MAX(CASE WHEN e.activity = 'Validate Stock' THEN 1 ELSE 0 END) = 0
ORDER BY o.order_value DESC;
```

```sql
SELECT
    e.case_id,
    e.activity,
    e.resource,
    r.role,
    r.department,
    CASE
        WHEN e.activity = 'Perform Credit Check' AND r.department != 'Finance' THEN ''
        WHEN e.activity = 'Validate Stock' AND r.department != 'Logistics' THEN ''
        WHEN e.activity = 'Ship Goods' AND r.role NOT LIKE '%Warehouse%' THEN ''
        ELSE ''
    END AS compliance_flag
FROM order_event_log e
LEFT JOIN resources r ON e.resource = r.resource_id
WHERE e.activity IN ('Perform Credit Check', 'Validate Stock', 'Ship Goods')
ORDER BY compliance_flag, e.case_id;
```

```sql
WITH order_timeline AS (
    SELECT
        e.case_id,
        o.customer_id,
        o.order_type,
        o.order_value,
        MIN(CASE WHEN e.activity = 'Register Order' THEN e.timestamp END) AS register_time,
        MIN(CASE WHEN e.activity = 'Receive Payment' THEN e.timestamp END) AS payment_time,
        MIN(CASE WHEN e.activity = 'Ship Goods' THEN e.timestamp END) AS ship_time
    FROM order_event_log e
    JOIN orders o ON e.case_id = o.case_id
    GROUP BY e.case_id, o.customer_id, o.order_type, o.order_value
)
SELECT
    case_id,
    customer_id,
    order_type,
    order_value,
    register_time,
    payment_time,
    ship_time,
    EXTRACT(EPOCH FROM (payment_time - register_time))/60 AS minutes_between_register_and_payment,
    CASE
        WHEN payment_time < ship_time THEN ''
        ELSE ''
    END AS payment_delivery_pattern
FROM order_timeline
WHERE payment_time IS NOT NULL
ORDER BY minutes_between_register_and_payment;
```
